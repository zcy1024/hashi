43pub async fn query_transaction_risk_level(
44    client: &Client,
45    api_key: &str,
46    tx_hash: &str,
47    blockchain: &str,
48) -> Result<i64, HashiScreenerError> {
49    let url = format!("{}/api/v4.2/transactions/", MERKLE_SCIENCE_BASE_URL);
50    query_risk_level(client, api_key, &url, tx_hash, blockchain).await
51}

63fn retry_policy() -> ExponentialBuilder {
64    // Recommended retry policy from merkle science docs:
65    // https://docs.merklescience.com/reference/retry-policy
66    ExponentialBuilder::default()
67        .with_min_delay(Duration::from_secs(30))
68        .with_max_delay(Duration::from_secs(3000))
69        .with_max_times(10)
70}

72#[derive(Debug)]
73enum ApiError {
74    Retryable(String),
75    Permanent(HashiScreenerError),
76}
77
78async fn query_risk_level(
79    client: &Client,
80    api_key: &str,
81    url: &str,
82    identifier: &str,
83    blockchain: &str,
84) -> Result<i64, HashiScreenerError> {
85    let request_body = MerkleScienceRequest {
86        identifier: identifier.to_string(),
87        blockchain: blockchain.to_string(),
88    };
89
90    let result = (|| async {
91        let response = client
92            .post(url)
93            .header("Accept", "application/json")
94            .header("Content-Type", "application/json")
95            .header("X-API-KEY", api_key)
96            .json(&request_body)
97            .send()
98            .await
99            .map_err(|e| ApiError::Retryable(format!("MerkleScience API request failed: {}", e)))?;
100
101        if response.status().is_success() {
102            let merkle_response: MerkleScienceResponse = response.json().await.map_err(|e| {
103                ApiError::Permanent(HashiScreenerError::InternalError(format!(
104                    "Failed to deserialize response: {}",
105                    e
106                )))
107            })?;
108
109            return merkle_response.risk_level.ok_or(ApiError::Permanent(
110                HashiScreenerError::InternalError(
111                    "Missing risk_level field in response".to_string(),
112                ),
113            ));
114        }
115
116        let status = response.status().as_u16();
117        let body = response
118            .text()
119            .await
120            .unwrap_or_else(|_| "Unknown error".to_string());
121
122        // Transaction not Indexed (400) , Rate limits (429) and server errors (5xx) are retryable.
123        if status == 400 || status == 429 || status >= 500 {
124            return Err(ApiError::Retryable(format!("HTTP {status}: {body}",)));
125        }
126
127        // Other client errors (4xx) are not retryable.
128        Err(ApiError::Permanent(HashiScreenerError::InternalError(
129            format!("HTTP {}: {}", status, body),
130        )))
131    })
132    .retry(retry_policy())
133    .when(|e| matches!(e, ApiError::Retryable(_)))
134    .notify(|err, dur| {
135        tracing::warn!(
136            error = ?err,
137            retry_after = ?dur,
138            url = %url,
139            identifier = %identifier,
140            blockchain = %blockchain,
141            "MerkleScience API - Retrying Request",
142        );
143    })
144    .await;
145
146    match result {
147        Ok(risk_level) => Ok(risk_level),
148        Err(ApiError::Permanent(e)) => {
149            tracing::error!(
150                error = %e,
151                url = %url,
152                identifier = %identifier,
153                blockchain = %blockchain,
154                "MerkleScience API - Permanent Error",
155            );
156            Err(e)
157        }
158        Err(ApiError::Retryable(msg)) => {
159            tracing::error!(
160                error = %msg,
161                url = %url,
162                identifier = %identifier,
163                blockchain = %blockchain,
164                "MerkleScience API - Retries Exhausted",
165            );
166            Err(HashiScreenerError::InternalError(msg))
167        }
168    }
169}
