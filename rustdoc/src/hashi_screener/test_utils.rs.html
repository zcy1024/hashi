4//! Test utilities for spinning up a mock screener gRPC server.
5
6use std::collections::HashSet;
7use std::net::SocketAddr;
8use std::sync::Arc;
9
10use hashi_types::proto::screener::ApproveRequest;
11use hashi_types::proto::screener::ApproveResponse;
12use hashi_types::proto::screener::screener_service_server::ScreenerService;
13use hashi_types::proto::screener::screener_service_server::ScreenerServiceServer;
14use sui_futures::service::Service;
15use tonic::Request;
16use tonic::Response;
17use tonic::Status;
18use tonic::transport::Server;
19
20/// A mock screener service that auto-approves all requests.
21pub struct MockScreenerService;
22
23#[tonic::async_trait]
24impl ScreenerService for MockScreenerService {
25    async fn approve(
26        &self,
27        _request: Request<ApproveRequest>,
28    ) -> Result<Response<ApproveResponse>, Status> {
29        Ok(Response::new(ApproveResponse { approved: true }))
30    }
31}
32
33/// A configurable mock that rejects requests involving blocked addresses.
34pub struct ConfigurableMockScreenerService {
35    pub blocked_addresses: Arc<HashSet<String>>,
36}
37
38#[tonic::async_trait]
39impl ScreenerService for ConfigurableMockScreenerService {
40    async fn approve(
41        &self,
42        request: Request<ApproveRequest>,
43    ) -> Result<Response<ApproveResponse>, Status> {
44        let req = request.into_inner();
45        let approved = !self.blocked_addresses.contains(&req.destination_address);
46        Ok(Response::new(ApproveResponse { approved }))
47    }
48}
49
50/// Start a mock screener gRPC server that auto-approves all requests.
51/// Returns the address and a handle to abort the server.
52pub async fn start_mock_screener_server() -> (SocketAddr, Service) {
53    start_screener_server_impl(ScreenerServiceServer::new(MockScreenerService)).await
54}
55
56/// Start a configurable mock screener gRPC server that rejects blocked addresses.
57pub async fn start_configurable_mock_screener_server(
58    blocked_addresses: HashSet<String>,
59) -> (SocketAddr, Service) {
60    let service = ConfigurableMockScreenerService {
61        blocked_addresses: Arc::new(blocked_addresses),
62    };
63    start_screener_server_impl(ScreenerServiceServer::new(service)).await
64}
65
66async fn start_screener_server_impl(
67    service: ScreenerServiceServer<impl ScreenerService>,
68) -> (SocketAddr, Service) {
69    let listener = tokio::net::TcpListener::bind("127.0.0.1:0").await.unwrap();
70    let addr = listener.local_addr().unwrap();
71
72    let service = Service::new().spawn_aborting(async move {
73        Server::builder()
74            .add_service(service)
75            .serve_with_incoming(tokio_stream::wrappers::TcpListenerStream::new(listener))
76            .await?;
77        Ok(())
78    });
79
80    // Brief pause to let the server start accepting connections
81    tokio::time::sleep(std::time::Duration::from_millis(50)).await;
82
83    (addr, service)
84}
