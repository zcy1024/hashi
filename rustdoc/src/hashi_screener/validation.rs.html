4use std::str::FromStr;
5
6use bitcoin::Address as BitcoinAddress;
7use bitcoin::Txid;
8use bitcoin::address::NetworkUnchecked;
9use sui_sdk_types::Address as SuiAddress;
10use sui_sdk_types::Digest;
11
12use crate::error::HashiScreenerError;
13
14pub fn validate_btc_tx_hash(hash: &str) -> Result<Txid, HashiScreenerError> {
15    Txid::from_str(hash).map_err(|_| {
16        HashiScreenerError::ValidationError(format!("invalid Bitcoin transaction hash: '{}'", hash))
17    })
18}
19
20pub fn validate_btc_address(
21    address: &str,
22) -> Result<BitcoinAddress<NetworkUnchecked>, HashiScreenerError> {
23    BitcoinAddress::from_str(address).map_err(|_| {
24        HashiScreenerError::ValidationError(format!("invalid Bitcoin address: '{}'", address))
25    })
26}
