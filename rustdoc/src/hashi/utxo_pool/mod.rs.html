4//! UTXO pool management and coin selection for Bitcoin withdrawal transactions.
5//!
6//! This module provides a coin selection algorithm that simultaneously:
7//! - Selects UTXOs to fund one or more withdrawal requests in a single transaction.
8//! - Amortises UTXO pool maintenance by aggressively consolidating inputs back
9//!   into a small target number of UTXOs via a change output.
10//!
11//! The miner fee is split equally among the selected withdrawal requests and
12//! deducted from each request's output amount. The fee must not exceed a
13//! configurable per-request cap (`CoinSelectionParams::max_fee_per_request`),
14//! and is sourced entirely from the request amounts. When spending
15//! unconfirmed change (CPFP), the per-request deduction also includes any
16//! deficit needed to bring the ancestor package fee rate up to the target.
17//! The fund balance decreases by `sum(request.amounts)` for confirmed inputs
18//! and by `sum(request.amounts) + cpfp_deficit` when ancestor boosting is
19//! needed.
20

38/// Base weight of a segwit transaction input (non-witness portion).
39///
40/// nSequence(4) + prevout_txid(32) + prevout_vout(4) + script_sig_len(1)
41///   = 41 bytes × 4 = 164 WU
42const TXIN_BASE_WEIGHT: Weight = Weight::from_wu(164);
43
44/// Base weight of a transaction output (amount + script-length varint).
45///
46/// amount(8) + script_len_varint(1) = 9 bytes × 4 = 36 WU.
47/// The scriptPubKey weight is added separately.
48const TXOUT_BASE_WEIGHT: Weight = Weight::from_wu(36);
49
50/// Minimum value (sat) for a P2PKH or P2SH output to be above the dust
51/// relay threshold at Bitcoin's default dust relay fee of 3 sat/vByte.
52/// This is the legacy dust limit and the highest of the three thresholds.
53#[allow(dead_code)]
54const DUST_RELAY_MIN_VALUE: u64 = 546;
55
56/// Minimum value (sat) for a P2TR (or P2WSH) output to be above the dust
57/// relay threshold at Bitcoin's default dust relay fee of 3 sat/vByte.
58/// Outputs below this value will not be relayed by default Bitcoin nodes.
59pub const TR_DUST_RELAY_MIN_VALUE: u64 = 330;
60
61/// Minimum value (sat) for a P2WPKH output to be above the dust relay
62/// threshold at Bitcoin's default dust relay fee of 3 sat/vByte.
63const WPKH_DUST_RELAY_MIN_VALUE: u64 = 294;

65// ── Spend Path ───────────────────────────────────────────────────────────────
66
67/// Describes the spend path for a UTXO, used to calculate the witness
68/// satisfaction weight and thus the full input weight contribution to a
69/// transaction.
70#[derive(Clone, Debug)]
71pub enum SpendPath {
72    /// Taproot script-path 2-of-2 multisig spend.
73    ///
74    /// Witness items:
75    ///   items_count(1) + sig1_len(1) + sig1(64) + sig2_len(1) + sig2(64)
76    ///     + script_len(1) + script(68) + control_block_len(1) + control_block(33)
77    ///   = 234 WU
78    TaprootScriptPath2of2,
79
80    /// Taproot key-path spend (single x-only Schnorr signature).
81    ///
82    /// Witness items:
83    ///   items_count(1) + sig_len(1) + sig(64) = 66 WU
84    TaprootKeyPath,
85
86    /// Custom witness satisfaction weight, for non-standard spend paths.
87    Custom(Weight),
88}
89
90impl SpendPath {
91    /// Returns the witness-only satisfaction weight.
92    pub fn satisfaction_weight(&self) -> Weight {
93        match self {
94            SpendPath::TaprootScriptPath2of2 => Weight::from_wu(234),
95            SpendPath::TaprootKeyPath => Weight::from_wu(66),
96            SpendPath::Custom(w) => *w,
97        }
98    }
99
100    /// Returns the total input weight: `TXIN_BASE_WEIGHT` (164 WU) + satisfaction weight.
101    pub fn input_weight(&self) -> Weight {
102        TXIN_BASE_WEIGHT
103            .checked_add(self.satisfaction_weight())
104            .expect("input weight overflow")
105    }
106}

108// ── UTXO Status ──────────────────────────────────────────────────────────────
109
110/// An ancestor transaction in an unconfirmed UTXO's chain.
111///
112/// Each entry represents a transaction that must confirm before the
113/// UTXO is fully settled. The chain is ordered from the UTXO's
114/// creating transaction (index 0) back to the oldest unconfirmed
115/// ancestor.
116#[derive(Clone, Debug)]
117pub struct AncestorTx {
118    /// Number of confirmations. `0` means the transaction is in the
119    /// mempool only (not yet included in a block). Values `1..N`
120    /// mean it has been included but has not yet reached the
121    /// finality threshold.
122    pub confirmations: u32,
123    /// The total weight of this transaction.
124    pub tx_weight: Weight,
125    /// The total fee paid by this transaction in satoshis.
126    pub tx_fee: u64,
127}
128
129/// The confirmation status of a UTXO.
130#[derive(Clone, Debug)]
131pub enum UtxoStatus {
132    /// The UTXO and all of its ancestors have reached the finality
133    /// threshold (e.g., 6 confirmations). Safe to spend without any
134    /// concern for reorgs or rebroadcasting.
135    Confirmed,
136
137    /// The UTXO has an unconfirmed or insufficiently confirmed
138    /// ancestor chain.
139    ///
140    /// `chain[0]` is the transaction that created this UTXO.
141    /// `chain[1]` is its parent (if unconfirmed), and so on back to
142    /// the oldest unconfirmed ancestor. The chain is empty only for
143    /// `Confirmed` UTXOs (which use the `Confirmed` variant instead).
144    ///
145    /// This information is used to:
146    /// - Compute the **package fee rate** for CPFP: the effective
147    ///   rate of a new transaction spending this UTXO is
148    ///   `(sum(chain.tx_fee) + new_fee) / (sum(chain.tx_weight) + new_weight)`.
149    /// - Enforce a **maximum chain depth** policy (e.g., never chain
150    ///   deeper than 3 unconfirmed transactions).
151    /// - Determine how much extra fee a child transaction must pay
152    ///   to boost underpaying ancestors.
153    Pending {
154        /// The ancestor chain, from this UTXO's creating transaction
155        /// back to the oldest unconfirmed ancestor.
156        chain: Vec<AncestorTx>,
157    },
158}
159
160impl UtxoStatus {
161    /// Returns the full depth of the ancestor chain (both mempool and
162    /// insufficiently confirmed). `0` for confirmed UTXOs,
163    /// `chain.len()` for pending ones.
164    pub fn chain_depth(&self) -> usize {
165        match self {
166            UtxoStatus::Confirmed => 0,
167            UtxoStatus::Pending { chain } => chain.len(),
168        }
169    }
170
171    /// Returns the number of ancestors still in the mempool (0
172    /// confirmations). This is the depth that matters for Bitcoin
173    /// relay policy limits. Ancestors with 1+ confirmations are
174    /// already in blocks and don't count toward the mempool chain
175    /// limit.
176    pub fn mempool_chain_depth(&self) -> usize {
177        match self {
178            UtxoStatus::Confirmed => 0,
179            UtxoStatus::Pending { chain } => chain.iter().filter(|a| a.confirmations == 0).count(),
180        }
181    }
182
183    /// Returns the total weight of all unconfirmed ancestors
184    /// (transactions with 0 confirmations).
185    pub fn unconfirmed_ancestor_weight(&self) -> Weight {
186        match self {
187            UtxoStatus::Confirmed => Weight::ZERO,
188            UtxoStatus::Pending { chain } => chain
189                .iter()
190                .filter(|a| a.confirmations == 0)
191                .map(|a| a.tx_weight)
192                .sum(),
193        }
194    }
195
196    /// Returns the total fee paid by all unconfirmed ancestors
197    /// (transactions with 0 confirmations).
198    pub fn unconfirmed_ancestor_fee(&self) -> u64 {
199        match self {
200            UtxoStatus::Confirmed => 0,
201            UtxoStatus::Pending { chain } => chain
202                .iter()
203                .filter(|a| a.confirmations == 0)
204                .map(|a| a.tx_fee)
205                .sum(),
206        }
207    }
208}

228/// A withdrawal request candidate for inclusion in a Bitcoin transaction.
229///
230/// This is the coin-selection view of a withdrawal; it contains only the

252// ── Algorithm Parameters ─────────────────────────────────────────────────────
253
254/// Tuning parameters for the coin selection algorithm.
255#[derive(Clone, Debug)]
256pub struct CoinSelectionParams {
257    /// Upper bound on transaction weight. The algorithm will not propose a
258    /// transaction that exceeds this limit.
259    pub max_tx_weight: Weight,
260
261    /// Hard cap on the total number of transaction inputs, covering both the
262    /// primary inputs selected in Step 2 and any consolidation inputs added in
263    /// Step 3.
264    pub max_inputs: usize,
265
266    /// Maximum number of withdrawal requests that may be batched into a single
267    /// transaction.
268    pub max_withdrawal_requests: usize,
269
270    /// Maximum miner fee in satoshis that may be deducted from any single
271    /// request's amount. The batch is rejected if `total_fee / N` would exceed
272    /// this value for the selected set of N requests. Sourced from
273    /// [`crate::onchain::types::Config::worst_case_network_fee`], which is
274    /// derived from `high_fee_rate_threshold` and the worst-case transaction
275    /// shape.
276    pub max_fee_per_request: u64,
277
278    /// Absolute minimum fee rate (floor). The actual fee rate passed to
279    /// [`select_coins`] should be clamped to at least this value before the
280    /// call; it is also used as the lower bound when computing the available
281    /// consolidation budget.
282    pub min_fee_rate: FeeRate,
283
284    /// Expected long-term average fee rate (e.g., 10 sat/vB, matching
285    /// Bitcoin Core's `-consolidatefeerate` default). Divides the fee
286    /// environment into two regimes:
287    ///
288    /// - **Below this rate:** consolidation is cheap. The algorithm
289    ///   aggressively pulls in extra inputs (up to `max_inputs`) to
290    ///   reduce the UTXO set while fees are favourable.
291    /// - **At or above this rate:** input selection is aggressively
292    ///   minimized. Consolidation is capped at `input_budget` extra
293    ///   inputs per request, and no pool consolidation is performed
294    ///   beyond what is strictly required to fund the withdrawals.
295    pub long_term_fee_rate: FeeRate,
296
297    /// The fee rate threshold at which the algorithm aggressively
298    /// minimizes inputs and performs no pool consolidation.
299    /// `max_fee_per_request` is derived from this rate applied to the
300    /// worst-case transaction shape. Also used as the reference point
301    /// for scaling the consolidation budget: `max_fee_per_request` is
302    /// proportionally scaled from this rate to `long_term_fee_rate`
303    /// when computing the low-fee consolidation budget.
304    pub high_fee_rate_threshold: FeeRate,
305
306    /// Maximum number of extra consolidation inputs per request when the
307    /// fee rate is at or above `long_term_fee_rate`. At high fee rates,
308    /// consolidation is capped at `input_budget × N` extra inputs; below
309    /// the long-term rate the cap is lifted (bounded only by `max_inputs`).
310    /// Matches the on-chain `input_budget` config parameter.
311    pub input_budget: usize,
312
313    /// Maximum number of mempool-only (0-confirmation) ancestors a
314    /// UTXO may have and still be eligible as an input. This
315    /// constrains how deep the chain of unconfirmed transactions can
316    /// grow, staying within Bitcoin's relay policy limits.
317    ///
318    /// Only ancestors with 0 confirmations count toward this limit.
319    /// Ancestors already included in a block (1+ confirmations, even
320    /// if below the 6-confirmation finality threshold) do not count
321    /// — they are settled from a mempool policy perspective.
322    ///
323    /// A value of 0 means only confirmed UTXOs (or those whose
324    /// entire ancestor chain has at least 1 confirmation) are
325    /// eligible. A value of 3 allows spending UTXOs with up to 3
326    /// mempool-only ancestors.
327    pub max_mempool_chain_depth: usize,
328
329    /// Bitcoin address to receive the consolidation/change output.
330    pub change_address: BitcoinAddress,
331}
332
333impl CoinSelectionParams {
334    /// Default maximum transaction weight (400 kWU). This matches
335    /// Bitcoin Core's `MAX_STANDARD_TX_WEIGHT` policy limit — the
336    /// maximum weight a transaction can have and still be relayed by
337    /// default nodes. It is 10% of the 4 MWU block weight consensus
338    /// limit. Transactions exceeding this are valid per consensus but
339    /// will not propagate through the standard P2P network.
340    pub const DEFAULT_MAX_TX_WEIGHT: Weight = Weight::from_wu(400_000);
341
342    /// Default maximum consolidation inputs beyond the minimum
343    /// required to fund withdrawals.
344    pub const DEFAULT_MAX_INPUTS: usize = 500;
345
346    /// Default maximum number of withdrawal requests per batch. A
347    /// batch of 50 P2TR outputs is ~9.2 kWU (~2.3% of the 400 kWU
348    /// policy limit), so the weight limit is never the binding
349    /// constraint. Larger batches amortize the fixed transaction
350    /// overhead across more requests, reducing per-request fees.
351    pub const DEFAULT_MAX_WITHDRAWAL_REQUESTS: usize = 50;
352
353    /// Default minimum fee rate floor (1 sat/vB), matching Bitcoin
354    /// Core's minimum relay fee.
355    pub const DEFAULT_MIN_FEE_RATE: FeeRate = FeeRate::from_sat_per_vb_unchecked(1);
356
357    /// Default long-term average fee rate (10 sat/vB), matching
358    /// Bitcoin Core's `-consolidatefeerate` default. Below this rate,
359    /// the algorithm aggressively consolidates; at or above, it
360    /// minimizes inputs.
361    pub const DEFAULT_LONG_TERM_FEE_RATE: FeeRate = FeeRate::from_sat_per_vb_unchecked(10);
362
363    /// Default high fee rate threshold (30 sat/vB). Matches Bitcoin
364    /// Core's CoinGrinder activation threshold (3× the long-term
365    /// consolidation fee rate). At or above this rate, the algorithm
366    /// aggressively minimizes inputs and performs no pool
367    /// consolidation.
368    pub const DEFAULT_HIGH_FEE_RATE_THRESHOLD: FeeRate = FeeRate::from_sat_per_vb_unchecked(30);
369
370    /// Default input budget per request at or above the long-term fee
371    /// rate (10 extra inputs per request). Matches the on-chain
372    /// `input_budget` config parameter.
373    pub const DEFAULT_INPUT_BUDGET: usize = 10;
374
375    /// Default maximum mempool chain depth (5). Allows spending
376    /// change from up to 5 levels of mempool-only ancestors. At
377    /// 5-minute batching intervals this covers ~25 minutes of
378    /// batches before a confirmation is needed, well within
379    /// Bitcoin's cluster mempool limit of 64 and the pre-cluster
380    /// limit of 25.
381    pub const DEFAULT_MAX_MEMPOOL_CHAIN_DEPTH: usize = 5;
382
383    /// Creates a new `CoinSelectionParams` with sensible defaults.
384    ///
385    /// The `change_address` must be provided; all other parameters
386    /// use their `DEFAULT_*` values and can be overridden via struct
387    /// update syntax:
388    ///
389    /// ```ignore
390    /// let params = CoinSelectionParams {
391    ///     max_withdrawal_requests: 20,
392    ///     ..CoinSelectionParams::new(change_addr)
393    /// };
394    /// ```
395    ///
396    /// `max_fee_per_request` is derived from
397    /// `DEFAULT_HIGH_FEE_RATE_THRESHOLD` and a worst-case transaction
398    /// shape of `DEFAULT_INPUT_BUDGET` inputs, mirroring the on-chain
399    /// `worst_case_network_fee` formula.
400    pub fn new(change_address: BitcoinAddress) -> Self {
401        // Mirror the on-chain worst_case_network_fee formula:
402        //   tx_vbytes = TX_FIXED_VB + (input_budget * INPUT_VB) + (OUTPUT_BUDGET * OUTPUT_VB)
403        //   max_fee_per_request = high_fee_rate_threshold * tx_vbytes
404        //
405        // Using the on-chain constants: TX_FIXED_VB=11, INPUT_VB=100,
406        // OUTPUT_BUDGET=2, OUTPUT_VB=43.
407        const TX_FIXED_VB: u64 = 11;
408        const INPUT_VB: u64 = 100;
409        const OUTPUT_BUDGET: u64 = 2;
410        const OUTPUT_VB: u64 = 43;
411
412        let tx_vbytes = TX_FIXED_VB
413            + (Self::DEFAULT_INPUT_BUDGET as u64 * INPUT_VB)
414            + (OUTPUT_BUDGET * OUTPUT_VB);
415        let max_fee_per_request =
416            Self::DEFAULT_HIGH_FEE_RATE_THRESHOLD.to_sat_per_vb_floor() * tx_vbytes;
417
418        Self {
419            max_tx_weight: Self::DEFAULT_MAX_TX_WEIGHT,
420            max_inputs: Self::DEFAULT_MAX_INPUTS,
421            max_withdrawal_requests: Self::DEFAULT_MAX_WITHDRAWAL_REQUESTS,
422            max_fee_per_request,
423            min_fee_rate: Self::DEFAULT_MIN_FEE_RATE,
424            long_term_fee_rate: Self::DEFAULT_LONG_TERM_FEE_RATE,
425            high_fee_rate_threshold: Self::DEFAULT_HIGH_FEE_RATE_THRESHOLD,
426            input_budget: Self::DEFAULT_INPUT_BUDGET,
427            max_mempool_chain_depth: Self::DEFAULT_MAX_MEMPOOL_CHAIN_DEPTH,
428            change_address,
429        }
430    }
431}

435/// A single withdrawal output in the proposed transaction.
436#[derive(Clone, Debug)]
437pub struct WithdrawalOutput {
438    /// The ID of the withdrawal request this output satisfies.
439    pub request_id: Address,
440    /// Raw witness-program bytes of the recipient Bitcoin address.

447/// The result produced by a successful call to [`select_coins`].
448#[derive(Debug)]
449pub struct CoinSelectionResult {
450    /// UTXOs selected as transaction inputs.
451    ///
452    /// Includes both the minimum inputs required to fund the selected
453    /// withdrawal requests and any additional consolidation inputs pulled in
454    /// for pool maintenance.
455    pub inputs: Vec<UtxoCandidate>,
456
457    /// Withdrawal outputs, one per selected request, in the same order as
458    /// `selected_requests`.
459    pub withdrawal_outputs: Vec<WithdrawalOutput>,
460
461    /// Change output amount in satoshis, or `None` if the residual value
462    /// after outputs and fees falls below the dust threshold. When `Some`,
463    /// the change is sent to the `change_address` passed to [`select_coins`].
464    pub change: Option<u64>,
465
466    /// Total miner fee in satoshis:
467    /// `sum(inputs.amount) − sum(withdrawal_outputs.amount) − change.unwrap_or(0)`.
468    ///
469    /// Computed as `floor(total_weight_fee / N) × N`, where `total_weight_fee`
470    /// is the fee at `fee_rate` for the final transaction weight and N is the
471    /// number of selected requests. The `total_weight_fee % N` remainder is
472    /// implicitly donated to the miner and is included here (it is captured in
473    /// the conservation identity but not charged to any individual request).
474    pub fee: u64,
475
476    /// The withdrawal requests included in this transaction, in the same
477    /// order as `withdrawal_outputs`.
478    pub selected_requests: Vec<WithdrawalRequest>,
479}

483/// Errors that can be returned by [`select_coins`].
484#[derive(Debug, Error)]
485pub enum CoinSelectionError {
486    /// The UTXO pool is empty; there are no inputs available to fund any
487    /// transaction.
488    #[error("UTXO pool is empty")]
489    EmptyPool,
490
491    /// No withdrawal requests were provided to the algorithm.
492    #[error("no withdrawal requests provided")]
493    NoRequests,
494
495    /// The final per-request fee share exceeds `params.max_fee_per_request`.
496    ///
497    /// This is a safety check performed after all inputs and outputs are
498    /// finalised (step 5). It should not occur when parameters are
499    /// well-formed (i.e. `fee_rate` has been clamped to `params.high_fee_rate_threshold`
500    /// before calling [`select_coins`] and `params.max_fee_per_request` was
501    /// derived from `params.high_fee_rate_threshold`).
502    #[error(
503        "per-request deduction {fee_per_request} sat exceeds the configured cap \
504         {max_fee_per_request} sat (total deduction {total_deduction} sat across \
505         {n_requests} requests)"
506    )]
507    FeeExceedsCap {
508        /// `ceil(total_deduction / N)` — the equal share charged to each
509        /// request.
510        fee_per_request: u64,
511        /// The configured per-request fee cap (`params.max_fee_per_request`).
512        max_fee_per_request: u64,
513        /// Total deduction in satoshis (miner fee + CPFP deficit + dust
514        /// padding).
515        total_deduction: u64,
516        /// Number of selected requests (N).
517        n_requests: usize,
518    },
519
520    /// The total available UTXO value is insufficient to cover the total
521    /// requested withdrawal amounts. All fees come from request amounts, so
522    /// inputs only need to cover `sum(request.amount)`; no extra for fees.
523    #[error(
524        "insufficient funds: {available} sat available across confirmed and \
525         pending UTXOs, {required} sat required to cover selected withdrawal \
526         amounts"
527    )]
528    InsufficientFunds {
529        /// Total satoshis available across confirmed and pending UTXOs.
530        available: u64,
531        /// Sum of the selected withdrawal request amounts.
532        required: u64,
533    },
534
535    /// The transaction weight exceeds `params.max_tx_weight`. Step 1 takes
536    /// `min(requests.len(), max_withdrawal_requests)` oldest requests; Step 2
537    /// checks the resulting weight and returns this error if it is too high.
538    #[error(
539        "single-request transaction weight {weight:#} exceeds the configured \
540         maximum {max_weight:#}"
541    )]
542    ExceedsMaxWeight {
543        /// Estimated weight of a single-request transaction.
544        weight: Weight,
545        /// Configured maximum (`params.max_tx_weight`).
546        max_weight: Weight,
547    },
548
549    /// A selected request's amount is smaller than the per-request fee share,
550    /// which would produce a zero-value or negative output.
551    ///
552    /// This can occur when a request amount is less than
553    /// `ceil(total_fee / N)`. Callers should ensure request amounts exceed
554    /// `params.max_fee_per_request` before submitting them.
555    #[error(
556        "withdrawal request {request_id} amount {amount} sat is less than the \
557         per-request fee share {fee_per_request} sat"
558    )]
559    RequestAmountTooSmall {
560        /// ID of the offending withdrawal request.
561        request_id: Address,
562        /// The request's withdrawal amount.
563        amount: u64,
564        /// The fee share that would be deducted.
565        fee_per_request: u64,
566    },
567}

569// ── Private helpers ───────────────────────────────────────────────────────────
570
571/// Output weight for a withdrawal request's recipient.
572///
573/// Assumes recipient addresses have been validated upstream. Uses the
574/// cheaper P2WPKH weight for 20-byte witness programs and the P2TR
575/// weight for 32-byte programs. Any other length conservatively uses
576/// the P2TR weight (the more expensive of the two).
577fn output_weight_for_recipient(request: &WithdrawalRequest) -> Weight {
578    match request.recipient.len() {
579        // P2WPKH: OP_0 OP_PUSHBYTES_20 <20 bytes> → 22-byte scriptPubKey → 124 WU
580        20 => TXOUT_BASE_WEIGHT + Weight::from_wu(22 * 4),
581        // P2TR (or unknown): OP_1 OP_PUSHBYTES_32 <32 bytes> → 34-byte scriptPubKey → 172 WU
582        _ => TXOUT_BASE_WEIGHT + Weight::from_wu(34 * 4),
583    }
584}
585
586/// Output weight for an arbitrary scriptPubKey.
587///
588/// Assumes the script-length varint fits in one byte, which holds for all
589/// standard scripts (length < 252).
590fn output_weight_for_script(script: &bitcoin::Script) -> Weight {
591    Weight::from_wu((8u64 + 1 + script.len() as u64) * 4)
592}
593
594/// Minimum change output value (sat) above Bitcoin's dust relay threshold for
595/// the given change scriptPubKey.
596///
597/// Uses the standard script lengths to identify the output type:
598/// - 22 bytes → P2WPKH (294 sat at 3 sat/vByte min relay fee)
599/// - 34 bytes → P2TR / P2WSH (330 sat)
600/// - anything else → conservative P2TR value (330 sat)
601fn dust_threshold_for_change(script: &bitcoin::Script) -> u64 {
602    match script.len() {
603        22 => WPKH_DUST_RELAY_MIN_VALUE,
604        _ => TR_DUST_RELAY_MIN_VALUE,
605    }
606}

608/// Fee in satoshis for a given weight at the given fee rate.
609///
610/// Uses `FeeRate::fee_wu` which rounds up to the next whole satoshi to
611/// avoid underpaying relay fees.
612fn fee_for_weight(fee_rate: FeeRate, weight: Weight) -> u64 {
613    fee_rate
614        .fee_wu(weight)
615        .map(|a| a.to_sat())
616        .expect("fee computation overflow: fee rate or weight is unreasonably large")
617}

619/// Select UTXOs and plan a batched withdrawal transaction.
620///
621/// Uses a `TransactionBuilder` to construct the transaction
622/// incrementally, with CPFP-aware fee management for spending
623/// unconfirmed change outputs.
624///
625/// # Algorithm
626///
627/// 1. **Request selection** — up to `max_withdrawal_requests` requests
628///    are taken oldest-first by timestamp.
629///
630/// 2. **Input selection** — UTXOs are sorted largest-first and selected
631///    greedily until `input_total >= total_requested`. Both confirmed
632///    and pending UTXOs are eligible, subject to the
633///    `max_mempool_chain_depth` filter.
634///
635/// 3. **Consolidation** — if there is change (excess input value) and
636///    the fee rate is favourable, extra *confirmed* UTXOs are pulled in
637///    smallest-first to reduce the UTXO set. Aggressiveness depends on
638///    the fee environment:
639///    - Below `long_term_fee_rate`: up to `input_budget × N` extras.
640///    - Between long-term and `high_fee_rate_threshold`: up to
641///      `input_budget × N / 2` extras.
642///    - At or above high threshold: no consolidation.
643///
644/// 4. **Fee allocation** — `finalize_fees` computes the total deduction
645///    (own fee + CPFP deficit + dust padding), checks it against the
646///    per-request cap, verifies each output stays above dust, assigns
647///    net amounts, and derives the actual miner fee from conservation.
648///
649/// # Preconditions
650///
651/// The caller must provide a **consistent, non-overlapping UTXO set**.
652/// See the module-level documentation for details.
653pub fn select_coins(
654    utxos: &[UtxoCandidate],
655    requests: &[WithdrawalRequest],
656    params: &CoinSelectionParams,
657    fee_rate: FeeRate,
658) -> Result<CoinSelectionResult, CoinSelectionError> {
659    // ── Early validation ────────────────────────────────────────────────────
660    if utxos.is_empty() {
661        return Err(CoinSelectionError::EmptyPool);
662    }
663    if requests.is_empty() {
664        return Err(CoinSelectionError::NoRequests);
665    }
666    if params.max_withdrawal_requests == 0 {
667        return Err(CoinSelectionError::NoRequests);
668    }
669
670    // clamp fee rate floor
671    let fee_rate = fee_rate.max(params.min_fee_rate);
672
673    let mut builder = TransactionBuilder {
674        fee_rate,
675        params,
676        inputs: Vec::new(),
677        outputs: Vec::new(),
678        raw_change: None,
679        final_change: None,
680    };
681
682    // ── Step 1: Request selection ───────────────────────────────────────────
683    // Take up to max_withdrawal_requests oldest requests.
684
685    let mut requests: Vec<&WithdrawalRequest> = requests.iter().collect();
686    requests.sort_by_key(|r| r.timestamp_ms);
687
688    for r in requests.iter().take(params.max_withdrawal_requests) {
689        builder.outputs.push(PendingOutput {
690            request: r,
691            net_amount: 0,
692        });
693    }
694
695    // ── Step 2: Input selection (largest-first) ────────────────────────────
696    //
697    // Sort all UTXOs by descending value, then by id for determinism.
698    // Both confirmed and pending UTXOs are eligible — pending ones may
699    // require CPFP fee boosting, accounted for in Step 4. UTXOs with
700    // too many mempool-only (0-confirmation) ancestors are excluded to
701    // stay within Bitcoin's relay policy limits.
702    let mut pool: Vec<&UtxoCandidate> = utxos
703        .iter()
704        .filter(|u| {
705            u.status.mempool_chain_depth() <= params.max_mempool_chain_depth
706                // Also ensure that if this UTXO were to be selected, the resulting unconfirmed
707                // chain depth would be less than the max relay limit
708                && u.status.mempool_chain_depth() < MAX_ANCESTOR_DEPTH
709        })
710        .collect();
711    pool.sort_by(|a, b| b.amount.cmp(&a.amount).then_with(|| a.id.cmp(&b.id)));
712
713    let total_requested = builder.total_requested();
714    let mut input_total = 0u64;
715
716    for utxo in &pool {
717        if builder.inputs.len() >= params.max_inputs {
718            break;
719        }
720        if input_total >= total_requested {
721            break;
722        }
723
724        builder.inputs.push(utxo);
725        input_total += utxo.amount;
726    }
727
728    if input_total < total_requested {
729        return Err(CoinSelectionError::InsufficientFunds {
730            available: pool.iter().map(|u| u.amount).sum(),
731            required: total_requested,
732        });
733    }
734
735    let raw_change = builder.compute_raw_change();
736
737    // Verify the base transaction (before consolidation) is within fee
738    // and weight limits. Bail out early if even the minimum inputs
739    // exceed limits.
740    builder.check_fees()?;
741    builder.check_weight()?;
742
743    // ── Step 3: Consolidation ──────────────────────────────────────────
744    //
745    // Pull in extra *confirmed* inputs (smallest-first) to reduce the
746    // UTXO set toward a single output. Only confirmed UTXOs are
747    // eligible — pending UTXOs would add unaccounted ancestors to the
748    // CPFP calculation. Consolidation only runs when there is already
749    // a change output (raw_change > 0).
750    //
751    // The aggressiveness depends on the fee environment:
752    // - Below long-term rate: up to `input_budget × N` extras.
753    // - Between long-term and high threshold: up to
754    //   `input_budget × N / 2` extras.
755    // - At or above high threshold: no consolidation.
756    let max_consolidation = if fee_rate < params.long_term_fee_rate {
757        (params.max_inputs.saturating_sub(builder.inputs.len()))
758            .min(builder.outputs.len() * params.input_budget)
759    } else if fee_rate < params.high_fee_rate_threshold {
760        (params.max_inputs.saturating_sub(builder.inputs.len()))
761            .min(builder.outputs.len() * params.input_budget / 2)
762    } else {
763        0
764    };
765
766    if max_consolidation > 0 && raw_change > 0 {
767        let mut remaining: Vec<&UtxoCandidate> = pool
768            .iter()
769            .filter(|u| {
770                matches!(u.status, UtxoStatus::Confirmed)
771                    && !builder.inputs.iter().any(|s| s.id == u.id)
772            })
773            .copied()
774            .collect();
775        remaining.sort_by_key(|u| u.amount);
776
777        for utxo in remaining.into_iter().take(max_consolidation) {
778            if builder.inputs.len() >= params.max_inputs {
779                break;
780            }
781            builder.inputs.push(utxo);
782            builder.compute_raw_change();
783
784            // If adding this input pushed fees or weight over the
785            // limit, undo it and stop consolidating. This is a
786            // greedy heuristic: inputs are sorted smallest-first so
787            // weight increases monotonically, but a larger input
788            // could theoretically eliminate dust padding and lower
789            // total_deduction enough to pass. We accept this
790            // approximation for simplicity.
791            if builder.check_fees().is_err() || builder.exceeds_max_weight() {
792                builder.inputs.pop();
793                builder.compute_raw_change();
794                break;
795            }
796        }
797    }
798
799    // ── Step 4: Fee allocation ──────────────────────────────────────────
800    //
801    // Recompute raw_change (consolidation may have added inputs),
802    // then finalize: check fee caps, assign net output amounts,
803    // compute final change (with dust padding), and derive the
804    // actual miner fee from conservation.
805    builder.compute_raw_change();
806    let miner_fee = builder.finalize_fees()?;
807
808    Ok(CoinSelectionResult {
809        inputs: builder.inputs.iter().map(|u| (*u).clone()).collect(),
810        withdrawal_outputs: builder
811            .outputs
812            .iter()
813            .map(|o| WithdrawalOutput {
814                request_id: o.request.id,
815                recipient: o.request.recipient.clone(),
816                amount: o.net_amount,
817            })
818            .collect(),
819        change: builder.final_change,
820        fee: miner_fee,
821        selected_requests: builder.outputs.iter().map(|o| o.request.clone()).collect(),
822    })
823}

825struct TransactionBuilder<'a> {
826    // Fixed from the Params
827    fee_rate: FeeRate,
828    params: &'a CoinSelectionParams,
829
830    // Values that are built up
831    inputs: Vec<&'a UtxoCandidate>,
832    outputs: Vec<PendingOutput<'a>>,
833    /// The raw excess of input value over requested value, before any
834    /// dust padding. `None` means not yet computed; `Some(0)` means
835    /// exact match (no change output); `Some(n)` means `n` sats of
836    /// change.
837    raw_change: Option<u64>,
838    final_change: Option<u64>,
839}
840
841struct PendingOutput<'a> {
842    request: &'a WithdrawalRequest,
843    net_amount: u64,
844}
845
846impl<'a> PendingOutput<'a> {
847    /// The dust relay threshold for this output's recipient address type.
848    fn dust_threshold(&self) -> u64 {
849        match self.request.recipient.len() {
850            20 => WPKH_DUST_RELAY_MIN_VALUE,
851            _ => TR_DUST_RELAY_MIN_VALUE,
852        }
853    }
854}
855
856impl<'a> TransactionBuilder<'a> {
857    /// Compute the total transaction weight based on the current inputs,
858    /// outputs, and change state.
859    ///
860    /// When `raw_change` is `None` (not yet computed), the change output is
861    /// conservatively assumed to be present. This overestimates weight
862    /// slightly, which is the safe direction: a weight check that passes
863    /// with a change output included will still pass if no change output
864    /// ends up being emitted.
865    fn weight(&self) -> Weight {
866        let input_count = self.inputs.len();
867        let has_change = self.raw_change.is_none_or(|v| v > 0);
868        let output_count = self.outputs.len() + if has_change { 1 } else { 0 };
869
870        // Non-witness fixed fields (scaled ×4):
871        //   nVersion(4) + nLockTime(4) = 8 bytes × 4 = 32 WU
872        //   vin_count varint + vout_count varint (scaled ×4)
873        //
874        // Witness fixed fields (scaled ×1):
875        //   segwit_marker(1) + segwit_flag(1) = 2 WU
876        let fixed = Weight::from_wu(32 + 2)
877            + varint_weight(input_count as u64)
878            + varint_weight(output_count as u64);
879
880        // Sum of all input weights (base + witness satisfaction).
881        let inputs: Weight = self
882            .inputs
883            .iter()
884            .map(|u| u.spend_path.input_weight())
885            .sum();
886
887        // Sum of all withdrawal output weights.
888        let outputs: Weight = self
889            .outputs
890            .iter()
891            .map(|o| output_weight_for_recipient(o.request))
892            .sum();
893
894        // Change output weight (if present).
895        let change = if has_change {
896            output_weight_for_script(&self.change_script())
897        } else {
898            Weight::ZERO
899        };
900
901        fixed + inputs + outputs + change
902    }
903
904    /// The total value of all selected inputs.
905    fn input_total(&self) -> u64 {
906        self.inputs.iter().map(|u| u.amount).sum()
907    }
908
909    /// Compute and set the raw change amount based on the current
910    /// inputs and requested outputs. Raw change is the excess of input
911    /// value over the total requested amount, before any dust padding.
912    ///
913    /// - `Some(0)` means exact match (no change output).
914    /// - `Some(n)` where `n > 0` means `n` sats of raw change.
915    ///
916    /// Panics if `input_total < total_requested` — the caller must
917    /// ensure sufficient inputs have been selected before calling.
918    fn compute_raw_change(&mut self) -> u64 {
919        let input_total = self.input_total();
920        let total_requested = self.total_requested();
921        assert!(
922            input_total >= total_requested,
923            "input_total ({input_total}) < total_requested ({total_requested})"
924        );
925        let raw_change = input_total - total_requested;
926        self.raw_change = Some(raw_change);
927        raw_change
928    }
929
930    /// Whether the transaction weight exceeds the configured maximum.
931    fn exceeds_max_weight(&self) -> bool {
932        self.weight() > self.params.max_tx_weight
933    }
934
935    /// Check that the transaction weight is within the configured
936    /// maximum. Returns an error if exceeded.
937    fn check_weight(&self) -> Result<(), CoinSelectionError> {
938        if self.exceeds_max_weight() {
939            Err(CoinSelectionError::ExceedsMaxWeight {
940                weight: self.weight(),
941                max_weight: self.params.max_tx_weight,
942            })
943        } else {
944            Ok(())
945        }
946    }
947
948    /// The total amount deducted from requests: the transaction's own
949    /// fee, any CPFP deficit for underpaying ancestors, and any dust
950    /// padding needed to bring a sub-dust change output up to the
951    /// relay threshold.
952    fn total_deduction(&self) -> u64 {
953        self.required_fee() + self.dust_padding()
954    }
955
956    /// The amount of dust padding needed for the change output. If
957    /// the raw change is positive but below the dust threshold, it
958    /// must be padded up — and that padding cost comes from the
959    /// requests. Returns 0 if there is no change or the change is
960    /// already above dust.
961    fn dust_padding(&self) -> u64 {
962        match self.raw_change {
963            Some(v) if v > 0 && v < self.change_dust_threshold() => {
964                self.change_dust_threshold() - v
965            }
966            _ => 0,
967        }
968    }
969
970    /// Compute the CPFP deficit: the extra fee this transaction must
971    /// pay to ensure that the unconfirmed ancestor transactions
972    /// effectively pay the target `fee_rate`.
973    ///
974    /// ```text
975    /// needed_ancestor_fee = fee_rate × ancestor_weight
976    /// deficit = needed_ancestor_fee - ancestor_fee
977    /// ```
978    ///
979    /// This transaction's own fee (for its own weight) is handled
980    /// separately by the normal fee calculation. The deficit is purely
981    /// the shortfall in what the ancestors should have paid.
982    ///
983    /// Only mempool-only ancestors (0 confirmations) are included —
984    /// ancestors already in a block don't need boosting.
985    ///
986    /// Returns 0 if the ancestors already pay enough.
987    fn cpfp_deficit(&self) -> u64 {
988        let ancestor_weight: Weight = self
989            .inputs
990            .iter()
991            .map(|u| u.status.unconfirmed_ancestor_weight())
992            .sum();
993        let ancestor_fee: u64 = self
994            .inputs
995            .iter()
996            .map(|u| u.status.unconfirmed_ancestor_fee())
997            .sum();
998
999        let needed_ancestor_fee = fee_for_weight(self.fee_rate, ancestor_weight);
1000        needed_ancestor_fee.saturating_sub(ancestor_fee)
1001    }
1002
1003    /// The total fee this transaction must pay: its own weight-based
1004    /// fee plus any CPFP deficit to boost underpaying ancestors.
1005    fn required_fee(&self) -> u64 {
1006        let own_fee = fee_for_weight(self.fee_rate, self.weight());
1007        own_fee + self.cpfp_deficit()
1008    }
1009
1010    /// The sum of all withdrawal request amounts.
1011    fn total_requested(&self) -> u64 {
1012        self.outputs.iter().map(|o| o.request.amount).sum()
1013    }
1014
1015    /// The change script for the pool's change address.
1016    fn change_script(&self) -> bitcoin::ScriptBuf {
1017        self.params.change_address.script_pubkey()
1018    }
1019
1020    /// The dust threshold for change outputs based on the change
1021    /// address script type.
1022    fn change_dust_threshold(&self) -> u64 {
1023        dust_threshold_for_change(&self.change_script())
1024    }
1025
1026    /// Check that the current fee deductions are within bounds:
1027    /// the per-request cap isn't exceeded, and every request can
1028    /// absorb its share and still produce a non-dust output.
1029    ///
1030    /// This is a read-only check — it does not mutate the builder.
1031    /// Call it early to bail out before consolidation if the base
1032    /// transaction already exceeds limits, and again during
1033    /// consolidation to detect when adding inputs pushes fees over.
1034    fn check_fees(&self) -> Result<(), CoinSelectionError> {
1035        let n = self.outputs.len() as u64;
1036        if n == 0 {
1037            return Ok(());
1038        }
1039
1040        let total_deduction = self.total_deduction();
1041        let deduction_per_request = total_deduction.div_ceil(n);
1042
1043        if deduction_per_request > self.params.max_fee_per_request {
1044            return Err(CoinSelectionError::FeeExceedsCap {
1045                fee_per_request: deduction_per_request,
1046                max_fee_per_request: self.params.max_fee_per_request,
1047                total_deduction,
1048                n_requests: n as usize,
1049            });
1050        }
1051
1052        for output in self.outputs.iter() {
1053            if output.request.amount < deduction_per_request + output.dust_threshold() {
1054                return Err(CoinSelectionError::RequestAmountTooSmall {
1055                    request_id: output.request.id,
1056                    amount: output.request.amount,
1057                    fee_per_request: deduction_per_request,
1058                });
1059            }
1060        }
1061
1062        Ok(())
1063    }
1064
1065    /// Perform final fee checks, assign net output amounts, compute
1066    /// final change (with dust padding), and return the actual miner
1067    /// fee.
1068    ///
1069    /// Requires `compute_raw_change` to have been called first.
1070    /// Returns the miner fee derived from the conservation identity:
1071    /// `input_total - sum(net_amounts) - final_change`.
1072    fn finalize_fees(&mut self) -> Result<u64, CoinSelectionError> {
1073        let raw_change = self
1074            .raw_change
1075            .expect("finalize_fees called before compute_raw_change");
1076
1077        // Run the fee checks.
1078        self.check_fees()?;
1079
1080        // Assign net amounts using ceiling division so the miner
1081        // always receives at least the required fee.
1082        let n = self.outputs.len() as u64;
1083        let deduction_per_request = self.total_deduction().div_ceil(n);
1084        for output in &mut self.outputs {
1085            output.net_amount = output.request.amount - deduction_per_request;
1086        }
1087
1088        // Compute final change (padded to dust threshold if needed).
1089        self.final_change = if raw_change > 0 {
1090            Some(raw_change.max(self.change_dust_threshold()))
1091        } else {
1092            None
1093        };
1094
1095        // Derive the actual miner fee from the conservation identity:
1096        //   input_total = sum(net_amounts) + final_change + miner_fee
1097        let output_total: u64 = self.outputs.iter().map(|o| o.net_amount).sum();
1098        let change = self.final_change.unwrap_or(0);
1099        let miner_fee = self.input_total() - output_total - change;
1100
1101        // With ceiling division, the miner always gets at least the
1102        // required fee.
1103        assert!(
1104            miner_fee >= self.required_fee(),
1105            "miner_fee ({miner_fee}) < required_fee ({})",
1106            self.required_fee()
1107        );
1108
1109        Ok(miner_fee)
1110    }
1111}

1113/// Weight of a Bitcoin varint encoding for `n`, counted as non-witness
1114/// data (scaled ×4).
1115///
1116/// Varint encoding sizes:
1117/// - 0–252: 1 byte
1118/// - 253–0xFFFF: 3 bytes
1119/// - 0x10000–0xFFFFFFFF: 5 bytes
1120/// - 0x100000000–: 9 bytes
1121fn varint_weight(n: u64) -> Weight {
1122    let bytes = match n {
1123        0..=252 => 1u64,
1124        253..=0xFFFF => 3,
1125        0x10000..=0xFFFFFFFF => 5,
1126        _ => 9,
1127    };
1128    Weight::from_wu(bytes * 4)
1129}
