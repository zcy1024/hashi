59impl fmt::Debug for CommitteeSet {
60    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
61        // Render tls_public_key_to_address with base64 keys
62        let tls_key_map: BTreeMap<String, &Address> = self
63            .tls_public_key_to_address
64            .iter()
65            .map(|(k, v)| (base64ct::Base64::encode_string(k), v))
66            .collect();
67
68        // Render tls_private_key as redacted with public key
69        let tls_private_key_display = self.tls_private_key.as_ref().map(|key| {
70            format!(
71                "<redacted, public_key: {}>",
72                base64ct::Base64::encode_string(key.verifying_key().as_bytes())
73            )
74        });
75
76        f.debug_struct("CommitteeSet")
77            .field("members_id", &self.members_id)
78            .field("members", &self.members)
79            .field("tls_public_key_to_address", &tls_key_map)
80            .field("epoch", &self.epoch)
81            .field("pending_epoch_change", &self.pending_epoch_change)
82            .field(
83                "mpc_public_key",
84                &Base64("MpcPublicKey", &self.mpc_public_key),
85            )
86            .field("committees_id", &self.committees_id)
87            .field("committees", &self.committees)
88            .field("tls_private_key", &tls_private_key_display)
89            .field(
90                "grpc_max_decoding_message_size",
91                &self.grpc_max_decoding_message_size,
92            )
93            .field("clients", &format_args!("<{} clients>", self.clients.len()))
94            .finish()
95    }
96}

161    // Set the tls private key to use when constructing tls configs for clients to other validators
162    pub fn set_tls_private_key(&mut self, tls_private_key: ed25519_dalek::SigningKey) -> &mut Self {
163        self.tls_private_key = Some(tls_private_key);

168    pub fn set_grpc_max_decoding_message_size(&mut self, limit: usize) -> &mut Self {
169        self.grpc_max_decoding_message_size = Some(limit);
170        self.update_all_clients();
171        self
172    }
173
174    pub fn set_members(&mut self, members: BTreeMap<Address, MemberInfo>) -> &mut Self {
175        self.tls_public_key_to_address = members
176            .values()
177            .filter_map(|info| {
178                info.tls_public_key
179                    .as_ref()
180                    .map(|pubkey| (*pubkey.as_bytes(), info.validator_address))
181            })
182            .collect();
183        self.members = members;
184        self.update_all_clients();
185        self
186    }
187
188    fn update_all_clients(&mut self) {
189        self.clients = self
190            .members
191            .values()
192            .filter_map(|info| {
193                if let (Some(addr), Some(public_key)) = (info.endpoint_url(), info.tls_public_key())
194                {
195                    Some((info.validator_address, addr, public_key))
196                } else {
197                    None
198                }
199            })
200            .filter_map(|(validator, endpoint_url, tls_public_key)| {
201                let tls_config = if let Some(tls_private_key) = &self.tls_private_key {
202                    crate::tls::make_client_config_with_client_auth(tls_private_key, tls_public_key)
203                } else {
204                    crate::tls::make_client_config(tls_public_key)
205                };
206                let mut client = Client::new(endpoint_url, tls_config)
207                    .inspect_err(|e| tracing::debug!("unable to build client for {validator}: {e}"))
208                    .ok()?;
209                if let Some(limit) = self.grpc_max_decoding_message_size {
210                    client = client.max_decoding_message_size(limit);
211                }
212                Some((validator, client))
213            })
214            .collect();
215    }
216
217    pub fn update_validator(&mut self, info: MemberInfo) {
218        let validator = info.validator_address;
219        let info_entry = self.members.entry(validator);
220
221        // remove old tls public key mapping
222        if let std::collections::btree_map::Entry::Occupied(entry) = &info_entry
223            && let Some(tls_public_key) = &entry.get().tls_public_key
224        {
225            self.tls_public_key_to_address
226                .remove(tls_public_key.as_bytes());
227        }
228
229        // insert new tls public key mapping
230        if let Some(tls_public_key) = &info.tls_public_key {
231            self.tls_public_key_to_address
232                .insert(*tls_public_key.as_bytes(), validator);
233        }
234
235        // update client
236        self.clients.remove(&validator);
237        if let Some(endpoint_url) = &info.endpoint_url
238            && let Some(tls_public_key) = &info.tls_public_key
239        {
240            let tls_config = if let Some(tls_private_key) = &self.tls_private_key {
241                crate::tls::make_client_config_with_client_auth(tls_private_key, tls_public_key)
242            } else {
243                crate::tls::make_client_config(tls_public_key)
244            };
245            if let Ok(mut client) = Client::new(endpoint_url, tls_config)
246                .inspect_err(|e| tracing::debug!("unable to build client for {validator}: {e}"))
247            {
248                if let Some(limit) = self.grpc_max_decoding_message_size {
249                    client = client.max_decoding_message_size(limit);
250                }
251                self.clients.insert(validator, client);
252            }
253        }
254
255        // replace info
256        match info_entry {
257            std::collections::btree_map::Entry::Occupied(mut entry) => {
258                entry.insert(info);
259            }
260            std::collections::btree_map::Entry::Vacant(entry) => {
261                entry.insert(info);
262            }
263        }
264    }
265
266    pub fn set_epoch(&mut self, epoch: u64) -> &mut Self {
267        self.epoch = epoch;
268        self
269    }
270
271    pub fn set_pending_epoch_change(&mut self, pending_epoch_change: Option<u64>) -> &mut Self {
272        self.pending_epoch_change = pending_epoch_change;
273        self
274    }
275
276    pub fn set_mpc_public_key(&mut self, mpc_public_key: Vec<u8>) -> &mut Self {
277        assert!(self.mpc_public_key.is_empty() || self.mpc_public_key == mpc_public_key);
278        self.mpc_public_key = mpc_public_key;
279        self
280    }
281
282    pub fn set_committees(&mut self, committees: BTreeMap<u64, Committee>) -> &mut Self {
283        self.committees = committees;
284        self
285    }
286
287    pub fn lookup_address_by_tls_public_key(
288        &self,
289        tls_public_key: &ed25519_dalek::VerifyingKey,
290    ) -> Option<Address> {
291        self.tls_public_key_to_address
292            .get(tls_public_key.as_bytes())
293            .copied()
294    }

297#[derive(Clone)]
298pub struct MemberInfo {
299    /// Sui Validator Address of this node
300    pub validator_address: Address,
301
302    /// Sui Address of an operations account
303    pub operator_address: Address,
304
305    /// bls12381 public key to be used in the next epoch.
306    ///
307    /// The public key for this node which is active in the current epoch can
308    /// be found in the `Committee` struct.
309    ///
310    /// This public key can be rotated but will only take effect at the
311    /// beginning of the next epoch.
312    pub next_epoch_public_key: BLS12381PublicKey,
313
314    /// The publicly reachable URL where the `hashi` service for this validator
315    /// can be reached.
316    ///
317    /// This URL can be rotated and any such updates will take effect
318    /// immediately.
319    pub endpoint_url: Option<http::Uri>,
320
321    /// ed25519 public key used to verify TLS self-signed x509 certs
322    ///
323    /// This public key can be rotated and any such updates will take effect
324    /// immediately.
325    pub tls_public_key: Option<ed25519_dalek::VerifyingKey>,
326
327    /// A 32-byte ristretto255 Ristretto encryption public key (ristretto255
328    /// RistrettoPoint) for MPC ECIES, to be used in the next epoch.
329    ///
330    /// This public key can be rotated but will only take effect at the
331    /// beginning of the next epoch.
332    pub next_epoch_encryption_public_key: Option<EncryptionPublicKey>,
333}
334
335impl fmt::Debug for MemberInfo {
336    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
337        let encryption_key_bytes = self
338            .next_epoch_encryption_public_key
339            .as_ref()
340            .map(|k| k.as_element().to_byte_array());
341
342        f.debug_struct("MemberInfo")
343            .field("validator_address", &self.validator_address)
344            .field("operator_address", &self.operator_address)
345            .field(
346                "next_epoch_public_key",
347                &Base64("BLS12381PublicKey", self.next_epoch_public_key.as_bytes()),
348            )
349            .field("endpoint_url", &self.endpoint_url)
350            .field(
351                "tls_public_key",
352                &self
353                    .tls_public_key
354                    .as_ref()
355                    .map(|k| Base64("Ed25519PublicKey", k.as_bytes())),
356            )
357            .field(
358                "next_epoch_encryption_public_key",
359                &encryption_key_bytes
360                    .as_ref()
361                    .map(|b| Base64("EncryptionPublicKey", b.as_slice())),
362            )
363            .finish()
364    }
365}

447#[derive(Debug)]
448pub struct Config {
449    pub config: BTreeMap<String, ConfigValue>,
450    pub enabled_versions: BTreeSet<u64>,
451    pub upgrade_cap: Option<UpgradeCap>,
452}
453
454// These constants mirror the values in btc_config.move and must be kept in sync.
455const DUST_RELAY_MIN_VALUE: u64 = 546;
456const MIN_RELAY_FEE_RATE: u64 = 1;
457const INPUT_VB: u64 = 100;
458const OUTPUT_VB: u64 = 43;
459const OUTPUT_BUDGET: u64 = 2;
460const TX_FIXED_VB: u64 = 11;
461
462impl Config {
463    pub fn withdrawal_fee_btc(&self) -> u64 {
464        match self.config.get("withdrawal_fee_btc") {
465            Some(ConfigValue::U64(v)) => (*v).max(DUST_RELAY_MIN_VALUE),
466            _ => DUST_RELAY_MIN_VALUE,
467        }
468    }
469
470    pub fn max_fee_rate(&self) -> u64 {
471        match self.config.get("max_fee_rate") {
472            Some(ConfigValue::U64(v)) => (*v).max(MIN_RELAY_FEE_RATE),
473            _ => MIN_RELAY_FEE_RATE,
474        }
475    }
476
477    pub fn input_budget(&self) -> u64 {
478        match self.config.get("input_budget") {
479            Some(ConfigValue::U64(v)) => (*v).max(1),
480            _ => 1,
481        }
482    }
483
484    /// Worst-case network (miner) fee for a withdrawal transaction,
485    /// mirroring the formula in config.move.
486    pub fn worst_case_network_fee(&self) -> u64 {
487        let tx_vbytes =
488            TX_FIXED_VB + (self.input_budget() * INPUT_VB) + (OUTPUT_BUDGET * OUTPUT_VB);
489        self.max_fee_rate() * tx_vbytes
490    }
491
492    /// Computed withdrawal minimum, mirroring the formula in config.move.
493    pub fn withdrawal_minimum(&self) -> u64 {
494        self.withdrawal_fee_btc() + self.worst_case_network_fee() + DUST_RELAY_MIN_VALUE
495    }
496
497    pub fn paused(&self) -> bool {
498        matches!(self.config.get("paused"), Some(ConfigValue::Bool(true)))
499    }
500
501    pub fn bitcoin_chain_id(&self) -> Option<Address> {
502        match self.config.get("bitcoin_chain_id") {
503            Some(ConfigValue::Address(v)) => Some(*v),
504            _ => None,
505        }
506    }
507
508    pub fn bitcoin_confirmation_threshold(&self) -> u32 {
509        match self.config.get("bitcoin_confirmation_threshold") {
510            Some(ConfigValue::U64(v)) => u32::try_from(*v).unwrap_or(u32::MAX),
511            _ => 6,
512        }
513    }
514}

592/// Withdrawal lifecycle status, mirroring the Move `WithdrawalStatus` enum.
593#[derive(Clone, Debug, PartialEq, serde_derive::Serialize)]
594pub enum WithdrawalStatus {
595    Requested,
596    Approved,
597    Processing { pending_withdrawal_id: Address },
598    Signed { pending_withdrawal_id: Address },
599    Confirmed { txid: Address },
600}
601
602impl WithdrawalStatus {
603    /// Returns true if the status is `Approved`.
604    pub fn is_approved(&self) -> bool {
605        matches!(self, Self::Approved)
606    }
607
608    /// Returns true if the status is `Requested`.
609    pub fn is_requested(&self) -> bool {
610        matches!(self, Self::Requested)
611    }

698            vout: id.vout,
699        }

706    pub utxo: Utxo,
707    /// The withdrawal_id that produced this UTXO as a change output; None if
708    /// confirmed (deposit or previously promoted change).

712}

715pub struct UtxoPool {
716    pub(super) utxo_records_id: Address,
717    pub(super) utxo_records: BTreeMap<UtxoId, UtxoRecord>,
718    pub(super) spent_utxos_id: Address,
719    pub(super) spent_utxos: BTreeMap<UtxoId, u64>,
720}

722impl UtxoPool {
723    pub fn utxo_records_id(&self) -> &Address {
724        &self.utxo_records_id
725    }
726
727    pub fn utxo_records(&self) -> &BTreeMap<UtxoId, UtxoRecord> {
728        &self.utxo_records
729    }
730
731    /// Returns all UTXOs that are available (not locked) for coin selection,
732    /// regardless of whether they are confirmed.
733    pub fn active_utxos(&self) -> impl Iterator<Item = (&UtxoId, &Utxo)> {
734        self.utxo_records
735            .iter()
736            .filter(|(_, r)| r.locked_by.is_none())
737            .map(|(id, r)| (id, &r.utxo))
738    }

741        &self.spent_utxos_id
742    }

745        &self.spent_utxos
746    }

750pub struct TreasuryCap {

752    pub id: Address,
753    pub supply: u64,
754}
755
756impl TreasuryCap {
757    pub fn try_from_contents(type_tag: &TypeTag, contents: &[u8]) -> Option<Self> {
758        let TypeTag::Struct(struct_tag) = type_tag else {
759            return None;
760        };
761
762        if struct_tag.address() == &Address::TWO
763            && struct_tag.module() == "coin"
764            && struct_tag.name() == "TreasuryCap"
765            && let [coin_type] = struct_tag.type_params()
766            && contents.len() == Address::LENGTH + std::mem::size_of::<u64>()
767        {
768            let id = Address::new((&contents[..Address::LENGTH]).try_into().unwrap());
769            let supply = u64::from_le_bytes((&contents[Address::LENGTH..]).try_into().unwrap());
770            Some(Self {
771                coin_type: coin_type.to_owned(),
772                id,
773                supply,
774            })
775        } else {
776            None
777        }
778    }
