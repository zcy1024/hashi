4//! Core backup logic for creating and restoring encrypted backup archives.
5//!
6//! This module handles the mechanics of building backup manifests, encrypting
7//! files into age-wrapped tar archives, and extracting them. CLI-specific
8//! orchestration (config loading, recipient resolution, user output) lives in
9//! [`crate::cli::commands::backup`].

28pub const BACKUP_MANIFEST_FILE_NAME: &str = "hashi-config-backup-manifest.toml";
29pub const DB_SNAPSHOT_TAR_PREFIX: &str = "hashi-db-snapshot";

31/// Open `path` for writing with mode `0o600`, failing if anything already
32/// exists there. The `AlreadyExists` case is mapped to a clear "refusing to
33/// overwrite" error so callers don't have to repeat the same pattern.
34fn create_file_strict(path: &Path) -> Result<File> {
35    OpenOptions::new()
36        .write(true)
37        .create_new(true)
38        .mode(0o600)
39        .open(path)
40        .map_err(|e| match e.kind() {
41            ErrorKind::AlreadyExists => {
42                anyhow::anyhow!("Refusing to overwrite existing file: {}", path.display())
43            }
44            _ => anyhow::Error::from(e).context(format!("Failed to create {}", path.display())),
45        })
46}
47
48/// Create directory `path` non-recursively, failing if anything already
49/// exists at that exact location. Caller is responsible for ensuring the
50/// parent exists.
51fn create_dir_strict(path: &Path) -> Result<()> {
52    fs::DirBuilder::new()
53        .recursive(false)
54        .create(path)
55        .map_err(|e| match e.kind() {
56            ErrorKind::AlreadyExists => {
57                anyhow::anyhow!(
58                    "Refusing to overwrite existing directory: {}",
59                    path.display()
60                )
61            }
62            _ => anyhow::Error::from(e)
63                .context(format!("Failed to create directory {}", path.display())),
64        })
65}

79pub fn build_backup_manifest(
80    files: &[PathBuf],
81    db_original_path: &Path,
82    db_snapshot_dir: &Path,
83) -> Result<BackupManifest> {
84    let mut archive_names = HashSet::new();
85    // Reserve the db snapshot prefix so a user file with the same basename
86    // gets disambiguated instead of silently colliding with the db entry.
87    archive_names.insert(DB_SNAPSHOT_TAR_PREFIX.to_string());
88    let mut manifest_paths = Vec::new();
89
90    for file in files {
91        let base_name = file
92            .file_name()
93            .ok_or_else(|| {
94                anyhow::anyhow!("Backup input does not have a file name: {}", file.display())
95            })?
96            .to_string_lossy();
97
98        let archive_name = if archive_names.contains(base_name.as_ref()) {
99            let stem = Path::new(base_name.as_ref())
100                .file_stem()
101                .unwrap_or_default()
102                .to_string_lossy();
103            let ext = Path::new(base_name.as_ref())
104                .extension()
105                .map(|e| format!(".{}", e.to_string_lossy()))
106                .unwrap_or_default();
107
108            let mut suffix = 2u32;
109            loop {
110                let candidate = format!("{stem}-{suffix}{ext}");
111                if archive_names.insert(candidate.clone()) {
112                    info!(
113                        original = %file.display(),
114                        renamed = %candidate,
115                        "Archive name collision for {base_name}",
116                    );
117                    break PathBuf::from(candidate);
118                }
119                suffix += 1;
120            }
121        } else {
122            archive_names.insert(base_name.to_string());
123            PathBuf::from(base_name.as_ref())
124        };
125
126        manifest_paths.push(BackupManifestEntry {
127            archive_name,
128            original_path: file.clone(),
129        });
130    }
131
132    manifest_paths.push(BackupManifestEntry {
133        archive_name: PathBuf::from(DB_SNAPSHOT_TAR_PREFIX),
134        original_path: db_original_path.to_path_buf(),
135    });
136
137    Ok(BackupManifest {
138        paths: manifest_paths,
139        db_archive_entries: collect_db_archive_entries(
140            db_snapshot_dir,
141            Path::new(DB_SNAPSHOT_TAR_PREFIX),
142        )?,
143    })
144}

146fn collect_db_archive_entries(src_dir: &Path, tar_prefix: &Path) -> Result<Vec<PathBuf>> {
147    let mut archive_entries = Vec::new();
148
149    for entry in walkdir::WalkDir::new(src_dir).min_depth(1) {
150        let entry = entry.with_context(|| {
151            format!(
152                "Failed to walk database snapshot directory {}",
153                src_dir.display()
154            )
155        })?;
156
157        if entry.file_type().is_symlink() {
158            anyhow::bail!(
159                "Refusing to archive symlink inside database snapshot directory: {}",
160                entry.path().display()
161            );
162        }
163
164        let relative = entry
165            .path()
166            .strip_prefix(src_dir)
167            .expect("walkdir entry is always under src_dir");
168        archive_entries.push(tar_prefix.join(relative));
169    }
170
171    archive_entries.sort();
172    Ok(archive_entries)
173}

175pub fn encrypt_files_to_age_archive(
176    manifest: &BackupManifest,
177    db_snapshot_dir: &Path,
178    recipient: &dyn age::Recipient,
179    output_path: &Path,
180) -> Result<()> {
181    let output = create_file_strict(output_path)?;
182    let encryptor = Encryptor::with_recipients(std::iter::once(recipient))?;
183    let mut encrypted = encryptor.wrap_output(output)?;
184    {
185        let mut archive = tar::Builder::new(&mut encrypted);
186        append_backup_manifest(&mut archive, manifest)?;
187
188        for entry in &manifest.paths {
189            if entry.archive_name == Path::new(DB_SNAPSHOT_TAR_PREFIX) {
190                continue;
191            }
192            archive.append_path_with_name(&entry.original_path, &entry.archive_name)?;
193            info!(
194                original = %entry.original_path.display(),
195                archive_name = %entry.archive_name.display(),
196                "Added file to backup archive",
197            );
198        }
199
200        append_dir_recursive(
201            &mut archive,
202            db_snapshot_dir,
203            Path::new(DB_SNAPSHOT_TAR_PREFIX),
204        )?;
205        info!("Added database snapshot to backup archive");
206
207        archive.finish()?;
208    }
209    encrypted.finish()?;
210
211    Ok(())
212}

214/// Recursively append all files under `src_dir` into the tar archive under
215/// `tar_prefix`. For example, if `src_dir` contains `file.sst` and
216/// `tar_prefix` is `db`, the archive entry will be `db/file.sst`.
217///
218/// Symlinks are rejected: fjall does not create them in its data directories,
219/// so the presence of one is either tampering or misconfiguration, and the
220/// restore path does not currently handle symlink tar entries correctly.
221fn append_dir_recursive<W: std::io::Write>(
222    archive: &mut tar::Builder<W>,
223    src_dir: &Path,
224    tar_prefix: &Path,
225) -> Result<()> {
226    for entry in walkdir::WalkDir::new(src_dir).min_depth(1) {
227        let entry = entry.with_context(|| {
228            format!(
229                "Failed to walk database snapshot directory {}",
230                src_dir.display()
231            )
232        })?;
233
234        if entry.file_type().is_symlink() {
235            anyhow::bail!(
236                "Refusing to archive symlink inside database snapshot directory: {}",
237                entry.path().display()
238            );
239        }
240
241        let relative = entry
242            .path()
243            .strip_prefix(src_dir)
244            .expect("walkdir entry is always under src_dir");
245        let archive_path = tar_prefix.join(relative);
246
247        if entry.file_type().is_dir() {
248            archive.append_dir(&archive_path, entry.path())?;
249        } else {
250            archive.append_path_with_name(entry.path(), &archive_path)?;
251        }
252    }
253    Ok(())
254}

256pub fn encrypted_backup_file_name() -> PathBuf {
257    // ISO 8601 basic format in UTC, e.g. 20260409T230419Z. Compact, sorts
258    // lexicographically, and contains no characters that need escaping on any
259    // common filesystem.
260    let timestamp = jiff::Timestamp::now()
261        .to_zoned(jiff::tz::TimeZone::UTC)
262        .strftime("%Y%m%dT%H%M%SZ")
263        .to_string();
264    PathBuf::from(format!("hashi-config-backup-{timestamp}.tar.age"))
265}

267fn append_backup_manifest<W: std::io::Write>(
268    archive: &mut tar::Builder<W>,
269    manifest: &BackupManifest,
270) -> Result<()> {
271    let manifest_bytes = toml::to_string_pretty(manifest)?.into_bytes();
272    let mut header = tar::Header::new_gnu();
273    header.set_size(manifest_bytes.len() as u64);
274    header.set_mode(0o644);
275    header.set_cksum();
276    archive.append_data(
277        &mut header,
278        BACKUP_MANIFEST_FILE_NAME,
279        manifest_bytes.as_slice(),
280    )?;
281    Ok(())
282}

284/// Determine the directory name to extract a backup tarball into.
285///
286/// Strips the `.tar.age` or `.age` suffix from the tarball's file name, so
287/// `hashi-config-backup-20260409T230419Z.tar.age` becomes
288/// `hashi-config-backup-20260409T230419Z`. An input without one of those
289/// suffixes is rejected rather than silently used verbatim, to avoid
290/// surprising extraction directory names when users point at the wrong file.
291pub fn extract_dir_name(backup_tarball: &Path) -> Result<PathBuf> {
292    let file_name = backup_tarball
293        .file_name()
294        .and_then(|name| name.to_str())
295        .ok_or_else(|| {
296            anyhow::anyhow!(
297                "Backup tarball path has no file name: {}",
298                backup_tarball.display()
299            )
300        })?;
301
302    let stem = file_name
303        .strip_suffix(".tar.age")
304        .or_else(|| file_name.strip_suffix(".age"))
305        .ok_or_else(|| {
306            anyhow::anyhow!(
307                "Backup tarball must have a .tar.age or .age suffix: {}",
308                backup_tarball.display()
309            )
310        })?;
311
312    // Require the stem to be exactly one plain directory component when
313    // joined under the user's output dir. Catches both the empty case (zero
314    // components) and traversal attempts like `../../tmp/pwn.tar.age` (a
315    // `ParentDir` component, not `Normal`).
316    let stem_path = Path::new(stem);
317    let mut components = stem_path.components();
318    let only = components.next();
319    let extra = components.next();
320    if extra.is_some() || !matches!(only, Some(Component::Normal(_))) {
321        anyhow::bail!(
322            "Backup tarball file name must be a single path component without separators or `..`: {}",
323            backup_tarball.display()
324        );
325    }
326
327    Ok(PathBuf::from(stem))
328}

330pub fn read_backup_manifest<R: Read>(
331    mut entry: tar::Entry<'_, R>,
332) -> Result<(BackupManifest, String)> {
333    let path = entry.path()?.into_owned();
334    let file_name = path.file_name().ok_or_else(|| {
335        anyhow::anyhow!(
336            "First tar entry does not have a file name: {}",
337            path.display()
338        )
339    })?;
340    if file_name != BACKUP_MANIFEST_FILE_NAME {
341        anyhow::bail!(
342            "Expected backup manifest {} as the first tar entry, found {}",
343            BACKUP_MANIFEST_FILE_NAME,
344            path.display()
345        );
346    }
347
348    let mut manifest_toml = String::new();
349    entry.read_to_string(&mut manifest_toml)?;
350    let manifest: BackupManifest = toml::from_str(&manifest_toml)?;
351
352    Ok((manifest, manifest_toml))
353}

355pub fn write_manifest_to_extract_dir(extract_dir: &Path, manifest_toml: &str) -> Result<()> {
356    let manifest_path = extract_dir.join(BACKUP_MANIFEST_FILE_NAME);
357    let mut file = create_file_strict(&manifest_path)?;
358    io::Write::write_all(&mut file, manifest_toml.as_bytes())
359        .with_context(|| format!("Failed to write manifest to {}", manifest_path.display()))?;
360    info!(path = %manifest_path.display(), "Restored manifest");
361    Ok(())
362}

364pub fn restore_backup_entries<R: Read>(
365    entries: tar::Entries<'_, R>,
366    output_dir: &Path,
367    manifest: &BackupManifest,
368) -> Result<()> {
369    let db_prefix = Path::new(DB_SNAPSHOT_TAR_PREFIX);
370    let expected_files: HashSet<PathBuf> = manifest
371        .paths
372        .iter()
373        .filter(|entry| entry.archive_name != db_prefix)
374        .map(|entry| entry.archive_name.clone())
375        .collect();
376    let expected_db_entries: HashSet<PathBuf> =
377        manifest.db_archive_entries.iter().cloned().collect();
378    let mut restored_db_entries = HashSet::new();
379    let mut restored_count: usize = 0;
380
381    for entry in entries {
382        let mut entry = entry?;
383        let archive_path = entry.path()?.into_owned();
384
385        if archive_path.starts_with(db_prefix) {
386            validate_db_archive_path(&archive_path)?;
387            if !expected_db_entries.contains(&archive_path) {
388                anyhow::bail!(
389                    "Backup archive contains unexpected database entry: {}",
390                    archive_path.display()
391                );
392            }
393            if !restored_db_entries.insert(archive_path.clone()) {
394                anyhow::bail!(
395                    "Backup archive contains duplicate database entry: {}",
396                    archive_path.display()
397                );
398            }
399            restore_db_entry(&mut entry, &archive_path, output_dir)?;
400        } else {
401            restore_config_entry(&mut entry, &archive_path, output_dir, &expected_files)?;
402            restored_count += 1;
403        }
404    }
405
406    if restored_count != expected_files.len() {
407        anyhow::bail!(
408            "Backup archive is missing file entries: expected {}, restored {}",
409            expected_files.len(),
410            restored_count
411        );
412    }
413
414    if restored_db_entries != expected_db_entries {
415        let mut missing_db_entries: Vec<_> = expected_db_entries
416            .difference(&restored_db_entries)
417            .cloned()
418            .collect();
419        missing_db_entries.sort();
420        anyhow::bail!(
421            "Backup archive is missing database entries: {}",
422            missing_db_entries
423                .into_iter()
424                .map(|path| path.display().to_string())
425                .collect::<Vec<_>>()
426                .join(", ")
427        );
428    }
429
430    Ok(())
431}

433/// Extract a single tar entry that lives under the db/ prefix into
434/// `output_dir`, preserving the relative directory structure. Directory
435/// entries are created (recursively, since the tar may stream them in any
436/// order); file entries are written with mode 0o600.
437fn validate_db_archive_path(archive_path: &Path) -> Result<()> {
438    let mut components = archive_path.components();
439    if !matches!(components.next(), Some(Component::Normal(prefix)) if prefix == DB_SNAPSHOT_TAR_PREFIX)
440    {
441        anyhow::bail!(
442            "Database entry must live under {}: {}",
443            DB_SNAPSHOT_TAR_PREFIX,
444            archive_path.display()
445        );
446    }
447
448    if archive_path
449        .components()
450        .skip(1)
451        .any(|component| !matches!(component, Component::Normal(_)))
452    {
453        anyhow::bail!(
454            "Database entry must not contain `..`, `.`, or absolute path components: {}",
455            archive_path.display()
456        );
457    }
458
459    Ok(())
460}

462fn restore_db_entry<R: Read>(
463    entry: &mut tar::Entry<'_, R>,
464    archive_path: &Path,
465    output_dir: &Path,
466) -> Result<()> {
467    let output_path = output_dir.join(archive_path);
468    if entry.header().entry_type() == tar::EntryType::Directory {
469        fs::create_dir_all(&output_path)
470            .with_context(|| format!("Failed to create directory {}", output_path.display()))?;
471        return Ok(());
472    }
473
474    if let Some(parent) = output_path.parent() {
475        fs::create_dir_all(parent)
476            .with_context(|| format!("Failed to create directory {}", parent.display()))?;
477    }
478    let mut output_file = create_file_strict(&output_path)?;
479    io::copy(entry, &mut output_file).with_context(|| {
480        format!(
481            "Failed to write restored file contents to {}",
482            output_path.display()
483        )
484    })?;
485    Ok(())
486}

488/// Extract a single config-file tar entry into `output_dir`. Config entries
489/// must be regular files at the tar root and must appear in the manifest;
490/// anything else is rejected so a tampered archive can't sneak unexpected
491/// files past the restore.
492fn restore_config_entry<R: Read>(
493    entry: &mut tar::Entry<'_, R>,
494    archive_path: &Path,
495    output_dir: &Path,
496    expected_files: &HashSet<PathBuf>,
497) -> Result<()> {
498    let archive_name = PathBuf::from(archive_path.file_name().ok_or_else(|| {
499        anyhow::anyhow!(
500            "Backup entry does not have a file name: {}",
501            archive_path.display()
502        )
503    })?);
504
505    let entry_type = entry.header().entry_type();
506    if entry_type != tar::EntryType::Regular {
507        anyhow::bail!(
508            "Backup entry {} has unexpected type {entry_type:?}; only regular files are supported",
509            archive_name.display(),
510        );
511    }
512
513    if archive_path != archive_name {
514        anyhow::bail!(
515            "Backup entry must be at the tar root: {}",
516            archive_path.display()
517        );
518    }
519
520    if !expected_files.contains(&archive_name) {
521        anyhow::bail!(
522            "Backup archive contains unexpected file: {}",
523            archive_name.display()
524        );
525    }
526
527    let output_path = output_dir.join(&archive_name);
528    let mut output_file = create_file_strict(&output_path)?;
529    io::copy(entry, &mut output_file).with_context(|| {
530        format!(
531            "Failed to write restored file contents to {}",
532            output_path.display()
533        )
534    })?;
535    info!(
536        archive_name = %archive_name.display(),
537        output = %output_path.display(),
538        "Restored file",
539    );
540    Ok(())
541}

543/// Copy each restored config file from the extract directory to its original
544/// path as recorded in the manifest. Refuses to overwrite anything.
545///
546/// `extract_dir` is where the tarball was unpacked; this function joins it
547/// with each entry's `archive_name` to find the source. The DB entry is
548/// skipped — restoring the snapshot dir is handled by
549/// [`copy_db_snapshot_to_original_path`].
550pub fn copy_restored_files_to_original_paths(
551    extract_dir: &Path,
552    manifest: &BackupManifest,
553) -> Result<()> {
554    let db_prefix = Path::new(DB_SNAPSHOT_TAR_PREFIX);
555
556    for file in manifest
557        .paths
558        .iter()
559        .filter(|entry| entry.archive_name != db_prefix)
560    {
561        let restored_path = extract_dir.join(&file.archive_name);
562
563        if let Some(parent) = file.original_path.parent() {
564            fs::create_dir_all(parent).with_context(|| {
565                format!("Failed to create parent directory {}", parent.display())
566            })?;
567        }
568
569        let mut source = File::open(&restored_path)
570            .with_context(|| format!("Failed to open {}", restored_path.display()))?;
571        // Custom AlreadyExists message: "original path" hints that the
572        // collision is with a path the manifest pointed at, not a freshly
573        // chosen output location.
574        let mut dest = OpenOptions::new()
575            .write(true)
576            .create_new(true)
577            .mode(0o600)
578            .open(&file.original_path)
579            .map_err(|e| match e.kind() {
580                ErrorKind::AlreadyExists => anyhow::anyhow!(
581                    "Refusing to overwrite existing original path: {}",
582                    file.original_path.display()
583                ),
584                _ => anyhow::Error::from(e)
585                    .context(format!("Failed to create {}", file.original_path.display())),
586            })?;
587        io::copy(&mut source, &mut dest).with_context(|| {
588            format!(
589                "Failed to copy {} to {}",
590                restored_path.display(),
591                file.original_path.display()
592            )
593        })?;
594        info!(
595            from = %restored_path.display(),
596            to = %file.original_path.display(),
597            "Copied restored file to original path",
598        );
599    }
600
601    Ok(())
602}

604/// Copy the extracted database snapshot directory to its original path.
605///
606/// Looks up the DB entry in the manifest to determine the original path,
607/// then recursively copies the extracted `db/` directory into a sibling
608/// staging directory and renames it into place on success. The rename is
609/// atomic on a single filesystem, so a crash mid-copy leaves the staging
610/// directory behind without ever creating a half-populated DB at `dest`.
611/// The staging directory is created with `tempdir_in(dest.parent())`, so it
612/// lives on the same filesystem as `dest`; that same-filesystem placement is
613/// required because `rename` does not work across mount points.
614///
615/// Fails if `dest` already exists.
616pub fn copy_db_snapshot_to_original_path(
617    extract_dir: &Path,
618    manifest: &BackupManifest,
619) -> Result<()> {
620    let db_prefix = Path::new(DB_SNAPSHOT_TAR_PREFIX);
621    let db_entry = manifest
622        .paths
623        .iter()
624        .find(|entry| entry.archive_name == db_prefix)
625        .ok_or_else(|| anyhow::anyhow!("Backup manifest does not contain a database entry"))?;
626
627    let source = extract_dir.join(DB_SNAPSHOT_TAR_PREFIX);
628    let dest = &db_entry.original_path;
629
630    // Pre-flight: refuse early if the destination already exists. This is
631    // checked again implicitly by the final rename, but failing here gives a
632    // clear error before doing all the copy work.
633    if dest
634        .try_exists()
635        .with_context(|| format!("Failed to stat database destination {}", dest.display()))?
636    {
637        anyhow::bail!(
638            "Refusing to overwrite existing database directory: {}",
639            dest.display()
640        );
641    }
642
643    let parent = dest.parent().ok_or_else(|| {
644        anyhow::anyhow!(
645            "Database destination has no parent directory: {}",
646            dest.display()
647        )
648    })?;
649    fs::create_dir_all(parent)
650        .with_context(|| format!("Failed to create parent directory {}", parent.display()))?;
651
652    // Stage into a sibling temp directory so the entire copy is invisible
653    // until the final rename. Using a sibling guarantees we're on the same
654    // filesystem, which is required for `rename` to be atomic.
655    let staging = tempfile::Builder::new()
656        .prefix(".hashi-db-restore-")
657        .tempdir_in(parent)
658        .with_context(|| {
659            format!(
660                "Failed to create db staging directory in {}",
661                parent.display()
662            )
663        })?;
664
665    copy_dir_recursive_strict(&source, staging.path()).with_context(|| {
666        format!(
667            "Failed to copy database snapshot from {} to staging directory {}",
668            source.display(),
669            staging.path().display()
670        )
671    })?;
672
673    // Convert the TempDir into an owned path before rename so we can do
674    // explicit best-effort cleanup ourselves if the rename fails.
675    let staging_path = staging.keep();
676    // `staging_path` was created as a sibling of `dest`, so this rename stays
677    // on one filesystem rather than crossing mounts; `fs::rename` does not
678    // work across mount points.
679    fs::rename(&staging_path, dest).map_err(|e| {
680        // Best-effort cleanup of the staging directory if the rename failed,
681        // since `into_path` consumed the auto-deleting guard.
682        let _ = fs::remove_dir_all(&staging_path);
683        match e.kind() {
684            ErrorKind::AlreadyExists => anyhow::anyhow!(
685                "Refusing to overwrite existing database directory: {}",
686                dest.display()
687            ),
688            _ => anyhow::Error::from(e).context(format!(
689                "Failed to move staged database into place at {}",
690                dest.display()
691            )),
692        }
693    })?;
694
695    info!(
696        from = %source.display(),
697        to = %dest.display(),
698        "Copied database snapshot to original path",
699    );
700
701    Ok(())
702}
703
704/// Recursively copy a directory tree from `src` to `dest` without ever
705/// overwriting existing files.
706///
707/// - The root `dest` is assumed to have already been created by the caller.
708/// - Subdirectories below the root use `create_dir` (non-recursive), so any
709///   collision surfaces as an error rather than merging into existing state.
710/// - Files are opened with `create_new(true)` and mode `0o600`, matching the
711///   protection `copy_restored_files_to_original_paths` applies to config
712///   files. fjall on-disk files are not sensitive individually, but we keep
713///   the mode consistent so the restored DB never advertises looser
714///   permissions than the backup did.
715/// - Symlinks are rejected outright.
716fn copy_dir_recursive_strict(src: &Path, dest: &Path) -> Result<()> {
717    for entry in walkdir::WalkDir::new(src).min_depth(1) {
718        let entry = entry?;
719
720        if entry.file_type().is_symlink() {
721            anyhow::bail!(
722                "Refusing to restore symlink from database snapshot: {}",
723                entry.path().display()
724            );
725        }
726
727        let relative = entry.path().strip_prefix(src).expect("walkdir under src");
728        let target = dest.join(relative);
729
730        if entry.file_type().is_dir() {
731            create_dir_strict(&target)?;
732        } else {
733            let mut source_file = File::open(entry.path())
734                .with_context(|| format!("Failed to open {}", entry.path().display()))?;
735            let mut dest_file = create_file_strict(&target)?;
736            io::copy(&mut source_file, &mut dest_file).with_context(|| {
737                format!(
738                    "Failed to copy {} to {}",
739                    entry.path().display(),
740                    target.display()
741                )
742            })?;
743        }
744    }
745    Ok(())
746}

769    fn append_regular_file<W: std::io::Write>(
770        archive: &mut tar::Builder<W>,
771        path: &str,
772        contents: &[u8],
773    ) {
774        let mut header = tar::Header::new_gnu();
775        header.set_entry_type(tar::EntryType::Regular);
776        header.set_mode(0o600);
777        header.set_size(contents.len() as u64);
778        header.set_cksum();
779        archive.append_data(&mut header, path, contents).unwrap();
780    }

782    fn build_archive_bytes(manifest: &BackupManifest, db_files: &[(&str, &[u8])]) -> Vec<u8> {
783        let mut tar_bytes = Vec::new();
784        {
785            let mut archive = tar::Builder::new(&mut tar_bytes);
786            append_backup_manifest(&mut archive, manifest).unwrap();
787            for (path, contents) in db_files {
788                append_regular_file(&mut archive, path, contents);
789            }
790            archive.finish().unwrap();
791        }
792        tar_bytes
793    }
