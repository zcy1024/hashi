104    async fn run(mut self) {
105        info!("Starting leader service");
106
107        // Wait for DKG to complete before processing any checkpoints.
108        let mpc_handle = self.inner.mpc_handle().expect("MpcHandle not initialized");
109        info!("Waiting for MPC key to become available...");
110        mpc_handle.wait_for_key_ready().await;
111        info!("MPC key is ready, starting leader loop");
112
113        let mut checkpoint_rx = self.inner.onchain_state().subscribe_checkpoint();
114
115        loop {
116            trace!("Waiting for next checkpoint or task completion...");
117            tokio::select! {
118                wait_result = checkpoint_rx.changed() => {
119                    if let Err(e) = wait_result {
120                        error!("Error waiting for checkpoint change: {e}");
121                        break;
122                    }
123                    let (checkpoint_height, checkpoint_timestamp_ms) = {
124                        let checkpoint_info = checkpoint_rx.borrow_and_update();
125                        (checkpoint_info.height, checkpoint_info.timestamp_ms)
126                    };
127
128                    let is_leader = self.is_current_leader(checkpoint_height);
129                    self.inner.metrics.is_leader.set(i64::from(is_leader));
130                    if is_leader {
131                        debug!("Checkpoint {checkpoint_height}: We are the leader node");
132                    } else {
133                        trace!("We are not the leader node");
134                        continue;
135                    }
136
137                    // TODO: Remove this drain once the leader loop is restructured
138                    // to avoid the checkpoint branch blocking task draining.
139                    while let Some(result) = self.deposit_tasks.try_join_next() {
140                        self.handle_completed_deposit_task(result);
141                    }
142
143                    self.process_deposit_requests(checkpoint_timestamp_ms);
144                    self.process_unapproved_withdrawal_requests(checkpoint_timestamp_ms);
145                    self.process_approved_withdrawal_requests(checkpoint_timestamp_ms);
146                    self.process_unsigned_pending_withdrawals();
147                    self.process_signed_pending_withdrawals();
148                    self.check_delete_proposals(checkpoint_timestamp_ms);
149                }
150                Some(result) = self.deposit_tasks.join_next() => {
151                    self.handle_completed_deposit_task(result);
152                    // Drain any other completed tasks while we're here.
153                    while let Some(result) = self.deposit_tasks.try_join_next() {
154                        self.handle_completed_deposit_task(result);
155                    }
156                }
157                Some(result) = self.withdrawal_signing_tasks.join_next() => {
158                    self.handle_completed_withdrawal_signing_task(result);
159                }
160                Some(result) = self.withdrawal_broadcast_tasks.join_next() => {
161                    self.handle_completed_withdrawal_broadcast_task(result);
162                }
163                Some(result) = OptionFuture::from(self.withdrawal_approval_task.as_mut()) => {
164                    self.withdrawal_approval_task = None;
165                    Self::log_task_result("withdrawal_approval", result);
166                }
167                Some(result) = OptionFuture::from(self.withdrawal_commitment_task.as_mut()) => {
168                    self.withdrawal_commitment_task = None;
169                    Self::log_task_result("withdrawal_commitment", result);
170                }
171                Some(result) = OptionFuture::from(self.deposit_gc_task.as_mut()) => {
172                    self.deposit_gc_task = None;
173                    Self::log_task_result("deposit_gc", result);
174                }
175                Some(result) = OptionFuture::from(self.proposal_gc_task.as_mut()) => {
176                    self.proposal_gc_task = None;
177                    Self::log_task_result("proposal_gc", result);
178                }
179
180            }
181        }
182    }

185        &mut self,
186        result: Result<(Address, anyhow::Result<()>), tokio::task::JoinError>,
187    ) {
188        let mapped = match result {
189            Ok((deposit_id, inner)) => {
190                self.inflight_deposits.remove(&deposit_id);
191                Ok(inner)
192            }
193            Err(e) => Err(e),
194        };
195        Self::log_task_result("deposit", mapped);
196    }

227        match result {
228            Ok(Ok(())) => {}
229            Ok(Err(err)) => error!("{label} task failed: {err:#}"),

235    pub fn is_current_leader(&self, checkpoint_height: u64) -> bool {
236        match self.inner.config.force_run_as_leader() {
237            ForceRunAsLeader::Always => return true,
238            ForceRunAsLeader::Never => return false,
239            ForceRunAsLeader::Default => (),
240        }
241
242        let Some(committee) = self.inner.onchain_state().current_committee() else {
243            // TODO: do we need to do anything when bootstrapping? At genesis there is no committee.
244            return false;
245        };
246        let this_validator_address = self
247            .inner
248            .config
249            .validator_address()
250            .expect("No configured validator address");
251        let Some(this_validator_idx) = committee
252            .index_of(&this_validator_address)
253            .map(|i| i as u64)
254        else {
255            // We are not in the committee yet, so we cannot be the leader
256            return false;
257        };
258        let num_validators = committee.members().len() as u64;
259
260        let current_turn = checkpoint_height / NUM_CONSECUTIVE_LEADER_CHECKPOINTS;
261        let is_leader = (current_turn % num_validators) == this_validator_idx;
262
263        trace!("Node index {this_validator_idx} is leader node: {is_leader}");
264        is_leader
265    }

267    fn process_deposit_requests(&mut self, checkpoint_timestamp_ms: u64) {
268        debug!("Entering process_deposit_requests");
269        let mut deposit_requests = self.inner.onchain_state().deposit_requests();
270        // Sort deposit_requests by timestamp, from earliest to latest
271        deposit_requests.sort_by_key(|r| r.timestamp_ms);
272        let deposit_ids: Vec<Address> = deposit_requests.iter().map(|r| r.id).collect();
273        self.deposit_retry_tracker.prune(&deposit_ids);
274        // TODO: If we keep AbortHandles for deposit tasks, explicitly abort tasks whose
275        // deposit IDs are no longer present here so they do not linger until timeout.
276        self.inflight_deposits
277            .retain(|deposit_id| deposit_ids.contains(deposit_id));
278
279        debug!("Processing {} deposit requests", deposit_requests.len());
280
281        let max_concurrent = self.inner.config.max_concurrent_leader_job_tasks();
282        for deposit_request in deposit_requests.iter().cloned() {
283            if self.deposit_tasks.len() >= max_concurrent {
284                break;
285            }
286            if self.inflight_deposits.contains(&deposit_request.id) {
287                continue;
288            }
289            if self
290                .deposit_retry_tracker
291                .should_skip(&deposit_request.id, checkpoint_timestamp_ms)
292            {
293                continue;
294            }
295
296            let deposit_id = deposit_request.id;
297            let inner = self.inner.clone();
298            let deposit_retry_tracker = self.deposit_retry_tracker.clone();
299
300            self.inflight_deposits.insert(deposit_id);
301            self.deposit_tasks.spawn(async move {
302                let task_result = tokio::time::timeout(
303                    LEADER_TASK_TIMEOUT,
304                    Self::process_deposit_request(
305                        inner.clone(),
306                        deposit_retry_tracker.clone(),
307                        deposit_request,
308                        checkpoint_timestamp_ms,
309                    ),
310                )
311                .await;
312
313                let (result, failure_kind) = match task_result {
314                    Ok(result) => (result, None),
315                    Err(_) => {
316                        let kind = DepositRequestErrorKind::TimedOut;
317                        inner
318                            .metrics
319                            .leader_retries_total
320                            .with_label_values(&["deposit", &format!("{kind:?}")])
321                            .inc();
322                        deposit_retry_tracker.record_failure(
323                            kind,
324                            deposit_id,
325                            checkpoint_timestamp_ms,
326                        );
327                        (
328                            Err(anyhow::anyhow!(
329                                "deposit {deposit_id} timed out after {LEADER_TASK_TIMEOUT:?}"
330                            )),
331                            Some(kind),
332                        )
333                    }
334                };
335
336                if result.is_err() && failure_kind.is_none() {
337                    let kind = DepositRequestErrorKind::TaskFailed;
338                    inner
339                        .metrics
340                        .leader_retries_total
341                        .with_label_values(&["deposit", &format!("{kind:?}")])
342                        .inc();
343                    deposit_retry_tracker.record_failure(kind, deposit_id, checkpoint_timestamp_ms);
344                }
345
346                (deposit_id, result)
347            });
348        }
349
350        self.inner
351            .metrics
352            .leader_items_in_backoff
353            .with_label_values(&["deposit"])
354            .set(
355                self.deposit_retry_tracker
356                    .in_backoff_count(checkpoint_timestamp_ms) as i64,
357            );
358
359        self.check_delete_expired_deposit_requests(&deposit_requests, checkpoint_timestamp_ms);
360    }

362    async fn process_deposit_request(
363        inner: Arc<Hashi>,
364        deposit_retry_tracker: RetryTracker<DepositRequestErrorKind>,
365        deposit_request: DepositRequest,
366        checkpoint_timestamp_ms: u64,
367    ) -> anyhow::Result<()> {
368        info!(deposit_request_id = %deposit_request.id, "Processing deposit request");
369
370        // Validate deposit_request before asking for signatures
371        match inner.validate_deposit_request(&deposit_request).await {
372            Ok(()) => {}
373            Err(e) => {
374                debug!(request_id = ?deposit_request.id, "Deposit validation failed: {e}");
375                let kind = e.kind();
376                inner
377                    .metrics
378                    .leader_retries_total
379                    .with_label_values(&["deposit", &format!("{kind:?}")])
380                    .inc();
381                deposit_retry_tracker.record_failure(
382                    kind,
383                    deposit_request.id,
384                    checkpoint_timestamp_ms,
385                );
386                return Ok(());
387            }
388        }
389
390        info!(deposit_request_id = %deposit_request.id, "Deposit request validated successfully");
391
392        let proto_request = deposit_request.to_proto();
393        let members = inner
394            .onchain_state()
395            .current_committee_members()
396            .expect("No current committee members");
397
398        let committee = inner
399            .onchain_state()
400            .current_committee()
401            .expect("No current committee");
402
403        let required_weight = certificate_threshold(committee.total_weight());
404
405        // Fan out signature requests to all members in parallel.
406        let mut sig_tasks = JoinSet::new();
407        for member in members {
408            let inner = inner.clone();
409            let proto_request = proto_request.clone();
410            sig_tasks.spawn(async move {
411                Self::request_deposit_confirmation_signature(&inner, proto_request, &member).await
412            });
413        }
414
415        // Collect signatures, stopping once we reach quorum.
416        let confirmation_message = DepositConfirmationMessage {
417            request_id: deposit_request.id,
418            utxo: deposit_request.utxo.clone(),
419        };
420        let mut aggregator = BlsSignatureAggregator::new(&committee, confirmation_message);
421        while let Some(result) = sig_tasks.join_next().await {
422            let Ok(Some(sig)) = result else { continue };
423            if let Err(e) = aggregator.add_signature(sig) {
424                error!(deposit_request_id = %deposit_request.id, "Failed to add deposit signature: {e}");
425            }
426            if aggregator.weight() >= required_weight {
427                break;
428            }
429        }
430
431        if aggregator.weight() < required_weight {
432            anyhow::bail!(
433                "Aggregate weight of signatures {} is less than required weight {required_weight}",
434                aggregator.weight()
435            );
436        }
437
438        let signed_message = aggregator.finish()?;
439        let mut executor = SuiTxExecutor::from_hashi(inner.clone())?;
440        executor
441            .execute_confirm_deposit(&deposit_request, signed_message)
442            .await
443            .inspect(|()| {
444                inner
445                    .metrics
446                    .sui_tx_submissions_total
447                    .with_label_values(&["confirm_deposit", "success"])
448                    .inc();
449                inner.metrics.deposits_confirmed_total.inc();
450                info!(deposit_request_id = %deposit_request.id, "Successfully submitted deposit confirmation");
451            })
452            .inspect_err(|_| {
453                inner
454                    .metrics
455                    .sui_tx_submissions_total
456                    .with_label_values(&["confirm_deposit", "failure"])
457                    .inc();
458            })?;
459        deposit_retry_tracker.clear(&deposit_request.id);
460        Ok(())
461    }

463    async fn request_deposit_confirmation_signature(
464        inner: &Arc<Hashi>,
465        proto_request: SignDepositConfirmationRequest,
466        member: &CommitteeMember,
467    ) -> Option<MemberSignature> {
468        let validator_address = member.validator_address();
469        trace!(
470            "Requesting deposit confirmation signature from {}",
471            validator_address
472        );
473
474        let mut rpc_client = inner
475            .onchain_state()
476            .bridge_service_client(&validator_address)
477            .or_else(|| {
478                error!(
479                    "Cannot find client for validator address: {:?}",
480                    validator_address
481                );
482                None
483            })?;
484
485        let response = rpc_client
486            .sign_deposit_confirmation(proto_request.clone())
487            .await
488            .inspect_err(|e| {
489                error!(
490                    "Failed to get deposit confirmation signature from {}: {e}",
491                    validator_address
492                );
493            })
494            .ok()?;
495
496        trace!(
497            "Retrieved deposit confirmation signature from {}",
498            validator_address
499        );
500
501        response
502            .into_inner()
503            .member_signature
504            .ok_or_else(|| anyhow::anyhow!("No member_signature in response"))
505            .and_then(parse_member_signature)
506            .inspect_err(|e| {
507                error!(
508                    "Failed to parse member signature from response from {}: {e}",
509                    validator_address
510                );
511            })
512            .ok()
513    }

519    fn process_unapproved_withdrawal_requests(&mut self, checkpoint_timestamp_ms: u64) {
520        debug!("Entering process_unapproved_withdrawal_requests");
521
522        if self.withdrawal_approval_task.is_some() {
523            debug!("Withdrawal approval task already in-flight, skipping");
524            return;
525        }
526
527        let mut unapproved: Vec<_> = self
528            .inner
529            .onchain_state()
530            .withdrawal_requests()
531            .into_iter()
532            .filter(|r| r.status.is_requested())
533            .collect();
534        unapproved.sort_by_key(|r| r.timestamp_ms);
535
536        let unapproved_ids: Vec<Address> = unapproved.iter().map(|r| r.id).collect();
537        self.withdrawal_approval_retry_tracker
538            .prune(&unapproved_ids);
539
540        let to_process: Vec<_> = unapproved
541            .into_iter()
542            .filter(|r| {
543                !self
544                    .withdrawal_approval_retry_tracker
545                    .should_skip(&r.id, checkpoint_timestamp_ms)
546            })
547            .collect();
548
549        self.inner
550            .metrics
551            .leader_items_in_backoff
552            .with_label_values(&["withdrawal_approval"])
553            .set(
554                self.withdrawal_approval_retry_tracker
555                    .in_backoff_count(checkpoint_timestamp_ms) as i64,
556            );
557
558        if to_process.is_empty() {
559            return;
560        }
561
562        let inner = self.inner.clone();
563        let retry_tracker = self.withdrawal_approval_retry_tracker.clone();
564
565        self.withdrawal_approval_task = Some(tokio::task::spawn(async move {
566            Self::process_unapproved_withdrawal_requests_task(
567                inner,
568                retry_tracker,
569                to_process,
570                checkpoint_timestamp_ms,
571            )
572            .await
573        }));
574    }

576    async fn process_unapproved_withdrawal_requests_task(
577        inner: Arc<Hashi>,
578        retry_tracker: RetryTracker<WithdrawalApprovalErrorKind>,
579        to_process: Vec<WithdrawalRequest>,
580        checkpoint_timestamp_ms: u64,
581    ) -> anyhow::Result<()> {
582        let max_concurrent = inner.config.max_concurrent_leader_job_tasks();
583
584        let this_validator_address = inner
585            .config
586            .validator_address()
587            .expect("No configured validator address");
588
589        let members = inner
590            .onchain_state()
591            .current_committee_members()
592            .expect("No current committee members");
593
594        let committee = inner
595            .onchain_state()
596            .current_committee()
597            .expect("No current committee");
598
599        let mut tasks = JoinSet::new();
600        let mut certified: Vec<(Address, CommitteeSignature)> = Vec::new();
601
602        for request in to_process {
603            if tasks.len() >= max_concurrent {
604                // Wait for one to finish before spawning more.
605                if let Some(result) = tasks.join_next().await {
606                    match &result {
607                        Err(err) if err.is_panic() => {
608                            error!("Withdrawal approval task panicked: {err}")
609                        }
610                        Err(err) => error!("Withdrawal approval task failed to join: {err}"),
611                        Ok(_) => {}
612                    }
613                    if let Ok((_request_id, Ok(Some(cert)))) = result {
614                        certified.push(cert);
615                    }
616                }
617            }
618
619            let inner = inner.clone();
620            let retry_tracker = retry_tracker.clone();
621            let members = members.clone();
622            let committee = committee.clone();
623            tasks.spawn(async move {
624                let request_id = request.id;
625                let task_result = tokio::time::timeout(
626                    LEADER_TASK_TIMEOUT,
627                    Self::process_unapproved_withdrawal_request(
628                        inner.clone(),
629                        retry_tracker.clone(),
630                        request,
631                        checkpoint_timestamp_ms,
632                        this_validator_address,
633                        &members,
634                        &committee,
635                    ),
636                )
637                .await;
638
639                let (result, failure_kind) = match task_result {
640                    Ok(result) => (result, None),
641                    Err(_) => {
642                        let kind = WithdrawalApprovalErrorKind::TimedOut;
643                        inner
644                            .metrics
645                            .leader_retries_total
646                            .with_label_values(&["withdrawal_approval", &format!("{kind:?}")])
647                            .inc();
648                        retry_tracker.record_failure(kind, request_id, checkpoint_timestamp_ms);
649                        (Ok(None), Some(kind))
650                    }
651                };
652
653                if result.is_err() && failure_kind.is_none() {
654                    let kind = WithdrawalApprovalErrorKind::TaskFailed;
655                    inner
656                        .metrics
657                        .leader_retries_total
658                        .with_label_values(&["withdrawal_approval", &format!("{kind:?}")])
659                        .inc();
660                    retry_tracker.record_failure(kind, request_id, checkpoint_timestamp_ms);
661                }
662                if let Err(err) = &result {
663                    error!(
664                        withdrawal_request_id = %request_id,
665                        "Withdrawal approval failed: {err:#}"
666                    );
667                }
668
669                (request_id, result)
670            });
671        }
672
673        while let Some(result) = tasks.join_next().await {
674            match &result {
675                Err(err) if err.is_panic() => error!("Withdrawal approval task panicked: {err}"),
676                Err(err) => error!("Withdrawal approval task failed to join: {err}"),
677                Ok(_) => {}
678            }
679            if let Ok((_request_id, Ok(Some(cert)))) = result {
680                certified.push(cert);
681            }
682        }
683
684        if certified.is_empty() {
685            return Ok(());
686        }
687
688        Self::submit_approve_withdrawal_requests_with_retry(&inner, certified).await;
689        Ok(())
690    }

692    async fn process_unapproved_withdrawal_request(
693        inner: Arc<Hashi>,
694        retry_tracker: RetryTracker<WithdrawalApprovalErrorKind>,
695        request: WithdrawalRequest,
696        checkpoint_timestamp_ms: u64,
697        this_validator_address: Address,
698        members: &[CommitteeMember],
699        committee: &hashi_types::committee::Committee,
700    ) -> anyhow::Result<Option<(Address, CommitteeSignature)>> {
701        let approval = WithdrawalRequestApproval {
702            request_id: request.id,
703        };
704
705        // Validate, screen, and sign locally first
706        let local_sig = match inner
707            .validate_and_sign_withdrawal_request_approval(&approval)
708            .await
709        {
710            Ok(sig) => {
711                retry_tracker.clear(&request.id);
712                parse_member_signature(sig).unwrap()
713            }
714            Err(e) => {
715                let kind = e.kind();
716                inner
717                    .metrics
718                    .leader_retries_total
719                    .with_label_values(&["withdrawal_approval", &format!("{kind:?}")])
720                    .inc();
721                retry_tracker.record_failure(kind, request.id, checkpoint_timestamp_ms);
722                return Ok(None);
723            }
724        };
725
726        let proto_request = approval.to_proto();
727        let required_weight = certificate_threshold(committee.total_weight());
728
729        let mut aggregator = BlsSignatureAggregator::new(committee, approval);
730        if let Err(e) = aggregator.add_signature(local_sig) {
731            error!(withdrawal_request_id = %request.id, "Failed to add local approval signature: {e}");
732        }
733
734        // Fan out signature requests to remote members in parallel.
735        let mut sig_tasks = JoinSet::new();
736        for member in members {
737            if member.validator_address() == this_validator_address {
738                continue;
739            }
740            let inner = inner.clone();
741            let proto_request = proto_request.clone();
742            let member = member.clone();
743            sig_tasks.spawn(async move {
744                Self::request_withdrawal_approval_signature(&inner, proto_request, &member).await
745            });
746        }
747
748        // Collect signatures, stopping once we reach quorum.
749        while let Some(result) = sig_tasks.join_next().await {
750            let Ok(Some(sig)) = result else { continue };
751            if let Err(e) = aggregator.add_signature(sig) {
752                error!(withdrawal_request_id = %request.id, "Failed to add approval signature: {e}");
753            }
754            if aggregator.weight() >= required_weight {
755                break;
756            }
757        }
758
759        let weight = aggregator.weight();
760        if weight < required_weight {
761            inner
762                .metrics
763                .leader_retries_total
764                .with_label_values(&["withdrawal_approval", "FailedQuorum"])
765                .inc();
766            retry_tracker.record_failure(
767                WithdrawalApprovalErrorKind::FailedQuorum,
768                request.id,
769                checkpoint_timestamp_ms,
770            );
771            error!(withdrawal_request_id = %request.id, "Insufficient approval signatures: weight {weight} < {required_weight}");
772            return Ok(None);
773        }
774
775        match aggregator.finish() {
776            Ok(signed) => Ok(Some((request.id, signed.committee_signature().clone()))),
777            Err(e) => {
778                error!(withdrawal_request_id = %request.id, "Failed to build approval certificate: {e}");
779                Ok(None)
780            }
781        }
782    }

784    async fn submit_approve_withdrawal_requests_with_retry(
785        inner: &Arc<Hashi>,
786        mut certified: Vec<(Address, CommitteeSignature)>,
787    ) {
788        loop {
789            let approvals: Vec<(Address, &CommitteeSignature)> =
790                certified.iter().map(|(id, cert)| (*id, cert)).collect();
791
792            let result = Self::submit_approve_withdrawal_requests(inner, &approvals)
793                .await
794                .inspect(|()| {
795                    inner
796                        .metrics
797                        .sui_tx_submissions_total
798                        .with_label_values(&["approve_withdrawal", "success"])
799                        .inc();
800                })
801                .inspect_err(|_| {
802                    inner
803                        .metrics
804                        .sui_tx_submissions_total
805                        .with_label_values(&["approve_withdrawal", "failure"])
806                        .inc();
807                });
808            let Err(e) = result else { return };
809
810            let err_msg = format!("{e}");
811            error!("approve_request PTB failed: {err_msg}");
812
813            // Try to identify which request caused the failure by checking
814            // which ones no longer exist in the queue (canceled).
815            let before_len = certified.len();
816            certified.retain(|(id, _)| inner.onchain_state().withdrawal_request(id).is_some());
817
818            if certified.len() == before_len {
819                error!("Could not identify failed request, aborting retry");
820                return;
821            }
822            if certified.is_empty() {
823                return;
824            }
825
826            info!(
827                "Retrying approve_request with {} remaining requests",
828                certified.len()
829            );
830        }
831    }

837    fn process_approved_withdrawal_requests(&mut self, checkpoint_timestamp_ms: u64) {
838        debug!("Entering process_approved_withdrawal_requests");
839
840        if self.withdrawal_commitment_task.is_some() {
841            debug!("Withdrawal commitment task already in-flight, skipping");
842            return;
843        }
844
845        let mut approved: Vec<_> = self
846            .inner
847            .onchain_state()
848            .withdrawal_requests()
849            .into_iter()
850            .filter(|r| r.status.is_approved())
851            .collect();
852        approved.sort_by_key(|r| r.timestamp_ms);
853
854        if self
855            .withdrawal_commitment_retry_tracker
856            .should_skip(checkpoint_timestamp_ms)
857        {
858            self.inner
859                .metrics
860                .leader_items_in_backoff
861                .with_label_values(&["withdrawal_commitment"])
862                .set(
863                    self.withdrawal_commitment_retry_tracker
864                        .in_backoff_count(checkpoint_timestamp_ms) as i64,
865                );
866            return;
867        }
868
869        // Collect all approved requests and process them as a single batch.
870        // The coin selection algorithm picks up to
871        // max_withdrawal_requests oldest requests from the slice.
872        let batch: Vec<WithdrawalRequest> = approved.into_iter().collect();
873
874        self.inner
875            .metrics
876            .leader_items_in_backoff
877            .with_label_values(&["withdrawal_commitment"])
878            .set(
879                self.withdrawal_commitment_retry_tracker
880                    .in_backoff_count(checkpoint_timestamp_ms) as i64,
881            );
882
883        if batch.is_empty() {
884            return;
885        }
886
887        let max_batch = self.inner.config.withdrawal_max_batch_size();
888        let delay_ms = self.inner.config.withdrawal_batching_delay_ms();
889
890        let batch_is_full = batch.len() >= max_batch;
891        let oldest_has_waited = batch
892            .first()
893            .is_some_and(|r| checkpoint_timestamp_ms >= r.timestamp_ms + delay_ms);
894
895        if !batch_is_full && !oldest_has_waited {
896            debug!(
897                "Holding {} approved request(s): oldest is {}ms old, \
898                 waiting for {}ms delay or {} requests",
899                batch.len(),
900                checkpoint_timestamp_ms.saturating_sub(batch[0].timestamp_ms),
901                delay_ms,
902                max_batch,
903            );
904            return;
905        }
906
907        let inner = self.inner.clone();
908        let retry_tracker = self.withdrawal_commitment_retry_tracker.clone();
909
910        self.withdrawal_commitment_task = Some(tokio::task::spawn(async move {
911            let task_result = tokio::time::timeout(
912                LEADER_TASK_TIMEOUT,
913                Self::process_approved_withdrawal_request_batch(
914                    inner.clone(),
915                    retry_tracker.clone(),
916                    batch,
917                    checkpoint_timestamp_ms,
918                ),
919            )
920            .await;
921
922            match task_result {
923                Ok(result) => result,
924                Err(_) => {
925                    let kind = WithdrawalCommitmentErrorKind::TimedOut;
926                    inner
927                        .metrics
928                        .leader_retries_total
929                        .with_label_values(&["withdrawal_commitment", &format!("{kind:?}")])
930                        .inc();
931                    Err(anyhow::anyhow!(
932                        "withdrawal commitment timed out after {LEADER_TASK_TIMEOUT:?}"
933                    ))
934                }
935            }
936        }));
937    }

939    async fn process_approved_withdrawal_request_batch(
940        inner: Arc<Hashi>,
941        retry_tracker: GlobalRetryTracker<WithdrawalCommitmentErrorKind>,
942        requests: Vec<WithdrawalRequest>,
943        checkpoint_timestamp_ms: u64,
944    ) -> anyhow::Result<()> {
945        info!(
946            withdrawal_request_ids = ?requests.iter().map(|r| r.id).collect::<Vec<_>>(),
947            "Processing batch of {} approved withdrawal request(s)",
948            requests.len(),
949        );
950
951        // Build the withdrawal tx commitment for the batch.
952        let approval = match inner.build_withdrawal_tx_commitment(&requests).await {
953            Ok(approval) => {
954                retry_tracker.clear();
955                approval
956            }
957            Err(e) => {
958                let kind = e.kind();
959                inner
960                    .metrics
961                    .leader_retries_total
962                    .with_label_values(&["withdrawal_commitment", &format!("{kind:?}")])
963                    .inc();
964                retry_tracker.record_failure(kind, checkpoint_timestamp_ms);
965                return Ok(());
966            }
967        };
968
969        // Fan out to committee for BLS signatures over the commitment message
970        let members = inner
971            .onchain_state()
972            .current_committee_members()
973            .expect("No current committee members");
974
975        let committee = inner
976            .onchain_state()
977            .current_committee()
978            .expect("No current committee");
979
980        let required_weight = certificate_threshold(committee.total_weight());
981        let proto_request = approval.to_proto();
982
983        // Fan out signature requests to all members in parallel.
984        let mut sig_tasks = JoinSet::new();
985        for member in members {
986            let inner = inner.clone();
987            let proto_request = proto_request.clone();
988            sig_tasks.spawn(async move {
989                Self::request_withdrawal_tx_commitment_signature(&inner, proto_request, &member)
990                    .await
991            });
992        }
993
994        // Collect signatures, stopping once we reach quorum.
995        let mut aggregator = BlsSignatureAggregator::new(&committee, approval.clone());
996        while let Some(result) = sig_tasks.join_next().await {
997            let Ok(Some(sig)) = result else { continue };
998            if let Err(e) = aggregator.add_signature(sig) {
999                error!("Failed to add withdrawal commitment signature: {e}");
1000            }
1001            if aggregator.weight() >= required_weight {
1002                break;
1003            }
1004        }
1005
1006        if aggregator.weight() < required_weight {
1007            inner
1008                .metrics
1009                .leader_retries_total
1010                .with_label_values(&["withdrawal_commitment", "FailedQuorum"])
1011                .inc();
1012            retry_tracker.record_failure(
1013                WithdrawalCommitmentErrorKind::FailedQuorum,
1014                checkpoint_timestamp_ms,
1015            );
1016            error!(
1017                "Insufficient withdrawal commitment signatures: weight {} < {required_weight}",
1018                aggregator.weight()
1019            );
1020            return Ok(());
1021        }
1022
1023        let signed_approval = match aggregator.finish() {
1024            Ok(signed_approval) => signed_approval,
1025            Err(e) => {
1026                error!("Failed to build withdrawal commitment certificate: {e}");
1027                return Ok(());
1028            }
1029        };
1030
1031        // Submit commit_withdrawal_tx to Sui
1032        Self::submit_commit_withdrawal_tx(&inner, &approval, signed_approval.committee_signature())
1033            .await
1034            .inspect(|()| {
1035                inner
1036                    .metrics
1037                    .sui_tx_submissions_total
1038                    .with_label_values(&["commit_withdrawal", "success"])
1039                    .inc();
1040            })
1041            .inspect_err(|e| {
1042                inner
1043                    .metrics
1044                    .sui_tx_submissions_total
1045                    .with_label_values(&["commit_withdrawal", "failure"])
1046                    .inc();
1047                error!("Failed to submit commit_withdrawal_tx: {e}");
1048            })?;
1049
1050        Ok(())
1051    }

1099    async fn process_unsigned_pending_withdrawal(
1100        inner: Arc<Hashi>,
1101        pending: PendingWithdrawal,
1102    ) -> anyhow::Result<()> {
1103        // If the pending withdrawal is from a previous epoch, reassign its presig
1104        // indices from the new epoch's counter before signing.
1105        // TODO: Batch multiple stale-epoch withdrawals into a single PTB.
1106        let current_epoch = inner.onchain_state().epoch();
1107        if pending.epoch != current_epoch {
1108            info!(
1109                pending_withdrawal_id = %pending.id,
1110                "Pending withdrawal from epoch {} (current {}), reassigning presig indices",
1111                pending.epoch, current_epoch,
1112            );
1113            let mut executor = SuiTxExecutor::from_hashi(inner.clone())?;
1114            executor
1115                .execute_allocate_presigs_for_pending_withdrawal(pending.id)
1116                .await?;
1117            info!(
1118                pending_withdrawal_id = %pending.id,
1119                "Presig indices reassigned, will sign on next checkpoint"
1120            );
1121            // Return and let the next checkpoint iteration pick up the updated state.
1122            return Ok(());
1123        }
1124        info!(pending_withdrawal_id = %pending.id, "MPC signing pending withdrawal");
1125
1126        let members = inner
1127            .onchain_state()
1128            .current_committee_members()
1129            .expect("No current committee members");
1130
1131        // 1. Request signed withdrawal tx witnesses from committee members.
1132        // MPC signing requires all threshold members to participate simultaneously
1133        // via P2P, so we must fan out requests in parallel.
1134        let signatures_by_input =
1135            Self::collect_withdrawal_tx_signatures(&inner, &pending.id, &members)
1136                .await
1137                .ok_or_else(|| {
1138                    anyhow::anyhow!("Failed to collect MPC signatures for {:?}", pending.id)
1139                })?;
1140
1141        // 2. Extract raw signature bytes for on-chain storage
1142        let witness_signatures: Vec<Vec<u8>> = signatures_by_input
1143            .iter()
1144            .map(|s| s.to_byte_array().to_vec())
1145            .collect();
1146
1147        // 3. Build the WithdrawalTxSigning and get BLS certificate via fan-out
1148        let signed_message = WithdrawalTxSigning {
1149            withdrawal_id: pending.id,
1150            request_ids: pending.request_ids.clone(),
1151            signatures: witness_signatures.clone(),
1152        };
1153
1154        let committee = inner
1155            .onchain_state()
1156            .current_committee()
1157            .expect("No current committee");
1158
1159        let required_weight = certificate_threshold(committee.total_weight());
1160        let proto_request = signed_message.to_proto();
1161
1162        let mut sig_tasks = JoinSet::new();
1163        for member in members {
1164            let inner = inner.clone();
1165            let proto_request = proto_request.clone();
1166            sig_tasks.spawn(async move {
1167                Self::request_withdrawal_tx_signing_signature(&inner, proto_request, &member).await
1168            });
1169        }
1170
1171        let mut aggregator = BlsSignatureAggregator::new(&committee, signed_message.clone());
1172        while let Some(result) = sig_tasks.join_next().await {
1173            let Ok(Some(sig)) = result else { continue };
1174            if let Err(e) = aggregator.add_signature(sig) {
1175                error!(pending_withdrawal_id = %pending.id, "Failed to add withdrawal sign message signature: {e}");
1176            }
1177            if aggregator.weight() >= required_weight {
1178                break;
1179            }
1180        }
1181
1182        let weight = aggregator.weight();
1183        if weight < required_weight {
1184            anyhow::bail!(
1185                "Insufficient signatures for sign_withdrawal: weight {weight} < {required_weight}"
1186            );
1187        }
1188
1189        let signed = aggregator.finish()?;
1190
1191        // 4. Submit sign_withdrawal to Sui (writes signatures on-chain).
1192        // Broadcast + confirm happens via process_signed_pending_withdrawals on the next tick.
1193        Self::submit_sign_withdrawal(
1194            &inner,
1195            &pending.id,
1196            &pending.request_ids.clone(),
1197            &witness_signatures,
1198            signed.committee_signature(),
1199        )
1200        .await
1201        .inspect(|()| {
1202            inner
1203                .metrics
1204                .sui_tx_submissions_total
1205                .with_label_values(&["sign_withdrawal", "success"])
1206                .inc();
1207        })
1208        .inspect_err(|_| {
1209            inner
1210                .metrics
1211                .sui_tx_submissions_total
1212                .with_label_values(&["sign_withdrawal", "failure"])
1213                .inc();
1214        })?;
1215
1216        Ok(())
1217    }

1265    /// Check BTC tx status, broadcast/re-broadcast if needed, confirm when
1266    /// enough BTC confirmations are reached.
1267    async fn handle_signed_withdrawal(
1268        inner: Arc<Hashi>,
1269        pending: PendingWithdrawal,
1270    ) -> anyhow::Result<()> {
1271        let confirmation_threshold = inner.onchain_state().bitcoin_confirmation_threshold();
1272        let txid = bitcoin::Txid::from_byte_array(pending.txid.into());
1273
1274        match inner.btc_monitor().get_transaction_status(txid).await {
1275            Ok(TxStatus::Confirmed { confirmations })
1276                if confirmations >= confirmation_threshold =>
1277            {
1278                info!(pending_withdrawal_id = %pending.id, bitcoin_txid = %txid, "Withdrawal tx confirmed with {confirmations} confirmations, proceeding to on-chain confirmation");
1279                Self::confirm_withdrawal_on_sui(&inner, &pending).await?;
1280            }
1281            Ok(TxStatus::Confirmed { confirmations }) => {
1282                debug!(pending_withdrawal_id = %pending.id, bitcoin_txid = %txid, "Withdrawal tx has {confirmations}/{confirmation_threshold} confirmations, waiting for more");
1283            }
1284            Ok(TxStatus::InMempool) => {
1285                debug!(pending_withdrawal_id = %pending.id, bitcoin_txid = %txid, "Withdrawal tx in mempool, waiting for confirmations");
1286            }
1287            Ok(TxStatus::NotFound) => {
1288                Self::rebuild_and_broadcast_withdrawal_btc_tx(&inner, &pending, txid).await;
1289            }
1290            Err(e) => {
1291                anyhow::bail!(
1292                    "failed to query transaction status for pending withdrawal {}: {e}",
1293                    pending.id
1294                );
1295            }
1296        }
1297        Ok(())
1298    }

1300    /// Rebuild a fully signed Bitcoin transaction from on-chain PendingWithdrawal

1304        pending: &PendingWithdrawal,
1305        txid: bitcoin::Txid,

1308
1309        let tx = match Self::rebuild_signed_tx_from_onchain(inner, pending) {
1310            Ok(tx) => tx,
1311            Err(e) => {
1312                error!(pending_withdrawal_id = %pending.id, bitcoin_txid = %txid, "Failed to rebuild signed withdrawal tx: {e}");
1313                return;
1314            }
1315        };
1316
1317        match inner.btc_monitor().broadcast_transaction(tx).await {
1318            Ok(()) => {
1319                info!(pending_withdrawal_id = %pending.id, bitcoin_txid = %txid, "Re-broadcast withdrawal tx");
1320            }
1321            Err(e) => {
1322                error!(pending_withdrawal_id = %pending.id, bitcoin_txid = %txid, "Failed to re-broadcast withdrawal tx: {e}");
1323            }
1324        }
1325    }

1327    /// Rebuild a fully signed Bitcoin transaction from on-chain PendingWithdrawal
1328    fn rebuild_signed_tx_from_onchain(
1329        inner: &Arc<Hashi>,
1330        pending: &PendingWithdrawal,
1331    ) -> anyhow::Result<bitcoin::Transaction> {
1332        let raw_sigs = pending
1333            .signatures
1334            .as_ref()
1335            .ok_or_else(|| anyhow::anyhow!("No signatures on pending withdrawal"))?;
1336
1337        let mut tx = inner.build_unsigned_withdrawal_tx(&pending.inputs, &pending.all_outputs())?;
1338
1339        anyhow::ensure!(
1340            raw_sigs.len() == tx.input.len(),
1341            "Signature count mismatch: tx has {} inputs, on-chain has {} signatures",
1342            tx.input.len(),
1343            raw_sigs.len()
1344        );
1345        anyhow::ensure!(
1346            tx.input.len() == pending.inputs.len(),
1347            "Input count mismatch: tx has {} inputs, pending has {}",
1348            tx.input.len(),
1349            pending.inputs.len()
1350        );
1351
1352        let hashi_pubkey = inner.get_hashi_pubkey()?;
1353        for ((input, pending_input), sig_bytes) in
1354            tx.input.iter_mut().zip(pending.inputs.iter()).zip(raw_sigs)
1355        {
1356            let pubkey =
1357                inner.deposit_pubkey(&hashi_pubkey, pending_input.derivation_path.as_ref())?;
1358            let (script, control_block, _) =
1359                bitcoin_utils::single_key_taproot_script_path_spend_artifacts(&pubkey);
1360            let mut witness = bitcoin::Witness::new();
1361            witness.push(sig_bytes);
1362            witness.push(script.to_bytes());
1363            witness.push(control_block.serialize());
1364            input.witness = witness;
1365        }
1366
1367        Ok(tx)
1368    }

1403    async fn collect_withdrawal_confirmation_signature(
1404        inner: &Arc<Hashi>,
1405        pending_id: Address,
1406        members: &[CommitteeMember],
1407    ) -> anyhow::Result<CommitteeSignature> {
1408        let committee = inner
1409            .onchain_state()
1410            .current_committee()
1411            .expect("No current committee");
1412        let confirmation = crate::withdrawals::WithdrawalConfirmation {
1413            withdrawal_id: pending_id,
1414        };
1415
1416        let required_weight = certificate_threshold(committee.total_weight());
1417
1418        let mut sig_tasks = JoinSet::new();
1419        for member in members {
1420            let inner = inner.clone();
1421            let member = member.clone();
1422            sig_tasks.spawn(async move {
1423                Self::request_withdrawal_confirmation_signature(&inner, pending_id, &member).await
1424            });
1425        }
1426
1427        let mut aggregator = BlsSignatureAggregator::new(&committee, confirmation);
1428        while let Some(result) = sig_tasks.join_next().await {
1429            let Ok(Some(sig)) = result else { continue };
1430            if let Err(e) = aggregator.add_signature(sig) {
1431                error!(pending_withdrawal_id = %pending_id, "Failed to add withdrawal confirmation signature: {e}");
1432            }
1433            if aggregator.weight() >= required_weight {
1434                break;
1435            }
1436        }
1437
1438        let weight = aggregator.weight();
1439        if weight < required_weight {
1440            anyhow::bail!(
1441                "Insufficient withdrawal confirmation signatures for pending {:?}: weight {weight} < {required_weight}",
1442                pending_id
1443            );
1444        }
1445
1446        Ok(aggregator.finish()?.into_parts().0)
1447    }
