107    #[tracing::instrument(name = "leader", skip_all)]
108    async fn run(mut self) {
109        info!("Starting leader service");
110
111        // Wait for DKG to complete before processing any checkpoints.
112        let mpc_handle = self.inner.mpc_handle().expect("MpcHandle not initialized");
113        info!("Waiting for MPC key to become available...");
114        mpc_handle.wait_for_key_ready().await;
115        info!("MPC key is ready, starting leader loop");
116
117        let mut checkpoint_rx = self.inner.onchain_state().subscribe_checkpoint();
118        let mut btc_block_rx = self.inner.btc_monitor().subscribe_block_height();
119
120        loop {
121            trace!("Waiting for next checkpoint or task completion...");
122            tokio::select! {
123                wait_result = checkpoint_rx.changed() => {
124                    if let Err(e) = wait_result {
125                        error!("Error waiting for checkpoint change: {e}");
126                        break;
127                    }
128                    let (checkpoint_height, checkpoint_timestamp_ms) = {
129                        let checkpoint_info = checkpoint_rx.borrow_and_update();
130                        (checkpoint_info.height, checkpoint_info.timestamp_ms)
131                    };
132
133                    let is_leader = self.is_current_leader(checkpoint_height);
134                    self.inner.metrics.is_leader.set(i64::from(is_leader));
135                    if is_leader {
136                        debug!("Checkpoint {checkpoint_height}: We are the leader node");
137                    } else {
138                        trace!("We are not the leader node");
139                        continue;
140                    }
141
142                    self.process_unapproved_withdrawal_requests(checkpoint_timestamp_ms);
143                    self.process_approved_withdrawal_requests(checkpoint_timestamp_ms);
144                    self.process_unsigned_withdrawal_txns();
145                    self.process_signed_withdrawal_txns();
146                    self.check_delete_proposals(checkpoint_timestamp_ms);
147
148                    if !self.pending_deposit_requests.is_empty() {
149                        self.process_deposit_requests();
150                    }
151                }
152                wait_result = btc_block_rx.changed() => {
153                    if let Err(e) = wait_result {
154                        error!("Error waiting for Bitcoin block height change: {e}");
155                        break;
156                    }
157                    let block_height = *btc_block_rx.borrow_and_update();
158                    let (checkpoint_height, checkpoint_timestamp_ms) = {
159                        let checkpoint_info = checkpoint_rx.borrow();
160                        (checkpoint_info.height, checkpoint_info.timestamp_ms)
161                    };
162
163                    if !self.is_current_leader(checkpoint_height) {
164                        continue;
165                    }
166
167                    debug!("New Bitcoin block {block_height}: processing deposit requests");
168
169                    self.check_delete_expired_deposit_requests(checkpoint_timestamp_ms);
170                    self.reload_pending_deposit_requests();
171                    self.process_deposit_requests();
172                }
173                Some(result) = self.deposit_tasks.join_next() => {
174                    self.handle_completed_deposit_task(result);
175                    while let Some(result) = self.deposit_tasks.try_join_next() {
176                        self.handle_completed_deposit_task(result);
177                    }
178                }
179                Some(result) = self.withdrawal_signing_tasks.join_next() => {
180                    self.handle_completed_withdrawal_signing_task(result);
181                }
182                Some(result) = self.withdrawal_broadcast_tasks.join_next() => {
183                    self.handle_completed_withdrawal_broadcast_task(result);
184                }
185                Some(result) = OptionFuture::from(self.withdrawal_approval_task.as_mut()) => {
186                    self.withdrawal_approval_task = None;
187                    Self::log_task_result("withdrawal_approval", result);
188                }
189                Some(result) = OptionFuture::from(self.withdrawal_commitment_task.as_mut()) => {
190                    self.withdrawal_commitment_task = None;
191                    Self::log_task_result("withdrawal_commitment", result);
192                }
193                Some(result) = OptionFuture::from(self.deposit_gc_task.as_mut()) => {
194                    self.deposit_gc_task = None;
195                    Self::log_task_result("deposit_gc", result);
196                }
197                Some(result) = OptionFuture::from(self.proposal_gc_task.as_mut()) => {
198                    self.proposal_gc_task = None;
199                    Self::log_task_result("proposal_gc", result);
200                }
201
202            }
203        }
204    }

206    fn handle_completed_deposit_task(
207        &mut self,
208        result: Result<(Address, Result<(), DepositError>), tokio::task::JoinError>,
209    ) {
210        match result {
211            Ok((deposit_id, result)) => {
212                self.inflight_deposits.remove(&deposit_id);
213                self.pending_deposit_requests
214                    .retain(|request| request.id != deposit_id);
215                match result {
216                    Ok(()) => {
217                        info!(deposit_id = %deposit_id, "Deposit processed successfully");
218                    }
219                    Err(err) => match err.kind() {
220                        DepositErrorKind::RetryOnNextBlock => {
221                            warn!(deposit_id = %deposit_id, "Deferring deposit until next block: {err:#}");
222                        }
223                        DepositErrorKind::NeverRetry => {
224                            self.never_retry_deposit_ids.insert(deposit_id);
225                            self.inner
226                                .metrics
227                                .never_retry_deposit_ids
228                                .set(self.never_retry_deposit_ids.len() as i64);
229                            warn!(deposit_id = %deposit_id, "Marking deposit as never retry: {err:#}");
230                        }
231                    },
232                }
233            }
234            Err(err) if err.is_panic() => error!("deposit task panicked: {err}"),
235            Err(err) => error!("deposit task failed to join: {err}"),
236        }
237    }

276    pub fn is_current_leader(&self, checkpoint_height: u64) -> bool {
277        if self.inner.onchain_state().state().hashi().config.paused() {
278            debug!("Bridge is paused, not acting as leader");
279            return false;
280        }
281
282        match self.inner.config.force_run_as_leader() {
283            ForceRunAsLeader::Always => return true,
284            ForceRunAsLeader::Never => return false,
285            ForceRunAsLeader::Default => (),
286        }
287
288        let Some(committee) = self.inner.onchain_state().current_committee() else {
289            // TODO: do we need to do anything when bootstrapping? At genesis there is no committee.
290            return false;
291        };
292        let this_validator_address = self
293            .inner
294            .config
295            .validator_address()
296            .expect("No configured validator address");
297        let Some(this_validator_idx) = committee
298            .index_of(&this_validator_address)
299            .map(|i| i as u64)
300        else {
301            // We are not in the committee yet, so we cannot be the leader
302            return false;
303        };
304        let num_validators = committee.members().len() as u64;
305
306        let current_turn = checkpoint_height / NUM_CONSECUTIVE_LEADER_CHECKPOINTS;
307        let is_leader = (current_turn % num_validators) == this_validator_idx;
308
309        trace!("Node index {this_validator_idx} is leader node: {is_leader}");
310        is_leader
311    }

313    fn reload_pending_deposit_requests(&mut self) {
314        let mut deposit_requests = self.inner.onchain_state().deposit_requests();
315        deposit_requests.sort_by_key(|r| r.timestamp_ms);
316        let deposit_ids: HashSet<Address> =
317            deposit_requests.iter().map(|request| request.id).collect();
318        self.inflight_deposits
319            .retain(|deposit_id| deposit_ids.contains(deposit_id));
320        self.never_retry_deposit_ids
321            .retain(|deposit_id| deposit_ids.contains(deposit_id));
322        self.inner
323            .metrics
324            .never_retry_deposit_ids
325            .set(self.never_retry_deposit_ids.len() as i64);
326        self.pending_deposit_requests = deposit_requests
327            .into_iter()
328            .filter(|request| !self.never_retry_deposit_ids.contains(&request.id))
329            .collect();
330        debug!(
331            pending_deposits = self.pending_deposit_requests.len(),
332            never_retry_deposits = self.never_retry_deposit_ids.len(),
333            "Reloaded pending deposit worklist"
334        );
335    }

347    fn process_deposit_requests(&mut self) {
348        if self.inner.onchain_state().state().hashi().config.paused() || self.is_reconfiguring() {
349            self.deposit_tasks.abort_all();
350            self.pending_deposit_requests.clear();
351            self.inflight_deposits.clear();
352            return;
353        }
354
355        let max_concurrent = self.inner.config.max_concurrent_leader_job_tasks();
356        for deposit_request in self.pending_deposit_requests.clone() {
357            if self.deposit_tasks.len() >= max_concurrent {
358                break;
359            }
360            let deposit_id = deposit_request.id;
361            if self.inflight_deposits.contains(&deposit_id) {
362                continue;
363            }
364
365            let inner = self.inner.clone();
366
367            self.inflight_deposits.insert(deposit_id);
368            self.deposit_tasks.spawn(async move {
369                let result = tokio::time::timeout(
370                    LEADER_TASK_TIMEOUT,
371                    Self::process_deposit_request(inner, deposit_request),
372                )
373                .await;
374
375                let result = match result {
376                    Ok(result) => result,
377                    Err(_) => Err(DepositError::TimedOut(LEADER_TASK_TIMEOUT)),
378                };
379
380                (deposit_id, result)
381            });
382        }
383    }

385    #[tracing::instrument(level = "info", skip_all, fields(deposit_id = %deposit_request.id))]
386    async fn process_deposit_request(
387        inner: Arc<Hashi>,
388        deposit_request: DepositRequest,
389    ) -> Result<(), DepositError> {
390        info!("Processing deposit request");
391
392        // Validate deposit_request before asking for signatures
393        inner
394            .validate_deposit_request(&deposit_request)
395            .await
396            .inspect_err(|err| debug!("Deposit validation failed: {err}"))?;
397
398        info!("Deposit request validated successfully");
399
400        let proto_request = deposit_request_to_proto(&deposit_request);
401        let members = inner
402            .onchain_state()
403            .current_committee_members()
404            .expect("No current committee members");
405
406        let committee = inner
407            .onchain_state()
408            .current_committee()
409            .expect("No current committee");
410
411        let required_weight = certificate_threshold(committee.total_weight());
412
413        // Fan out signature requests to all members in parallel.
414        let mut sig_tasks = JoinSet::new();
415        for member in members {
416            let inner = inner.clone();
417            let proto_request = proto_request.clone();
418            sig_tasks.spawn(async move {
419                Self::request_deposit_confirmation_signature(&inner, proto_request, &member).await
420            });
421        }
422
423        // Collect signatures, stopping once we reach quorum.
424        let confirmation_message = DepositConfirmationMessage {
425            request_id: deposit_request.id,
426            utxo: deposit_request.utxo.clone(),
427        };
428        let mut aggregator = BlsSignatureAggregator::new(&committee, confirmation_message);
429        while let Some(result) = sig_tasks.join_next().await {
430            let Ok(Some(sig)) = result else { continue };
431            if let Err(e) = aggregator.add_signature(sig) {
432                error!("Failed to add deposit signature: {e}");
433            }
434            if aggregator.weight() >= required_weight {
435                break;
436            }
437        }
438
439        if aggregator.weight() < required_weight {
440            return Err(DepositError::FailedQuorum {
441                weight: aggregator.weight(),
442                required_weight,
443            });
444        }
445
446        let signed_message = match aggregator.finish() {
447            Ok(signed_message) => signed_message,
448            Err(err) => return Err(DepositError::CertificateBuildFailed(err.into())),
449        };
450        let mut executor = match SuiTxExecutor::from_hashi(inner.clone()) {
451            Ok(executor) => executor,
452            Err(err) => return Err(DepositError::ExecutorInitFailed(err)),
453        };
454        executor
455            .execute_confirm_deposit(&deposit_request, signed_message)
456            .await
457            .inspect(|()| {
458                inner
459                    .metrics
460                    .sui_tx_submissions_total
461                    .with_label_values(&["confirm_deposit", "success"])
462                    .inc();
463                inner.metrics.deposits_confirmed_total.inc();
464                info!("Successfully submitted deposit confirmation");
465            })
466            .inspect_err(|e| {
467                error!("Failed to submit deposit confirmation: {e}");
468                inner
469                    .metrics
470                    .sui_tx_submissions_total
471                    .with_label_values(&["confirm_deposit", "failure"])
472                    .inc();
473            })
474            .map_err(DepositError::ConfirmDepositFailed)?;
475        Ok(())
476    }

478    #[tracing::instrument(level = "debug", skip_all, fields(validator = %member.validator_address()))]
479    async fn request_deposit_confirmation_signature(
480        inner: &Arc<Hashi>,
481        proto_request: SignDepositConfirmationRequest,
482        member: &CommitteeMember,
483    ) -> Option<MemberSignature> {
484        let validator_address = member.validator_address();
485        trace!("Requesting deposit confirmation signature");
486
487        let mut rpc_client = inner
488            .onchain_state()
489            .bridge_service_client(&validator_address)
490            .or_else(|| {
491                error!(
492                    "Cannot find client for validator address: {:?}",
493                    validator_address
494                );
495                None
496            })?;
497
498        let response = rpc_client
499            .sign_deposit_confirmation(proto_request.clone())
500            .await
501            .inspect_err(|e| {
502                error!(
503                    "Failed to get deposit confirmation signature from {}: {e}",
504                    validator_address
505                );
506            })
507            .ok()?;
508
509        trace!(
510            "Retrieved deposit confirmation signature from {}",
511            validator_address
512        );
513
514        response
515            .into_inner()
516            .member_signature
517            .ok_or_else(|| anyhow::anyhow!("No member_signature in response"))
518            .and_then(parse_member_signature)
519            .inspect_err(|e| {
520                error!(
521                    "Failed to parse member signature from response from {}: {e}",
522                    validator_address
523                );
524            })
525            .ok()

532    fn process_unapproved_withdrawal_requests(&mut self, checkpoint_timestamp_ms: u64) {
533        debug!("Entering process_unapproved_withdrawal_requests");
534        if self.is_reconfiguring() {
535            debug!("Reconfig in progress, skipping withdrawal approval processing");
536            return;
537        }
538
539        if self.withdrawal_approval_task.is_some() {
540            debug!("Withdrawal approval task already in-flight, skipping");
541            return;
542        }
543
544        let mut unapproved: Vec<_> = self
545            .inner
546            .onchain_state()
547            .withdrawal_requests()
548            .into_iter()
549            .filter(|r| r.status.is_requested())
550            .collect();
551        unapproved.sort_by_key(|r| r.timestamp_ms);
552
553        let unapproved_ids: Vec<Address> = unapproved.iter().map(|r| r.id).collect();
554        self.withdrawal_approval_retry_tracker
555            .prune(&unapproved_ids);
556
557        let to_process: Vec<_> = unapproved
558            .into_iter()
559            .filter(|r| {
560                !self
561                    .withdrawal_approval_retry_tracker
562                    .should_skip(&r.id, checkpoint_timestamp_ms)
563            })
564            .collect();
565
566        self.inner
567            .metrics
568            .leader_items_in_backoff
569            .with_label_values(&["withdrawal_approval"])
570            .set(
571                self.withdrawal_approval_retry_tracker
572                    .in_backoff_count(checkpoint_timestamp_ms) as i64,
573            );
574
575        if to_process.is_empty() {
576            return;
577        }
578
579        let inner = self.inner.clone();
580        let retry_tracker = self.withdrawal_approval_retry_tracker.clone();
581
582        self.withdrawal_approval_task =
583            Some(AbortOnDropHandle::new(tokio::task::spawn(async move {
584                Self::process_unapproved_withdrawal_requests_task(
585                    inner,
586                    retry_tracker,
587                    to_process,
588                    checkpoint_timestamp_ms,
589                )
590                .await
591            })));
592    }

594    #[tracing::instrument(level = "info", skip_all, fields(batch_size = to_process.len()))]
595    async fn process_unapproved_withdrawal_requests_task(
596        inner: Arc<Hashi>,
597        retry_tracker: RetryTracker<WithdrawalApprovalErrorKind>,
598        to_process: Vec<WithdrawalRequest>,
599        checkpoint_timestamp_ms: u64,
600    ) -> anyhow::Result<()> {
601        let max_concurrent = inner.config.max_concurrent_leader_job_tasks();
602
603        let this_validator_address = inner
604            .config
605            .validator_address()
606            .expect("No configured validator address");
607
608        let members = inner
609            .onchain_state()
610            .current_committee_members()
611            .expect("No current committee members");
612
613        let committee = inner
614            .onchain_state()
615            .current_committee()
616            .expect("No current committee");
617
618        let mut tasks = JoinSet::new();
619        let mut certified: Vec<(Address, CommitteeSignature)> = Vec::new();
620
621        for request in to_process {
622            if tasks.len() >= max_concurrent {
623                // Wait for one to finish before spawning more.
624                if let Some(result) = tasks.join_next().await {
625                    match &result {
626                        Err(err) if err.is_panic() => {
627                            error!("Withdrawal approval task panicked: {err}")
628                        }
629                        Err(err) => error!("Withdrawal approval task failed to join: {err}"),
630                        Ok(_) => {}
631                    }
632                    if let Ok((_request_id, Ok(Some(cert)))) = result {
633                        certified.push(cert);
634                    }
635                }
636            }
637
638            let inner = inner.clone();
639            let retry_tracker = retry_tracker.clone();
640            let members = members.clone();
641            let committee = committee.clone();
642            tasks.spawn(async move {
643                let request_id = request.id;
644                let task_result = tokio::time::timeout(
645                    LEADER_TASK_TIMEOUT,
646                    Self::process_unapproved_withdrawal_request(
647                        inner.clone(),
648                        retry_tracker.clone(),
649                        request,
650                        checkpoint_timestamp_ms,
651                        this_validator_address,
652                        &members,
653                        &committee,
654                    ),
655                )
656                .await;
657
658                let (result, failure_kind) = match task_result {
659                    Ok(result) => (result, None),
660                    Err(_) => {
661                        let kind = WithdrawalApprovalErrorKind::TimedOut;
662                        inner
663                            .metrics
664                            .leader_retries_total
665                            .with_label_values(&["withdrawal_approval", &format!("{kind:?}")])
666                            .inc();
667                        retry_tracker.record_failure(kind, request_id, checkpoint_timestamp_ms);
668                        (Ok(None), Some(kind))
669                    }
670                };
671
672                if result.is_err() && failure_kind.is_none() {
673                    let kind = WithdrawalApprovalErrorKind::TaskFailed;
674                    inner
675                        .metrics
676                        .leader_retries_total
677                        .with_label_values(&["withdrawal_approval", &format!("{kind:?}")])
678                        .inc();
679                    retry_tracker.record_failure(kind, request_id, checkpoint_timestamp_ms);
680                }
681                if let Err(err) = &result {
682                    error!(request_id = %request_id, "Withdrawal approval failed: {err:#}");
683                }
684
685                (request_id, result)
686            });
687        }
688
689        while let Some(result) = tasks.join_next().await {
690            match &result {
691                Err(err) if err.is_panic() => error!("Withdrawal approval task panicked: {err}"),
692                Err(err) => error!("Withdrawal approval task failed to join: {err}"),
693                Ok(_) => {}
694            }
695            if let Ok((_request_id, Ok(Some(cert)))) = result {
696                certified.push(cert);
697            }
698        }
699
700        if certified.is_empty() {
701            return Ok(());
702        }
703
704        Self::submit_approve_withdrawal_requests_with_retry(&inner, certified).await;
705        Ok(())
706    }

708    #[tracing::instrument(level = "info", skip_all, fields(request_id = %request.id))]
709    async fn process_unapproved_withdrawal_request(
710        inner: Arc<Hashi>,
711        retry_tracker: RetryTracker<WithdrawalApprovalErrorKind>,
712        request: WithdrawalRequest,
713        checkpoint_timestamp_ms: u64,
714        this_validator_address: Address,
715        members: &[CommitteeMember],
716        committee: &hashi_types::committee::Committee,
717    ) -> anyhow::Result<Option<(Address, CommitteeSignature)>> {
718        let approval = WithdrawalRequestApproval {
719            request_id: request.id,
720        };
721
722        // Validate, screen, and sign locally first
723        let local_sig = match inner
724            .validate_and_sign_withdrawal_request_approval(&approval)
725            .await
726        {
727            Ok(sig) => {
728                retry_tracker.clear(&request.id);
729                parse_member_signature(sig).unwrap()
730            }
731            Err(e) => {
732                let kind = e.kind();
733                inner
734                    .metrics
735                    .leader_retries_total
736                    .with_label_values(&["withdrawal_approval", &format!("{kind:?}")])
737                    .inc();
738                retry_tracker.record_failure(kind, request.id, checkpoint_timestamp_ms);
739                return Ok(None);
740            }
741        };
742
743        let proto_request = approval.to_proto();
744        let required_weight = certificate_threshold(committee.total_weight());
745
746        let mut aggregator = BlsSignatureAggregator::new(committee, approval);
747        if let Err(e) = aggregator.add_signature(local_sig) {
748            error!("Failed to add local approval signature: {e}");
749        }
750
751        // Fan out signature requests to remote members in parallel.
752        let mut sig_tasks = JoinSet::new();
753        for member in members {
754            if member.validator_address() == this_validator_address {
755                continue;
756            }
757            let inner = inner.clone();
758            let proto_request = proto_request.clone();
759            let member = member.clone();
760            sig_tasks.spawn(async move {
761                Self::request_withdrawal_approval_signature(&inner, proto_request, &member).await
762            });
763        }
764
765        // Collect signatures, stopping once we reach quorum.
766        while let Some(result) = sig_tasks.join_next().await {
767            let Ok(Some(sig)) = result else { continue };
768            if let Err(e) = aggregator.add_signature(sig) {
769                error!("Failed to add approval signature: {e}");
770            }
771            if aggregator.weight() >= required_weight {
772                break;
773            }
774        }
775
776        let weight = aggregator.weight();
777        if weight < required_weight {
778            inner
779                .metrics
780                .leader_retries_total
781                .with_label_values(&["withdrawal_approval", "FailedQuorum"])
782                .inc();
783            retry_tracker.record_failure(
784                WithdrawalApprovalErrorKind::FailedQuorum,
785                request.id,
786                checkpoint_timestamp_ms,
787            );
788            error!("Insufficient approval signatures: weight {weight} < {required_weight}");
789            return Ok(None);
790        }
791
792        match aggregator.finish() {
793            Ok(signed) => Ok(Some((request.id, signed.committee_signature().clone()))),
794            Err(e) => {
795                error!("Failed to build approval certificate: {e}");
796                Ok(None)
797            }
798        }
799    }

801    async fn submit_approve_withdrawal_requests_with_retry(
802        inner: &Arc<Hashi>,
803        mut certified: Vec<(Address, CommitteeSignature)>,
804    ) {
805        loop {
806            let approvals: Vec<(Address, &CommitteeSignature)> =
807                certified.iter().map(|(id, cert)| (*id, cert)).collect();
808
809            let result = Self::submit_approve_withdrawal_requests(inner, &approvals)
810                .await
811                .inspect(|()| {
812                    inner
813                        .metrics
814                        .sui_tx_submissions_total
815                        .with_label_values(&["approve_withdrawal", "success"])
816                        .inc();
817                })
818                .inspect_err(|_| {
819                    inner
820                        .metrics
821                        .sui_tx_submissions_total
822                        .with_label_values(&["approve_withdrawal", "failure"])
823                        .inc();
824                });
825            let Err(e) = result else { return };
826
827            let err_msg = format!("{e}");
828            error!("approve_request PTB failed: {err_msg}");
829
830            // Try to identify which request caused the failure by checking
831            // which ones no longer exist in the queue (canceled).
832            let before_len = certified.len();
833            certified.retain(|(id, _)| inner.onchain_state().withdrawal_request(id).is_some());
834
835            if certified.len() == before_len {
836                error!("Could not identify failed request, aborting retry");
837                return;
838            }
839            if certified.is_empty() {
840                return;
841            }
842
843            info!(
844                "Retrying approve_request with {} remaining requests",
845                certified.len()
846            );
847        }
848    }

854    fn process_approved_withdrawal_requests(&mut self, checkpoint_timestamp_ms: u64) {
855        debug!("Entering process_approved_withdrawal_requests");
856        if self.is_reconfiguring() {
857            debug!("Reconfig in progress, skipping withdrawal commitment processing");
858            return;
859        }
860
861        if self.withdrawal_commitment_task.is_some() {
862            debug!("Withdrawal commitment task already in-flight, skipping");
863            return;
864        }
865
866        let mut approved: Vec<_> = self
867            .inner
868            .onchain_state()
869            .withdrawal_requests()
870            .into_iter()
871            .filter(|r| r.status.is_approved())
872            .collect();
873        approved.sort_by_key(|r| r.timestamp_ms);
874
875        if self
876            .withdrawal_commitment_retry_tracker
877            .should_skip(checkpoint_timestamp_ms)
878        {
879            self.inner
880                .metrics
881                .leader_items_in_backoff
882                .with_label_values(&["withdrawal_commitment"])
883                .set(
884                    self.withdrawal_commitment_retry_tracker
885                        .in_backoff_count(checkpoint_timestamp_ms) as i64,
886                );
887            return;
888        }
889
890        // Collect all approved requests and process them as a single batch.
891        // The coin selection algorithm picks up to
892        // max_withdrawal_requests oldest requests from the slice.
893        let batch: Vec<WithdrawalRequest> = approved.into_iter().collect();
894
895        self.inner
896            .metrics
897            .leader_items_in_backoff
898            .with_label_values(&["withdrawal_commitment"])
899            .set(
900                self.withdrawal_commitment_retry_tracker
901                    .in_backoff_count(checkpoint_timestamp_ms) as i64,
902            );
903
904        if batch.is_empty() {
905            return;
906        }
907
908        let max_batch = self.inner.config.withdrawal_max_batch_size();
909        let delay_ms = self.inner.config.withdrawal_batching_delay_ms();
910
911        let batch_is_full = batch.len() >= max_batch;
912        let oldest_has_waited = batch
913            .first()
914            .is_some_and(|r| checkpoint_timestamp_ms >= r.timestamp_ms + delay_ms);
915
916        if !batch_is_full && !oldest_has_waited {
917            debug!(
918                "Holding {} approved request(s): oldest is {}ms old, \
919                 waiting for {}ms delay or {} requests",
920                batch.len(),
921                checkpoint_timestamp_ms.saturating_sub(batch[0].timestamp_ms),
922                delay_ms,
923                max_batch,
924            );
925            return;
926        }
927
928        let inner = self.inner.clone();
929        let retry_tracker = self.withdrawal_commitment_retry_tracker.clone();
930
931        self.withdrawal_commitment_task =
932            Some(AbortOnDropHandle::new(tokio::task::spawn(async move {
933                let task_result = tokio::time::timeout(
934                    LEADER_TASK_TIMEOUT,
935                    Self::process_approved_withdrawal_request_batch(
936                        inner.clone(),
937                        retry_tracker.clone(),
938                        batch,
939                        checkpoint_timestamp_ms,
940                    ),
941                )
942                .await;
943
944                match task_result {
945                    Ok(result) => result,
946                    Err(_) => {
947                        let kind = WithdrawalCommitmentErrorKind::TimedOut;
948                        inner
949                            .metrics
950                            .leader_retries_total
951                            .with_label_values(&["withdrawal_commitment", &format!("{kind:?}")])
952                            .inc();
953                        Err(anyhow::anyhow!(
954                            "withdrawal commitment timed out after {LEADER_TASK_TIMEOUT:?}"
955                        ))
956                    }
957                }
958            })));
959    }

961    #[tracing::instrument(level = "info", skip_all, fields(batch_size = requests.len()))]
962    async fn process_approved_withdrawal_request_batch(
963        inner: Arc<Hashi>,
964        retry_tracker: GlobalRetryTracker<WithdrawalCommitmentErrorKind>,
965        requests: Vec<WithdrawalRequest>,
966        checkpoint_timestamp_ms: u64,
967    ) -> anyhow::Result<()> {
968        info!(
969            withdrawal_request_ids = ?requests.iter().map(|r| r.id).collect::<Vec<_>>(),
970            "Processing batch of {} approved withdrawal request(s)",
971            requests.len(),
972        );
973
974        // Build the withdrawal tx commitment for the batch.
975        let approval = match inner.build_withdrawal_tx_commitment(&requests).await {
976            Ok(approval) => {
977                retry_tracker.clear();
978                approval
979            }
980            Err(e) => {
981                let kind = e.kind();
982                inner
983                    .metrics
984                    .leader_retries_total
985                    .with_label_values(&["withdrawal_commitment", &format!("{kind:?}")])
986                    .inc();
987                retry_tracker.record_failure(kind, checkpoint_timestamp_ms);
988                return Ok(());
989            }
990        };
991
992        // Fan out to committee for BLS signatures over the commitment message
993        let members = inner
994            .onchain_state()
995            .current_committee_members()
996            .expect("No current committee members");
997
998        let committee = inner
999            .onchain_state()
1000            .current_committee()
1001            .expect("No current committee");
1002
1003        let required_weight = certificate_threshold(committee.total_weight());
1004        let proto_request = approval.to_proto();
1005
1006        // Fan out signature requests to all members in parallel.
1007        let mut sig_tasks = JoinSet::new();
1008        for member in members {
1009            let inner = inner.clone();
1010            let proto_request = proto_request.clone();
1011            sig_tasks.spawn(async move {
1012                Self::request_withdrawal_tx_commitment_signature(&inner, proto_request, &member)
1013                    .await
1014            });
1015        }
1016
1017        // Collect signatures, stopping once we reach quorum.
1018        let mut aggregator = BlsSignatureAggregator::new(&committee, approval.clone());
1019        while let Some(result) = sig_tasks.join_next().await {
1020            let Ok(Some(sig)) = result else { continue };
1021            if let Err(e) = aggregator.add_signature(sig) {
1022                error!("Failed to add withdrawal commitment signature: {e}");
1023            }
1024            if aggregator.weight() >= required_weight {
1025                break;
1026            }
1027        }
1028
1029        if aggregator.weight() < required_weight {
1030            inner
1031                .metrics
1032                .leader_retries_total
1033                .with_label_values(&["withdrawal_commitment", "FailedQuorum"])
1034                .inc();
1035            retry_tracker.record_failure(
1036                WithdrawalCommitmentErrorKind::FailedQuorum,
1037                checkpoint_timestamp_ms,
1038            );
1039            error!(
1040                "Insufficient withdrawal commitment signatures: weight {} < {required_weight}",
1041                aggregator.weight()
1042            );
1043            return Ok(());
1044        }
1045
1046        let signed_approval = match aggregator.finish() {
1047            Ok(signed_approval) => signed_approval,
1048            Err(e) => {
1049                error!("Failed to build withdrawal commitment certificate: {e}");
1050                return Ok(());
1051            }
1052        };
1053
1054        // Proactively trigger a presig refill if this commit will allocate
1055        // indices beyond the current pool.
1056        {
1057            let num_inputs = approval.selected_utxos.len() as u64;
1058            let num_consumed = inner.onchain_state().state().hashi().num_consumed_presigs;
1059            let needed_end = num_consumed + num_inputs;
1060            if let Some(signing_manager) = inner.current_signing_manager() {
1061                let available_end = signing_manager.available_presig_end_index();
1062                if needed_end > available_end {
1063                    info!(
1064                        "Presig pool may be insufficient for this withdrawal: \
1065                         need index {needed_end}, pool ends at {available_end}. \
1066                         Triggering proactive refill.",
1067                    );
1068                    signing_manager.trigger_refill();
1069                }
1070            }
1071        }
1072
1073        // Submit commit_withdrawal_tx to Sui
1074        Self::submit_commit_withdrawal_tx(&inner, &approval, signed_approval.committee_signature())
1075            .await
1076            .inspect(|()| {
1077                inner
1078                    .metrics
1079                    .sui_tx_submissions_total
1080                    .with_label_values(&["commit_withdrawal", "success"])
1081                    .inc();
1082            })
1083            .inspect_err(|e| {
1084                inner
1085                    .metrics
1086                    .sui_tx_submissions_total
1087                    .with_label_values(&["commit_withdrawal", "failure"])
1088                    .inc();
1089                error!("Failed to submit commit_withdrawal_tx: {e}");
1090            })?;
1091
1092        Ok(())
1093    }

1146    #[tracing::instrument(level = "info", skip_all, fields(withdrawal_txn_id = %txn.id))]
1147    async fn process_unsigned_withdrawal_txn(
1148        inner: Arc<Hashi>,
1149        txn: WithdrawalTransaction,
1150    ) -> anyhow::Result<()> {
1151        // If the withdrawal transaction is from a previous epoch, reassign its presig
1152        // indices from the new epoch's counter before signing.
1153        // TODO: Batch multiple stale-epoch withdrawals into a single PTB.
1154        let current_epoch = inner.onchain_state().epoch();
1155        if txn.epoch != current_epoch {
1156            info!(
1157                "Withdrawal transaction from epoch {} (current {}), reassigning presig indices",
1158                txn.epoch, current_epoch,
1159            );
1160            let mut executor = SuiTxExecutor::from_hashi(inner.clone())?;
1161            executor
1162                .execute_allocate_presigs_for_withdrawal_txn(txn.id)
1163                .await?;
1164            info!("Presig indices reassigned, will sign on next checkpoint");
1165            // Return and let the next checkpoint iteration pick up the updated state.
1166            return Ok(());
1167        }
1168        info!("MPC signing withdrawal transaction");
1169
1170        let members = inner
1171            .onchain_state()
1172            .current_committee_members()
1173            .expect("No current committee members");
1174
1175        // 1. Request signed withdrawal tx witnesses from committee members.
1176        // MPC signing requires all threshold members to participate simultaneously
1177        // via P2P, so we must fan out requests in parallel.
1178        let signatures_by_input = Self::collect_withdrawal_tx_signatures(&inner, &txn.id, &members)
1179            .await
1180            .ok_or_else(|| anyhow::anyhow!("Failed to collect MPC signatures for {:?}", txn.id))?;
1181
1182        // 2. Extract raw signature bytes for on-chain storage
1183        let witness_signatures: Vec<Vec<u8>> = signatures_by_input
1184            .iter()
1185            .map(|s| s.to_byte_array().to_vec())
1186            .collect();
1187
1188        // 3. Build the WithdrawalTxSigning and get BLS certificate via fan-out
1189        let signed_message = WithdrawalTxSigning {
1190            withdrawal_id: txn.id,
1191            request_ids: txn.request_ids.clone(),
1192            signatures: witness_signatures.clone(),
1193        };
1194
1195        let committee = inner
1196            .onchain_state()
1197            .current_committee()
1198            .expect("No current committee");
1199
1200        let required_weight = certificate_threshold(committee.total_weight());
1201        let proto_request = signed_message.to_proto();
1202
1203        let mut sig_tasks = JoinSet::new();
1204        for member in members {
1205            let inner = inner.clone();
1206            let proto_request = proto_request.clone();
1207            sig_tasks.spawn(async move {
1208                Self::request_withdrawal_tx_signing_signature(&inner, proto_request, &member).await
1209            });
1210        }
1211
1212        let mut aggregator = BlsSignatureAggregator::new(&committee, signed_message.clone());
1213        while let Some(result) = sig_tasks.join_next().await {
1214            let Ok(Some(sig)) = result else { continue };
1215            if let Err(e) = aggregator.add_signature(sig) {
1216                error!(withdrawal_txn_id = %txn.id, "Failed to add withdrawal sign message signature: {e}");
1217            }
1218            if aggregator.weight() >= required_weight {
1219                break;
1220            }
1221        }
1222
1223        let weight = aggregator.weight();
1224        if weight < required_weight {
1225            anyhow::bail!(
1226                "Insufficient signatures for sign_withdrawal: weight {weight} < {required_weight}"
1227            );
1228        }
1229
1230        let signed = aggregator.finish()?;
1231
1232        // 4. Submit sign_withdrawal to Sui (writes signatures on-chain).
1233        // Broadcast + confirm happens via process_signed_withdrawal_txns on the next tick.
1234        Self::submit_sign_withdrawal(
1235            &inner,
1236            &txn.id,
1237            &txn.request_ids.clone(),
1238            &witness_signatures,
1239            signed.committee_signature(),
1240        )
1241        .await
1242        .inspect(|()| {
1243            inner
1244                .metrics
1245                .sui_tx_submissions_total
1246                .with_label_values(&["sign_withdrawal", "success"])
1247                .inc();
1248        })
1249        .inspect_err(|_| {
1250            inner
1251                .metrics
1252                .sui_tx_submissions_total
1253                .with_label_values(&["sign_withdrawal", "failure"])
1254                .inc();
1255        })?;
1256
1257        Ok(())
1258    }

1264    fn process_signed_withdrawal_txns(&mut self) {
1265        debug!("Entering process_signed_withdrawal_txns");
1266        let mut withdrawal_txns = self.inner.onchain_state().withdrawal_txns();
1267        withdrawal_txns.retain(|p| p.signatures.is_some());
1268        withdrawal_txns.sort_by_key(|p| p.timestamp_ms);
1269
1270        let pending_ids: Vec<Address> = withdrawal_txns.iter().map(|p| p.id).collect();
1271        self.inflight_withdrawal_broadcasts
1272            .retain(|id| pending_ids.contains(id));
1273
1274        let max_concurrent = self.inner.config.max_concurrent_leader_job_tasks();
1275        for txn in withdrawal_txns {
1276            if self.withdrawal_broadcast_tasks.len() >= max_concurrent {
1277                break;
1278            }
1279            if self.inflight_withdrawal_broadcasts.contains(&txn.id) {
1280                continue;
1281            }
1282
1283            let txn_id = txn.id;
1284            let inner = self.inner.clone();
1285
1286            self.inflight_withdrawal_broadcasts.insert(txn_id);
1287            self.withdrawal_broadcast_tasks.spawn(async move {
1288                let result = tokio::time::timeout(
1289                    LEADER_TASK_TIMEOUT,
1290                    Self::handle_signed_withdrawal(inner, txn),
1291                )
1292                .await;
1293
1294                let result = match result {

1377    /// Rebuild a fully signed Bitcoin transaction from on-chain WithdrawalTransaction
1378    fn rebuild_signed_tx_from_onchain(
1379        inner: &Arc<Hashi>,
1380        txn: &WithdrawalTransaction,
1381    ) -> anyhow::Result<bitcoin::Transaction> {
1382        let raw_sigs = txn
1383            .signatures
1384            .as_ref()
1385            .ok_or_else(|| anyhow::anyhow!("No signatures on withdrawal transaction"))?;
1386
1387        let mut tx = inner.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs())?;
1388
1389        anyhow::ensure!(
1390            raw_sigs.len() == tx.input.len(),
1391            "Signature count mismatch: tx has {} inputs, on-chain has {} signatures",
1392            tx.input.len(),
1393            raw_sigs.len()
1394        );
1395        anyhow::ensure!(
1396            tx.input.len() == txn.inputs.len(),
1397            "Input count mismatch: tx has {} inputs, txn has {}",
1398            tx.input.len(),
1399            txn.inputs.len()
1400        );
1401
1402        let hashi_pubkey = inner.get_hashi_pubkey()?;
1403        for ((input, txn_input), sig_bytes) in
1404            tx.input.iter_mut().zip(txn.inputs.iter()).zip(raw_sigs)
1405        {
1406            let pubkey = inner.deposit_pubkey(&hashi_pubkey, txn_input.derivation_path.as_ref())?;
1407            let (script, control_block, _) =
1408                bitcoin_utils::single_key_taproot_script_path_spend_artifacts(&pubkey);
1409            let mut witness = bitcoin::Witness::new();
1410            witness.push(sig_bytes);
1411            witness.push(script.to_bytes());
1412            witness.push(control_block.serialize());
1413            input.witness = witness;
1414        }
1415
1416        Ok(tx)
1417    }

1453    #[tracing::instrument(level = "debug", skip_all, fields(withdrawal_txn_id = %withdrawal_txn_id))]
1454    async fn collect_withdrawal_confirmation_signature(
1455        inner: &Arc<Hashi>,
1456        withdrawal_txn_id: Address,
1457        members: &[CommitteeMember],
1458    ) -> anyhow::Result<CommitteeSignature> {
1459        let committee = inner
1460            .onchain_state()
1461            .current_committee()
1462            .expect("No current committee");
1463        let confirmation = crate::withdrawals::WithdrawalConfirmation {
1464            withdrawal_id: withdrawal_txn_id,
1465        };
1466
1467        let required_weight = certificate_threshold(committee.total_weight());
1468
1469        let mut sig_tasks = JoinSet::new();
1470        for member in members {
1471            let inner = inner.clone();
1472            let member = member.clone();
1473            sig_tasks.spawn(async move {
1474                Self::request_withdrawal_confirmation_signature(&inner, withdrawal_txn_id, &member)
1475                    .await
1476            });
1477        }
1478
1479        let mut aggregator = BlsSignatureAggregator::new(&committee, confirmation);
1480        while let Some(result) = sig_tasks.join_next().await {
1481            let Ok(Some(sig)) = result else { continue };
1482            if let Err(e) = aggregator.add_signature(sig) {
1483                error!("Failed to add withdrawal confirmation signature: {e}");
1484            }
1485            if aggregator.weight() >= required_weight {
1486                break;
1487            }
1488        }
1489
1490        let weight = aggregator.weight();
1491        if weight < required_weight {
1492            anyhow::bail!(
1493                "Insufficient withdrawal confirmation signatures for {:?}: weight {weight} < {required_weight}",
1494                withdrawal_txn_id
1495            );
1496        }
1497
1498        Ok(aggregator.finish()?.into_parts().0)
1499    }
