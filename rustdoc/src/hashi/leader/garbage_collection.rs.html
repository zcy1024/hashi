18// const MAX_DEPOSIT_REQUEST_AGE_MS: u64 = 1000 * 60 * 60 * 24 * 3; // 3 days
19// const DEPOSIT_REQUEST_DELETE_DELAY_MS: u64 = 1000 * 60 * 60 * 24; // 1 day
20const MAX_DEPOSIT_REQUEST_AGE_MS: u64 = 1000 * 60 * 60 * 24; // 1 days
21const DEPOSIT_REQUEST_DELETE_DELAY_MS: u64 = 1000 * 60; // 1 minute
22const MAX_DEPOSIT_REQUEST_DELETIONS_PER_GC: usize = 500;
23
24const MAX_PROPOSAL_AGE_MS: u64 = 1000 * 60 * 60 * 24 * 7; // 7 days
25const PROPOSAL_DELETE_DELAY_MS: u64 = 1000 * 60 * 60 * 24; // 1 day
26
27impl LeaderService {
28    /// Check for and delete expired deposit requests.
29    /// Deposit requests are sorted by timestamp and deleted if they are older than
30    /// MAX_DEPOSIT_REQUEST_AGE_MS.
31    pub(crate) fn check_delete_expired_deposit_requests(&mut self, checkpoint_timestamp_ms: u64) {
32        if self.deposit_gc_task.is_some() {
33            debug!("Deposit GC task already in-flight, skipping");
34            return;
35        }
36
37        let mut deposit_requests = self.inner.onchain_state().deposit_requests();
38        deposit_requests.sort_by_key(|r| r.timestamp_ms);
39
40        let Some(oldest_request) = deposit_requests.first() else {
41            return;
42        };
43
44        if checkpoint_timestamp_ms
45            < oldest_request.timestamp_ms
46                + MAX_DEPOSIT_REQUEST_AGE_MS
47                + DEPOSIT_REQUEST_DELETE_DELAY_MS
48        {
49            return;
50        }
51
52        let expired_requests: Vec<_> = deposit_requests
53            .iter()
54            .filter(|r| checkpoint_timestamp_ms > r.timestamp_ms + MAX_DEPOSIT_REQUEST_AGE_MS)
55            .take(MAX_DEPOSIT_REQUEST_DELETIONS_PER_GC)
56            .cloned()
57            .collect();
58        if expired_requests.is_empty() {
59            return;
60        }
61
62        info!(
63            "Scheduling deletion of {} expired deposit requests",
64            expired_requests.len()
65        );
66
67        let inner = self.inner.clone();
68        self.deposit_gc_task = Some(AbortOnDropHandle::new(tokio::task::spawn(async move {
69            Self::delete_expired_deposit_requests(inner, expired_requests).await
70        })));
71    }
72
73    async fn delete_expired_deposit_requests(
74        inner: Arc<crate::Hashi>,
75        expired_requests: Vec<DepositRequest>,
76    ) -> anyhow::Result<()> {
77        let count = expired_requests.len();
78        let mut executor = SuiTxExecutor::from_hashi(inner)?;
79        executor
80            .execute_delete_expired_deposit_requests(&expired_requests)
81            .await?;
82        info!("Successfully deleted {count} expired deposit requests");
83        Ok(())
84    }

86    /// Check for and delete expired proposals.
87    /// Proposals are sorted by timestamp and deleted if they are older than MAX_PROPOSAL_AGE_MS.
88    pub(crate) fn check_delete_proposals(&mut self, checkpoint_timestamp_ms: u64) {
89        debug!("Entering check_delete_proposals");
90
91        if self.proposal_gc_task.is_some() {
92            debug!("Proposal GC task already in-flight, skipping");
93            return;
94        }
95
96        let mut proposals = self.inner.onchain_state().proposals();
97        // Sort proposals by timestamp, from earliest to latest
98        proposals.sort_by_key(|p| p.timestamp_ms);
99
100        // Check if it's time to delete
101        let Some(oldest_proposal) = proposals.first() else {
102            return;
103        };
104
105        // If there aren't any proposals at least 8 days old (7 days expiry + 1 day delay), don't do anything
106        if checkpoint_timestamp_ms
107            < oldest_proposal.timestamp_ms + MAX_PROPOSAL_AGE_MS + PROPOSAL_DELETE_DELAY_MS
108        {
109            return;
110        }
111
112        // Find all expired proposals (older than 7 days)
113        let expired_proposals: Vec<_> = proposals
114            .iter()
115            .filter(|p| checkpoint_timestamp_ms > p.timestamp_ms + MAX_PROPOSAL_AGE_MS)
116            .cloned()
117            .collect();
118
119        if expired_proposals.is_empty() {
120            return;
121        }
122
123        info!(
124            "Scheduling deletion of {} expired proposals",
125            expired_proposals.len()
126        );
127
128        let inner = self.inner.clone();
129        self.proposal_gc_task = Some(AbortOnDropHandle::new(tokio::task::spawn(async move {
130            Self::delete_expired_proposals(inner, expired_proposals).await
131        })));
132    }
133
134    async fn delete_expired_proposals(
135        inner: Arc<crate::Hashi>,
136        expired_proposals: Vec<Proposal>,
137    ) -> anyhow::Result<()> {
138        use sui_sdk_types::Identifier;
139        use sui_sdk_types::StructTag;
140        use sui_sdk_types::TypeTag;
141        use sui_transaction_builder::Function;
142        use sui_transaction_builder::ObjectInput;
143        use sui_transaction_builder::TransactionBuilder;
144
145        let mut executor = SuiTxExecutor::from_hashi(inner.clone())?;
146        let hashi_ids = inner.config.hashi_ids();
147
148        let mut builder = TransactionBuilder::new();
149
150        let hashi_arg = builder.object(
151            ObjectInput::new(hashi_ids.hashi_object_id)
152                .as_shared()
153                .with_mutable(true),
154        );
155
156        // Clock object (0x6) - immutable shared object
157        let clock_arg = builder.object(
158            ObjectInput::new(Address::from_static("0x6"))
159                .as_shared()
160                .with_mutable(false),
161        );
162
163        // Add a move call for each expired proposal
164        for proposal in &expired_proposals {
165            let proposal_id_arg = builder.pure(&proposal.id);
166
167            // Get the type argument for the proposal
168            let type_arg = match &proposal.proposal_type {
169                ProposalType::UpdateConfig => TypeTag::Struct(Box::new(StructTag::new(
170                    hashi_ids.package_id,
171                    Identifier::from_static("update_config"),
172                    Identifier::from_static("UpdateConfig"),
173                    vec![],
174                ))),
175                ProposalType::EnableVersion => TypeTag::Struct(Box::new(StructTag::new(
176                    hashi_ids.package_id,
177                    Identifier::from_static("enable_version"),
178                    Identifier::from_static("EnableVersion"),
179                    vec![],
180                ))),
181                ProposalType::DisableVersion => TypeTag::Struct(Box::new(StructTag::new(
182                    hashi_ids.package_id,
183                    Identifier::from_static("disable_version"),
184                    Identifier::from_static("DisableVersion"),
185                    vec![],
186                ))),
187                ProposalType::Upgrade => TypeTag::Struct(Box::new(StructTag::new(
188                    hashi_ids.package_id,
189                    Identifier::from_static("upgrade"),
190                    Identifier::from_static("Upgrade"),
191                    vec![],
192                ))),
193                ProposalType::EmergencyPause => TypeTag::Struct(Box::new(StructTag::new(
194                    hashi_ids.package_id,
195                    Identifier::from_static("emergency_pause"),
196                    Identifier::from_static("EmergencyPause"),
197                    vec![],
198                ))),
199                ProposalType::Unknown(type_name) => {
200                    error!(
201                        "Cannot delete proposal {:?} with unknown type: {}",
202                        proposal.id, type_name
203                    );
204                    continue;
205                }
206            };
207
208            builder.move_call(
209                Function::new(
210                    hashi_ids.package_id,
211                    Identifier::from_static("proposal"),
212                    Identifier::from_static("delete_expired"),
213                )
214                .with_type_args(vec![type_arg]),
215                vec![hashi_arg, proposal_id_arg, clock_arg],
216            );
217        }
218
219        let response = executor.execute(builder).await?;
220        if !response.transaction().effects().status().success() {
221            anyhow::bail!("Transaction failed to delete expired proposals");
222        }
223        info!(
224            "Successfully deleted {} expired proposals",
225            expired_proposals.len()
226        );
227        Ok(())
228    }
