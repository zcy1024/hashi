17const MAX_DEPOSIT_REQUEST_AGE_MS: u64 = 1000 * 60 * 60 * 24 * 3; // 3 days
18const DEPOSIT_REQUEST_DELETE_DELAY_MS: u64 = 1000 * 60 * 60 * 24; // 1 day
19const MAX_DEPOSIT_REQUEST_DELETIONS_PER_GC: usize = 500;
20
21const MAX_PROPOSAL_AGE_MS: u64 = 1000 * 60 * 60 * 24 * 7; // 7 days
22const PROPOSAL_DELETE_DELAY_MS: u64 = 1000 * 60 * 60 * 24; // 1 day
23
24impl LeaderService {
25    /// Check for and delete expired deposit requests.
26    /// Deposit requests must be sorted by timestamp, and will be deleted if they are older
27    /// than MAX_DEPOSIT_REQUEST_AGE_MS.
28    pub(crate) fn check_delete_expired_deposit_requests(
29        &mut self,
30        deposit_requests: &[DepositRequest],
31        checkpoint_timestamp_ms: u64,
32    ) {
33        if self.deposit_gc_task.is_some() {
34            debug!("Deposit GC task already in-flight, skipping");
35            return;
36        }
37
38        let Some(oldest_request) = deposit_requests.first() else {
39            return;
40        };
41
42        if checkpoint_timestamp_ms
43            < oldest_request.timestamp_ms
44                + MAX_DEPOSIT_REQUEST_AGE_MS
45                + DEPOSIT_REQUEST_DELETE_DELAY_MS
46        {
47            return;
48        }
49
50        let expired_requests: Vec<_> = deposit_requests
51            .iter()
52            .filter(|r| checkpoint_timestamp_ms > r.timestamp_ms + MAX_DEPOSIT_REQUEST_AGE_MS)
53            .take(MAX_DEPOSIT_REQUEST_DELETIONS_PER_GC)
54            .cloned()
55            .collect();
56        if expired_requests.is_empty() {
57            return;
58        }
59
60        info!(
61            "Scheduling deletion of {} expired deposit requests",
62            expired_requests.len()
63        );
64
65        let inner = self.inner.clone();
66        self.deposit_gc_task = Some(tokio::task::spawn(async move {
67            Self::delete_expired_deposit_requests(inner, expired_requests).await
68        }));
69    }
70
71    async fn delete_expired_deposit_requests(
72        inner: Arc<crate::Hashi>,
73        expired_requests: Vec<DepositRequest>,
74    ) -> anyhow::Result<()> {
75        let count = expired_requests.len();
76        let mut executor = SuiTxExecutor::from_hashi(inner)?;
77        executor
78            .execute_delete_expired_deposit_requests(&expired_requests)
79            .await?;
80        info!("Successfully deleted {count} expired deposit requests");
81        Ok(())
82    }

84    /// Check for and delete expired proposals.
85    /// Proposals are sorted by timestamp and deleted if they are older than MAX_PROPOSAL_AGE_MS.
86    pub(crate) fn check_delete_proposals(&mut self, checkpoint_timestamp_ms: u64) {
87        debug!("Entering check_delete_proposals");
88
89        if self.proposal_gc_task.is_some() {
90            debug!("Proposal GC task already in-flight, skipping");
91            return;
92        }
93
94        let mut proposals = self.inner.onchain_state().proposals();
95        // Sort proposals by timestamp, from earliest to latest
96        proposals.sort_by_key(|p| p.timestamp_ms);
97
98        // Check if it's time to delete
99        let Some(oldest_proposal) = proposals.first() else {
100            return;
101        };
102
103        // If there aren't any proposals at least 8 days old (7 days expiry + 1 day delay), don't do anything
104        if checkpoint_timestamp_ms
105            < oldest_proposal.timestamp_ms + MAX_PROPOSAL_AGE_MS + PROPOSAL_DELETE_DELAY_MS
106        {
107            return;
108        }
109
110        // Find all expired proposals (older than 7 days)
111        let expired_proposals: Vec<_> = proposals
112            .iter()
113            .filter(|p| checkpoint_timestamp_ms > p.timestamp_ms + MAX_PROPOSAL_AGE_MS)
114            .cloned()
115            .collect();
116
117        if expired_proposals.is_empty() {
118            return;
119        }
120
121        info!(
122            "Scheduling deletion of {} expired proposals",
123            expired_proposals.len()
124        );
125
126        let inner = self.inner.clone();
127        self.proposal_gc_task = Some(tokio::task::spawn(async move {
128            Self::delete_expired_proposals(inner, expired_proposals).await
129        }));
130    }
131
132    async fn delete_expired_proposals(
133        inner: Arc<crate::Hashi>,
134        expired_proposals: Vec<Proposal>,
135    ) -> anyhow::Result<()> {
136        use sui_sdk_types::Identifier;
137        use sui_sdk_types::StructTag;
138        use sui_sdk_types::TypeTag;
139        use sui_transaction_builder::Function;
140        use sui_transaction_builder::ObjectInput;
141        use sui_transaction_builder::TransactionBuilder;
142
143        let mut executor = SuiTxExecutor::from_hashi(inner.clone())?;
144        let hashi_ids = inner.config.hashi_ids();
145
146        let mut builder = TransactionBuilder::new();
147
148        let hashi_arg = builder.object(
149            ObjectInput::new(hashi_ids.hashi_object_id)
150                .as_shared()
151                .with_mutable(true),
152        );
153
154        // Clock object (0x6) - immutable shared object
155        let clock_arg = builder.object(
156            ObjectInput::new(Address::from_static("0x6"))
157                .as_shared()
158                .with_mutable(false),
159        );
160
161        // Add a move call for each expired proposal
162        for proposal in &expired_proposals {
163            let proposal_id_arg = builder.pure(&proposal.id);
164
165            // Get the type argument for the proposal
166            let type_arg = match &proposal.proposal_type {
167                ProposalType::UpdateConfig => TypeTag::Struct(Box::new(StructTag::new(
168                    hashi_ids.package_id,
169                    Identifier::from_static("update_config"),
170                    Identifier::from_static("UpdateConfig"),
171                    vec![],
172                ))),
173                ProposalType::EnableVersion => TypeTag::Struct(Box::new(StructTag::new(
174                    hashi_ids.package_id,
175                    Identifier::from_static("enable_version"),
176                    Identifier::from_static("EnableVersion"),
177                    vec![],
178                ))),
179                ProposalType::DisableVersion => TypeTag::Struct(Box::new(StructTag::new(
180                    hashi_ids.package_id,
181                    Identifier::from_static("disable_version"),
182                    Identifier::from_static("DisableVersion"),
183                    vec![],
184                ))),
185                ProposalType::Upgrade => TypeTag::Struct(Box::new(StructTag::new(
186                    hashi_ids.package_id,
187                    Identifier::from_static("upgrade"),
188                    Identifier::from_static("Upgrade"),
189                    vec![],
190                ))),
191                ProposalType::Unknown(type_name) => {
192                    error!(
193                        "Cannot delete proposal {:?} with unknown type: {}",
194                        proposal.id, type_name
195                    );
196                    continue;
197                }
198            };
199
200            builder.move_call(
201                Function::new(
202                    hashi_ids.package_id,
203                    Identifier::from_static("proposal"),
204                    Identifier::from_static("delete_expired"),
205                )
206                .with_type_args(vec![type_arg]),
207                vec![hashi_arg, proposal_id_arg, clock_arg],
208            );
209        }
210
211        let response = executor.execute(builder).await?;
212        if !response.transaction().effects().status().success() {
213            anyhow::bail!("Transaction failed to delete expired proposals");
214        }
215        info!(
216            "Successfully deleted {} expired proposals",
217            expired_proposals.len()
218        );
219        Ok(())
220    }
