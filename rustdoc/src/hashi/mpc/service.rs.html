41const RETRY_INTERVAL: Duration = Duration::from_secs(10);

59impl MpcHandle {
60    pub async fn wait_for_key_ready(&self) -> G {
61        let mut rx = self.key_ready_rx.clone();
62        loop {
63            {
64                let value = rx.borrow();
65                if let Some(pk) = value.as_ref() {
66                    return *pk;
67                }
68            }
69            if rx.changed().await.is_err() {
70                std::future::pending().await
71            }
72        }
73    }
74
75    pub fn public_key(&self) -> Option<G> {
76        *self.key_ready_rx.borrow()
77    }
78}

111    #[tracing::instrument(name = "mpc_service", skip_all)]
112    async fn run(mut self) {
113        let pending = self.get_pending_epoch_change();
114        let is_in_committee = self.inner.is_in_current_committee();
115        info!(
116            "MPC service starting: pending_epoch_change={pending:?}, \
117             is_in_current_committee={is_in_committee}",
118        );
119        if let Some(epoch) = pending {
120            info!("Entering handle_reconfig for epoch {epoch}");
121            self.handle_reconfig(epoch).await;
122        } else if self.is_awaiting_genesis() {
123            // No committee has been formed yet (epoch 0, no committee for epoch 0).
124            // Wait for enough validators to register then trigger genesis reconfig.
125            info!("No initial committee yet; waiting for enough validators to register...");
126            self.try_submit_genesis_reconfig().await;
127        } else if self.inner.is_in_current_committee() {
128            loop {
129                self.sync_if_stale().await;
130                let epoch = self.inner.onchain_state().epoch();
131                if self.inner.signing_manager_for(epoch).is_some() {
132                    break;
133                }
134                tokio::time::sleep(RETRY_INTERVAL).await;
135            }
136        } else {
137            info!("Node is not in the current committee, waiting for reconfig notification...");
138        }
139        let mut notifications = self.inner.onchain_state().subscribe();
140        let mut checkpoint_rx = self.inner.onchain_state().subscribe_checkpoint();
141        let mut reconcile_tick = tokio::time::interval(RECONCILE_TICK);
142        reconcile_tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
143        loop {
144            // Check for pending reconfig before blocking on `recv()`.
145            if let Some(epoch) = self.get_pending_epoch_change() {
146                self.handle_reconfig(epoch).await;
147                continue;
148            }
149            tokio::select! {
150                notification = notifications.recv() => {
151                    match notification {
152                        Ok(notification) => match notification {
153                            Notification::StartReconfig(epoch) => {
154                                self.handle_reconfig(epoch).await;
155                            }
156                            Notification::SuiEpochChanged(sui_epoch) => {
157                                self.try_submit_start_reconfig(sui_epoch).await;
158                            }
159                            _ => {}
160                        },
161                        Err(e) => {
162                            error!("MPC notification recv error: {e:?}, resubscribing");
163                            notifications = self.inner.onchain_state().subscribe();
164                        }
165                    }
166                }
167                Ok(()) = checkpoint_rx.changed() => {
168                    self.sync_if_stale().await;
169                }
170                _ = reconcile_tick.tick() => {
171                    self.sync_if_stale().await;
172                }
173                Ok(()) = self.refill_rx.changed() => {
174                    let next_batch = *self.refill_rx.borrow();
175                    for attempt in 1..=MAX_PROTOCOL_ATTEMPTS {
176                        match self.refill_presignatures(next_batch).await {
177                            Ok(()) => break,
178                            Err(e) => {
179                                error!(
180                                    "Presignature refill attempt {attempt}/{MAX_PROTOCOL_ATTEMPTS} failed: {e}"
181                                );
182                                if attempt < MAX_PROTOCOL_ATTEMPTS {
183                                    tokio::time::sleep(RETRY_INTERVAL).await;
184                                }
185                            }
186                        }
187                    }
188                }
189            }
190        }
191    }

193    async fn sleep_if_still_pending(&self, epoch: u64) {
194        if self.get_pending_epoch_change() == Some(epoch) {
195            tokio::time::sleep(RETRY_INTERVAL).await;
196        }
197    }

208    /// Returns true if no committee has ever been formed (genesis state).
209    fn is_awaiting_genesis(&self) -> bool {
210        let state = self.inner.onchain_state().state();
211        let committees = &state.hashi().committees;
212        committees.epoch() == 0 && committees.current_committee().is_none()
213    }

215    /// Wait for enough validators to register, then submit `start_reconfig`
216    /// to form the initial committee. Blocks until a pending epoch change
217    /// appears (either from our own submission or another node's).
218    async fn try_submit_genesis_reconfig(&self) {
219        loop {
220            if self.get_pending_epoch_change().is_some() {
221                return;
222            }
223            // Attempt to submit start_reconfig. This will fail on-chain if
224            // not enough validators have registered (95% stake threshold).
225            let result = async {
226                let mut executor =
227                    crate::sui_tx_executor::SuiTxExecutor::from_hashi(self.inner.clone())?;
228                executor.execute_start_reconfig().await
229            };
230            match result.await {
231                Ok(()) => {
232                    info!("Genesis start_reconfig submitted successfully");
233                    return;
234                }
235                Err(e) => {
236                    debug!("Genesis start_reconfig not yet possible: {e}");
237                    // Poll for pending epoch change while waiting, in case
238                    // another node submitted start_reconfig.
239                    let polls = (RETRY_INTERVAL.as_millis()
240                        / START_RECONFIG_POLL_INTERVAL.as_millis())
241                        as u32;
242                    for _ in 0..polls {
243                        if self.get_pending_epoch_change().is_some() {
244                            return;
245                        }
246                        tokio::time::sleep(START_RECONFIG_POLL_INTERVAL).await;
247                    }
248                }
249            }
250        }
251    }

253    async fn recover_mpc_state(&self) -> anyhow::Result<MpcOutput> {
254        let onchain_state = self.inner.onchain_state().clone();
255        let epoch = onchain_state.epoch();
256        let protocol_type = onchain_state
257            .fetch_certs(epoch, None)
258            .await?
259            .map(|(pt, _)| pt);
260        let onchain_mpc_key = onchain_state.mpc_public_key();
261        info!(
262            "recover_mpc_state: epoch={epoch}, protocol_type={protocol_type:?}, \
263             onchain_mpc_key_len={}",
264            onchain_mpc_key.len(),
265        );
266        let output = match protocol_type {
267            Some(hashi_types::move_types::ProtocolType::KeyRotation) => {
268                self.setup_key_rotation(epoch)?;
269                self.run_key_rotation(epoch).await
270            }
271            _ => {
272                self.setup_initial_dkg(epoch)?;
273                self.run_dkg(epoch).await
274            }
275        }?;
276        info!(
277            "recover_mpc_state: recovered vk={}",
278            hex::encode(output.public_key.to_byte_array())
279        );
280        Ok(output)
281    }

283    #[tracing::instrument(level = "info", skip_all, fields(target_epoch))]
284    async fn run_dkg(&self, target_epoch: u64) -> anyhow::Result<MpcOutput> {
285        let onchain_state = self.inner.onchain_state().clone();
286        let mpc_manager = self
287            .inner
288            .mpc_manager()
289            .expect("MpcManager must be set before run_dkg");
290        let signer = self.inner.config.operator_private_key()?;
291        let p2p_channel = RpcP2PChannel::new(onchain_state.clone(), target_epoch, MPC_LABEL_DKG);
292        let mut tob_channel = SuiTobChannel::new(
293            self.inner.config.hashi_ids(),
294            onchain_state,
295            target_epoch,
296            None,
297            signer,
298        );
299        let output = MpcManager::run_dkg(
300            &mpc_manager,
301            &p2p_channel,
302            &mut tob_channel,
303            &self.inner.metrics,
304        )
305        .await
306        .map_err(|e| anyhow::anyhow!("DKG failed: {e}"))?;
307        Ok(output)
308    }

310    async fn generate_presignatures(
311        &self,
312        epoch: u64,
313        batch_index: u32,
314    ) -> anyhow::Result<(Committee, Presignatures)> {
315        let onchain_state = self.inner.onchain_state().clone();
316        let committee = onchain_state
317            .state()
318            .hashi()
319            .committees
320            .committees()
321            .get(&epoch)
322            .ok_or_else(|| anyhow::anyhow!("No committee found for epoch {}", epoch))?
323            .clone();
324        let mpc_manager = self
325            .inner
326            .mpc_manager()
327            .ok_or_else(|| anyhow::anyhow!("MpcManager not initialized"))?;
328        let signer = self.inner.config.operator_private_key()?;
329        let p2p_channel =
330            RpcP2PChannel::new(onchain_state.clone(), epoch, MPC_LABEL_NONCE_GENERATION);
331        let mut tob_channel = SuiTobChannel::new(
332            self.inner.config.hashi_ids(),
333            onchain_state,
334            epoch,
335            Some(batch_index),
336            signer,
337        );
338        let metrics = &self.inner.metrics;
339        let _timer = metrics
340            .mpc_total_duration_seconds
341            .with_label_values(&[MPC_LABEL_NONCE_GENERATION])
342            .start_timer();
343        let nonce_result = MpcManager::run_nonce_generation(
344            &mpc_manager,
345            batch_index,
346            &p2p_channel,
347            &mut tob_channel,
348            metrics,
349        )
350        .await;
351        drop(_timer);
352        let nonce_outputs =
353            nonce_result.map_err(|e| anyhow::anyhow!("Nonce generation failed: {e}"))?;
354        let (batch_size_per_weight, f) = {
355            let mgr = mpc_manager.read().unwrap();
356            (
357                mgr.batch_size_per_weight,
358                mgr.mpc_config.max_faulty as usize,
359            )
360        };
361        let _timer = metrics
362            .mpc_presig_conversion_duration_seconds
363            .with_label_values(&[MPC_LABEL_NONCE_GENERATION])
364            .start_timer();
365        let presignatures = Presignatures::new(nonce_outputs, batch_size_per_weight, f)
366            .map_err(|e| anyhow::anyhow!("Failed to create presignatures: {e}"))?;
367        drop(_timer);
368        Ok((committee, presignatures))
369    }

371    async fn prepare_signing(&self, epoch: u64, output: &MpcOutput) -> anyhow::Result<()> {
372        let (committee, presignatures) = self.generate_presignatures(epoch, 0).await?;
373        let address = self.inner.config.validator_address()?;
374        let signing_manager = SigningManager::new(
375            address,
376            committee,
377            output.threshold,
378            output.key_shares.clone(),
379            output.public_key,
380            presignatures,
381            0, // batch_index
382            0, // batch_start_index
383            PRESIG_REFILL_DIVISOR,
384            self.refill_tx.clone(),
385        );
386        self.inner.store_signing_manager(signing_manager);
387        Ok(())
388    }

390    async fn recover_presigning_state(&self, output: &MpcOutput) -> anyhow::Result<()> {
391        let (num_consumed, epoch, committee) = {
392            let state = self.inner.onchain_state().state();
393            let hashi = state.hashi();
394            let num_consumed = hashi.num_consumed_presigs;
395            let epoch = hashi.committees.epoch();
396            let committee = hashi
397                .committees
398                .committees()
399                .get(&epoch)
400                .ok_or_else(|| anyhow::anyhow!("No committee found for epoch {epoch}"))?
401                .clone();
402            (num_consumed, epoch, committee)
403        };
404        let mpc_manager = self
405            .inner
406            .mpc_manager()
407            .ok_or_else(|| anyhow::anyhow!("MpcManager not initialized"))?;
408        let (batch_size_per_weight, f) = {
409            let mgr = mpc_manager.read().unwrap();
410            (
411                mgr.batch_size_per_weight,
412                mgr.mpc_config.max_faulty as usize,
413            )
414        };
415        // Walk through batches to find the one containing `num_consumed`.
416        // Each batch can have a different size with unequal committee weights.
417        let mut batch_start = 0u64;
418        let mut batch_index = 0u32;
419        let presignatures = loop {
420            let presigs = self
421                .recover_presignatures_from_certs(
422                    &mpc_manager,
423                    epoch,
424                    batch_index,
425                    batch_size_per_weight,
426                    f,
427                )
428                .await?;
429            let size = presigs.len() as u64;
430            if num_consumed < batch_start + size {
431                break presigs;
432            }
433            batch_start += size;
434            batch_index += 1;
435        };
436        let batch_size = presignatures.len();
437        let address = self.inner.config.validator_address()?;
438        let signing_manager = SigningManager::new(
439            address,
440            committee,
441            output.threshold,
442            output.key_shares.clone(),
443            output.public_key,
444            presignatures,
445            batch_index,
446            batch_start,
447            PRESIG_REFILL_DIVISOR,
448            self.refill_tx.clone(),
449        );
450        self.inner.store_signing_manager(signing_manager);
451        info!(
452            "Recovered presigning state: batch_index={batch_index}, \
453             batch_start={batch_start}, batch_size={batch_size} \
454             (num_consumed_presigs={num_consumed})."
455        );
456        Ok(())
457    }

459    async fn sync_if_stale(&self) {
460        if self.is_awaiting_genesis() || !self.inner.is_in_current_committee() {
461            return;
462        }
463        let epoch = self.inner.onchain_state().epoch();
464        if self.inner.signing_manager_for(epoch).is_some() {
465            return;
466        }
467        let _guard = match self.reconciling.try_lock() {
468            Ok(g) => g,
469            Err(_) => return,
470        };
471        let epoch = self.inner.onchain_state().epoch();
472        if self.inner.signing_manager_for(epoch).is_some() {
473            return;
474        }
475        info!("sync_if_stale: rebuilding SigningManager for epoch {epoch}");
476        let output = match self.recover_mpc_state().await {
477            Ok(output) => output,
478            Err(e) => {
479                error!("sync_if_stale: recover_mpc_state failed for epoch {epoch}: {e}");
480                return;
481            }
482        };
483        match self.recover_presigning_state(&output).await {
484            Ok(()) => {
485                info!("sync_if_stale: recovered epoch {epoch} via certs path");
486            }
487            Err(certs_err) => {
488                info!(
489                    "sync_if_stale: certs path failed for epoch {epoch} ({certs_err}), \
490                     falling back to protocol"
491                );
492                if let Err(e) = self.prepare_signing(epoch, &output).await {
493                    error!("sync_if_stale: protocol fallback failed for epoch {epoch}: {e}");
494                    return;
495                }
496                info!("sync_if_stale: recovered epoch {epoch} via protocol path");
497            }
498        }
499        let _ = self.key_ready_tx.send(Some(output.public_key));
500    }

502    async fn refill_presignatures(&self, batch_index: u32) -> anyhow::Result<()> {
503        let epoch = self.inner.onchain_state().epoch();
504        let signing_manager = self
505            .inner
506            .signing_manager_for(epoch)
507            .ok_or_else(|| anyhow::anyhow!("SigningManager not available for epoch {epoch}"))?;
508        let (_, presignatures) = self.generate_presignatures(epoch, batch_index).await?;
509        if self.inner.onchain_state().epoch() != epoch {
510            return Err(anyhow::anyhow!("Epoch changed during presignature refill"));
511        }
512        signing_manager.set_next_batch(presignatures);
513        Ok(())
514    }

516    async fn recover_presignatures_from_certs(
517        &self,
518        mpc_manager: &Arc<std::sync::RwLock<MpcManager>>,
519        epoch: u64,
520        batch_index: u32,
521        batch_size_per_weight: u16,
522        f: usize,
523    ) -> anyhow::Result<Presignatures> {
524        let onchain_state = self.inner.onchain_state().clone();
525        let (_, certs) = onchain_state
526            .fetch_certs(epoch, Some(batch_index))
527            .await?
528            .ok_or_else(|| {
529                anyhow::anyhow!(
530                    "No nonce gen certificates on TOB for epoch {epoch} batch {batch_index}"
531                )
532            })?;
533        let p2p_channel = RpcP2PChannel::new(
534            self.inner.onchain_state().clone(),
535            epoch,
536            MPC_LABEL_NONCE_GENERATION,
537        );
538        let outputs = MpcManager::reconstruct_presignatures_with_complaint_recovery(
539            mpc_manager,
540            epoch,
541            batch_index,
542            &certs,
543            &p2p_channel,
544        )
545        .await?;
546        if outputs.is_empty() {
547            return Err(anyhow::anyhow!(
548                "No valid nonce outputs after reconstruction for epoch {epoch} batch {batch_index}"
549            ));
550        }
551        Presignatures::new(outputs, batch_size_per_weight, f)
552            .map_err(|e| anyhow::anyhow!("Failed to create presignatures: {e}"))
553    }

555    async fn try_submit_start_reconfig(&self, sui_epoch: u64) {
556        if self.get_pending_epoch_change().is_some() {
557            return;
558        }
559        let hashi_epoch = self
560            .inner
561            .onchain_state()
562            .state()
563            .hashi()
564            .committees
565            .epoch();
566        if hashi_epoch >= sui_epoch {
567            return;
568        }
569        for attempt in 1..=MAX_PROTOCOL_ATTEMPTS {
570            let result = async {
571                let mut executor =
572                    crate::sui_tx_executor::SuiTxExecutor::from_hashi(self.inner.clone())?;
573                executor.execute_start_reconfig().await
574            };
575            match result.await {
576                Ok(()) => {
577                    return;
578                }
579                Err(e) => {
580                    warn!("start_reconfig attempt {attempt}/{MAX_PROTOCOL_ATTEMPTS} failed: {e}");
581                    if attempt < MAX_PROTOCOL_ATTEMPTS {
582                        // Poll for pending epoch change while waiting, so we can
583                        // return early if another node submitted start_reconfig.
584                        let polls = (RETRY_INTERVAL.as_millis()
585                            / START_RECONFIG_POLL_INTERVAL.as_millis())
586                            as u32;
587                        for _ in 0..polls {
588                            if self.get_pending_epoch_change().is_some() {
589                                return;
590                            }
591                            tokio::time::sleep(START_RECONFIG_POLL_INTERVAL).await;
592                        }
593                    }
594                }
595            }
596        }
597    }

599    #[tracing::instrument(level = "info", skip_all, fields(target_epoch))]
600    async fn handle_reconfig(&self, target_epoch: u64) {
601        let run_dkg = self
602            .inner
603            .onchain_state()
604            .state()
605            .hashi()
606            .committees
607            .mpc_public_key()
608            .is_empty();
609        let protocol_label = if run_dkg {
610            MPC_LABEL_DKG
611        } else {
612            MPC_LABEL_KEY_ROTATION
613        };
614        let metrics = &self.inner.metrics;
615        let _reconfig_timer = metrics
616            .mpc_reconfig_total_duration_seconds
617            .with_label_values(&[protocol_label])
618            .start_timer();
619        info!("handle_reconfig: epoch={target_epoch}, run_dkg={run_dkg}, entering retry loop",);
620        let output = loop {
621            if self.get_pending_epoch_change() != Some(target_epoch) {
622                info!("handle_reconfig: epoch {target_epoch} no longer pending, aborting",);
623                return;
624            }
625            let setup_result = if run_dkg {
626                self.setup_initial_dkg(target_epoch)
627            } else {
628                self.setup_key_rotation(target_epoch)
629            };
630            if let Err(e) = setup_result {
631                error!(
632                    "Failed to set up MPC manager for epoch {}: {e}, retrying...",
633                    target_epoch
634                );
635                self.sleep_if_still_pending(target_epoch).await;
636                continue;
637            }
638            let _timer = metrics
639                .mpc_total_duration_seconds
640                .with_label_values(&[protocol_label])
641                .start_timer();
642            let result = if run_dkg {
643                tokio::time::timeout(MPC_PROTOCOL_TIMEOUT, self.run_dkg(target_epoch))
644                    .await
645                    .unwrap_or_else(|_| {
646                        Err(anyhow::anyhow!(
647                            "DKG timed out after {MPC_PROTOCOL_TIMEOUT:?}"
648                        ))
649                    })
650            } else {
651                tokio::time::timeout(MPC_PROTOCOL_TIMEOUT, self.run_key_rotation(target_epoch))
652                    .await
653                    .unwrap_or_else(|_| {
654                        Err(anyhow::anyhow!(
655                            "Key rotation timed out after {MPC_PROTOCOL_TIMEOUT:?}"
656                        ))
657                    })
658            };
659            drop(_timer);
660            match result {
661                Ok(output) => break output,
662                Err(e) => {
663                    error!(
664                        "MPC protocol for epoch {} failed: {e}, retrying...",
665                        target_epoch
666                    );
667                    self.sleep_if_still_pending(target_epoch).await;
668                }
669            }
670        };
671        let _ = self.key_ready_tx.send(Some(output.public_key));
672        info!("MPC key ready for epoch {target_epoch}, submitting end_reconfig");
673        let _end_reconfig_timer = metrics
674            .mpc_end_reconfig_duration_seconds
675            .with_label_values(&[protocol_label])
676            .start_timer();
677        loop {
678            if self.get_pending_epoch_change() != Some(target_epoch) {
679                break;
680            }
681            match self.submit_end_reconfig(target_epoch, &output).await {
682                Ok(()) => break,
683                Err(e) => {
684                    warn!(
685                        "submit_end_reconfig for epoch {} failed: {e}, retrying...",
686                        target_epoch
687                    );
688                    self.sleep_if_still_pending(target_epoch).await;
689                }
690            }
691        }
692        drop(_end_reconfig_timer);
693        info!("end_reconfig complete for epoch {target_epoch}, running prepare_signing");
694        if let Err(e) = self.inner.db.prune_messages_below(target_epoch) {
695            error!("Failed to prune old MPC messages below epoch {target_epoch}: {e}");
696        }
697        let _prepare_signing_timer = metrics
698            .mpc_prepare_signing_duration_seconds
699            .with_label_values(&[protocol_label])
700            .start_timer();
701        for attempt in 1..=MAX_PROTOCOL_ATTEMPTS {
702            match self.prepare_signing(target_epoch, &output).await {
703                Ok(()) => break,
704                Err(e) => {
705                    error!(
706                        "prepare_signing attempt {attempt}/{MAX_PROTOCOL_ATTEMPTS} \
707                         for epoch {target_epoch}: {e}"
708                    );
709                    if attempt < MAX_PROTOCOL_ATTEMPTS {
710                        tokio::time::sleep(RETRY_INTERVAL).await;
711                    } else {
712                        error!(
713                            "All prepare_signing attempts exhausted for epoch {target_epoch}. \
714                             Node cannot sign until next recovery trigger."
715                        );
716                    }
717                }
718            }
719        }
720        drop(_prepare_signing_timer);
721        drop(_reconfig_timer);
722    }

740    #[tracing::instrument(level = "info", skip_all, fields(target_epoch))]
741    async fn run_key_rotation(&self, target_epoch: u64) -> anyhow::Result<MpcOutput> {
742        let onchain_state = self.inner.onchain_state().clone();
743        let mpc_manager = self
744            .inner
745            .mpc_manager()
746            .ok_or_else(|| anyhow::anyhow!("MpcManager not initialized for key rotation"))?;
747        let source_epoch = mpc_manager.read().unwrap().source_epoch;
748        let onchain_mpc_key = hex::encode(onchain_state.mpc_public_key());
749        let onchain_epoch = onchain_state.epoch();
750        info!(
751            "run_key_rotation: target_epoch={target_epoch}, source_epoch={source_epoch}, \
752             onchain_epoch={onchain_epoch}, onchain_mpc_key={onchain_mpc_key}",
753        );
754        let previous_certs = fetch_certificates(&onchain_state, source_epoch, None)
755            .await
756            .map_err(|e| anyhow::anyhow!("Failed to fetch previous certificates: {e}"))?;
757        let previous_certs: Vec<CertificateV1> =
758            previous_certs.into_iter().map(|(_, cert)| cert).collect();
759        info!(
760            "run_key_rotation: fetched {} certs for source_epoch={source_epoch}",
761            previous_certs.len(),
762        );
763        let signer = self.inner.config.operator_private_key()?;
764        let p2p_channel =
765            RpcP2PChannel::new(onchain_state.clone(), target_epoch, MPC_LABEL_KEY_ROTATION);
766        let mut tob_channel = SuiTobChannel::new(
767            self.inner.config.hashi_ids(),
768            onchain_state,
769            target_epoch,
770            None,
771            signer,
772        );
773        let output = MpcManager::run_key_rotation(
774            &mpc_manager,
775            &previous_certs,
776            &p2p_channel,
777            &mut tob_channel,
778            &self.inner.metrics,
779        )
780        .await
781        .map_err(|e| anyhow::anyhow!("Key rotation failed: {e}"))?;
782        Ok(output)
783    }

785    async fn submit_end_reconfig(&self, epoch: u64, output: &MpcOutput) -> anyhow::Result<()> {
786        let mpc_public_key =
787            bcs::to_bytes(&output.public_key).expect("public key serialization should succeed");
788        let target_committee = self
789            .inner
790            .onchain_state()
791            .state()
792            .hashi()
793            .committees
794            .committees()
795            .get(&epoch)
796            .cloned()
797            .ok_or_else(|| anyhow::anyhow!("no committee found for epoch {}", epoch))?;
798        let message = ReconfigCompletionMessage {
799            epoch,
800            mpc_public_key: mpc_public_key.clone(),
801        };
802        let signing_key = self
803            .inner
804            .config
805            .protocol_private_key()
806            .ok_or_else(|| anyhow::anyhow!("no protocol_private_key configured"))?;
807        let my_address = self.inner.config.validator_address()?;
808        let my_sig = signing_key.sign(epoch, my_address, &message);
809        self.inner
810            .store_reconfig_signature(epoch, my_sig.signature().as_bytes().to_vec());
811        let cert = loop {
812            if self.get_pending_epoch_change() != Some(epoch) {
813                return Err(anyhow::anyhow!("epoch {} no longer pending", epoch));
814            }
815            match self
816                .collect_reconfig_signatures(epoch, &mpc_public_key, &target_committee)
817                .await
818            {
819                Ok(cert) => break cert,
820                Err(e) => {
821                    warn!(
822                        "Signature collection for epoch {} failed: {e}, retrying...",
823                        epoch
824                    );
825                    self.sleep_if_still_pending(epoch).await;
826                }
827            }
828        };
829        loop {
830            if self.get_pending_epoch_change() != Some(epoch) {
831                return Err(anyhow::anyhow!("epoch {} no longer pending", epoch));
832            }
833            let result = async {
834                let mut executor =
835                    crate::sui_tx_executor::SuiTxExecutor::from_hashi(self.inner.clone())?;
836                executor
837                    .execute_end_reconfig(&mpc_public_key, cert.committee_signature())
838                    .await
839            };
840            match result.await {
841                Ok(()) => return Ok(()),
842                Err(e) => {
843                    warn!(
844                        "end_reconfig submission for epoch {} failed: {e}, retrying...",
845                        epoch
846                    );
847                    self.sleep_if_still_pending(epoch).await;
848                }
849            }
850        }
851    }
852
853    async fn collect_reconfig_signatures(
854        &self,
855        epoch: u64,
856        mpc_public_key: &[u8],
857        committee: &Committee,
858    ) -> anyhow::Result<hashi_types::committee::SignedMessage<ReconfigCompletionMessage>> {
859        let message = ReconfigCompletionMessage {
860            epoch,
861            mpc_public_key: mpc_public_key.to_vec(),
862        };
863        let my_address = self.inner.config.validator_address()?;
864        let my_sig_bytes = self
865            .inner
866            .get_reconfig_signature(epoch)
867            .expect("own signature must be stored before collecting");
868        let my_sig =
869            BLS12381Signature::from_bytes(&my_sig_bytes).expect("stored signature must be valid");
870        let mut aggregator = BlsSignatureAggregator::new(committee, message.clone());
871        aggregator
872            .add_signature_from(my_address, my_sig)
873            .map_err(|e| anyhow::anyhow!("failed to add own signature: {e}"))?;
874        let required_weight = certificate_threshold(committee.total_weight());
875        while aggregator.weight() < required_weight {
876            if self.get_pending_epoch_change() != Some(epoch) {
877                return Err(anyhow::anyhow!(
878                    "epoch {epoch} no longer pending during signature collection"
879                ));
880            }
881            let other_members: Vec<_> = committee
882                .members()
883                .iter()
884                .filter(|m| m.validator_address() != my_address)
885                .collect();
886            let futures = other_members.iter().map(|member| {
887                let address = member.validator_address();
888                async move {
889                    let result = tokio::time::timeout(RPC_TIMEOUT, async {
890                        let client = self
891                            .inner
892                            .onchain_state()
893                            .state()
894                            .hashi()
895                            .committees
896                            .client(&address)
897                            .ok_or_else(|| anyhow::anyhow!("client not found for {}", address))?;
898                        client
899                            .get_reconfig_completion_signature(epoch)
900                            .await
901                            .map_err(|e| anyhow::anyhow!("RPC failed: {e}"))
902                    })
903                    .await
904                    .unwrap_or_else(|_| Err(anyhow::anyhow!("RPC timed out")));
905                    (
906                        address,
907                        result.and_then(|opt| {
908                            opt.ok_or_else(|| anyhow::anyhow!("signature not ready"))
909                        }),
910                    )
911                }
912            });
913            let results = join_all(futures).await;
914            for (address, result) in results {
915                if let Ok(sig_bytes) = result {
916                    match BLS12381Signature::from_bytes(&sig_bytes) {
917                        Ok(sig) => {
918                            if let Err(e) = aggregator.add_signature_from(address, sig) {
919                                info!("Signature from {} rejected: {e}", address);
920                            }
921                        }
922                        Err(e) => {
923                            info!("Invalid signature bytes from {}: {e}", address);
924                        }
925                    }
926                }
927            }
928            if aggregator.weight() < required_weight {
929                tokio::time::sleep(RETRY_INTERVAL).await;
930            }
931        }
932        aggregator
933            .finish()
934            .map_err(|e| anyhow::anyhow!("failed to finalize certificate: {e}"))
935    }
