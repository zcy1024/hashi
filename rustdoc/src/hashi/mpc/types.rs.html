41#[derive(Clone, Debug, Serialize, Deserialize)]
42pub struct MpcConfig {
43    pub epoch: u64,
44    pub nodes: Nodes<EncryptionGroupElement>,
45    /// Threshold for signing (t)
46    pub threshold: u16,
47    /// Maximum number of faulty validators (f)
48    pub max_faulty: u16,
49}
50
51impl MpcConfig {
52    pub fn new(
53        epoch: u64,
54        nodes: Nodes<EncryptionGroupElement>,
55        threshold: u16,
56        max_faulty: u16,
57    ) -> Result<Self, MpcError> {
58        if threshold <= max_faulty {
59            return Err(MpcError::InvalidThreshold(
60                "threshold must be greater than max_faulty".into(),
61            ));
62        }
63        let total_weight = nodes.total_weight();
64        if threshold + 2 * max_faulty > total_weight {
65            return Err(MpcError::InvalidThreshold(format!(
66                "t + 2f ({}) must be <= total weight ({})",
67                threshold + 2 * max_faulty,
68                total_weight
69            )));
70        }
71        Ok(Self {
72            epoch,
73            nodes,
74            threshold,
75            max_faulty,
76        })
77    }
78}

84// Unique MPC protocol instance identifier (per epoch & chain).
85#[derive(Clone, Debug, Hash, Eq, PartialEq, Serialize, Deserialize)]
86pub enum ProtocolType {
87    Dkg,
88    KeyRotation,
89    NonceGeneration { batch_index: u32 },
90    Signing { message_hash: MessageHash },
91}
92
93impl SessionId {
94    pub fn new(chain_id: &str, epoch: u64, protocol_identifer: &ProtocolType) -> Self {
95        let oracle = RandomOracle::new(DOMAIN_HASHI);
96        SessionId(oracle.evaluate(&(chain_id, epoch, protocol_identifer)))
97    }
98
99    pub fn dealer_session_id(&self, dealer: &Address) -> SessionId {
100        let oracle = RandomOracle::new(&hex::encode(self.0));
101        SessionId(oracle.evaluate(&dealer))
102    }
103
104    pub fn nonce_dealer_session_id(
105        chain_id: &str,
106        epoch: u64,
107        batch_index: u32,
108        dealer: &Address,
109    ) -> SessionId {
110        let base = Self::new(
111            chain_id,
112            epoch,
113            &ProtocolType::NonceGeneration { batch_index },
114        );
115        base.dealer_session_id(dealer)
116    }
117
118    pub fn rotation_session_id(&self, dealer: &Address, share_index: ShareIndex) -> SessionId {
119        let oracle = RandomOracle::new(&hex::encode(self.0));
120        SessionId(oracle.evaluate(&(dealer, share_index.get())))
121    }
122
123    pub fn to_vec(&self) -> Vec<u8> {
124        self.0.to_vec()
125    }
126}

248pub struct DealerMessagesHash {
249    pub dealer_address: Address,
250    pub messages_hash: MessageHash,
251}
252
253impl DealerMessagesHash {
254    pub fn from_onchain_cert(
255        cert: &DealerSubmissionV1,
256        epoch: u64,
257    ) -> Result<DealerCertificate, MpcError> {
258        let hash_bytes: [u8; 32] =
259            cert.message
260                .messages_hash
261                .as_slice()
262                .try_into()
263                .map_err(|_| MpcError::InvalidMessage {
264                    sender: cert.message.dealer_address,
265                    reason: "invalid messages_hash length".into(),
266                })?;
267
268        let message = Self {
269            dealer_address: cert.message.dealer_address,
270            messages_hash: hash_bytes.into(),
271        };
272        let signed_message = SignedMessage::new(
273            epoch,
274            message,
275            &cert.signature.signature,
276            &cert.signature.signers_bitmap,
277        )
278        .map_err(|e| MpcError::InvalidCertificate(e.to_string()))?;
279        Ok(signed_message)
280    }
281}

287pub enum CertificateV1 {

289    Rotation(DealerCertificate),
290    NonceGeneration {

296impl CertificateV1 {
297    pub fn new(
298        protocol_type: hashi_types::move_types::ProtocolType,
299        batch_index: Option<u32>,
300        cert: DealerCertificate,
301    ) -> Self {
302        match protocol_type {
303            hashi_types::move_types::ProtocolType::Dkg => CertificateV1::Dkg(cert),
304            hashi_types::move_types::ProtocolType::KeyRotation => CertificateV1::Rotation(cert),
305            hashi_types::move_types::ProtocolType::NonceGeneration => {
306                CertificateV1::NonceGeneration {
307                    batch_index: batch_index.expect("batch_index required for NonceGeneration"),
308                    cert,
309                }
310            }
311        }
312    }
313
314    pub fn epoch(&self) -> u64 {
315        match self {
316            CertificateV1::Dkg(cert) | CertificateV1::Rotation(cert) => cert.epoch(),
317            CertificateV1::NonceGeneration { cert, .. } => cert.epoch(),
318        }
319    }
320
321    pub fn dealer_address(&self) -> Address {
322        match self {
323            CertificateV1::Dkg(cert) | CertificateV1::Rotation(cert) => {
324                cert.message().dealer_address
325            }
326            CertificateV1::NonceGeneration { cert, .. } => cert.message().dealer_address,
327        }
328    }
329
330    pub fn signature_bytes(&self) -> &[u8] {
331        match self {
332            CertificateV1::Dkg(cert) | CertificateV1::Rotation(cert) => cert.signature_bytes(),
333            CertificateV1::NonceGeneration { cert, .. } => cert.signature_bytes(),
334        }
335    }
336
337    pub fn signers_bitmap_bytes(&self) -> &[u8] {
338        match self {
339            CertificateV1::Dkg(cert) | CertificateV1::Rotation(cert) => cert.signers_bitmap_bytes(),
340            CertificateV1::NonceGeneration { cert, .. } => cert.signers_bitmap_bytes(),
341        }
342    }
343
344    pub fn signers(
345        &self,
346        committee: &Committee,
347    ) -> Result<Vec<Address>, sui_crypto::SignatureError> {
348        match self {
349            CertificateV1::Dkg(cert) | CertificateV1::Rotation(cert) => cert.signers(committee),
350            CertificateV1::NonceGeneration { cert, .. } => cert.signers(committee),
351        }
352    }
353
354    pub fn weight(&self, committee: &Committee) -> Result<u64, sui_crypto::SignatureError> {
355        match self {
356            CertificateV1::Dkg(cert) | CertificateV1::Rotation(cert) => cert.weight(committee),
357            CertificateV1::NonceGeneration { cert, .. } => cert.weight(committee),
358        }
359    }
360
361    pub fn is_signer(
362        &self,
363        address: &Address,
364        committee: &Committee,
365    ) -> Result<bool, sui_crypto::SignatureError> {
366        match self {
367            CertificateV1::Dkg(cert) | CertificateV1::Rotation(cert) => {
368                cert.is_signer(address, committee)
369            }
370            CertificateV1::NonceGeneration { cert, .. } => cert.is_signer(address, committee),
371        }
372    }
373
374    pub fn message(&self) -> &DealerMessagesHash {
375        match self {
376            CertificateV1::Dkg(cert) | CertificateV1::Rotation(cert) => cert.message(),
377            CertificateV1::NonceGeneration { cert, .. } => cert.message(),
378        }
379    }
380
381    pub fn protocol_type(&self) -> ProtocolType {
382        match self {
383            CertificateV1::Dkg(_) => ProtocolType::Dkg,
384            CertificateV1::Rotation(_) => ProtocolType::KeyRotation,
385            CertificateV1::NonceGeneration { batch_index, .. } => ProtocolType::NonceGeneration {
386                batch_index: *batch_index,
387            },
388        }
389    }
390}

406    InvalidMessage { sender: Address, reason: String },
407
408    #[error("Protocol timeout after {seconds} seconds")]
409    Timeout { seconds: u64 },
410
411    #[error("Not enough approvals: need {needed}, got {got}")]
412    NotEnoughApprovals { needed: usize, got: usize },
413
414    #[error("Certificate verification failed: {0}")]
415    InvalidCertificate(String),
416
417    #[error("Broadcast channel error: {0}")]
418    BroadcastError(String),

420    #[error("Pairwise communication error: {0}")]
421    PairwiseCommunicationError(String),
422
423    #[error("Storage error: {0}")]
424    StorageError(String),

426    #[error("Cryptographic error: {0}")]
427    CryptoError(String),
428
429    #[error("Not found: {0}")]
430    NotFound(String),

432    #[error("Not ready: {0}")]
433    NotReady(String),

436    ProtocolFailed(String),

709    #[test]
710    fn test_from_onchain_cert_success() {
711        let mut rng = rand::thread_rng();
712        let epoch = 100u64;
713
714        // Create committee with 3 members
715        let signing_keys: Vec<_> = (0..3)
716            .map(|_| Bls12381PrivateKey::generate(&mut rng))
717            .collect();
718        let encryption_keys: Vec<_> = (0..3)
719            .map(|_| EncryptionPrivateKey::new(&mut rng))
720            .collect();
721        let members: Vec<_> = (0..3)
722            .map(|i| {
723                CommitteeMember::new(
724                    Address::new([i as u8; 32]),
725                    signing_keys[i].public_key(),
726                    EncryptionPublicKey::from_private_key(&encryption_keys[i]),
727                    1,
728                )
729            })
730            .collect();
731        let committee = Committee::new(members, epoch);
732
733        // Create a DealerMessagesHash
734        let dealer_address = Address::new([0u8; 32]);
735        let messages_hash: [u8; 32] = [42u8; 32];
736        let dkg_message = DealerMessagesHash {
737            dealer_address,
738            messages_hash: messages_hash.into(),
739        };
740
741        // Sign with committee members to create a valid certificate
742        let mut aggregator = BlsSignatureAggregator::new(&committee, dkg_message.clone());
743        for (i, key) in signing_keys.iter().enumerate() {
744            let addr = Address::new([i as u8; 32]);
745            let sig = key.sign(epoch, addr, &dkg_message);
746            aggregator.add_signature(sig).unwrap();
747        }
748        let signed_message = aggregator.finish().unwrap();
749
750        // Convert to on-chain format
751        let onchain_cert = DealerSubmissionV1 {
752            message: DealerMessagesHashV1 {
753                dealer_address,
754                messages_hash: messages_hash.to_vec(),
755            },
756            signature: MoveCommitteeSignature {
757                epoch,
758                signature: signed_message.signature_bytes().to_vec(),
759                signers_bitmap: signed_message.signers_bitmap_bytes().to_vec(),
760            },
761        };
762
763        // Parse back using from_onchain_cert
764        let result = DealerMessagesHash::from_onchain_cert(&onchain_cert, epoch);
765        assert!(
766            result.is_ok(),
767            "Should parse valid certificate: {:?}",
768            result.err()
769        );
770
771        let parsed = result.unwrap();
772        assert_eq!(parsed.message().dealer_address, dealer_address);
773        assert_eq!(
774            <MessageHash as AsRef<[u8; 32]>>::as_ref(&parsed.message().messages_hash),
775            &messages_hash
776        );
777    }
