25#[tonic::async_trait]
26impl MpcService for HttpService {
27    #[tracing::instrument(skip(self, request))]
28    async fn send_messages(
29        &self,
30        request: tonic::Request<SendMessagesRequest>,
31    ) -> Result<tonic::Response<SendMessagesResponse>, Status> {
32        let sender = authenticate_caller(&request)?;
33        let external_request = request.into_inner();
34        let internal_request = types::SendMessagesRequest::try_from(&external_request)
35            .map_err(|e| Status::invalid_argument(e.to_string()))?;
36        let label = match &internal_request.messages {
37            types::Messages::Dkg(_) => crate::metrics::MPC_LABEL_DKG,
38            types::Messages::Rotation(_) => crate::metrics::MPC_LABEL_KEY_ROTATION,
39            types::Messages::NonceGeneration(_) => crate::metrics::MPC_LABEL_NONCE_GENERATION,
40        };
41        let mpc_manager = self.mpc_manager()?;
42        let _timer = self
43            .metrics()
44            .mpc_rpc_handler_process_duration_seconds
45            .with_label_values(&[label])
46            .start_timer();
47        let response = spawn_blocking(move || -> Result<_, Status> {
48            let mut mgr = mpc_manager.write().unwrap();
49            validate_epoch(mgr.mpc_config.epoch, external_request.epoch)?;
50            mgr.handle_send_messages_request(sender, &internal_request)
51                .map_err(|e| {
52                    if matches!(&e, MpcError::NotReady(_)) {
53                        tracing::info!("send_messages from {sender:?}: {e}");
54                    } else {
55                        tracing::warn!("send_messages from {sender:?} failed: {e}");
56                    }
57                    mpc_error_to_status(e)
58                })
59        })
60        .await?;
61        drop(_timer);
62        Ok(tonic::Response::new(SendMessagesResponse::from(&response)))
63    }

65    #[tracing::instrument(skip(self, request))]
66    async fn retrieve_messages(
67        &self,
68        request: tonic::Request<RetrieveMessagesRequest>,
69    ) -> Result<tonic::Response<RetrieveMessagesResponse>, Status> {
70        authenticate_caller(&request)?;
71        let external_request = request.into_inner();
72        let internal_request = types::RetrieveMessagesRequest::try_from(&external_request)
73            .map_err(|e| Status::invalid_argument(e.to_string()))?;
74        let response = {
75            let mpc_manager = self.mpc_manager()?;
76            let mgr = mpc_manager.read().unwrap();
77            validate_epoch_current_or_source(
78                mgr.mpc_config.epoch,
79                mgr.source_epoch,
80                internal_request.epoch,
81            )?;
82            mgr.handle_retrieve_messages_request(&internal_request)
83                .map_err(|e| {
84                    match &e {
85                        MpcError::NotFound(_) => {
86                            tracing::debug!("retrieve_messages: {e}");
87                        }
88                        _ => {
89                            tracing::warn!("retrieve_messages failed: {e}");
90                        }
91                    }
92                    mpc_error_to_status(e)
93                })?
94        };
95        Ok(tonic::Response::new(RetrieveMessagesResponse::from(
96            &response,
97        )))
98    }

121        })
122        .await?;
123        Ok(tonic::Response::new(ComplainResponse::from(&response)))

150    async fn get_reconfig_completion_signature(
151        &self,
152        request: tonic::Request<GetReconfigCompletionSignatureRequest>,
153    ) -> Result<tonic::Response<GetReconfigCompletionSignatureResponse>, Status> {
154        authenticate_caller(&request)?;
155        let external_request = request.into_inner();
156        let epoch = external_request
157            .epoch
158            .ok_or_else(|| Status::invalid_argument("epoch: missing required field"))?;
159        let signature = self.get_reconfig_signature(epoch).map(Into::into);
160        Ok(tonic::Response::new(
161            GetReconfigCompletionSignatureResponse { signature },
162        ))
163    }

165    #[tracing::instrument(skip(self, request))]
166    async fn get_partial_signatures(
167        &self,
168        request: tonic::Request<GetPartialSignaturesRequest>,
169    ) -> Result<tonic::Response<GetPartialSignaturesResponse>, Status> {
170        authenticate_caller(&request)?;
171        let external_request = request.into_inner();
172        let internal_request = types::GetPartialSignaturesRequest::try_from(&external_request)
173            .map_err(|e| Status::invalid_argument(e.to_string()))?;
174        let request_epoch = external_request
175            .epoch
176            .ok_or_else(|| Status::invalid_argument("epoch: missing required field"))?;
177        let response = {
178            let signing_manager = self.signing_manager_for(request_epoch)?;
179            signing_manager
180                .handle_get_partial_signatures_request(&internal_request)
181                .map_err(|e| {
182                    match &e {
183                        SigningError::NotFound(_) => {
184                            tracing::debug!("get_partial_signatures: {e}");
185                        }
186                        _ => {
187                            tracing::warn!("get_partial_signatures failed: {e}");
188                        }
189                    }
190                    signing_error_to_status(e)
191                })?
192        };
193        Ok(tonic::Response::new(GetPartialSignaturesResponse::from(
194            &response,
195        )))
196    }

199fn authenticate_caller<T>(request: &tonic::Request<T>) -> Result<Address, Status> {
200    request
201        .extensions()
202        .get::<Address>()
203        .copied()
204        .ok_or_else(|| Status::permission_denied("unknown validator"))
205}
206
207fn validate_epoch(expected: u64, request_epoch: Option<u64>) -> Result<(), Status> {
208    let epoch =
209        request_epoch.ok_or_else(|| Status::invalid_argument("epoch: missing required field"))?;
210    if epoch != expected {
211        return Err(Status::failed_precondition(format!(
212            "epoch mismatch: expected {expected}, got {epoch}"
213        )));
214    }
215    Ok(())
216}
217
218fn validate_epoch_current_or_source(
219    current_epoch: u64,
220    source_epoch: u64,
221    request_epoch: u64,
222) -> Result<(), Status> {
223    if request_epoch != current_epoch && request_epoch != source_epoch {
224        return Err(Status::failed_precondition(format!(
225            "epoch mismatch: expected {current_epoch} or {source_epoch}, got {request_epoch}"
226        )));
227    }
228    Ok(())
229}
230
231fn signing_error_to_status(err: SigningError) -> Status {
232    match &err {
233        SigningError::InvalidMessage { .. } => Status::invalid_argument(err.to_string()),
234        SigningError::NotFound(_) => Status::not_found(err.to_string()),
235        SigningError::CryptoError(_) => Status::internal(err.to_string()),
236        SigningError::Timeout { .. } => Status::deadline_exceeded(err.to_string()),
237        SigningError::TooManyInvalidSignatures { .. } => {
238            Status::failed_precondition(err.to_string())
239        }
240        SigningError::PoolExhausted => Status::resource_exhausted(err.to_string()),
241    }
242}
243
244fn mpc_error_to_status(err: MpcError) -> Status {
245    use types::MpcError::*;
246    match &err {
247        InvalidThreshold(_) | InvalidMessage { .. } | InvalidCertificate(_) => {
248            Status::invalid_argument(err.to_string())
249        }
250        Timeout { .. } => Status::deadline_exceeded(err.to_string()),
251        NotEnoughParticipants { .. }
252        | NotEnoughApprovals { .. }
253        | InvalidConfig(_)
254        | NotReady(_) => Status::failed_precondition(err.to_string()),
255        NotFound(_) => Status::not_found(err.to_string()),
256        _ => Status::internal(err.to_string()),
257    }
