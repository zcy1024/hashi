24pub struct RpcP2PChannel {
25    onchain_state: OnchainState,
26    epoch: u64,
27    protocol_label: &'static str,
28}
29
30impl RpcP2PChannel {
31    pub fn new(onchain_state: OnchainState, epoch: u64, protocol_label: &'static str) -> Self {
32        Self {
33            onchain_state,
34            epoch,
35            protocol_label,
36        }
37    }
38
39    fn get_client(&self, address: &Address) -> ChannelResult<Client> {
40        self.onchain_state
41            .state()
42            .hashi()
43            .committees
44            .client(address)
45            .ok_or(ChannelError::ClientNotFound(*address))
46    }
47
48    /// Wrap a protobuf message in a `tonic::Request` tagged with the MPC
49    /// protocol label, so the server-side metrics layer can attribute
50    /// traffic to the originating protocol.
51    fn build_request<T>(&self, message: T) -> tonic::Request<T> {
52        let mut req = tonic::Request::new(message);
53        req.metadata_mut().insert(
54            MPC_PROTOCOL_METADATA_KEY,
55            MetadataValue::from_static(self.protocol_label),
56        );
57        req
58    }
59}
60
61#[async_trait]
62impl P2PChannel for RpcP2PChannel {
63    async fn send_messages(
64        &self,
65        recipient: &Address,
66        request: &SendMessagesRequest,
67    ) -> ChannelResult<SendMessagesResponse> {
68        let client = self.get_client(recipient)?;
69        let proto_request = self.build_request(request.to_proto(self.epoch));
70        let response = client
71            .mpc_service_client()
72            .send_messages(proto_request)
73            .await
74            .map_err(|e| ChannelError::RequestFailed(e.to_string()))?;
75        SendMessagesResponse::try_from(response.get_ref())
76            .map_err(|e| ChannelError::RequestFailed(e.to_string()))
77    }
78
