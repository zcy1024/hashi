24const POLL_INTERVAL: Duration = Duration::from_millis(500);
25const TX_CONFIRMATION_TIMEOUT: Duration = Duration::from_secs(30);

27#[derive(Debug, Error)]
28pub enum TobError {
29    #[error("Sui RPC error: {0}")]
30    RpcError(String),
31
32    #[error("Invalid certificate data: {0}")]
33    InvalidCertificate(String),
34
35    #[error("Invalid state: {0}")]
36    InvalidState(String),
37}
38
39impl From<TobError> for ChannelError {
40    fn from(e: TobError) -> Self {
41        match e {
42            TobError::RpcError(msg) => ChannelError::RequestFailed(msg),
43            _ => ChannelError::Other(e.to_string()),
44        }
45    }
46}

48pub struct SuiTobChannel {
49    hashi_ids: HashiIds,
50    onchain_state: OnchainState,
51    epoch: u64,
52    batch_index: Option<u32>,
53    signer: Ed25519PrivateKey,
54    /// Dealers we've already returned certificates for
55    seen_dealers: HashSet<Address>,
56    /// Cached certificates not yet returned
57    pending_certs: VecDeque<CertificateV1>,
58}
59
60impl SuiTobChannel {
61    pub fn new(
62        hashi_ids: HashiIds,
63        onchain_state: OnchainState,
64        epoch: u64,
65        batch_index: Option<u32>,
66        signer: Ed25519PrivateKey,
67    ) -> Self {
68        Self {
69            hashi_ids,
70            onchain_state,
71            epoch,
72            batch_index,
73            signer,
74            seen_dealers: HashSet::new(),
75            pending_certs: VecDeque::new(),
76        }
77    }
78
79    fn create_executor(&self) -> SuiTxExecutor {
80        SuiTxExecutor::new(
81            self.onchain_state.client(),
82            self.signer.clone(),
83            self.hashi_ids,
84        )
85        .with_timeout(TX_CONFIRMATION_TIMEOUT)
86    }
87}

89pub async fn fetch_certificates(
90    onchain_state: &OnchainState,
91    epoch: u64,
92    batch_index: Option<u32>,
93) -> Result<Vec<(Address, CertificateV1)>, TobError> {
94    let Some((protocol_type, raw_certs)) = onchain_state
95        .fetch_certs(epoch, batch_index)
96        .await
97        .map_err(|e| TobError::RpcError(e.to_string()))?
98    else {
99        return Ok(vec![]);
100    };
101    let mut certificates = Vec::with_capacity(raw_certs.len());
102    for (dealer, cert) in raw_certs {
103        let inner_cert = DealerMessagesHash::from_onchain_cert(&cert, epoch)
104            .map_err(|e| TobError::InvalidCertificate(e.to_string()))?;
105        let cert = CertificateV1::new(protocol_type, batch_index, inner_cert);
106        certificates.push((dealer, cert));
107    }
108    Ok(certificates)
109}

111#[async_trait]
112impl OrderedBroadcastChannel<CertificateV1> for SuiTobChannel {
113    async fn publish(&self, cert: CertificateV1) -> ChannelResult<()> {
114        let dealer = cert.dealer_address();
115        let existing = fetch_certificates(&self.onchain_state, self.epoch, self.batch_index)
116            .await
117            .map_err(ChannelError::from)?;
118        if existing.iter().any(|(d, _)| *d == dealer) {
119            return Ok(());
120        }
121
122        let mut executor = self.create_executor();
123        executor
124            .execute_submit_certificate(&cert)
125            .await
126            .map_err(|e| ChannelError::Other(e.to_string()))
127    }
128
129    async fn receive(&mut self) -> ChannelResult<CertificateV1> {
130        loop {
131            if let Some(cert) = self.pending_certs.pop_front() {
132                return Ok(cert);
133            }
134            // TODO: Optimize by checking table size first to avoid redundant fetches.
135            let all_certs = fetch_certificates(&self.onchain_state, self.epoch, self.batch_index)
136                .await
137                .map_err(ChannelError::from)?;
138            for (dealer, cert) in all_certs {
139                if !self.seen_dealers.contains(&dealer) {
140                    self.seen_dealers.insert(dealer);
141                    self.pending_certs.push_back(cert);
142                }
143            }
144            if self.pending_certs.is_empty() {
145                tokio::time::sleep(POLL_INTERVAL).await;
146            }
147        }
148    }
149
150    async fn certified_dealers(&mut self) -> Vec<Address> {
151        if let Ok(all_certs) =
152            fetch_certificates(&self.onchain_state, self.epoch, self.batch_index).await
153        {
154            for (dealer, cert) in all_certs {
155                if !self.seen_dealers.contains(&dealer) {
156                    self.seen_dealers.insert(dealer);
157                    self.pending_certs.push_back(cert);
158                }
159            }
160        }
161        self.seen_dealers.iter().copied().collect()
162    }
163}
