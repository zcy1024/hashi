4//! Retry and timeout utilities
5
6use super::ChannelError;
7use super::ChannelResult;
8use backon::ExponentialBuilder;
9use backon::Retryable;
10use futures::future::join_all;
11use std::future::Future;
12use std::time::Duration;
13use sui_sdk_types::Address;
14
15// TODO: Use lower thresholds for unit tests.
16pub const RETRY_MIN_DELAY: Duration = Duration::from_millis(100);
17pub const RETRY_MAX_DELAY: Duration = Duration::from_secs(2);
18pub const MAX_RETRIES: usize = 10;
19pub const CALL_TIMEOUT: Duration = Duration::from_secs(30);
20
21pub async fn with_timeout_and_retry<T, F, Fut>(mut f: F) -> ChannelResult<T>
22where
23    F: FnMut() -> Fut,
24    Fut: Future<Output = ChannelResult<T>>,
25{
26    (move || with_timeout(f())).retry(retry_policy()).await
27}
28
29fn retry_policy() -> ExponentialBuilder {
30    ExponentialBuilder::default()
31        .with_min_delay(RETRY_MIN_DELAY)
32        .with_max_delay(RETRY_MAX_DELAY)
33        .with_max_times(MAX_RETRIES)
34}
35
36async fn with_timeout<T>(fut: impl Future<Output = ChannelResult<T>>) -> ChannelResult<T> {
37    match tokio::time::timeout(CALL_TIMEOUT, fut).await {
38        Ok(result) => result,
39        Err(_) => Err(ChannelError::Timeout),
40    }
41}
42
43pub async fn send_to_many<I, Req, Resp, F, Fut>(
44    recipients: I,
45    request: Req,
46    send: F,
47) -> Vec<(Address, ChannelResult<Resp>)>
48where
49    I: IntoIterator<Item = Address>,
50    Req: Clone + Send + Sync,
51    Resp: Send,
52    F: Fn(Address, Req) -> Fut + Clone + Send + Sync,
53    Fut: Future<Output = ChannelResult<Resp>> + Send,
54{
55    join_all(recipients.into_iter().map(|addr| {
56        let req = request.clone();
57        let send = send.clone();
58        async move {
59            let result = with_timeout_and_retry(|| send(addr, req.clone())).await;
60            (addr, result)
61        }
62    }))
63    .await
64}
