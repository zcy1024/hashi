28    //
29    // key: big endian u64 for the epoch the key is used for

40    //

43    rotation_messages: Keyspace,
44
45    // Column Family used to store nonce messages for presignature generation.
46    //
47    // key: (big endian u64 epoch) + (big endian u32 batch_index) + (32-byte validator address)
48    // value: BCS-serialized batch_avss::Message
49    nonce_messages: Keyspace,
50}

52const ENCRYPTION_KEYS_CF_NAME: &str = "encryption_keys";
53const DEALER_MESSAGES_CF_NAME: &str = "dealer_messages";

57impl Database {
58    pub fn open(path: &Path) -> anyhow::Result<Self> {
59        // Preserve the underlying `fjall::Error` as the source so callers can
60        // `downcast_ref::<fjall::Error>()` and match on variants like
61        // `fjall::Error::Locked`. Using `anyhow!("...: {e}")` with a format
62        // string would stringify the error and drop the source chain.
63        let db = fjall::Database::builder(path).open().map_err(|e| {
64            anyhow::Error::new(e).context(format!("failed to open database at {}", path.display()))
65        })?;
66        let encryption_keys =
67            db.keyspace(ENCRYPTION_KEYS_CF_NAME, KeyspaceCreateOptions::default)?;
68        let dealer_messages =
69            db.keyspace(DEALER_MESSAGES_CF_NAME, KeyspaceCreateOptions::default)?;
70        let rotation_messages =
71            db.keyspace(ROTATION_MESSAGES_CF_NAME, KeyspaceCreateOptions::default)?;
72        let nonce_messages = db.keyspace(NONCE_MESSAGES_CF_NAME, KeyspaceCreateOptions::default)?;
73        Ok(Self {
74            db,
75            encryption_keys,
76            dealer_messages,
77            rotation_messages,
78            nonce_messages,
79        })
80    }

82    /// Returns all keyspaces paired with their names. When adding a new
83    /// keyspace, add it here so that it is included in snapshots and backups.
84    ///
85    /// Returns a `Vec` rather than a fixed-size array so callers don't have
86    /// to know the keyspace count, and so adding a new keyspace is a one-line
87    /// change here. Only invoked on the snapshot path, so the per-call
88    /// allocation is irrelevant.
89    fn all_keyspaces(&self) -> Vec<(&str, &Keyspace)> {
90        vec![
91            (ENCRYPTION_KEYS_CF_NAME, &self.encryption_keys),
92            (DEALER_MESSAGES_CF_NAME, &self.dealer_messages),
93            (ROTATION_MESSAGES_CF_NAME, &self.rotation_messages),
94            (NONCE_MESSAGES_CF_NAME, &self.nonce_messages),
95        ]
96    }

98    /// Create a consistent snapshot of this database and write it as a new fjall
99    /// database at `dest`.
100    ///
101    /// Uses fjall's MVCC snapshot to get a point-in-time view of all keyspaces
102    /// while the source database remains open for writes. The destination is a
103    /// fully self-contained fjall database that can later be opened with
104    /// [`Database::open`].
105    ///
106    /// `dest` must not already exist. The function refuses to write into an
107    /// existing directory because fjall would otherwise happily merge the
108    /// snapshot rows into whatever's already there, producing a hybrid DB that
109    /// looks valid but isn't a true point-in-time snapshot.
110    pub fn snapshot_to_path(&self, dest: &Path) -> anyhow::Result<()> {
111        // Bound batch memory by serialized payload size rather than entry
112        // count: keyspaces in this DB hold values ranging from 32-byte
113        // scalars to multi-MB BCS-encoded MPC messages, so a fixed entry
114        // count would spike memory unpredictably for the large-value keyspaces.
115        //
116        // 16 MiB is a comfortable middle ground: small enough that the
117        // worst-case batch fits easily in RAM on any host that can run a
118        // validator, large enough to amortise per-batch journal commit cost.
119        const BATCH_BYTE_BUDGET: usize = 16 * 1024 * 1024;
120
121        if dest
122            .try_exists()
123            .with_context(|| format!("failed to stat snapshot destination {}", dest.display()))?
124        {
125            anyhow::bail!(
126                "snapshot destination {} already exists; refusing to merge into an existing directory",
127                dest.display()
128            );
129        }
130
131        let snapshot = self.db.snapshot();
132
133        let dest_db = fjall::Database::builder(dest).open().map_err(|e| {
134            anyhow::Error::new(e).context(format!(
135                "failed to open destination database at {}",
136                dest.display()
137            ))
138        })?;
139
140        for (name, source_ks) in self.all_keyspaces() {
141            let dest_ks = dest_db.keyspace(name, KeyspaceCreateOptions::default)?;
142            let mut batch = dest_db.batch();
143            let mut batch_bytes = 0usize;
144
145            for guard in snapshot.iter(source_ks) {
146                let (key, value) = guard.into_inner()?;
147                let entry_bytes = key.len() + value.len();
148                batch.insert(&dest_ks, &*key, &*value);
149                batch_bytes += entry_bytes;
150
151                if batch_bytes >= BATCH_BYTE_BUDGET {
152                    batch.commit()?;
153                    batch = dest_db.batch();
154                    batch_bytes = 0;
155                }
156            }
157
158            if batch_bytes > 0 {
159                batch.commit()?;
160            }
161        }
162
163        // `persist(SyncAll)` fsyncs the destination journal. Freshly inserted
164        // rows live in the active memtable and are recovered from the journal
165        // on next open, so the destination directory is self-contained and
166        // safe to archive after this call returns. We do not fsync the dest
167        // directory entry here: this function's caller immediately tars the
168        // directory in the same process, so a crash before archival just
169        // discards an incomplete backup and loses no committed state.
170        dest_db.persist(fjall::PersistMode::SyncAll)?;
171
172        Ok(())
173    }

175    /// Store encryption key for the given epoch.
176    ///
177    /// No-op if a key already exists for this epoch (idempotent for restart safety).
178    pub fn store_encryption_key(
179        &self,
180        epoch: u64,
181        encryption_key: &EncryptionPrivateKey,
182    ) -> Result<()> {
183        let key = epoch.to_be_bytes();
184        if !self.encryption_keys.contains_key(key)? {
185            let value = bcs::to_bytes(encryption_key).unwrap();
186            self.encryption_keys.insert(key, value)?;
187        }
188        Ok(())
189    }

191    pub fn latest_encryption_key_epoch(&self) -> Result<Option<u64>> {
192        let mut latest: Option<u64> = None;
193        for guard in self.encryption_keys.iter() {
194            let key = guard.key()?;
195            if let Ok(bytes) = <[u8; 8]>::try_from(key.as_ref()) {
196                let epoch = u64::from_be_bytes(bytes);
197                latest = Some(latest.map_or(epoch, |l: u64| l.max(epoch)));
198            }
199        }
200        Ok(latest)
201    }

203    pub fn get_encryption_key(&self, epoch: u64) -> Result<Option<EncryptionPrivateKey>> {
204        let key = epoch.to_be_bytes();
205        let bytes = match self.encryption_keys.get(key) {
206            Ok(Some(bytes)) => bytes,
207            Ok(None) => return Ok(None),
208            Err(e) => return Err(e),
209        };
210        let byte_array = (&*bytes).try_into().map_err(|_| {
211            fjall::Error::Io(std::io::Error::new(
212                std::io::ErrorKind::InvalidData,
213                "invalid point",
214            ))
215        })?;
216        let scalar = RistrettoScalar::from_byte_array(&byte_array).map_err(|_| {
217            fjall::Error::Io(std::io::Error::new(
218                std::io::ErrorKind::InvalidData,
219                "invalid point",
220            ))
221        })?;
222        Ok(Some(EncryptionPrivateKey::from(scalar)))
223    }

382    /// Prune all MPC keyspaces, deleting entries with `epoch < cutoff_epoch`.
383    pub fn prune_messages_below(&self, cutoff_epoch: u64) -> Result<()> {
384        prune_keyspace(&self.encryption_keys, cutoff_epoch)?;
385        prune_keyspace(&self.dealer_messages, cutoff_epoch)?;
386        prune_keyspace(&self.rotation_messages, cutoff_epoch)?;
387        prune_keyspace(&self.nonce_messages, cutoff_epoch)?;
388        Ok(())
389    }

392/// List all `(Address, T)` pairs from a keyspace where keys match the given prefix.
393/// Keys are expected to end with a 32-byte address suffix.
394fn list_messages_by_prefix<T: DeserializeOwned>(
395    keyspace: &Keyspace,
396    prefix: &[u8],
397) -> Result<Vec<(Address, T)>> {
398    let addr_len = 32;
399    let mut results = Vec::new();
400    for guard in keyspace.prefix(prefix) {
401        let (key, value) = guard.into_inner()?;
402        let addr_start = key.len().checked_sub(addr_len).ok_or_else(|| {
403            fjall::Error::Io(std::io::Error::new(
404                std::io::ErrorKind::InvalidData,
405                "key too short for address",
406            ))
407        })?;
408        let address_bytes: [u8; 32] = key[addr_start..].try_into().map_err(|_| {
409            fjall::Error::Io(std::io::Error::new(
410                std::io::ErrorKind::InvalidData,
411                "invalid key length",
412            ))
413        })?;
414        let address = Address::new(address_bytes);
415        let message: T = bcs::from_bytes(&value).map_err(|_| {
416            fjall::Error::Io(std::io::Error::new(
417                std::io::ErrorKind::InvalidData,
418                "invalid message data",
419            ))
420        })?;
421        results.push((address, message));
422    }
423    Ok(results)
424}

426/// Delete entries from `keyspace` whose leading big-endian u64 epoch is `< cutoff_epoch`.
427fn prune_keyspace(keyspace: &Keyspace, cutoff_epoch: u64) -> Result<()> {
428    let keys_to_delete: Vec<_> = keyspace
429        .iter()
430        .filter_map(|guard| {
431            let key = guard.key().ok()?;
432            let epoch_bytes: [u8; 8] = key.as_ref().get(..8)?.try_into().ok()?;
433            let epoch = u64::from_be_bytes(epoch_bytes);
434            (epoch < cutoff_epoch).then(|| key.to_vec())
435        })
436        .collect();
437    for key in keys_to_delete {
438        keyspace.remove(key)?;
439    }
440    Ok(())
441}
