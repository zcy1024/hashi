635                &["protocol"],
636                LATENCY_SEC_BUCKETS.to_vec(),
637                registry,
638            )
639            .unwrap(),
640        }
641    }

643    pub fn update_onchain_state(&self, state: &crate::onchain::OnchainState) {
644        self.latest_checkpoint_height
645            .set(state.latest_checkpoint_height() as i64);
646        self.latest_checkpoint_timestamp_ms
647            .set(state.latest_checkpoint_timestamp_ms() as i64);
648        self.sui_epoch.set(state.latest_checkpoint_epoch() as i64);
649
650        let guard = state.state();
651        let hashi = guard.hashi();
652
653        self.epoch.set(hashi.committees.epoch() as i64);
654        self.reconfig_in_progress
655            .set(if hashi.committees.pending_epoch_change().is_some() {
656                1
657            } else {
658                0
659            });
660        self.paused.set(if hashi.config.paused() { 1 } else { 0 });
661        self.deposit_queue_size
662            .set(hashi.deposit_queue.requests().len() as i64);
663        let (requested, approved) = hashi
664            .withdrawal_queue
665            .requests()
666            .values()
667            .partition::<Vec<_>, _>(|r| r.status.is_requested());
668        let (signed, pending): (Vec<_>, Vec<_>) = hashi
669            .withdrawal_queue
670            .withdrawal_txns()
671            .values()
672            .partition(|w| w.signatures.is_some());
673        self.withdrawal_queue_size
674            .with_label_values(&["requested"])
675            .set(requested.len() as i64);
676        self.withdrawal_queue_value
677            .with_label_values(&["requested", "BTC"])
678            .set(requested.iter().map(|r| r.btc_amount).sum::<u64>() as i64);
679        self.withdrawal_queue_size
680            .with_label_values(&["approved"])
681            .set(approved.len() as i64);
682        self.withdrawal_queue_value
683            .with_label_values(&["approved", "BTC"])
684            .set(approved.iter().map(|r| r.btc_amount).sum::<u64>() as i64);
685        self.withdrawal_queue_size
686            .with_label_values(&["pending"])
687            .set(pending.len() as i64);
688        self.withdrawal_queue_value
689            .with_label_values(&["pending", "BTC"])
690            .set(
691                pending
692                    .iter()
693                    .flat_map(|w| &w.withdrawal_outputs)
694                    .map(|o| o.amount)
695                    .sum::<u64>() as i64,
696            );
697        self.withdrawal_queue_size
698            .with_label_values(&["signed"])
699            .set(signed.len() as i64);
700        self.withdrawal_queue_value
701            .with_label_values(&["signed", "BTC"])
702            .set(
703                signed
704                    .iter()
705                    .flat_map(|w| &w.withdrawal_outputs)
706                    .map(|o| o.amount)
707                    .sum::<u64>() as i64,
708            );
709        // Track three views of utxo_records:
710        // - available:         all selectable UTXOs (locked_by = None), whether
711        //                      confirmed or not; this is the coin-selection pool
712        // - unconfirmed_change: subset of available whose producing withdrawal
713        //                      has not yet confirmed on Bitcoin (produced_by =
714        //                      Some); useful for gauging mempool chain depth
715        // - locked:            committed to a pending withdrawal, awaiting
716        //                      Bitcoin confirmation (locked_by = Some)
717        let mut available_count = 0i64;
718        let mut unconfirmed_change_count = 0i64;
719        let mut locked_count = 0i64;
720        let mut available_value = 0u64;
721        let mut unconfirmed_change_value = 0u64;
722        let mut locked_value = 0u64;
723        for record in hashi.utxo_pool.utxo_records().values() {
724            if record.locked_by.is_some() {
725                locked_count += 1;
726                locked_value += record.utxo.amount;
727            } else {
728                available_count += 1;
729                available_value += record.utxo.amount;
730                if record.produced_by.is_some() {
731                    unconfirmed_change_count += 1;
732                    unconfirmed_change_value += record.utxo.amount;
733                }
734            }
735        }
736        self.utxo_pool_size
737            .with_label_values(&["available"])
738            .set(available_count);
739        self.utxo_pool_size
740            .with_label_values(&["unconfirmed_change"])
741            .set(unconfirmed_change_count);
742        self.utxo_pool_size
743            .with_label_values(&["locked"])
744            .set(locked_count);
745        self.utxo_pool_value
746            .with_label_values(&["available"])
747            .set(available_value as i64);
748        self.utxo_pool_value
749            .with_label_values(&["unconfirmed_change"])
750            .set(unconfirmed_change_value as i64);
751        self.utxo_pool_value
752            .with_label_values(&["locked"])
753            .set(locked_value as i64);
754        {
755            use crate::onchain::types::ProposalType;
756            let mut counts = std::collections::HashMap::<&str, i64>::new();
757            for proposal in hashi.proposals.proposals().values() {
758                *counts.entry(proposal.proposal_type.as_str()).or_default() += 1;
759            }
760            for label in ProposalType::all_labels() {
761                self.proposals
762                    .with_label_values(&[label])
763                    .set(*counts.get(label).unwrap_or(&0));
764            }
765        }
766        self.num_consumed_presigs
767            .set(hashi.num_consumed_presigs as i64);
768        for (type_tag, cap) in &hashi.treasury.treasury_caps {
769            if let sui_sdk_types::TypeTag::Struct(struct_tag) = type_tag {
770                self.treasury_supply
771                    .with_label_values(&[struct_tag.name().as_str()])
772                    .set(cap.supply as i64);
773            }
774        }
775
776        self.package_version_enabled.reset();
777        for version in &hashi.config.enabled_versions {
778            let version_str = version.to_string();
779            let package_id_str = guard
780                .package_versions()
781                .get(version)
782                .map(|addr| addr.to_string())
783                .unwrap_or_else(|| "unknown".to_string());
784            self.package_version_enabled
785                .with_label_values(&[&version_str, &package_id_str])
786                .set(1);
787        }
788    }
789}

791/// Create a metric that measures the uptime from when this metric was constructed.
792/// The metric is labeled with:
793/// - 'version': binary version, generally be of the format: 'semver-gitrevision'
794/// - 'chain_identifier': the identifier of the network which this process is part of
795pub fn uptime_metric(
796    version: &'static str,
797    sui_chain_id: &str,
798    bitcoin_chain_id: &str,
799    package_id: &str,
800    hashi_object_id: &str,
801) -> Box<dyn prometheus::core::Collector> {
802    let opts = prometheus::opts!("uptime", "uptime of the node service in seconds")
803        .variable_label("version")
804        .variable_label("sui_chain_id")
805        .variable_label("bitcoin_chain_id")
806        .variable_label("package_id")
807        .variable_label("hashi_object_id");
808
809    let start_time = std::time::Instant::now();
810    let uptime = move || start_time.elapsed().as_secs();
811    let metric = prometheus_closure_metric::ClosureMetric::new(
812        opts,
813        prometheus_closure_metric::ValueType::Counter,
814        uptime,
815        &[
816            version,
817            sui_chain_id,
818            bitcoin_chain_id,
819            package_id,
820            hashi_object_id,
821        ],
822    )
823    .unwrap();
824
825    Box::new(metric)
826}

828const METRICS_ROUTE: &str = "/metrics";
829
830// Creates a new http server that has as a sole purpose to expose
831// an endpoint that prometheus agent can use to poll for the metrics.
832pub fn start_prometheus_server(
833    addr: std::net::SocketAddr,
834    registry: prometheus::Registry,
835) -> sui_http::ServerHandle {
836    let router = axum::Router::new()
837        .route(METRICS_ROUTE, axum::routing::get(metrics))
838        .with_state(registry);
839
840    sui_http::Builder::new().serve(addr, router).unwrap()
841}
842
843async fn metrics(
844    axum::extract::State(registry): axum::extract::State<prometheus::Registry>,
845) -> (http::StatusCode, String) {
846    let metrics_families = registry.gather();
847    match prometheus::TextEncoder.encode_to_string(&metrics_families) {
848        Ok(metrics) => (http::StatusCode::OK, metrics),
849        Err(error) => (
850            http::StatusCode::INTERNAL_SERVER_ERROR,
851            format!("unable to encode metrics: {error}"),
852        ),
853    }
