5//!
6//! Provides a reusable [`build_package`] + [`publish_and_init`] workflow that can be called

32/// Well-known Sui CoinRegistry shared object address (0xc).
33const COIN_REGISTRY_OBJECT_ID: Address = Address::from_static("0xc");
34
35/// Parameters for building a Move package.
36pub struct BuildParams<'a> {
37    /// Path to the `sui` CLI binary.
38    pub sui_binary: &'a Path,
39    /// Path to the Move package directory.
40    pub package_path: &'a Path,
41    /// Optional path to a `sui client.yaml` for dependency resolution.
42    pub client_config: Option<&'a Path>,
43    /// Network environment for the build (`"testnet"`, `"mainnet"`, etc.).
44    pub environment: Option<&'a str>,
45}
46
47/// JSON output produced by `sui move build --dump-bytecode-as-base64`.
48#[derive(serde::Deserialize)]
49struct MoveBuildOutput {
50    modules: Vec<String>,
51    dependencies: Vec<Address>,
52    digest: Vec<u8>,
53}
54
55/// Build a Move package and return the compiled [`sui_sdk_types::Publish`] payload.
56///
57/// Shells out to `sui move build --dump-bytecode-as-base64`, parses the JSON
58/// output, and decodes the base64-encoded module bytecodes.
59pub fn build_package(params: &BuildParams<'_>) -> Result<sui_sdk_types::Publish> {
60    let mut cmd = Command::new(params.sui_binary);
61    cmd.arg("move");
62
63    if let Some(config) = params.client_config {
64        cmd.arg("--client.config").arg(config);
65    }
66
67    cmd.arg("-p").arg(params.package_path).arg("build");
68
69    if let Some(env) = params.environment {
70        cmd.args(["-e", env]);
71    }
72
73    cmd.arg("--dump-bytecode-as-base64");
74
75    let output = cmd.output()?;
76
77    if !output.status.success() {
78        return Err(anyhow::anyhow!(
79            "sui move build failed
80stdout: {}
81stderr: {}",
82            output.stdout.escape_ascii(),
83            output.stderr.escape_ascii()
84        ));
85    }
86
87    let build_output: MoveBuildOutput = serde_json::from_slice(&output.stdout)?;
88    let modules = build_output
89        .modules
90        .into_iter()
91        .map(|b64| <base64ct::Base64 as base64ct::Encoding>::decode_vec(&b64))
92        .collect::<Result<Vec<_>, _>>()?;
93    let _digest = sui_sdk_types::Digest::from_bytes(build_output.digest)?;
94
95    Ok(sui_sdk_types::Publish {
96        modules,
97        dependencies: build_output.dependencies,
98    })
99}

101/// Publish the compiled package and run post-publish initialization.
102///
103/// Executes two transactions:
104///
105/// 1. **Publish** – publishes the modules, transfers the `UpgradeCap` to the sender.
106/// 2. **Init** – calls `hashi::finish_publish` to register BTC, the upgrade cap,
107///    and set the bitcoin chain ID.
108///
109/// Returns the [`HashiIds`] (package ID + Hashi shared-object ID) on success.
110pub async fn publish_and_init(
111    client: &mut Client,
112    signer: &Ed25519PrivateKey,
113    publish: sui_sdk_types::Publish,
114    bitcoin_chain_id: &str,
115) -> Result<HashiIds> {
116    let sender = signer.public_key().derive_address();
117
118    // Validate and convert bitcoin_chain_id to a Move-compatible address.
119    anyhow::ensure!(
120        network_from_chain_id(bitcoin_chain_id).is_some(),
121        "unrecognized bitcoin chain id: {bitcoin_chain_id}"
122    );
123    let block_hash = BlockHash::from_str(bitcoin_chain_id)?;
124    let bitcoin_chain_id_addr = Address::new(*block_hash.as_byte_array());
125
126    // ── Transaction 1: Publish ──────────────────────────────────────────
127    let mut builder = TransactionBuilder::new();
128    builder.set_sender(sender);
129
130    let upgrade_cap = builder.publish(publish.modules, publish.dependencies);
131    let sender_arg = builder.pure(&sender);
132    builder.transfer_objects(vec![upgrade_cap], sender_arg);
133
134    let transaction = builder.build(client).await?;
135    let signature = signer.sign_transaction(&transaction)?;
136
137    let response = client
138        .execute_transaction_and_wait_for_checkpoint(
139            ExecuteTransactionRequest::new(transaction.into())
140                .with_signatures(vec![signature.into()])
141                .with_read_mask(FieldMask::from_str("*")),
142            std::time::Duration::from_secs(30),
143        )
144        .await?
145        .into_inner();
146
147    anyhow::ensure!(
148        response.transaction().effects().status().success(),
149        "publish transaction failed"
150    );
151
152    // Extract IDs from effects ────────────────────────────────────────────
153
154    let package_id = response
155        .transaction()
156        .effects()
157        .changed_objects()
158        .iter()
159        .find(|o| o.object_type() == "package")
160        .ok_or_else(|| anyhow::anyhow!("package not found in publish effects"))?
161        .object_id()
162        .parse::<Address>()?;
163
164    let hashi_type = StructTag::new(
165        package_id,
166        Identifier::from_static("hashi"),
167        Identifier::from_static("Hashi"),
168        vec![],
169    )
170    .to_string();
171
172    let hashi_object_id = response
173        .transaction()
174        .effects()
175        .changed_objects()
176        .iter()
177        .find(|o| o.object_type() == hashi_type)
178        .ok_or_else(|| anyhow::anyhow!("Hashi shared object not found in publish effects"))?
179        .object_id()
180        .parse::<Address>()?;
181
182    let upgrade_cap_type = StructTag::from_str("0x2::package::UpgradeCap")?.to_string();
183    let upgrade_cap_id = response
184        .transaction()
185        .effects()
186        .changed_objects()
187        .iter()
188        .find(|o| o.object_type() == upgrade_cap_type)
189        .ok_or_else(|| anyhow::anyhow!("UpgradeCap not found in publish effects"))?
190        .object_id()
191        .parse::<Address>()?;
192
193    // ── Transaction 2: Init (finish_publish) ─────────────────────────────
194    let mut builder = TransactionBuilder::new();
195    builder.set_sender(sender);
196
197    let hashi_arg = builder.object(
198        ObjectInput::new(hashi_object_id)
199            .as_shared()
200            .with_mutable(true),
201    );
202    let upgrade_cap_arg = builder.object(ObjectInput::new(upgrade_cap_id).as_owned());
203    let bitcoin_chain_id_arg = builder.pure(&bitcoin_chain_id_addr);
204    let coin_registry_arg = builder.object(
205        ObjectInput::new(COIN_REGISTRY_OBJECT_ID)
206            .as_shared()
207            .with_mutable(true),
208    );
209
210    builder.move_call(
211        Function::new(
212            package_id,
213            Identifier::from_static("hashi"),
214            Identifier::from_static("finish_publish"),
215        ),
216        vec![
217            hashi_arg,
218            upgrade_cap_arg,
219            bitcoin_chain_id_arg,
220            coin_registry_arg,
221        ],
222    );
223
224    let transaction = builder.build(client).await?;
225    let signature = signer.sign_transaction(&transaction)?;
226
227    let response = client
228        .execute_transaction_and_wait_for_checkpoint(
229            ExecuteTransactionRequest::new(transaction.into())
230                .with_signatures(vec![signature.into()])
231                .with_read_mask(FieldMask::from_str("*")),
232            std::time::Duration::from_secs(30),
233        )
234        .await?
235        .into_inner();
236
237    anyhow::ensure!(
238        response.transaction().effects().status().success(),
239        "init transaction failed (finish_publish)"
240    );
241
242    Ok(HashiIds {
243        package_id,
244        hashi_object_id,
245    })
