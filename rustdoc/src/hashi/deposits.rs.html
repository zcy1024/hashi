22/// Derive a deposit address from a compressed MPC public key (33-byte ProjectivePoint),
23/// an optional derivation path, and the target Bitcoin network.
24pub fn derive_deposit_address(
25    mpc_key: &ProjectivePoint,
26    derivation_path: Option<&sui_sdk_types::Address>,
27    btc_network: bitcoin::Network,
28) -> anyhow::Result<bitcoin::Address> {
29    let xonly = if let Some(path) = derivation_path {
30        let derived = fastcrypto_tbls::threshold_schnorr::key_derivation::derive_verifying_key(
31            mpc_key,
32            &path.into_inner(),
33        );
34        XOnlyPublicKey::from_slice(&derived.to_byte_array()).context("valid 32-byte x-only key")?
35    } else {
36        let schnorr_pk = SchnorrPublicKey::try_from(mpc_key)
37            .context("Failed to convert MPC key to schnorr key")?;
38        XOnlyPublicKey::from_slice(&schnorr_pk.to_byte_array())
39            .context("Failed to parse x-only key")?
40    };
41
42    Ok(bitcoin_utils::single_key_taproot_script_path_address(
43        &xonly,
44        btc_network,
45    ))
46}

58    #[tracing::instrument(level = "debug", skip_all, fields(deposit_id = %deposit_request.id))]
59    pub async fn validate_deposit_request(
60        &self,
61        deposit_request: &DepositRequest,
62    ) -> Result<(), DepositError> {
63        self.validate_deposit_request_on_sui(deposit_request)?;
64        self.validate_deposit_request_on_bitcoin(deposit_request)
65            .await?;
66        self.screen_deposit(deposit_request).await?;
67        Ok(())
68    }

70    /// Run AML/Sanctions checks for the deposit request.
71    /// If no screener client is configured, checks are skipped.
72    #[tracing::instrument(level = "debug", skip_all, fields(deposit_id = %deposit_request.id))]
73    async fn screen_deposit(&self, deposit_request: &DepositRequest) -> Result<(), DepositError> {
74        let Some(screener) = self.screener_client() else {
75            tracing::debug!("AML checks skipped: no screener configured");
76            return Ok(());
77        };
78
79        // bitcoin
80        let source_tx_hash = deposit_request.utxo.id.txid.to_string();
81        let bitcoin_chain_id = self.config.bitcoin_chain_id().to_string();
82
83        // sui
84        let destination_address = deposit_request.id.to_string();
85        let sui_chain_id = self.config.sui_chain_id().to_string();
86
87        let approved = screener
88            .approve_deposit(
89                &source_tx_hash,
90                &destination_address,
91                &bitcoin_chain_id,
92                &sui_chain_id,
93            )
94            .await
95            .map_err(|e| DepositError::AmlServiceError(anyhow!(e)))?;
96
97        if !approved {
98            return Err(DepositError::AmlRejected(anyhow!(
99                "AML checks failed for source tx {source_tx_hash}, destination {destination_address}, bitcoin chain {bitcoin_chain_id}, sui chain {sui_chain_id}"
100            )));
101        }
102
103        Ok(())
104    }

106    /// Validate that the deposit request exists on Sui
107    #[tracing::instrument(level = "debug", skip_all, fields(deposit_id = %deposit_request.id))]
108    fn validate_deposit_request_on_sui(
109        &self,
110        deposit_request: &DepositRequest,
111    ) -> Result<(), DepositError> {
112        let state = self.onchain_state().state();
113        let deposit_queue = &state.hashi().deposit_queue;
114        match deposit_queue.requests().get(&deposit_request.id) {
115            None => {
116                return Err(DepositError::InvalidOnchainRequest(anyhow!(
117                    "Deposit request not found on Sui"
118                )));
119            }
120            Some(onchain_request) => {
121                if onchain_request != deposit_request {
122                    return Err(DepositError::InvalidOnchainRequest(anyhow!(
123                        "Deposit request fields do not match on-chain state"
124                    )));
125                }
126            }
127        }
128
129        let utxo_pool = &state.hashi().utxo_pool;
130        if utxo_pool
131            .utxo_records()
132            .contains_key(&deposit_request.utxo.id)
133            || utxo_pool
134                .spent_utxos()
135                .contains_key(&deposit_request.utxo.id)
136        {
137            return Err(DepositError::DuplicateOrSpentOnSui(anyhow!(
138                "UTXO {:?} is already active or spent",
139                deposit_request.utxo.id
140            )));
141        }
142
143        Ok(())
144    }
145
146    /// Validate that there is a txout on Bitcoin that matches the deposit request
147    #[tracing::instrument(
148        level = "debug",
149        skip_all,
150        fields(
151            deposit_id = %deposit_request.id,
152            bitcoin_txid = %deposit_request.utxo.id.txid,
153            vout = deposit_request.utxo.id.vout,
154        ),
155    )]
156    async fn validate_deposit_request_on_bitcoin(
157        &self,
158        deposit_request: &DepositRequest,
159    ) -> Result<(), DepositError> {
160        let outpoint = bitcoin::OutPoint {
161            txid: deposit_request.utxo.id.txid.into(),
162            vout: deposit_request.utxo.id.vout,
163        };
164        let txout = self
165            .btc_monitor()
166            .confirm_deposit(outpoint)
167            .await
168            .map_err(|e| match e {
169                DepositConfirmError::UtxoSpent { .. } => {
170                    self.metrics.deposits_rejected_utxo_spent.inc();
171                    DepositError::BitcoinUtxoSpent(anyhow!(e))
172                }
173                DepositConfirmError::Other(err) => DepositError::BitcoinConfirmFailed(err),
174            })?;
175        if txout.value.to_sat() != deposit_request.utxo.amount {
176            return Err(DepositError::DepositDataMismatch(anyhow!(
177                "Bitcoin deposit amount mismatch: got {}, onchain is {}",
178                deposit_request.utxo.amount,
179                txout.value.to_sat(),
180            )));
181        }
182
183        self.validate_deposit_request_derivation_path(&txout.script_pubkey, deposit_request)
184            .await?;
185        Ok(())
186    }

188    async fn validate_deposit_request_derivation_path(
189        &self,
190        script_pubkey: &ScriptBuf,
191        deposit_request: &DepositRequest,
192    ) -> Result<(), DepositError> {
193        let deposit_address =
194            bitcoin::Address::from_script(script_pubkey, self.config.bitcoin_network()).map_err(
195                |e| {
196                    DepositError::DepositDataMismatch(anyhow!(
197                        "Failed to extract address from script_pubkey: {e}"
198                    ))
199                },
200            )?;
201        let hashi_pubkey = self.get_hashi_pubkey().map_err(DepositError::NotReady)?;
202        let expected_address = self
203            .get_deposit_address(&hashi_pubkey, deposit_request.utxo.derivation_path.as_ref())
204            .map_err(DepositError::DepositDataMismatch)?;
205
206        if deposit_address != expected_address {
207            return Err(DepositError::DepositDataMismatch(anyhow!(
208                "Expected address {expected_address}, got address {deposit_address}",
209            )));
210        }
211
212        Ok(())
213    }

215    pub fn get_deposit_address(
216        &self,
217        hashi_pubkey: &XOnlyPublicKey,
218        derivation_path: Option<&sui_sdk_types::Address>,
219    ) -> anyhow::Result<bitcoin::Address> {
220        let pubkey = self.deposit_pubkey(hashi_pubkey, derivation_path)?;
221        Ok(self.bitcoin_address_from_pubkey(&pubkey))
222    }

224    pub(crate) fn deposit_pubkey(
225        &self,
226        hashi_pubkey: &XOnlyPublicKey,
227        derivation_path: Option<&sui_sdk_types::Address>,
228    ) -> anyhow::Result<XOnlyPublicKey> {
229        if let Some(path) = derivation_path {
230            // Prefer the local signing manager (available after DKG preparation).
231            // Fall back to the on-chain key, which is guaranteed present once the
232            // initial committee has formed and `end_reconfig` has been processed.
233            let verifying_key = self
234                .signing_verifying_key()
235                .map(Ok)
236                .unwrap_or_else(|| self.onchain_verifying_key_g())
237                .context("MPC public key not available yet")?;
238            let derived = fastcrypto_tbls::threshold_schnorr::key_derivation::derive_verifying_key(
239                &verifying_key,
240                &path.into_inner(),
241            );
242            let pubkey = XOnlyPublicKey::from_slice(&derived.to_byte_array())
243                .context("valid 32-byte x-only key")?;
244            Ok(pubkey)
245        } else {
246            Ok(*hashi_pubkey)
247        }
248    }

250    pub(crate) fn bitcoin_address_from_pubkey(&self, pubkey: &XOnlyPublicKey) -> bitcoin::Address {

254    pub fn get_hashi_pubkey(&self) -> anyhow::Result<XOnlyPublicKey> {
255        let g = self
256            .mpc_handle()
257            .context("MpcHandle not initialized")?
258            .public_key()
259            .context("MPC public key not available yet")?;
260        // Convert G (ProjectivePoint, 33 bytes compressed) to SchnorrPublicKey (32 bytes x-only)
261        let schnorr_pk = SchnorrPublicKey::try_from(&g)
262            .map_err(|e| anyhow!("invalid group element for schnorr key: {e}"))?;
263        Ok(XOnlyPublicKey::from_slice(&schnorr_pk.to_byte_array())?)
264    }

266    /// Read the MPC public key from on-chain state.
267    ///
268    /// Unlike `get_hashi_pubkey`, this does not depend on the node's local DKG
269    /// completion channel. The on-chain key is set during `end_reconfig` event
270    /// processing, so it is guaranteed to be present once the initial committee
271    /// has formed (i.e., after `HashiNetworkBuilder::build()` returns).
272    pub fn get_onchain_mpc_pubkey(&self) -> anyhow::Result<XOnlyPublicKey> {
273        let g = self.onchain_verifying_key_g()?;
274        let schnorr_pk = SchnorrPublicKey::try_from(&g)
275            .map_err(|e| anyhow!("invalid group element for schnorr key: {e}"))?;
276        Ok(XOnlyPublicKey::from_slice(&schnorr_pk.to_byte_array())?)
277    }

279    /// Deserialize the BCS-encoded MPC group element from on-chain state.
280    ///
281    /// The on-chain key is stored as `bcs::to_bytes(&G)` in the `CommitteeSet`
282    /// and is populated atomically with the `end_reconfig` event.
283    fn onchain_verifying_key_g(&self) -> anyhow::Result<G> {
284        let bytes = self.onchain_state().mpc_public_key();
285        anyhow::ensure!(
286            !bytes.is_empty(),
287            "MPC public key not yet available on-chain"
288        );
289        bcs::from_bytes(&bytes).context("failed to deserialize on-chain MPC public key")
290    }

292    fn sign_deposit_confirmation(
293        &self,
294        deposit_request: &DepositRequest,
295    ) -> anyhow::Result<MemberSignature> {
296        let epoch = self.onchain_state().epoch();
297        let validator_address = self
298            .config
299            .validator_address()
300            .map_err(|e| anyhow!("No validator address configured: {}", e))?;
301        let private_key = self
302            .config
303            .protocol_private_key()
304            .ok_or_else(|| anyhow!("No protocol private key configured"))?;
305        let public_key_bytes = private_key.public_key().as_bytes().to_vec().into();
306
307        let message = DepositConfirmationMessage {
308            request_id: deposit_request.id,
309            utxo: deposit_request.utxo.clone(),
310        };
311
312        let signature_bytes = private_key
313            .sign(epoch, validator_address, &message)
314            .signature()
315            .as_bytes()
316            .to_vec()
317            .into();
318
319        Ok(MemberSignature {
320            epoch: Some(epoch),
321            address: Some(validator_address.to_string()),
322            public_key: Some(public_key_bytes),
323            signature: Some(signature_bytes),
324        })
325    }

328#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
329pub(crate) enum DepositErrorKind {
330    RetryOnNextBlock,
331    NeverRetry,
332}
333
334#[derive(Debug, Error)]
335pub enum DepositError {
336    #[error("Failed to confirm Bitcoin deposit: {0}")]
337    BitcoinConfirmFailed(#[source] anyhow::Error),
338
339    #[error("Screener service error: {0}")]
340    AmlServiceError(#[source] anyhow::Error),
341
342    #[error("Not ready: {0}")]
343    NotReady(#[source] anyhow::Error),
344
345    #[error("Invalid on-chain deposit request: {0}")]
346    InvalidOnchainRequest(#[source] anyhow::Error),
347
348    #[error("UTXO is already active or spent on Sui: {0}")]
349    DuplicateOrSpentOnSui(#[source] anyhow::Error),
350
351    #[error("Deposit UTXO has already been spent on Bitcoin: {0}")]
352    BitcoinUtxoSpent(#[source] anyhow::Error),
353
354    #[error("Deposit data mismatch: {0}")]
355    DepositDataMismatch(#[source] anyhow::Error),
356
357    #[error("AML checks rejected deposit: {0}")]
358    AmlRejected(#[source] anyhow::Error),
359
360    #[error("Failed quorum: weight {weight} < {required_weight}")]
361    FailedQuorum { weight: u64, required_weight: u64 },
362
363    #[error("Failed to build deposit certificate: {0}")]
364    CertificateBuildFailed(#[source] anyhow::Error),
365
366    #[error("Failed to create Sui transaction executor: {0}")]
367    ExecutorInitFailed(#[source] anyhow::Error),
368
369    #[error("Failed to confirm deposit on Sui: {0}")]
370    ConfirmDepositFailed(#[source] anyhow::Error),
371
372    #[error("Deposit processing timed out after {0:?}")]
373    TimedOut(std::time::Duration),
374}
375
376impl DepositError {
377    pub(crate) fn kind(&self) -> DepositErrorKind {
378        match self {
379            Self::BitcoinConfirmFailed(_)
380            | Self::AmlServiceError(_)
381            | Self::NotReady(_)
382            | Self::FailedQuorum { .. }
383            | Self::CertificateBuildFailed(_)
384            | Self::ExecutorInitFailed(_)
385            | Self::ConfirmDepositFailed(_)
386            | Self::TimedOut(_) => DepositErrorKind::RetryOnNextBlock,
387            Self::InvalidOnchainRequest(_)
388            | Self::DuplicateOrSpentOnSui(_)
389            | Self::BitcoinUtxoSpent(_)
390            | Self::DepositDataMismatch(_)
391            | Self::AmlRejected(_) => DepositErrorKind::NeverRetry,
392        }
393    }
