4//! Reusable helpers for the package-upgrade governance flow.
5//!
6//! Covers the non-orchestrating pieces that both the CLI (`hashi proposal
7//! create upgrade` + `proposal execute`) and the e2e test harness need:
8//!
9//! - Building an upgrade package via `sui move build`
10//! - Constructing the `execute + publish + finalize` PTB
11//! - Constructing the generic `execute` PTB for non-upgrade proposals
12//! - Parsing the proposal ID out of a `ProposalCreatedEvent` in tx effects
13//! - Finding the new package ID in the effects of an upgrade transaction
14//!
15//! Orchestration (collecting votes from committee members, driving the full
16//! propose → vote → execute lifecycle end to end) still lives in the caller —
17//! the e2e harness has all four validator keys so it can drive it
18//! programmatically, while the CLI only has one operator key and drives it
19//! one step at a time.

34/// Relative path (from a package root) to the Move source file declaring the
35/// `PACKAGE_VERSION` constant.
36const PACKAGE_VERSION_SOURCE: &str = "sources/core/config/config.move";
37
38/// Parse the `PACKAGE_VERSION` constant from `sources/core/config/config.move`
39/// in a package source tree.
40///
41/// This is the version gate enforced by `config::assert_version_enabled`; it
42/// must be bumped in every upgrade or the new package will reject all entry
43/// points guarded by that check.
44pub fn read_package_version(package_path: &Path) -> Result<u64> {
45    let source_path = package_path.join(PACKAGE_VERSION_SOURCE);
46    let source = std::fs::read_to_string(&source_path).map_err(|e| {
47        anyhow!(
48            "failed to read {} to parse PACKAGE_VERSION: {e}",
49            source_path.display()
50        )
51    })?;
52
53    let line = source
54        .lines()
55        .find(|l| l.trim_start().starts_with("const PACKAGE_VERSION"))
56        .ok_or_else(|| {
57            anyhow!(
58                "PACKAGE_VERSION declaration not found in {}",
59                source_path.display()
60            )
61        })?;
62
63    let rhs = line
64        .split_once('=')
65        .and_then(|(_, rhs)| rhs.split(';').next())
66        .ok_or_else(|| anyhow!("malformed PACKAGE_VERSION line: {line:?}"))?
67        .trim();
68
69    rhs.parse::<u64>()
70        .map_err(|e| anyhow!("PACKAGE_VERSION value {rhs:?} is not a u64: {e}"))
71}

73/// Build an upgrade package by invoking `sui move build --dump-bytecode-as-base64`
74/// and parsing the resulting JSON.
75///
76/// `sui_binary` is the path (or PATH-resolvable name) of the `sui` executable
77/// to shell out to. `package_path` is the directory containing `Move.toml`.
78/// `client_config` is passed as `--client.config` when supplied, otherwise the
79/// `sui` binary's default client config is used.
80///
81/// Runs a pre-flight check that the package declares `PACKAGE_VERSION ==
82/// expected_version`, failing before shelling out if the constant wasn't
83/// bumped. Callers typically pass `current_highest_version + 1`.
84///
85/// Returns the compiled `Publish` (modules + dependencies) plus the package
86/// digest — the latter is what goes into the `Upgrade` proposal.
87pub fn build_upgrade_package(
88    sui_binary: &Path,
89    package_path: &Path,
90    client_config: Option<&Path>,
91    expected_version: u64,
92) -> Result<(Publish, Vec<u8>)> {
93    let declared_version = read_package_version(package_path)?;
94    anyhow::ensure!(
95        declared_version == expected_version,
96        "upgrade package declares PACKAGE_VERSION = {declared_version}, expected {expected_version} \
97         (must be exactly +1 of the currently published version)"
98    );
99
100    let mut cmd = std::process::Command::new(sui_binary);
101    cmd.arg("move");
102
103    if let Some(config) = client_config {
104        cmd.arg("--client.config").arg(config);
105    }
106
107    cmd.arg("-p")
108        .arg(package_path)
109        .arg("build")
110        .arg("-e")
111        .arg("testnet")
112        .arg("--dump-bytecode-as-base64");
113
114    let output = cmd.output()?;
115    anyhow::ensure!(
116        output.status.success(),
117        "sui move build failed:\nstdout: {}\nstderr: {}",
118        output.stdout.escape_ascii(),
119        output.stderr.escape_ascii()
120    );
121
122    #[derive(serde::Deserialize)]
123    struct MoveBuildOutput {
124        modules: Vec<String>,
125        dependencies: Vec<Address>,
126        digest: Vec<u8>,
127    }
128
129    let build_output: MoveBuildOutput = serde_json::from_slice(&output.stdout)?;
130    let digest = build_output.digest.clone();
131    let modules = build_output
132        .modules
133        .into_iter()
134        .map(|b64| <base64ct::Base64 as base64ct::Encoding>::decode_vec(&b64))
135        .collect::<Result<Vec<_>, _>>()?;
136
137    Ok((
138        Publish {
139            modules,
140            dependencies: build_output.dependencies,
141        },
142        digest,
143    ))
144}

146/// Build the PTB that executes an `Upgrade` proposal in a single transaction:
147/// `upgrade::execute` → `builder.upgrade(...)` → `upgrade::finalize_upgrade`.
148///
149/// The three steps must be in one PTB so the `UpgradeTicket` and
150/// `UpgradeReceipt` hot potatoes can be consumed without leaving the
151/// transaction.
152pub fn build_upgrade_execution_transaction(
153    hashi_ids: HashiIds,
154    proposal_id: Address,
155    compiled: Publish,
156) -> TransactionBuilder {
157    let mut builder = TransactionBuilder::new();
158    let hashi_arg = builder.object(
159        ObjectInput::new(hashi_ids.hashi_object_id)
160            .as_shared()
161            .with_mutable(true),
162    );
163    let proposal_id_arg = builder.pure(&proposal_id);
164    let clock_arg = builder.object(
165        ObjectInput::new(SUI_CLOCK_OBJECT_ID)
166            .as_shared()
167            .with_mutable(false),
168    );
169
170    // Step A: upgrade::execute → UpgradeTicket
171    let ticket = builder.move_call(
172        Function::new(
173            hashi_ids.package_id,
174            Identifier::from_static("upgrade"),
175            Identifier::from_static("execute"),
176        ),
177        vec![hashi_arg, proposal_id_arg, clock_arg],
178    );
179
180    // Step B: publish upgrade → UpgradeReceipt
181    let receipt = builder.upgrade(
182        compiled.modules,
183        compiled.dependencies,
184        hashi_ids.package_id,
185        ticket,
186    );
187
188    // Step C: finalize_upgrade — takes the receipt and swaps the package in-place.
189    // Needs a second mutable reference to the hashi object since the first one
190    // was consumed by `upgrade::execute`.
191    let hashi_arg2 = builder.object(
192        ObjectInput::new(hashi_ids.hashi_object_id)
193            .as_shared()
194            .with_mutable(true),
195    );
196    builder.move_call(
197        Function::new(
198            hashi_ids.package_id,
199            Identifier::from_static("upgrade"),
200            Identifier::from_static("finalize_upgrade"),
201        ),
202        vec![hashi_arg2, receipt],
203    );
204
205    builder
206}

208/// Build the PTB that executes a non-upgrade proposal (UpdateConfig,
209/// EnableVersion, DisableVersion, EmergencyPause).
210///
211/// Calls `<execute_package_id>::<proposal_module>::execute(hashi, proposal_id, clock)`.
212///
213/// `execute_package_id` is almost always `hashi_ids.package_id`, but may
214/// differ when disabling an old version after an upgrade: the `execute` call
215/// has to go through the NEW package (whose `PACKAGE_VERSION` differs from
216/// the version being disabled), not through the stored original
217/// `hashi_ids.package_id`.
218pub fn build_execute_proposal_transaction(
219    hashi_ids: HashiIds,
220    proposal_id: Address,
221    execute_package_id: Address,
222    proposal_module: &str,
223) -> Result<TransactionBuilder> {
224    let module = Identifier::new(proposal_module)
225        .map_err(|e| anyhow!("invalid proposal module {proposal_module:?}: {e}"))?;
226
227    let mut builder = TransactionBuilder::new();
228    let hashi_arg = builder.object(
229        ObjectInput::new(hashi_ids.hashi_object_id)
230            .as_shared()
231            .with_mutable(true),
232    );
233    let proposal_id_arg = builder.pure(&proposal_id);
234    let clock_arg = builder.object(
235        ObjectInput::new(SUI_CLOCK_OBJECT_ID)
236            .as_shared()
237            .with_mutable(false),
238    );
239
240    builder.move_call(
241        Function::new(
242            execute_package_id,
243            module,
244            Identifier::from_static("execute"),
245        ),
246        vec![hashi_arg, proposal_id_arg, clock_arg],
247    );
248
249    Ok(builder)
250}

252/// Extract the newly-created proposal's object ID from a transaction that
253/// called `<module>::propose(...)`. Looks for a single `ProposalCreatedEvent`
254/// in the transaction's emitted events.
255///
256/// The BCS payload of the event is `(proposal_id, timestamp_ms)`; we only
257/// return the proposal ID here.
258pub fn extract_proposal_id_from_response(response: &ExecuteTransactionResponse) -> Result<Address> {
259    let event = response
260        .transaction()
261        .events()
262        .events()
263        .iter()
264        .find(|e| e.contents().name().contains("ProposalCreatedEvent"))
265        .ok_or_else(|| anyhow!("ProposalCreatedEvent not found in transaction effects"))?;
266
267    let (id, _ts): (Address, u64) = bcs::from_bytes(event.contents().value())
268        .map_err(|e| anyhow!("failed to deserialize ProposalCreatedEvent payload: {e}"))?;
269    Ok(id)
270}

272/// Extract the new package ID from the effects of a successful upgrade
273/// transaction. The upgrade PTB creates exactly one `package` changed object.
274pub fn extract_new_package_id_from_response(
275    response: &ExecuteTransactionResponse,
276) -> Result<Address> {
277    response
278        .transaction()
279        .effects()
280        .changed_objects()
281        .iter()
282        .find(|o| o.object_type() == "package")
283        .ok_or_else(|| anyhow!("new package not found in upgrade effects"))?
284        .object_id()
285        .parse::<Address>()
286        .map_err(|e| anyhow!("failed to parse new package ID: {e}"))
287}
