21pub async fn run(action: DepositCommands, config: &CliConfig, tx_opts: &TxOptions) -> Result<()> {
22    match action {
23        DepositCommands::GenerateAddress { recipient } => {
24            generate_address(config, &recipient).await
25        }
26        DepositCommands::Request { txid, recipient } => {
27            request_all(config, tx_opts, &txid, recipient.as_deref()).await
28        }
29        DepositCommands::RequestSingle {
30            txid,
31            vout,
32            amount,
33            recipient,
34        } => request(config, tx_opts, &txid, vout, amount, recipient.as_deref()).await,
35        DepositCommands::Status { request_id } => status(config, &request_id).await,
36        DepositCommands::List {
37            output_format,
38            json,
39        } => {
40            let output_format = if json {
41                OutputFormat::Json
42            } else {
43                output_format
44            };
45            list(config, output_format).await
46        }
47    }
48}

50/// Parse raw on-chain MPC public key bytes and derive the deposit address.
51pub fn cli_derive_deposit_address(
52    mpc_pubkey_bytes: &[u8],
53    recipient: Option<&sui_sdk_types::Address>,
54    btc_network: bitcoin::Network,
55) -> Result<bitcoin::Address> {
56    use fastcrypto::groups::secp256k1::ProjectivePoint;
57    use fastcrypto::serde_helpers::ToFromByteArray;
58
59    let mpc_key = match mpc_pubkey_bytes.len() {
60        33 => <ProjectivePoint as ToFromByteArray<33>>::from_byte_array(
61            mpc_pubkey_bytes
62                .try_into()
63                .context("MPC key must be 33 bytes")?,
64        )
65        .context("Failed to parse MPC key as ProjectivePoint")?,
66        32 => {
67            if recipient.is_some() {
68                anyhow::bail!(
69                    "Key derivation requires the full 33-byte compressed MPC key, \
70                     but only 32-byte x-only key is available"
71                );
72            }
73            let xonly = XOnlyPublicKey::from_slice(mpc_pubkey_bytes)
74                .context("Failed to parse 32-byte MPC key")?;
75            return Ok(
76                hashi_types::guardian::bitcoin_utils::single_key_taproot_script_path_address(
77                    &xonly,
78                    btc_network,
79                ),
80            );
81        }
82        n => anyhow::bail!("Unexpected MPC public key length: {} bytes", n),
83    };
84
85    crate::deposits::derive_deposit_address(&mpc_key, recipient, btc_network)
86}

88async fn generate_address(config: &CliConfig, recipient: &str) -> Result<()> {
89    let client = HashiClient::new(config).await?;
90
91    let mpc_pubkey = client.fetch_mpc_public_key();
92    if mpc_pubkey.is_empty() {
93        anyhow::bail!("MPC public key not available on-chain. Has the committee completed DKG?");
94    }
95
96    let is_change = recipient.is_empty();
97    let recipient_addr = if is_change {
98        None
99    } else {
100        Some(
101            recipient
102                .parse::<sui_sdk_types::Address>()
103                .context("Invalid recipient Sui address")?,
104        )
105    };
106
107    let btc_network = crate::btc_monitor::config::parse_btc_network(
108        config.bitcoin.as_ref().and_then(|b| b.network.as_deref()),
109    )?;
110
111    let address = cli_derive_deposit_address(&mpc_pubkey, recipient_addr.as_ref(), btc_network)?;
112
113    let title = if is_change {
114        "Deposit Address (Change Address)"
115    } else {
116        "Deposit Address"
117    };
118
119    println!("\n{}", title.bold());
120    println!("{}", "━".repeat(50).dimmed());
121    println!("  {} {}", "Address:".bold(), address.to_string().green());
122    println!("  {} {:?}", "Network:".bold(), btc_network);
123    if !is_change {
124        println!("  {} {}", "hBTC Recipient:".bold(), recipient);
125    }
126    println!("{}", "━".repeat(50).dimmed());
127
128    Ok(())
129}

131async fn request(
132    config: &CliConfig,
133    _tx_opts: &TxOptions,
134    txid: &str,
135    vout: u32,
136    amount: u64,
137    recipient: Option<&str>,
138) -> Result<()> {
139    config.validate()?;
140
141    let hashi_ids = crate::config::HashiIds {
142        package_id: config.package_id(),
143        hashi_object_id: config.hashi_object_id(),
144    };
145
146    let signer = config
147        .load_keypair()?
148        .context("Keypair required for deposit request. Set keypair_path in config.")?;
149
150    let derivation_path = match recipient {
151        Some(r) => Some(
152            r.parse::<sui_sdk_types::Address>()
153                .context("Invalid recipient Sui address")?,
154        ),
155        None => {
156            let addr = signer.public_key().derive_address();
157            print_info(&format!(
158                "No --recipient specified, defaulting to signer address {}",
159                addr
160            ));
161            Some(addr)
162        }
163    };
164
165    let client = sui_rpc::Client::new(&config.sui_rpc_url)?;
166    let mut executor = crate::sui_tx_executor::SuiTxExecutor::new(client, signer, hashi_ids);
167
168    let parsed_txid: bitcoin::Txid = txid.parse().context("Invalid txid")?;
169    let txid_address = sui_sdk_types::Address::new(parsed_txid.to_byte_array());
170
171    print_info("Submitting deposit request on Sui...");
172
173    let request_id = executor
174        .execute_create_deposit_request(txid_address, vout, amount, derivation_path)
175        .await?;
176
177    print_success(&format!("Deposit request created: {}", request_id));
178
179    Ok(())
180}

182async fn request_all(
183    config: &CliConfig,
184    _tx_opts: &TxOptions,
185    txid: &str,
186    recipient: Option<&str>,
187) -> Result<()> {
188    config.validate()?;
189
190    let btc_rpc = config.require_btc_rpc_client()?;
191
192    let hashi_ids = crate::config::HashiIds {
193        package_id: config.package_id(),
194        hashi_object_id: config.hashi_object_id(),
195    };
196
197    let signer = config
198        .load_keypair()?
199        .context("Keypair required for deposit request. Set keypair_path in config.")?;
200
201    let derivation_path = match recipient {
202        Some(r) => Some(
203            r.parse::<sui_sdk_types::Address>()
204                .context("Invalid recipient Sui address")?,
205        ),
206        None => {
207            let addr = signer.public_key().derive_address();
208            print_info(&format!(
209                "No --recipient specified, defaulting to signer address {}",
210                addr
211            ));
212            Some(addr)
213        }
214    };
215
216    // Fetch the MPC public key and derive the deposit address
217    let client = HashiClient::new(config).await?;
218    let mpc_pubkey = client.fetch_mpc_public_key();
219    if mpc_pubkey.is_empty() {
220        anyhow::bail!("MPC public key not available on-chain. Has the committee completed DKG?");
221    }
222
223    let btc_network = crate::btc_monitor::config::parse_btc_network(
224        config.bitcoin.as_ref().and_then(|b| b.network.as_deref()),
225    )?;
226
227    let deposit_address =
228        cli_derive_deposit_address(&mpc_pubkey, derivation_path.as_ref(), btc_network)?;
229
230    // Look up the transaction and find all matching outputs
231    let parsed_txid: bitcoin::Txid = txid.parse().context("Invalid txid")?;
232
233    let raw_tx = btc_rpc
234        .get_raw_transaction(parsed_txid)
235        .map_err(anyhow::Error::from)
236        .and_then(|resp| resp.transaction().map_err(anyhow::Error::from))
237        .context("Failed to fetch transaction from Bitcoin RPC")?;
238
239    let matching_utxos: Vec<(u32, u64)> = raw_tx
240        .output
241        .iter()
242        .enumerate()
243        .filter(|(_, output)| output.script_pubkey == deposit_address.script_pubkey())
244        .map(|(i, output)| (i as u32, output.value.to_sat()))
245        .collect();
246
247    if matching_utxos.is_empty() {
248        anyhow::bail!(
249            "No outputs in transaction {} match the deposit address {}",
250            txid,
251            deposit_address
252        );
253    }
254
255    let total_sats: u64 = matching_utxos.iter().map(|(_, amount)| amount).sum();
256    println!("\n{}", "Found matching outputs".bold());
257    println!("{}", "━".repeat(50).dimmed());
258    for (vout, amount) in &matching_utxos {
259        println!(
260            "  vout {}: {} sats ({:.8} BTC)",
261            vout,
262            amount,
263            *amount as f64 / 1e8
264        );
265    }
266    println!(
267        "  {} {} outputs, {} total sats ({:.8} BTC)",
268        "Summary:".bold(),
269        matching_utxos.len(),
270        total_sats,
271        total_sats as f64 / 1e8
272    );
273    println!("{}", "━".repeat(50).dimmed());
274
275    // 3 dynamic-field ops per deposit × 1000 object-runtime cap = 333/PTB.
276    const CHUNK_SIZE: usize = 333;
277
278    let sui_client = sui_rpc::Client::new(&config.sui_rpc_url)?;
279    let mut executor = crate::sui_tx_executor::SuiTxExecutor::new(sui_client, signer, hashi_ids);
280
281    let txid_address =
282        sui_sdk_types::Address::new(bitcoin::hashes::Hash::to_byte_array(parsed_txid));
283
284    let total = matching_utxos.len();
285    let mut all_request_ids: Vec<sui_sdk_types::Address> = Vec::with_capacity(total);
286
287    for (chunk_idx, chunk) in matching_utxos.chunks(CHUNK_SIZE).enumerate() {
288        print_info(&format!(
289            "Submitting batch {}/{} ({} deposits) on Sui...",
290            chunk_idx + 1,
291            total.div_ceil(CHUNK_SIZE),
292            chunk.len()
293        ));
294
295        let request_ids = executor
296            .execute_create_deposit_requests_batch(txid_address, chunk, derivation_path)
297            .await?;
298        all_request_ids.extend(request_ids);
299    }
300
301    println!("\n{}", "Deposit requests created".bold().green());
302    println!("{}", "━".repeat(50).dimmed());
303    for (i, request_id) in all_request_ids.iter().enumerate() {
304        let (vout, amount) = matching_utxos[i];
305        println!("  vout {}: {} sats -> request {}", vout, amount, request_id);
306    }
307    println!("{}", "━".repeat(50).dimmed());
308
309    Ok(())
310}

312async fn status(config: &CliConfig, request_id: &str) -> Result<()> {
313    let client = HashiClient::new(config).await?;
314
315    let req_addr = request_id
316        .parse::<sui_sdk_types::Address>()
317        .context("Invalid request ID")?;
318
319    let deposits = client.fetch_deposit_requests();
320    let deposit = deposits.iter().find(|d| d.id == req_addr);
321
322    println!("\n{}", "Deposit Request Status".bold());
323    println!("{}", "━".repeat(60).dimmed());
324
325    let Some(dep) = deposit else {
326        print_info("Deposit request not found in pending queue (may be confirmed or expired).");
327        println!("{}", "━".repeat(60).dimmed());
328        return Ok(());
329    };
330
331    let txid = dep.utxo.id.txid;
332    println!(
333        "  {} {}",
334        "Request ID:".bold(),
335        display::format_address_full(&dep.id)
336    );
337    println!("  {} {}:{}", "UTXO:".bold(), txid, dep.utxo.id.vout);
338    println!("  {} {} sats", "Amount:".bold(), dep.utxo.amount);
339    println!(
340        "  {} {}",
341        "Requested:".bold(),
342        display::format_timestamp(dep.timestamp_ms)
343    );
344    println!("  {} {}", "Status:".bold(), "Pending".yellow());
345
346    // BTC context if available
347    if let Ok(Some(btc_rpc)) = config.btc_rpc_client() {
348        println!();
349        println!("  {}", "BTC Context:".bold());
350        match btc_rpc.get_raw_transaction_verbose(txid.into()) {
351            Ok(info) => {
352                let confirmations = info.confirmations.unwrap_or(0);
353                println!("    {} {}", "Confirmations:".bold(), confirmations);
354                if let Some(ref blockhash) = info.block_hash {
355                    println!("    {} {}", "Block:".bold(), blockhash);
356                }
357            }
358            Err(_) => {
359                println!("    {}", "(transaction not found on BTC node)".dimmed());
360            }
361        }
362    }
363
364    println!("{}", "━".repeat(60).dimmed());
365    Ok(())
366}

368async fn list(config: &CliConfig, output_format: OutputFormat) -> Result<()> {
369    let client = HashiClient::new(config).await?;
370
371    let deposits = client.fetch_deposit_requests();
372
373    match output_format {
374        OutputFormat::Json => {
375            let rows: Vec<_> = deposits
376                .iter()
377                .map(|dep| {
378                    serde_json::json!({
379                        "request_id": dep.id.to_string(),
380                        "amount_sats": dep.utxo.amount,
381                        "utxo": {
382                            "txid": dep.utxo.id.txid.to_string(),
383                            "vout": dep.utxo.id.vout,
384                        },
385                        "status": "pending",
386                        "caller": dep.sender.to_string(),
387                        "requested_ms": dep.timestamp_ms,
388                    })
389                })
390                .collect();
391
392            println!(
393                "{}",
394                serde_json::to_string_pretty(&serde_json::json!({
395                    "deposits": rows,
396                    "count": deposits.len(),
397                }))?
398            );
399        }
400        OutputFormat::HumanTable => {
401            println!("\n{}", "Deposit Requests".bold());
402            println!("{}", "━".repeat(100).dimmed());
403
404            if deposits.is_empty() {
405                print_info("No pending deposit requests.");
406            } else {
407                println!(
408                    "  {:<20} {:<14} {:<12} {:<10} {:<20} {}",
409                    "Request ID".bold(),
410                    "Amount (sats)".bold(),
411                    "UTXO".bold(),
412                    "Status".bold(),
413                    "Caller".bold(),
414                    "Requested".bold()
415                );
416                for dep in &deposits {
417                    println!(
418                        "  {:<20} {:<14} {}:{:<3} {:<10} {:<20} {}",
419                        display::format_address_full(&dep.id),
420                        dep.utxo.amount,
421                        dep.utxo.id.txid,
422                        dep.utxo.id.vout,
423                        "Pending".yellow(),
424                        display::format_address_full(&dep.sender),
425                        display::format_timestamp(dep.timestamp_ms)
426                    );
427                }
428                println!("\n  {} deposit(s)", deposits.len());
429            }
430
431            println!("{}", "━".repeat(100).dimmed());
432        }
433    }
434
435    Ok(())
