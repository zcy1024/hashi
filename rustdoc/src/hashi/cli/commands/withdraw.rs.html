21pub async fn run(action: WithdrawCommands, config: &CliConfig, tx_opts: &TxOptions) -> Result<()> {
22    match action {
23        WithdrawCommands::Request {
24            amount,
25            btc_address,
26        } => request(config, tx_opts, amount, &btc_address).await,
27        WithdrawCommands::Cancel { request_id } => cancel(config, tx_opts, &request_id).await,
28        WithdrawCommands::Status { request_id } => status(config, &request_id).await,
29        WithdrawCommands::List {
30            output_format,
31            json,
32        } => {
33            let output_format = if json {
34                OutputFormat::Json
35            } else {
36                output_format
37            };
38            list(config, output_format).await
39        }
40    }
41}

43async fn request(
44    config: &CliConfig,
45    _tx_opts: &TxOptions,
46    amount: u64,
47    btc_address: &str,
48) -> Result<()> {
49    config.validate()?;
50
51    let hashi_ids = crate::config::HashiIds {
52        package_id: config.package_id(),
53        hashi_object_id: config.hashi_object_id(),
54    };
55
56    let signer = config
57        .load_keypair()?
58        .context("Keypair required for withdrawal request. Set keypair_path in config.")?;
59
60    // Parse the BTC destination address and verify it matches the configured network
61    let btc_network = crate::btc_monitor::config::parse_btc_network(
62        config.bitcoin.as_ref().and_then(|b| b.network.as_deref()),
63    )?;
64    let btc_addr: bitcoin::Address<bitcoin::address::NetworkUnchecked> =
65        btc_address.parse().context("Invalid Bitcoin address")?;
66    let btc_addr = btc_addr
67        .require_network(btc_network)
68        .context("Withdrawal address does not match the configured Bitcoin network")?;
69    let destination_bytes = witness_program_from_address(&btc_addr)?;
70
71    print_info(&format!("Withdrawal amount: {} sats", amount));
72    print_info(&format!("BTC destination: {}", btc_address));
73
74    let client = sui_rpc::Client::new(&config.sui_rpc_url)?;
75    let mut executor = crate::sui_tx_executor::SuiTxExecutor::new(client, signer, hashi_ids);
76
77    print_info("Submitting withdrawal request on Sui...");
78
79    let request_id = executor
80        .execute_create_withdrawal_request(amount, destination_bytes)
81        .await?;
82
83    print_success(&format!("Withdrawal request created: {}", request_id));
84
85    Ok(())
86}

88async fn cancel(config: &CliConfig, _tx_opts: &TxOptions, request_id: &str) -> Result<()> {
89    config.validate()?;
90
91    let req_addr = request_id
92        .parse::<sui_sdk_types::Address>()
93        .context("Invalid request ID")?;
94
95    let signer = config
96        .load_keypair()?
97        .context("Keypair required to cancel withdrawal.")?;
98
99    let hashi_ids = crate::config::HashiIds {
100        package_id: config.package_id(),
101        hashi_object_id: config.hashi_object_id(),
102    };
103
104    let client = sui_rpc::Client::new(&config.sui_rpc_url)?;
105    let mut executor = crate::sui_tx_executor::SuiTxExecutor::new(client, signer, hashi_ids);
106
107    print_info("Cancelling withdrawal...");
108    executor.execute_cancel_withdrawal(&req_addr).await?;
109    print_success("Withdrawal cancelled.");
110
111    Ok(())
112}

114async fn status(config: &CliConfig, request_id: &str) -> Result<()> {
115    let client = HashiClient::new(config).await?;
116
117    let req_addr = request_id
118        .parse::<sui_sdk_types::Address>()
119        .context("Invalid request ID")?;
120
121    let withdrawal_requests = client.fetch_withdrawal_requests();
122    let pending_withdrawals = client.fetch_pending_withdrawals();
123
124    println!("\n{}", "Withdrawal Status".bold());
125    println!("{}", "━".repeat(60).dimmed());
126
127    // Check pending request queue first
128    if let Some(wr) = withdrawal_requests.iter().find(|w| w.id == req_addr) {
129        println!(
130            "  {} {}",
131            "Request ID:".bold(),
132            display::format_address_full(&wr.id)
133        );
134        println!("  {} {} sats", "Amount:".bold(), wr.btc_amount);
135        println!(
136            "  {} {}",
137            "BTC Address:".bold(),
138            hex::encode(&wr.bitcoin_address)
139        );
140        println!(
141            "  {} {}",
142            "Requester:".bold(),
143            display::format_address(&wr.sender)
144        );
145        println!(
146            "  {} {}",
147            "Requested:".bold(),
148            display::format_timestamp(wr.timestamp_ms)
149        );
150        println!();
151
152        let status_label = if wr.status.is_approved() {
153            "Approved".green()
154        } else {
155            "Requested".yellow()
156        };
157
158        let step = if wr.status.is_approved() { 2 } else { 1 };
159        println!("  {} {} ({}/6)", "Progress:".bold(), status_label, step);
160        println!(
161            "    {} Requested",
162            if step >= 1 {
163                "[done]".green()
164            } else {
165                "[    ]".dimmed()
166            }
167        );
168        println!(
169            "    {} Approved",
170            if step >= 2 {
171                "[done]".green()
172            } else {
173                "[    ]".dimmed()
174            }
175        );
176        println!("    {} Committed", "[    ]".dimmed());
177        println!("    {} Signed", "[    ]".dimmed());
178        println!("    {} Broadcast", "[    ]".dimmed());
179        println!("    {} Confirmed", "[    ]".dimmed());
180    }
181    // Check committed/signed pending withdrawals
182    else if let Some(pw) = pending_withdrawals
183        .iter()
184        .find(|p| p.request_ids.contains(&req_addr))
185    {
186        let txid_bytes: [u8; 32] = pw.id.into();
187        let txid = bitcoin::Txid::from_byte_array(txid_bytes);
188        let is_signed = pw.signatures.is_some();
189        let step = if is_signed { 4 } else { 3 };
190        let status_label = if is_signed {
191            "Signed".green()
192        } else {
193            "Committed".cyan()
194        };
195
196        println!(
197            "  {} {}",
198            "Request ID:".bold(),
199            display::format_address_full(&req_addr)
200        );
201        println!("  {} {}", "BTC txid:".bold(), txid);
202        println!();
203        println!("  {} {} ({}/6)", "Progress:".bold(), status_label, step);
204        println!("    {} Requested", "[done]".green());
205        println!("    {} Approved", "[done]".green());
206        println!("    {} Committed          txid: {}", "[done]".green(), txid);
207        println!(
208            "    {} Signed",
209            if is_signed {
210                "[done]".green()
211            } else {
212                "[    ]".dimmed()
213            }
214        );
215        println!("    {} Broadcast", "[    ]".dimmed());
216        println!("    {} Confirmed", "[    ]".dimmed());
217
218        // BTC context
219        if let Ok(Some(btc_rpc)) = config.btc_rpc_client() {
220            println!();
221            println!("  {}", "BTC Context:".bold());
222            use bitcoincore_rpc::RpcApi;
223            match btc_rpc.get_raw_transaction_info(&txid, None) {
224                Ok(info) => {
225                    let confirmations = info.confirmations.unwrap_or(0);
226                    let tx_status = if confirmations > 0 {
227                        "Confirmed".to_string()
228                    } else {
229                        "In Mempool".to_string()
230                    };
231                    println!("    {} {}", "TX Status:".bold(), tx_status);
232                    println!("    {} {}/6", "Confirmations:".bold(), confirmations);
233                }
234                Err(_) => {
235                    println!("    {}", "(transaction not found on BTC node)".dimmed());
236                }
237            }
238        }
239    } else {
240        print_info(
241            "Withdrawal request not found in pending queues (may be confirmed or cancelled).",
242        );
243    }
244
245    println!("{}", "━".repeat(60).dimmed());
246    Ok(())
247}

249async fn list(config: &CliConfig, output_format: OutputFormat) -> Result<()> {
250    let client = HashiClient::new(config).await?;
251
252    let requests = client.fetch_withdrawal_requests();
253    let pending = client.fetch_pending_withdrawals();
254
255    match output_format {
256        OutputFormat::Json => {
257            let queued: Vec<_> = requests
258                .iter()
259                .map(|wr| {
260                    serde_json::json!({
261                        "request_id": wr.id.to_string(),
262                        "amount_sats": wr.btc_amount,
263                        "status": if wr.status.is_approved() { "approved" } else { "requested" },
264                        "caller": wr.sender.to_string(),
265                        "requested_ms": wr.timestamp_ms,
266                    })
267                })
268                .collect();
269
270            let committed_or_signed: Vec<_> = pending
271                .iter()
272                .map(|pw| {
273                    let txid_bytes: [u8; 32] = pw.id.into();
274                    let txid = bitcoin::Txid::from_byte_array(txid_bytes);
275                    serde_json::json!({
276                        "txid": txid.to_string(),
277                        "status": if pw.signatures.is_some() { "signed" } else { "committed" },
278                        "request_count": pw.request_ids.len(),
279                    })
280                })
281                .collect();
282
283            println!(
284                "{}",
285                serde_json::to_string_pretty(&serde_json::json!({
286                    "queued": queued,
287                    "committed_or_signed": committed_or_signed,
288                    "queued_count": requests.len(),
289                    "committed_or_signed_count": pending.len(),
290                }))?
291            );
292        }
293        OutputFormat::HumanTable => {
294            println!("\n{}", "Withdrawal Requests".bold());
295            println!("{}", "━".repeat(100).dimmed());
296
297            if requests.is_empty() && pending.is_empty() {
298                print_info("No withdrawal requests found.");
299            } else {
300                if !requests.is_empty() {
301                    println!("  {}", "Queued:".bold().underline());
302                    println!(
303                        "  {:<20} {:<14} {:<10} {:<20} {}",
304                        "Request ID".bold(),
305                        "Amount (sats)".bold(),
306                        "Status".bold(),
307                        "Caller".bold(),
308                        "Requested".bold()
309                    );
310                    for wr in &requests {
311                        let status = if wr.status.is_approved() {
312                            "Approved"
313                        } else {
314                            "Requested"
315                        };
316                        println!(
317                            "  {:<20} {:<14} {:<10} {:<20} {}",
318                            display::format_address_full(&wr.id),
319                            wr.btc_amount,
320                            status,
321                            display::format_address_full(&wr.sender),
322                            display::format_timestamp(wr.timestamp_ms)
323                        );
324                    }
325                }
326
327                if !pending.is_empty() {
328                    if !requests.is_empty() {
329                        println!();
330                    }
331                    println!("  {}", "Committed/Signed:".bold().underline());
332                    for pw in &pending {
333                        let txid_bytes: [u8; 32] = pw.id.into();
334                        let txid = bitcoin::Txid::from_byte_array(txid_bytes);
335                        let status = if pw.signatures.is_some() {
336                            "Signed"
337                        } else {
338                            "Committed"
339                        };
340                        println!(
341                            "  txid: {}  status: {}  requests: {}",
342                            txid,
343                            status,
344                            pw.request_ids.len()
345                        );
346                    }
347                }
348
349                println!(
350                    "\n  {} queued, {} committed/signed",
351                    requests.len(),
352                    pending.len()
353                );
354            }
355
356            println!("{}", "━".repeat(100).dimmed());
357        }
358    }
359
360    Ok(())
