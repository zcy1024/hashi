25/// Print metadata if present
26fn print_metadata(metadata: &[(String, String)]) {
27    if !metadata.is_empty() {
28        println!("  {}", "Metadata:".bold());
29        for (key, value) in metadata {
30            println!("    {}: {}", key.dimmed(), value);
31        }
32    }
33}

35/// Print simulation (dry-run) results
36fn print_simulation_result(result: &SimulationResult) {
37    println!("\n{}", "🔍 Dry-run Results:".bold());
38    println!("  {} {}", "Sender:".dimmed(), result.sender.to_hex().cyan());
39    println!(
40        "  {} {} MIST",
41        "Gas Budget:".dimmed(),
42        result.gas_budget.to_string().cyan()
43    );
44    println!(
45        "  {} {} MIST/unit",
46        "Gas Price:".dimmed(),
47        result.gas_price.to_string().cyan()
48    );
49    let max_cost_sui = (result.gas_budget as f64) / 1_000_000_000.0;
50    println!(
51        "  {} {:.6} SUI",
52        "Max Cost:".dimmed(),
53        format!("{:.6}", max_cost_sui).yellow()
54    );
55    println!(
56        "\n  {} Transaction was simulated successfully. Use without --dry-run to execute.",
57        "✓".green()
58    );
59}

61/// Execute or simulate a transaction based on tx_opts.
62///
63/// Returns `Some(response)` when a real transaction was executed, and `None`
64/// on dry-run or when no keypair is configured.
65async fn execute_or_simulate(
66    client: &mut HashiClient,
67    tx: sui_transaction_builder::TransactionBuilder,
68    tx_opts: &TxOptions,
69) -> Result<Option<ExecuteTransactionResponse>> {
70    if !client.can_execute() {
71        print_warning("Transaction execution requires keypair configuration (--keypair).");
72        return Ok(None);
73    }
74
75    if tx_opts.dry_run {
76        print_info("Simulating transaction (dry-run)...");
77        let result = client.simulate(tx).await?;
78        print_simulation_result(&result);
79        return Ok(None);
80    }
81
82    print_info("Executing transaction...");
83    let response = client.execute(tx).await?;
84    let digest = response.transaction().digest();
85    println!(
86        "\n{} Transaction submitted: {}",
87        "✓".green(),
88        digest.to_string().cyan()
89    );
90    Ok(Some(response))
91}

93/// Print the newly-created proposal's ID after a `create_*_proposal` call,
94/// when the response is available (real execute, not dry-run).
95fn print_created_proposal_id(response: Option<&ExecuteTransactionResponse>) {
96    let Some(response) = response else {
97        return;
98    };
99    match crate::cli::upgrade::extract_proposal_id_from_response(response) {
100        Ok(id) => println!("  {} {}", "Proposal ID:".bold(), id.to_hex().cyan()),
101        Err(e) => {
102            tracing::warn!("Could not extract proposal ID from response: {e}");
103        }
104    }
105}

107/// List all active proposals
108pub async fn list_proposals(
109    config: &CliConfig,
110    type_filter: Option<String>,
111    detailed: bool,
112) -> Result<()> {
113    let client = HashiClient::new(config).await?;
114
115    print_info("Fetching proposals...");
116
117    let proposals = client.fetch_proposals();
118
119    if proposals.is_empty() {
120        println!("\n{}", "No active proposals found.".dimmed());
121        return Ok(());
122    }
123
124    // Filter by type if specified
125    let proposals: Vec<_> = if let Some(ref filter) = type_filter {
126        let filter_lower = filter.to_lowercase();
127        proposals
128            .into_iter()
129            .filter(|p| {
130                display::format_proposal_type(&p.proposal_type)
131                    .to_lowercase()
132                    .contains(&filter_lower)
133            })
134            .collect()
135    } else {
136        proposals
137    };
138
139    if proposals.is_empty() {
140        println!(
141            "\n{}",
142            format!(
143                "No proposals found matching type filter: {}",
144                type_filter.unwrap_or_default()
145            )
146            .dimmed()
147        );
148        return Ok(());
149    }
150
151    println!("\n📋 Active Proposals:\n");
152
153    if detailed {
154        // List mode skips the per-proposal vote/quorum fetch to avoid N extra
155        // network calls; use `proposal view <id>` for full vote progress.
156        for proposal in &proposals {
157            print_proposal_detailed(proposal, None, None);
158            println!();
159        }
160    } else {
161        #[derive(Tabled)]
162        struct ProposalRow {
163            #[tabled(rename = "ID")]
164            id: String,
165            #[tabled(rename = "Type")]
166            proposal_type: String,
167            #[tabled(rename = "Created")]
168            timestamp: String,
169        }
170
171        let rows: Vec<ProposalRow> = proposals
172            .iter()
173            .map(|p| ProposalRow {
174                id: display::format_address(&p.id),
175                proposal_type: display::format_proposal_type(&p.proposal_type),
176                timestamp: display::format_timestamp(p.timestamp_ms),
177            })
178            .collect();
179
180        let table = Table::new(rows).to_string();
181        println!("{}", table);
182    }
183
184    println!(
185        "\n{} {} proposal(s) found",
186        "ℹ".blue(),
187        proposals.len().to_string().bold()
188    );
189
190    Ok(())
191}

215/// Vote on a proposal, optionally chaining an execute if this vote pushes
216/// the proposal over quorum.
217pub async fn vote(
218    config: &CliConfig,
219    proposal_id: &str,
220    execute: bool,
221    tx_opts: &TxOptions,
222) -> Result<()> {
223    let mut client = HashiClient::new(config).await?;
224
225    let proposal_addr = Address::from_hex(proposal_id)
226        .with_context(|| format!("Invalid proposal ID: {}", proposal_id))?;
227
228    print_info(&format!("Fetching proposal {}...", proposal_id));
229
230    let proposal = client
231        .fetch_proposal(&proposal_addr)
232        .ok_or_else(|| anyhow::anyhow!("Proposal not found: {}", proposal_id))?;
233
234    let proposal_type_str = display::format_proposal_type(&proposal.proposal_type);
235
236    println!("\n{}", "Proposal Details:".bold());
237    println!("  Type: {}", proposal_type_str.cyan());
238
239    if !tx_opts.skip_confirm {
240        prompt_continue("vote on this proposal").await?;
241    }
242
243    print_info("Building vote transaction...");
244
245    // Infer the type tag from the on-chain proposal type
246    let type_arg = get_proposal_type_arg(client.hashi_ids().package_id, &proposal.proposal_type)?;
247    let tx = client.build_vote_transaction(proposal_addr, type_arg);
248
249    print_info(&format!(
250        "Transaction: proposal::vote<{}> on {}",
251        proposal_type_str, proposal_id
252    ));
253
254    let vote_response = execute_or_simulate(&mut client, tx, tx_opts).await?;
255
256    if !execute {
257        return Ok(());
258    }
259
260    // `--execute` was requested. Only meaningful after a real execute (not a
261    // dry-run / missing-keypair).
262    if vote_response.is_none() {
263        return Ok(());
264    }
265
266    // Upgrade proposals require the dedicated upgrade flow — the generic
267    // `<module>::execute` path can't construct an UpgradeTicket.
268    use crate::onchain::types::ProposalType;
269    if matches!(proposal.proposal_type, ProposalType::Upgrade) {
270        print_warning(
271            "--execute is not supported for Upgrade proposals; run the \
272             dedicated upgrade flow once quorum is reached.",
273        );
274        return Ok(());
275    }
276
277    // Re-fetch live vote state to see whether the vote we just submitted
278    // pushed us over quorum. `HashiClient`'s cached scrape is from CLI start,
279    // so this has to be the live `list_dynamic_fields` call.
280    let details = client
281        .fetch_proposal_details(proposal_addr)
282        .await
283        .context("failed to re-fetch proposal state after voting")?;
284    let committee = client
285        .fetch_current_committee()
286        .ok_or_else(|| anyhow::anyhow!("no committee available to compute quorum"))?;
287
288    let total_weight = committee.total_weight();
289    let voted_weight: u64 = details
290        .votes
291        .iter()
292        .map(|voter| {
293            committee
294                .members()
295                .iter()
296                .find(|m| m.validator_address() == *voter)
297                .map(|m| m.weight())
298                .unwrap_or(0)
299        })
300        .sum();
301    let threshold_weight = total_weight
302        .saturating_mul(details.quorum_threshold_bps)
303        .div_ceil(10_000);
304
305    if voted_weight < threshold_weight {
306        print_info(&format!(
307            "Quorum not reached yet ({voted_weight}/{threshold_weight} weight); \
308             skipping --execute."
309        ));
310        return Ok(());
311    }
312
313    print_info(&format!(
314        "Quorum reached ({voted_weight}/{threshold_weight} weight); executing..."
315    ));
316    let execute_tx =
317        client.build_execute_proposal_transaction(proposal_addr, &proposal.proposal_type)?;
318    print_info(&format!(
319        "Transaction: {}::execute on {}",
320        proposal.proposal_type.as_str(),
321        proposal_id
322    ));
323    execute_or_simulate(&mut client, execute_tx, tx_opts).await?;
324    Ok(())
325}

374        .fetch_proposal(&proposal_addr)

378    let proposal_type_str = display::format_proposal_type(proposal_type);

385        );
386    }
387
388    println!("\n{}", "Execute Proposal:".bold());
389    println!("  Type: {}", proposal_type_str.cyan());
390    println!("  ID:   {}", proposal_id);
391
392    if !tx_opts.skip_confirm {
393        prompt_continue("execute this proposal").await?;
394    }
395
396    let tx = client.build_execute_proposal_transaction(proposal_addr, proposal_type)?;
397
398    print_info(&format!(
399        "Transaction: {}::execute on {}",
400        proposal_type.as_str(),
401        proposal_id
402    ));
403
404    execute_or_simulate(&mut client, tx, tx_opts).await?;
405    Ok(())
406}

408/// Create an upgrade proposal.
409///
410/// Exactly one of `digest` or `package_path` must be `Some`. When
411/// `package_path` is provided, the CLI builds the package via `sui move build`
412/// and verifies that its `PACKAGE_VERSION` constant is exactly +1 of the
413/// currently published version (pre-flight check) before submitting the
414/// proposal. The `--digest` path skips that check and is retained only for
415/// callers with a pre-built package.
416pub async fn create_upgrade_proposal(
417    config: &CliConfig,
418    digest: Option<&str>,
419    package_path: Option<&std::path::Path>,
420    sui_binary: &std::path::Path,
421    sui_client_config: Option<&std::path::Path>,
422    metadata: Vec<(String, String)>,
423    tx_opts: &TxOptions,
424) -> Result<()> {
425    let mut client = HashiClient::new(config).await?;
426
427    let digest_bytes = match (digest, package_path) {
428        (Some(d), None) => {
429            print_warning(
430                "--digest skips pre-flight checks (PACKAGE_VERSION = current + 1). \
431                 Prefer --package-path.",
432            );
433            hex::decode(d.trim_start_matches("0x")).context("Invalid digest hex")?
434        }
435        (None, Some(path)) => {
436            let current_version = client.highest_package_version().context(
437                "could not determine current package version from on-chain state; \
438                 is the package deployed?",
439            )?;
440            let expected_version = current_version + 1;
441            print_info(&format!(
442                "Building upgrade package at {} (expecting PACKAGE_VERSION = {expected_version})",
443                path.display()
444            ));
445            let (_compiled, digest) = crate::cli::upgrade::build_upgrade_package(
446                sui_binary,
447                path,
448                sui_client_config,
449                expected_version,
450            )
451            .context("failed to build upgrade package")?;
452            digest
453        }
454        (None, None) => {
455            anyhow::bail!("must provide either --digest or --package-path");
456        }
457        (Some(_), Some(_)) => unreachable!("clap enforces mutual exclusion"),
458    };
459
460    println!("\n{}", "Creating Upgrade Proposal:".bold());
461    println!("  Digest: 0x{}", hex::encode(&digest_bytes));
462    print_metadata(&metadata);
463
464    if !tx_opts.skip_confirm {
465        prompt_continue("create this upgrade proposal").await?;
466    }
467
468    let tx = client.build_create_proposal_transaction(CreateProposalParams::Upgrade {
469        digest: digest_bytes,
470        metadata,
471    });
472
473    print_info("Transaction: upgrade::propose");
474    let response = execute_or_simulate(&mut client, tx, tx_opts).await?;
475    print_created_proposal_id(response.as_ref());
476    Ok(())
477}

512/// Parse a CLI config value string like "u64:1000" or "bool:true" into a ConfigValueParam.
513fn parse_config_value(s: &str) -> Result<hashi_types::move_types::ConfigValue> {
514    use hashi_types::move_types::ConfigValue;
515
516    let (type_prefix, raw) = s
517        .split_once(':')
518        .ok_or_else(|| anyhow::anyhow!("expected type:value format (e.g. u64:1000)"))?;
519
520    match type_prefix {
521        "u64" => Ok(ConfigValue::U64(raw.parse().context("invalid u64")?)),
522        "bool" => Ok(ConfigValue::Bool(raw.parse().context("invalid bool")?)),
523        "string" => Ok(ConfigValue::String(raw.to_string())),
524        "address" => Ok(ConfigValue::Address(
525            raw.parse().context("invalid address")?,
526        )),
527        other => anyhow::bail!(
528            "unknown type prefix '{}' (expected u64, bool, string, address)",
529            other
530        ),
531    }
532}

590fn print_proposal_detailed(
591    proposal: &Proposal,
592    details: Option<&crate::cli::client::ProposalDetails>,
593    committee: Option<&hashi_types::committee::Committee>,
594) {
595    println!("{}", "━".repeat(60).dimmed());
596    println!(
597        "  {} {}",
598        "ID:".bold(),
599        display::format_address_full(&proposal.id).cyan()
600    );
601    println!(
602        "  {} {}",
603        "Type:".bold(),
604        display::format_proposal_type(&proposal.proposal_type).green()
605    );
606    println!(
607        "  {} {}",
608        "Created:".bold(),
609        display::format_timestamp(proposal.timestamp_ms)
610    );
611
612    if let Some(details) = details {
613        println!(
614            "  {} {}",
615            "Creator:".bold(),
616            details.creator.to_hex().dimmed()
617        );
618
619        // Vote tally + quorum progress.
620        let total_weight = committee.map(|c| c.total_weight()).unwrap_or(0);
621        let voted_weight: u64 = details
622            .votes
623            .iter()
624            .map(|voter| {
625                committee
626                    .and_then(|c| c.members().iter().find(|m| m.validator_address() == *voter))
627                    .map(|m| m.weight())
628                    .unwrap_or(0)
629            })
630            .sum();
631        let threshold_weight = total_weight
632            .saturating_mul(details.quorum_threshold_bps)
633            .div_ceil(10_000);
634        let quorum_met = voted_weight >= threshold_weight && total_weight > 0;
635
636        let status = if quorum_met {
637            "QUORUM REACHED".green().bold()
638        } else {
639            format!(
640                "{}/{} weight ({} more needed)",
641                voted_weight,
642                threshold_weight,
643                threshold_weight.saturating_sub(voted_weight)
644            )
645            .yellow()
646        };
647        println!(
648            "  {} {} voter(s) — {} of total weight {} — {}",
649            "Votes:".bold(),
650            details.votes.len().to_string().cyan(),
651            voted_weight.to_string().cyan(),
652            total_weight.to_string().dimmed(),
653            status
654        );
655        println!(
656            "  {} {} bps ({:.2}%)",
657            "Threshold:".bold(),
658            details.quorum_threshold_bps,
659            details.quorum_threshold_bps as f64 / 100.0
660        );
661        if !details.votes.is_empty() {
662            println!("  {}", "Voters:".bold());
663            for voter in &details.votes {
664                println!("    - {}", voter.to_hex().dimmed());
665            }
666        }
667
668        if !details.metadata.contents.is_empty() {
669            println!("  {}", "Metadata:".bold());
670            for entry in &details.metadata.contents {
671                println!("    {}: {}", entry.key.dimmed(), entry.value);
672            }
673        }
674    }
675
676    println!("{}", "━".repeat(60).dimmed());
677}

679/// Pause for user acknowledgement. Press enter to proceed, Ctrl+C to cancel.
680async fn prompt_continue(action: &str) -> Result<()> {
681    use tokio::io::AsyncBufReadExt;
682    use tokio::io::BufReader;
683
684    println!(
685        "\n{}",
686        format!("Press enter to {action}, or Ctrl+C to cancel...").yellow()
687    );
688
689    let mut reader = BufReader::new(tokio::io::stdin());
