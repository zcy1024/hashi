24/// Print metadata if present
25fn print_metadata(metadata: &[(String, String)]) {
26    if !metadata.is_empty() {
27        println!("  {}", "Metadata:".bold());
28        for (key, value) in metadata {
29            println!("    {}: {}", key.dimmed(), value);
30        }
31    }
32}

34/// Print simulation (dry-run) results
35fn print_simulation_result(result: &SimulationResult) {
36    println!("\n{}", "🔍 Dry-run Results:".bold());
37    println!("  {} {}", "Sender:".dimmed(), result.sender.to_hex().cyan());
38    println!(
39        "  {} {} MIST",
40        "Gas Budget:".dimmed(),
41        result.gas_budget.to_string().cyan()
42    );
43    println!(
44        "  {} {} MIST/unit",
45        "Gas Price:".dimmed(),
46        result.gas_price.to_string().cyan()
47    );
48    let max_cost_sui = (result.gas_budget as f64) / 1_000_000_000.0;
49    println!(
50        "  {} {:.6} SUI",
51        "Max Cost:".dimmed(),
52        format!("{:.6}", max_cost_sui).yellow()
53    );
54    println!(
55        "\n  {} Transaction was simulated successfully. Use without --dry-run to execute.",
56        "✓".green()
57    );
58}

60/// Execute or simulate a transaction based on tx_opts
61///
62/// Returns Ok(true) if execution/simulation succeeded, Ok(false) if no keypair configured.
63async fn execute_or_simulate(
64    client: &mut HashiClient,
65    tx: sui_transaction_builder::TransactionBuilder,
66    tx_opts: &TxOptions,
67) -> Result<bool> {
68    if !client.can_execute() {
69        print_warning("Transaction execution requires keypair configuration (--keypair).");
70        return Ok(false);
71    }
72
73    if tx_opts.dry_run {
74        print_info("Simulating transaction (dry-run)...");
75        let result = client.simulate(tx).await?;
76        print_simulation_result(&result);
77    } else {
78        print_info("Executing transaction...");
79        let response = client.execute(tx).await?;
80        let digest = response.transaction().digest();
81        println!(
82            "\n{} Transaction submitted: {}",
83            "✓".green(),
84            digest.to_string().cyan()
85        );
86    }
87
88    Ok(true)
89}

91/// List all active proposals
92pub async fn list_proposals(
93    config: &CliConfig,
94    type_filter: Option<String>,
95    detailed: bool,
96) -> Result<()> {
97    let client = HashiClient::new(config).await?;
98
99    print_info("Fetching proposals...");
100
101    let proposals = client.fetch_proposals();
102
103    if proposals.is_empty() {
104        println!("\n{}", "No active proposals found.".dimmed());
105        return Ok(());
106    }
107
108    // Filter by type if specified
109    let proposals: Vec<_> = if let Some(ref filter) = type_filter {
110        let filter_lower = filter.to_lowercase();
111        proposals
112            .into_iter()
113            .filter(|p| {
114                display::format_proposal_type(&p.proposal_type)
115                    .to_lowercase()
116                    .contains(&filter_lower)
117            })
118            .collect()
119    } else {
120        proposals
121    };
122
123    if proposals.is_empty() {
124        println!(
125            "\n{}",
126            format!(
127                "No proposals found matching type filter: {}",
128                type_filter.unwrap_or_default()
129            )
130            .dimmed()
131        );
132        return Ok(());
133    }
134
135    println!("\n📋 Active Proposals:\n");
136
137    if detailed {
138        for proposal in &proposals {
139            print_proposal_detailed(proposal);
140            println!();
141        }
142    } else {
143        #[derive(Tabled)]
144        struct ProposalRow {
145            #[tabled(rename = "ID")]
146            id: String,
147            #[tabled(rename = "Type")]
148            proposal_type: String,
149            #[tabled(rename = "Created")]
150            timestamp: String,
151        }
152
153        let rows: Vec<ProposalRow> = proposals
154            .iter()
155            .map(|p| ProposalRow {
156                id: display::format_address(&p.id),
157                proposal_type: display::format_proposal_type(&p.proposal_type),
158                timestamp: display::format_timestamp(p.timestamp_ms),
159            })
160            .collect();
161
162        let table = Table::new(rows).to_string();
163        println!("{}", table);
164    }
165
166    println!(
167        "\n{} {} proposal(s) found",
168        "ℹ".blue(),
169        proposals.len().to_string().bold()
170    );
171
172    Ok(())
173}

175/// View details of a specific proposal
176pub async fn view_proposal(config: &CliConfig, proposal_id: &str) -> Result<()> {
177    let client = HashiClient::new(config).await?;
178
179    let proposal_addr = Address::from_hex(proposal_id)
180        .with_context(|| format!("Invalid proposal ID: {}", proposal_id))?;
181
182    print_info(&format!("Fetching proposal {}...", proposal_id));
183
184    let proposal = client
185        .fetch_proposal(&proposal_addr)
186        .ok_or_else(|| anyhow::anyhow!("Proposal not found: {}", proposal_id))?;
187
188    println!();
189    print_proposal_detailed(&proposal);
190
191    Ok(())
192}

194/// Vote on a proposal
195pub async fn vote(config: &CliConfig, proposal_id: &str, tx_opts: &TxOptions) -> Result<()> {
196    let mut client = HashiClient::new(config).await?;
197
198    let proposal_addr = Address::from_hex(proposal_id)
199        .with_context(|| format!("Invalid proposal ID: {}", proposal_id))?;
200
201    print_info(&format!("Fetching proposal {}...", proposal_id));
202
203    let proposal = client
204        .fetch_proposal(&proposal_addr)
205        .ok_or_else(|| anyhow::anyhow!("Proposal not found: {}", proposal_id))?;
206
207    let proposal_type_str = display::format_proposal_type(&proposal.proposal_type);
208
209    println!("\n{}", "Proposal Details:".bold());
210    println!("  Type: {}", proposal_type_str.cyan());
211
212    if !tx_opts.skip_confirm && !confirm_action("vote on this proposal").await? {
213        return Ok(());
214    }
215
216    print_info("Building vote transaction...");
217
218    // Infer the type tag from the on-chain proposal type
219    let type_arg = get_proposal_type_arg(client.hashi_ids().package_id, &proposal.proposal_type)?;
220    let tx = client.build_vote_transaction(proposal_addr, type_arg);
221
222    print_info(&format!(
223        "Transaction: proposal::vote<{}> on {}",
224        proposal_type_str, proposal_id
225    ));
226
227    execute_or_simulate(&mut client, tx, tx_opts).await?;
228    Ok(())
229}

268/// Create an upgrade proposal
269pub async fn create_upgrade_proposal(
270    config: &CliConfig,
271    digest: &str,
272    metadata: Vec<(String, String)>,
273    tx_opts: &TxOptions,
274) -> Result<()> {
275    let digest_bytes =
276        hex::decode(digest.trim_start_matches("0x")).context("Invalid digest hex")?;
277
278    println!("\n{}", "Creating Upgrade Proposal:".bold());
279    println!("  Digest: 0x{}", hex::encode(&digest_bytes));
280    print_metadata(&metadata);
281
282    if !tx_opts.skip_confirm && !confirm_action("create this upgrade proposal").await? {
283        return Ok(());
284    }
285
286    let mut client = HashiClient::new(config).await?;
287    let tx = client.build_create_proposal_transaction(CreateProposalParams::Upgrade {
288        digest: digest_bytes,
289        metadata,
290    });
291
292    print_info("Transaction: upgrade::propose");
293    execute_or_simulate(&mut client, tx, tx_opts).await?;
294    Ok(())
295}

329/// Parse a CLI config value string like "u64:1000" or "bool:true" into a ConfigValueParam.
330fn parse_config_value(s: &str) -> Result<hashi_types::move_types::ConfigValue> {
331    use hashi_types::move_types::ConfigValue;
332
333    let (type_prefix, raw) = s
334        .split_once(':')
335        .ok_or_else(|| anyhow::anyhow!("expected type:value format (e.g. u64:1000)"))?;
336
337    match type_prefix {
338        "u64" => Ok(ConfigValue::U64(raw.parse().context("invalid u64")?)),
339        "bool" => Ok(ConfigValue::Bool(raw.parse().context("invalid bool")?)),
340        "string" => Ok(ConfigValue::String(raw.to_string())),
341        "address" => Ok(ConfigValue::Address(
342            raw.parse().context("invalid address")?,
343        )),
344        other => anyhow::bail!(
345            "unknown type prefix '{}' (expected u64, bool, string, address)",
346            other
347        ),
348    }
349}

405fn print_proposal_detailed(proposal: &Proposal) {
406    println!("{}", "━".repeat(60).dimmed());

425async fn confirm_action(action: &str) -> Result<bool> {
426    use tokio::io::AsyncBufReadExt;
427    use tokio::io::BufReader;
428
429    println!(
430        "\n{}",
431        format!("Are you sure you want to {}? (y/N)", action).yellow()
432    );
433
434    let mut reader = BufReader::new(tokio::io::stdin());
435    let mut input = String::new();
436    reader.read_line(&mut input).await?;
437
438    if input.trim().eq_ignore_ascii_case("y") {
439        Ok(true)
440    } else {
441        print_warning("Cancelled.");
442        Ok(false)
443    }
