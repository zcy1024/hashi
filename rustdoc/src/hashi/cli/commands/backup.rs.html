27/// Save an encrypted backup of the current config, referenced files, and database
28pub fn save(
29    config: &CliConfig,
30    backup_age_pubkey_override: Option<String>,
31    output_dir: &Path,
32) -> Result<()> {
33    let recipient = backup_age_pubkey_override
34        .map(|value| BackupRecipient::from_str(&value))
35        .transpose()?
36        .or_else(|| config.backup_age_pubkey.clone())
37        .ok_or_else(|| {
38            anyhow::anyhow!(
39                "No age public key configured. Pass --backup-age-pubkey or set backup_age_pubkey in the config file."
40            )
41        })?;
42
43    // `backup_file_paths` enforces all the structural invariants we need
44    // (CLI config in use, node config path set, node config loadable, key
45    // file paths actually exist).
46    let files = config.backup_file_paths()?;
47
48    for file in &files {
49        if !file.exists() {
50            anyhow::bail!("Backup input does not exist: {}", file.display());
51        }
52    }
53
54    // The DB path lives in the node config and isn't part of `files`, so we
55    // load the node config a second time here. `backup_file_paths` already
56    // proved this load succeeds, so an error here would have to be a TOCTOU
57    // — fine to surface as-is.
58    let node_config_path = config
59        .node_config_path
60        .as_ref()
61        .expect("backup_file_paths verified node_config_path is set");
62    let node_config = crate::config::Config::load(node_config_path).with_context(|| {
63        format!(
64            "Failed to load node config from {}",
65            node_config_path.display()
66        )
67    })?;
68    let db_path = node_config.db.as_ref().ok_or_else(|| {
69        anyhow::anyhow!(
70            "Node config at {} does not specify a database path",
71            node_config_path.display()
72        )
73    })?;
74
75    // Refuse to silently create an empty DB at a typo'd path. `fjall` would
76    // otherwise happily `create_dir_all` on open and we'd ship a zero-row
77    // backup with no warning.
78    if !db_path
79        .try_exists()
80        .with_context(|| format!("Failed to stat database path {}", db_path.display()))?
81    {
82        anyhow::bail!(
83            "Database path {} does not exist. Fix the `db` field in {} before running backup save.",
84            db_path.display(),
85            node_config_path.display(),
86        );
87    }
88
89    // Open the database — fails with a clear message if the node is running.
90    // `Database::open` preserves `fjall::Error` as the source, so the
91    // downcast below matches on the underlying variant.
92    let db = crate::db::Database::open(db_path).map_err(|e| {
93        if e.downcast_ref::<fjall::Error>()
94            .is_some_and(|fe| matches!(fe, fjall::Error::Locked))
95        {
96            anyhow::anyhow!(
97                "Cannot open database at {}: it is locked by a running hashi node. \
98                 Stop the node before running backup save.",
99                db_path.display()
100            )
101        } else {
102            e.context(format!("Failed to open database at {}", db_path.display()))
103        }
104    })?;
105
106    // Snapshot the database into a subdirectory of a temp dir. `snapshot_to_path`
107    // requires a non-existent destination, so we create the parent tempdir and
108    // point at a yet-to-be-created child.
109    let tmp_parent = tempfile::Builder::new()
110        .prefix("hashi-db-snapshot-")
111        .tempdir()
112        .context("Failed to create temp directory for database snapshot")?;
113    let snapshot_path = tmp_parent.path().join("db");
114    db.snapshot_to_path(&snapshot_path)
115        .context("Failed to snapshot database")?;
116    drop(db);
117
118    info!(source = %db_path.display(), "Database snapshot created");
119
120    fs::create_dir_all(output_dir)
121        .with_context(|| format!("Failed to create output directory {}", output_dir.display()))?;
122
123    let manifest = backup::build_backup_manifest(&files, db_path, &snapshot_path)?;
124
125    info!(
126        file_count = files.len(),
127        %recipient,
128        "Backing up files + database",
129    );
130
131    let encryptor_recipient = build_encryptor_recipient(&recipient)?;
132
133    let output_path = output_dir.join(backup::encrypted_backup_file_name());
134    backup::encrypt_files_to_age_archive(
135        &manifest,
136        &snapshot_path,
137        encryptor_recipient.as_ref(),
138        &output_path,
139    )?;
140    drop(tmp_parent);
141
142    print_success(&format!("Backup completed: {}", output_path.display()));
143
144    Ok(())
145}

147/// Materialize a `BackupRecipient` into a concrete `age::Recipient` trait object
148/// suitable for passing to `Encryptor::with_recipients`.
149///
150/// For plugin recipients, this is where `age-plugin-<name>` is looked up on
151/// `$PATH`, so this call will fail if the plugin binary is not installed.
152fn build_encryptor_recipient(recipient: &BackupRecipient) -> Result<Box<dyn age::Recipient>> {
153    match recipient {
154        BackupRecipient::Native(r) => Ok(Box::new(r.clone())),
155        BackupRecipient::Plugin(r) => {
156            let plugin_name = r.plugin().to_string();
157            let recipient_plugin =
158                plugin::RecipientPluginV1::new(&plugin_name, std::slice::from_ref(r), &[], UiCallbacks)
159                    .with_context(|| {
160                        format!(
161                            "Failed to initialize age plugin '{plugin_name}'. Is `age-plugin-{plugin_name}` installed and on $PATH?"
162                        )
163                    })?;
164            Ok(Box::new(recipient_plugin))
165        }
166    }

169pub fn restore(
170    backup_tarball: &Path,
171    backup_age_identity: &Path,
172    output_dir: &Path,
173    copy_to_original_paths: bool,
174) -> Result<()> {
175    let identities = load_backup_identities(backup_age_identity)?;
176
177    let extract_dir = output_dir.join(backup::extract_dir_name(backup_tarball)?);
178    if extract_dir
179        .try_exists()
180        .with_context(|| format!("Failed to stat {}", extract_dir.display()))?
181    {
182        anyhow::bail!(
183            "Refusing to overwrite existing extract directory: {}",
184            extract_dir.display()
185        );
186    }
187    fs::create_dir_all(output_dir)
188        .with_context(|| format!("Failed to create output directory {}", output_dir.display()))?;
189
190    // Extract into a sibling staging directory and rename into place on
191    // success. A failure mid-extract leaves the staging dir behind (auto-
192    // cleaned on `TempDir` drop) so the user can retry without manual cleanup
193    // and the final `extract_dir` never appears half-populated.
194    let staging = tempfile::Builder::new()
195        .prefix(".hashi-restore-")
196        .tempdir_in(output_dir)
197        .with_context(|| {
198            format!(
199                "Failed to create staging directory in {}",
200                output_dir.display()
201            )
202        })?;
203
204    let input = File::open(backup_tarball)
205        .with_context(|| format!("Failed to open backup tarball {}", backup_tarball.display()))?;
206    let decryptor = Decryptor::new(input)
207        .with_context(|| format!("Failed to parse age header of {}", backup_tarball.display()))?;
208    let mut decrypted = decryptor
209        .decrypt(identities.iter().map(|i| i.as_ref() as &dyn age::Identity))
210        .with_context(|| {
211            format!(
212                "Failed to decrypt {}; is {} the correct age identity?",
213                backup_tarball.display(),
214                backup_age_identity.display()
215            )
216        })?;
217    let mut archive = tar::Archive::new(&mut decrypted);
218    let mut entries = archive.entries()?;
219    let manifest_entry = entries
220        .next()
221        .transpose()?
222        .ok_or_else(|| anyhow::anyhow!("Backup archive is empty"))?;
223    let (manifest, manifest_toml) = backup::read_backup_manifest(manifest_entry)?;
224    backup::restore_backup_entries(entries, staging.path(), &manifest)?;
225    // Manifest is written last so it acts as a marker that extraction finished
226    // successfully. Any earlier failure leaves the staging dir without a
227    // manifest, so partial state can never be confused for a complete restore.
228    backup::write_manifest_to_extract_dir(staging.path(), &manifest_toml)?;
229
230    // Promote the staged dir into its final location atomically. Same-
231    // filesystem rename is required for atomicity, which `tempdir_in` of a
232    // sibling guarantees.
233    let staging_path = staging.keep();
234    fs::rename(&staging_path, &extract_dir).map_err(|e| {
235        let _ = fs::remove_dir_all(&staging_path);
236        anyhow::Error::from(e).context(format!(
237            "Failed to move staged restore into place at {}",
238            extract_dir.display()
239        ))
240    })?;
241
242    if copy_to_original_paths {
243        backup::copy_restored_files_to_original_paths(&extract_dir, &manifest)?;
244        backup::copy_db_snapshot_to_original_path(&extract_dir, &manifest)?;
245    }
246
247    print_success(&format!(
248        "Restore completed from {} into {}",
249        backup_tarball.display(),
250        extract_dir.display()
251    ));
252
253    Ok(())
254}

256fn load_backup_identities(
257    backup_age_identity: &Path,
258) -> Result<Vec<Box<dyn age::Identity + Send + Sync>>> {
259    let path_str = backup_age_identity.to_str().ok_or_else(|| {
260        anyhow::anyhow!(
261            "Age identity path is not valid UTF-8: {}",
262            backup_age_identity.display()
263        )
264    })?;
265    IdentityFile::from_file(path_str.to_string())?
266        .with_callbacks(UiCallbacks)
267        .into_identities()
268        .map_err(Into::into)
269}

353    /// Run `save` with a freshly generated age identity and return everything `restore` needs.
354    fn save_with_fresh_identity(fixture: &TestFixture) -> SavedBackup {
355        let dir = tempfile::Builder::new().tempdir().unwrap();
356
357        let identity = x25519::Identity::generate();
358        let recipient = identity.to_public();

360        save(&fixture.config, Some(recipient.to_string()), dir.path()).unwrap();

362        let tarball = fs::read_dir(dir.path())

585    fn round_trip_preserves_db_contents_after_extraction() {
