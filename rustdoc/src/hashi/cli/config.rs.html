23impl FromStr for BackupRecipient {
24    type Err = anyhow::Error;
25
26    fn from_str(s: &str) -> Result<Self, Self::Err> {
27        if let Ok(recipient) = x25519::Recipient::from_str(s) {
28            return Ok(Self::Native(recipient));
29        }
30        match plugin::Recipient::from_str(s) {
31            Ok(recipient) => Ok(Self::Plugin(recipient)),
32            Err(plugin_err) => anyhow::bail!(
33                "failed to parse age recipient '{s}': not a valid x25519 recipient, and not a valid plugin recipient ({plugin_err})"
34            ),
35        }
36    }
37}
38
39impl fmt::Display for BackupRecipient {
40    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
41        match self {
42            Self::Native(r) => write!(f, "{r}"),
43            Self::Plugin(r) => write!(f, "{r}"),
44        }
45    }
46}
47
48impl fmt::Debug for BackupRecipient {
49    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
50        write!(f, "BackupRecipient({self})")
51    }
52}

128/// CLI overrides for Bitcoin configuration, from command-line flags.
129#[derive(Default)]
130pub struct BitcoinOverrides {
131    pub rpc_url: Option<String>,
132    pub rpc_user: Option<String>,
133    pub rpc_password: Option<String>,
134    pub network: Option<String>,
135    pub private_key: Option<PathBuf>,
136}
137
138impl CliConfig {
139    /// Load configuration from file and CLI overrides.
140    ///
141    /// When no explicit config path is provided, checks for a default config
142    /// file at `.hashi/localnet/hashi-cli.toml` (written by `hashi-localnet start`).
143    pub fn load(
144        config_path: Option<&Path>,
145        sui_rpc_url: Option<String>,
146        package_id: Option<String>,
147        hashi_object_id: Option<String>,
148        keypair_path: Option<PathBuf>,
149        btc_overrides: BitcoinOverrides,
150    ) -> Result<Self> {
151        let default_path = PathBuf::from(DEFAULT_CONFIG_PATH);
152        let mut config = if let Some(path) = config_path {
153            Self::load_from_file(path)?
154        } else if default_path.exists() {
155            Self::load_from_file(&default_path)?
156        } else {
157            Self::default()
158        };
159
160        // Apply CLI overrides (these always win)
161        if let Some(url) = sui_rpc_url {
162            config.sui_rpc_url = url;
163        }
164
165        if let Some(id) = package_id {
166            config.package_id = Some(
167                Address::from_hex(&id).with_context(|| format!("Invalid package ID: {}", id))?,
168            );
169        }
170
171        if let Some(id) = hashi_object_id {
172            config.hashi_object_id = Some(
173                Address::from_hex(&id)
174                    .with_context(|| format!("Invalid Hashi object ID: {}", id))?,
175            );
176        }
177
178        if let Some(path) = keypair_path {
179            config.keypair_path = Some(path);
180        }
181
182        // Apply BTC overrides
183        config.apply_btc_overrides(btc_overrides);
184
185        Ok(config)
186    }
187
188    /// Load configuration from a TOML file
189    fn load_from_file(path: &Path) -> Result<Self> {
190        let contents = std::fs::read_to_string(path)
191            .with_context(|| format!("Failed to read config file: {}", path.display()))?;
192
193        let mut config: Self = toml::from_str(&contents)
194            .with_context(|| format!("Failed to parse config file: {}", path.display()))?;
195        config.loaded_from_path = Some(path.to_path_buf());
196        Ok(config)
197    }
198
199    fn apply_btc_overrides(&mut self, overrides: BitcoinOverrides) {
200        let has_overrides = overrides.rpc_url.is_some()
201            || overrides.rpc_user.is_some()
202            || overrides.rpc_password.is_some()
203            || overrides.network.is_some()
204            || overrides.private_key.is_some();
205
206        if !has_overrides {
207            return;
208        }
209
210        let btc = self.bitcoin.get_or_insert_with(BitcoinConfig::default);
211        if let Some(url) = overrides.rpc_url {
212            btc.rpc_url = Some(url);
213        }
214        if let Some(user) = overrides.rpc_user {
215            btc.rpc_user = Some(user);
216        }
217        if let Some(password) = overrides.rpc_password {
218            btc.rpc_password = Some(password);
219        }
220        if let Some(network) = overrides.network {
221            btc.network = Some(network);
222        }
223        if let Some(key) = overrides.private_key {
224            btc.private_key_path = Some(key);
225        }
226    }

266    /// Validate that required configuration is present
267    pub fn validate(&self) -> Result<()> {
268        if self.package_id.is_none() {
269            anyhow::bail!("package_id is required. Set it via --package-id or in the config file.");
270        }
271        if self.hashi_object_id.is_none() {
272            anyhow::bail!(
273                "hashi_object_id is required. Set it via --hashi-object-id or in the config file."
274            );
275        }
276        Ok(())
277    }

290    /// Load the keypair from the configured path
291    ///
292    /// Returns `None` if no keypair path is configured.
293    /// Returns an error if the path is configured but the keypair cannot be loaded.
294    ///
295    /// Uses the shared `load_ed25519_private_key_from_path` from the hashi crate,
296    /// which supports DER and PEM formats.
297    pub fn load_keypair(&self) -> Result<Option<Ed25519PrivateKey>> {
298        let Some(ref path) = self.keypair_path else {
299            return Ok(None);
300        };
301
302        let pk = load_ed25519_private_key_from_path(path)
303            .with_context(|| format!("Failed to load keypair from {}", path.display()))?;
304
305        Ok(Some(pk))
306    }

308    /// All file paths which must be backed up to enable full node recovery
309    pub fn backup_file_paths(&self) -> Vec<PathBuf> {
310        let mut paths = Vec::new();
311
312        if let Some(path) = &self.loaded_from_path {
313            paths.push(path.clone());
314        }
315
316        if let Some(path) = &self.keypair_path {
317            paths.push(path.clone());
318        }
319
320        if let Some(bitcoin) = &self.bitcoin
321            && let Some(path) = &bitcoin.private_key_path
322        {
323            paths.push(path.clone());
324        }
325
326        paths
327    }

329    /// Get a Bitcoin RPC client from the config, if configured.
330    pub fn btc_rpc_client(&self) -> Result<Option<corepc_client::client_sync::v29::Client>> {
331        let Some(ref btc) = self.bitcoin else {
332            return Ok(None);
333        };
334        let Some(ref url) = btc.rpc_url else {
335            return Ok(None);
336        };
337
338        let auth = match (&btc.rpc_user, &btc.rpc_password) {
339            (Some(user), Some(pass)) => {
340                corepc_client::client_sync::Auth::UserPass(user.clone(), pass.clone())
341            }
342            _ => corepc_client::client_sync::Auth::None,
343        };
344
345        let client = crate::btc_monitor::config::new_rpc_client(url, auth)
346            .with_context(|| format!("Failed to connect to Bitcoin RPC at {}", url))?;
347        Ok(Some(client))
348    }
349
350    /// Require a Bitcoin RPC client, returning an error if not configured.
351    pub fn require_btc_rpc_client(&self) -> Result<corepc_client::client_sync::v29::Client> {
352        self.btc_rpc_client()?.ok_or_else(|| {
353            anyhow::anyhow!(
354                "Bitcoin RPC not configured. Set [bitcoin] in your config file or use --btc-rpc-url"
355            )
356        })
357    }

359    /// Get the path to the config file this was loaded from, for in-place updates.
360    pub fn save_to_file(&self, path: &Path) -> Result<()> {
361        let contents = toml::to_string_pretty(self).context("Failed to serialize config")?;
362        std::fs::write(path, contents)
363            .with_context(|| format!("Failed to write config to {}", path.display()))?;
364        Ok(())
365    }

368/// An age recipient that can be used as the target of a config backup.
369///
370/// Supports both native x25519 recipients (`age1...`) and plugin recipients
371/// (`age1<plugin-name>1...`, e.g. `age1yubikey1...`). Plugin recipients are only
372/// resolved against a plugin binary at encryption time, so storing one in the
373/// config does not require the plugin to be installed.
374#[derive(Clone)]
375pub enum BackupRecipient {
376    Native(x25519::Recipient),
377    Plugin(plugin::Recipient),
378}

380mod optional_age_recipient {
381    use super::BackupRecipient;
382    use serde::Deserialize;
383    use serde::Deserializer;
384    use serde::Serializer;
385    use std::str::FromStr;
386
387    pub fn serialize<S>(value: &Option<BackupRecipient>, serializer: S) -> Result<S::Ok, S::Error>
388    where
389        S: Serializer,
390    {
391        match value {
392            Some(recipient) => serializer.serialize_some(&recipient.to_string()),
393            None => serializer.serialize_none(),
394        }
395    }
396
397    pub fn deserialize<'de, D>(deserializer: D) -> Result<Option<BackupRecipient>, D::Error>
398    where
399        D: Deserializer<'de>,
400    {
401        let value = Option::<String>::deserialize(deserializer)?;
402        value
403            .map(|value| BackupRecipient::from_str(&value).map_err(serde::de::Error::custom))
404            .transpose()
405    }
