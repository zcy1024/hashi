19/// Bitcoin RPC and wallet configuration

38/// CLI Configuration
39#[derive(Debug, Clone, Serialize, Deserialize)]
40pub struct CliConfig {
41    /// Sui RPC endpoint URL
42    #[serde(default = "default_sui_rpc_url")]
43    pub sui_rpc_url: String,
44
45    /// Hashi package ID (original package, used for type resolution)
46    pub package_id: Option<Address>,
47
48    /// Hashi shared object ID
49    pub hashi_object_id: Option<Address>,
50
51    /// Path to the keypair file for signing transactions
52    pub keypair_path: Option<PathBuf>,
53
54    /// Optional: Gas coin object ID to use for transactions
55    pub gas_coin: Option<Address>,
56
57    /// Optional Bitcoin configuration for deposit/withdrawal commands
58    #[serde(default)]
59    pub bitcoin: Option<BitcoinConfig>,
60}

62fn default_sui_rpc_url() -> String {
63    "https://fullnode.mainnet.sui.io:443".to_string()
64}

66/// Default path for the CLI config file written by `hashi-localnet start`.
67const DEFAULT_CONFIG_PATH: &str = ".hashi/localnet/hashi-cli.toml";
68
69impl Default for CliConfig {
70    fn default() -> Self {
71        Self {
72            sui_rpc_url: default_sui_rpc_url(),
73            package_id: None,
74            hashi_object_id: None,
75            keypair_path: None,
76            gas_coin: None,
77            bitcoin: None,
78        }
79    }
80}

82/// CLI overrides for Bitcoin configuration, from command-line flags.
83#[derive(Default)]
84pub struct BitcoinOverrides {
85    pub rpc_url: Option<String>,
86    pub rpc_user: Option<String>,
87    pub rpc_password: Option<String>,
88    pub network: Option<String>,
89    pub private_key: Option<PathBuf>,
90}
91
92impl CliConfig {
93    /// Load configuration from file and CLI overrides.
94    ///
95    /// When no explicit config path is provided, checks for a default config
96    /// file at `.hashi/localnet/hashi-cli.toml` (written by `hashi-localnet start`).
97    pub fn load(
98        config_path: Option<&Path>,
99        sui_rpc_url: Option<String>,
100        package_id: Option<String>,
101        hashi_object_id: Option<String>,
102        keypair_path: Option<PathBuf>,
103        btc_overrides: BitcoinOverrides,
104    ) -> Result<Self> {
105        let default_path = PathBuf::from(DEFAULT_CONFIG_PATH);
106        let mut config = if let Some(path) = config_path {
107            Self::load_from_file(path)?
108        } else if default_path.exists() {
109            Self::load_from_file(&default_path)?
110        } else {
111            Self::default()
112        };
113
114        // Apply CLI overrides (these always win)
115        if let Some(url) = sui_rpc_url {
116            config.sui_rpc_url = url;
117        }
118
119        if let Some(id) = package_id {
120            config.package_id = Some(
121                Address::from_hex(&id).with_context(|| format!("Invalid package ID: {}", id))?,
122            );
123        }
124
125        if let Some(id) = hashi_object_id {
126            config.hashi_object_id = Some(
127                Address::from_hex(&id)
128                    .with_context(|| format!("Invalid Hashi object ID: {}", id))?,
129            );
130        }
131
132        if let Some(path) = keypair_path {
133            config.keypair_path = Some(path);
134        }
135
136        // Apply BTC overrides
137        config.apply_btc_overrides(btc_overrides);
138
139        Ok(config)
140    }
141
142    /// Load configuration from a TOML file
143    fn load_from_file(path: &Path) -> Result<Self> {
144        let contents = std::fs::read_to_string(path)
145            .with_context(|| format!("Failed to read config file: {}", path.display()))?;
146
147        toml::from_str(&contents)
148            .with_context(|| format!("Failed to parse config file: {}", path.display()))
149    }
150
151    fn apply_btc_overrides(&mut self, overrides: BitcoinOverrides) {
152        let has_overrides = overrides.rpc_url.is_some()
153            || overrides.rpc_user.is_some()
154            || overrides.rpc_password.is_some()
155            || overrides.network.is_some()
156            || overrides.private_key.is_some();
157
158        if !has_overrides {
159            return;
160        }
161
162        let btc = self.bitcoin.get_or_insert_with(BitcoinConfig::default);
163        if let Some(url) = overrides.rpc_url {
164            btc.rpc_url = Some(url);
165        }
166        if let Some(user) = overrides.rpc_user {
167            btc.rpc_user = Some(user);
168        }
169        if let Some(password) = overrides.rpc_password {
170            btc.rpc_password = Some(password);
171        }
172        if let Some(network) = overrides.network {
173            btc.network = Some(network);
174        }
175        if let Some(key) = overrides.private_key {
176            btc.private_key_path = Some(key);
177        }
178    }

215    /// Validate that required configuration is present
216    pub fn validate(&self) -> Result<()> {
217        if self.package_id.is_none() {
218            anyhow::bail!("package_id is required. Set it via --package-id or in the config file.");
219        }
220        if self.hashi_object_id.is_none() {
221            anyhow::bail!(
222                "hashi_object_id is required. Set it via --hashi-object-id or in the config file."
223            );
224        }
225        Ok(())
226    }

228    /// Get the package ID, panics if not set
229    pub fn package_id(&self) -> Address {
230        self.package_id.expect("package_id not configured")
231    }
232
233    /// Get the Hashi object ID, panics if not set
234    pub fn hashi_object_id(&self) -> Address {
235        self.hashi_object_id
236            .expect("hashi_object_id not configured")
237    }

239    /// Load the keypair from the configured path
240    ///
241    /// Returns `None` if no keypair path is configured.
242    /// Returns an error if the path is configured but the keypair cannot be loaded.
243    ///
244    /// Uses the shared `load_ed25519_private_key_from_path` from the hashi crate,
245    /// which supports DER and PEM formats.
246    pub fn load_keypair(&self) -> Result<Option<Ed25519PrivateKey>> {
247        let Some(ref path) = self.keypair_path else {
248            return Ok(None);
249        };
250
251        let pk = load_ed25519_private_key_from_path(path)
252            .with_context(|| format!("Failed to load keypair from {}", path.display()))?;
253
254        Ok(Some(pk))
255    }

257    /// Get a Bitcoin RPC client from the config, if configured.
258    pub fn btc_rpc_client(&self) -> Result<Option<bitcoincore_rpc::Client>> {
259        let Some(ref btc) = self.bitcoin else {
260            return Ok(None);
261        };
262        let Some(ref url) = btc.rpc_url else {
263            return Ok(None);
264        };
265
266        let auth = match (&btc.rpc_user, &btc.rpc_password) {
267            (Some(user), Some(pass)) => bitcoincore_rpc::Auth::UserPass(user.clone(), pass.clone()),
268            _ => bitcoincore_rpc::Auth::None,
269        };
270
271        let client = bitcoincore_rpc::Client::new(url, auth)
272            .with_context(|| format!("Failed to connect to Bitcoin RPC at {}", url))?;
273        Ok(Some(client))
274    }
275
276    /// Require a Bitcoin RPC client, returning an error if not configured.
277    pub fn require_btc_rpc_client(&self) -> Result<bitcoincore_rpc::Client> {
278        self.btc_rpc_client()?.ok_or_else(|| {
279            anyhow::anyhow!(
280                "Bitcoin RPC not configured. Set [bitcoin] in your config file or use --btc-rpc-url"
281            )
282        })
283    }

285    /// Get the path to the config file this was loaded from, for in-place updates.
286    pub fn save_to_file(&self, path: &Path) -> Result<()> {
287        let contents = toml::to_string_pretty(self).context("Failed to serialize config")?;
288        std::fs::write(path, contents)
289            .with_context(|| format!("Failed to write config to {}", path.display()))?;
290        Ok(())
291    }
