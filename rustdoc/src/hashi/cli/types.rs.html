4//! Type re-exports and display helpers for the CLI
5//!
6//! This module re-exports types from `crate::onchain::types` and provides
7//! display formatting utilities for CLI output.
8
9// Re-export onchain types used by the CLI
10pub use crate::onchain::types::Proposal;
11
12use crate::onchain::types::ProposalType;
13use sui_sdk_types::Address;
14
15/// Display formatting helpers
16pub mod display {
17    use super::*;
18
19    /// Format an address for display (truncated)
20    pub fn format_address(addr: &Address) -> String {
21        let hex = addr.to_hex();
22        if hex.len() > 16 {
23            format!("{}...{}", &hex[..10], &hex[hex.len() - 6..])
24        } else {
25            hex
26        }
27    }
28
29    /// Format a full address
30    pub fn format_address_full(addr: &Address) -> String {
31        addr.to_hex()
32    }
33
34    /// Format a timestamp in human-readable form
35    pub fn format_timestamp(timestamp_ms: u64) -> String {
36        let ts = jiff::Timestamp::from_millisecond(timestamp_ms as i64)
37            .unwrap_or(jiff::Timestamp::UNIX_EPOCH);
38        let dt = ts.to_zoned(jiff::tz::TimeZone::UTC).datetime();
39        format!(
40            "{:04}-{:02}-{:02} {:02}:{:02} UTC",
41            dt.year(),
42            dt.month() as u8,
43            dt.day(),
44            dt.hour(),
45            dt.minute()
46        )
47    }
48
49    /// Format proposal type for display (from on-chain type)
50    pub fn format_proposal_type(proposal_type: &ProposalType) -> String {
51        match proposal_type {
52            ProposalType::Upgrade => "Upgrade".to_string(),
53            ProposalType::UpdateConfig => "UpdateConfig".to_string(),
54            ProposalType::EnableVersion => "EnableVersion".to_string(),
55            ProposalType::DisableVersion => "DisableVersion".to_string(),
56            ProposalType::Unknown(s) => format!("Unknown({})", s),
57        }
58    }
59}
