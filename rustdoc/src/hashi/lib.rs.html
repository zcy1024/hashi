119    pub fn set_mpc_manager(&self, manager: mpc::MpcManager) {
120        match self.mpc_manager.get() {
121            Some(lock) => {
122                // RwLock::write only fails if poisoned (a thread panicked while holding the lock).
123                // Poisoning indicates a bug, so we propagate the panic rather than recover.
124                *lock.write().unwrap() = manager;
125            }
126            None => {
127                // First-time initialization (e.g. new committee member joining mid-rotation).
128                let _ = self.mpc_manager.set(Arc::new(RwLock::new(manager)));
129            }
130        }
131    }

203        let (onchain_state, service) = onchain::OnchainState::new(
204            self.config.sui_rpc.as_deref().unwrap(),
205            self.config.hashi_ids(),
206            self.config.tls_private_key().ok(),

217    pub fn create_mpc_manager(
218        &self,
219        epoch: u64,
220        protocol_type: mpc::types::ProtocolType,
221    ) -> anyhow::Result<mpc::MpcManager> {
222        let state = self.onchain_state().state();
223        let committee_set = &state.hashi().committees;
224        let session_id = mpc::SessionId::new(self.config.sui_chain_id(), epoch, &protocol_type);
225        let encryption_key = self.config.encryption_private_key()?;
226        self.db
227            .store_encryption_key(epoch, &encryption_key)
228            .map_err(|e| anyhow!("failed to store encryption key: {e}"))?;
229        let signing_key = self
230            .config
231            .protocol_private_key()
232            .ok_or_else(|| anyhow!("no protocol_private_key configured"))?;
233        let store = Box::new(storage::EpochPublicMessagesStore::new(
234            self.db.clone(),
235            epoch,
236        ));
237        let address = self.config.validator_address()?;
238        let chain_id = self.config.sui_chain_id();
239        let batch_size_per_weight =
240            if let Some(override_val) = self.config.test_batch_size_per_weight {
241                assert_test_only_config(
242                    chain_id,
243                    self.config.bitcoin_chain_id(),
244                    "test_batch_size_per_weight",
245                );
246                override_val
247            } else {
248                BATCH_SIZE_PER_WEIGHT
249            };
250        if self.config.test_corrupt_shares_for.is_some() {
251            assert_test_only_config(
252                chain_id,
253                self.config.bitcoin_chain_id(),
254                "test_corrupt_shares_for",
255            );
256        }
257        Ok(mpc::MpcManager::new(
258            address,
259            committee_set,
260            session_id,
261            encryption_key,
262            signing_key,
263            store,
264            WEIGHT_REDUCTION_ALLOWED_DELTA,
265            chain_id,
266            self.config.test_weight_divisor,
267            batch_size_per_weight,
268            self.config.test_corrupt_shares_for,
269        )?)
270    }

272    /// Verify the Sui RPC endpoint is on the expected chain.
273    async fn verify_sui_chain_id(&self) -> anyhow::Result<()> {
274        use sui_rpc::proto::sui::rpc::v2::GetServiceInfoRequest;
275
276        let sui_rpc_url = self.config.sui_rpc.as_deref().unwrap();
277        let mut client = sui_rpc::Client::new(sui_rpc_url)?;
278
279        let service_info = client
280            .ledger_client()
281            .get_service_info(GetServiceInfoRequest::default())
282            .await?
283            .into_inner();
284
285        let rpc_chain_id = service_info.chain_id();
286
287        let expected = self.config.sui_chain_id();
288
289        anyhow::ensure!(
290            rpc_chain_id == expected,
291            "Sui chain ID mismatch: local config has {expected}, \
292             but RPC endpoint reports {rpc_chain_id}"
293        );
294
295        tracing::info!("Sui chain ID verified: {expected}");
296        Ok(())
297    }

299    /// Verify the local config's `bitcoin_chain_id` matches the value stored on-chain.
300    fn verify_bitcoin_chain_id(&self) -> anyhow::Result<()> {
301        use bitcoin::hashes::Hash as _;
302        use std::str::FromStr;
303
304        let onchain_chain_id = self
305            .onchain_state()
306            .state()
307            .hashi()
308            .config
309            .bitcoin_chain_id()
310            .ok_or_else(|| anyhow!("bitcoin_chain_id not found in on-chain config"))?;
311
312        let local_chain_id = self.config.bitcoin_chain_id();
313        let block_hash = btc_monitor::config::BlockHash::from_str(local_chain_id)?;
314        let local_addr = sui_sdk_types::Address::new(*block_hash.as_byte_array());
315
316        anyhow::ensure!(
317            local_addr == onchain_chain_id,
318            "bitcoin chain ID mismatch: local config has {local_chain_id}, \
319             but on-chain value is {onchain_chain_id}"
320        );
321
322        tracing::info!("Bitcoin chain ID verified: {local_chain_id}");
323        Ok(())
324    }

326    /// Verify the connected bitcoind is on the expected network.
327    fn verify_bitcoind_network(&self) -> anyhow::Result<()> {
328        use bitcoincore_rpc::RpcApi;
329
330        let rpc = bitcoincore_rpc::Client::new(
331            self.config.bitcoin_rpc(),
332            self.config.bitcoin_rpc_auth(),
333        )?;
334
335        let info = rpc.get_blockchain_info()?;
336        let expected = self.config.bitcoin_network();
337
338        anyhow::ensure!(
339            info.chain == expected,
340            "bitcoind network mismatch: expected {expected:?}, but node reports {:?}",
341            info.chain
342        );
343
344        tracing::info!("Bitcoind network verified: {expected:?}");
345        Ok(())
346    }

348    fn initialize_btc_monitor(&self) -> anyhow::Result<Service> {
349        self.verify_bitcoind_network()?;
350
351        let monitor_config = crate::btc_monitor::config::MonitorConfig::builder()
352            .network(self.config.bitcoin_network())
353            .confirmation_threshold(self.onchain_state().bitcoin_confirmation_threshold())
354            .start_height(self.config.bitcoin_start_height())
355            .bitcoind_rpc_config(
356                self.config.bitcoin_rpc().to_string(),
357                self.config.bitcoin_rpc_auth(),
358            )
359            .trusted_peers(self.config.bitcoin_trusted_peers()?)
360            .data_dir(
361                self.config
362                    .db
363                    .as_deref()
364                    .expect("Db path is not set")
365                    .join("btc-monitor"),
366            )
367            .build();
368        let (client, service) =
369            crate::btc_monitor::monitor::Monitor::run(monitor_config, self.metrics.clone())
370                .expect("Failed to start BtcMonitor");
371        self.btc_monitor
372            .set(client)
373            .map_err(|_| anyhow!("BtcMonitor already initialized"))?;
374        Ok(service)
375    }

377    pub async fn start(self: Arc<Self>) -> anyhow::Result<Service> {
378        let screener = if let Some(endpoint) = self.config.screener_endpoint() {
379            match grpc::screener_client::ScreenerClient::new(endpoint) {
380                Ok(client) => {
381                    tracing::info!("Screener client configured for {}", client.endpoint());
382                    Some(client)
383                }
384                Err(e) => {
385                    tracing::warn!(
386                        "Failed to configure screener client for {}: {}",
387                        endpoint,
388                        e
389                    );
390                    None
391                }
392            }
393        } else {
394            tracing::warn!("No screener endpoint configured; AML screening will be skipped");
395            None
396        };
397
398        self.metrics
399            .screener_enabled
400            .set(if screener.is_some() { 1 } else { 0 });
401
402        self.screener_client
403            .set(screener)
404            .map_err(|_| anyhow!("Screener client already initialized"))?;
405
406        // Verify Sui RPC is on the expected chain before loading any state.
407        self.verify_sui_chain_id().await?;
408
409        // Initialize
410        let onchain_service = self.initialize_onchain_state().await?;
411
412        // Verify the local bitcoin_chain_id matches the on-chain value.
413        self.verify_bitcoin_chain_id()?;
414
415        // Sweep any SUI in the configured account to AB to enable parallelization of txns
416        sui_tx_executor::sweep_to_address_balance(&mut self.onchain_state().client(), &self.config)
417            .await?;
418
419        // Register validator (if not already registered) and update any stale metadata.
420        match sui_tx_executor::SuiTxExecutor::from_config(&self.config, self.onchain_state())?
421            .execute_register_or_update_validator(&self.config, None)
422            .await
423        {
424            Ok(true) => tracing::info!("Validator registered/updated on-chain"),
425            Ok(false) => tracing::debug!("Validator metadata is already up-to-date"),
426            Err(e) => tracing::warn!("Failed to register/update validator metadata: {e}"),
427        }
428
429        if self.is_in_current_committee() {
430            tracing::info!("Node is in the current committee; MPC service will recover state");
431        } else if self.onchain_state().epoch() == 0
432            && self.onchain_state().current_committee().is_none()
433        {
434            tracing::info!("No initial committee yet; MPC service will handle genesis bootstrap");
435        } else {
436            tracing::info!(
437                "Node is not in the current committee; skipping initial DKG manager creation"
438            );
439        }
440
441        let (mpc_service, mpc_handle) = mpc::MpcService::new(self.clone());
442        self.mpc_handle
443            .set(mpc_handle)
444            .expect("MpcHandle already set");
445
446        let btc_monitor_service = self.initialize_btc_monitor().map_err(|e| {
447            tracing::error!("Failed to initialize BtcMonitor: {e}");
448            e
449        })?;
450
451        // Start services
452        let (_http_addr, http_service) = grpc::HttpService::new(self.clone()).start().await;
453        let leader_service = leader::LeaderService::new(self.clone()).start();
454        let mpc_service = mpc_service.start();
455
456        let service = Service::new()
457            .merge(onchain_service)
458            .merge(btc_monitor_service)
459            .merge(http_service)
460            .merge(leader_service)
461            .merge(mpc_service);
462
463        Ok(service)
464    }

466    pub(crate) fn is_in_current_committee(&self) -> bool {
467        let address = match self.config.validator_address() {
468            Ok(a) => a,
469            Err(_) => return false,
470        };
471        self.onchain_state()
472            .current_committee()
473            .is_some_and(|c| c.index_of(&address).is_some())
474    }

497fn assert_test_only_config(sui_chain_id: &str, bitcoin_chain_id: &str, field_name: &str) {
498    assert!(
499        sui_chain_id != constants::SUI_MAINNET_CHAIN_ID
500            && sui_chain_id != constants::SUI_TESTNET_CHAIN_ID
501            && bitcoin_chain_id == constants::BITCOIN_REGTEST_CHAIN_ID,
502        "{field_name} is only allowed on regtest"
503    );
504}
