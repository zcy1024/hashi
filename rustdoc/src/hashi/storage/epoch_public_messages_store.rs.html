4use std::sync::Arc;
5
6use fastcrypto_tbls::threshold_schnorr::avss;
7use fastcrypto_tbls::threshold_schnorr::batch_avss;
8use sui_sdk_types::Address;
9
10use crate::db::Database;
11use crate::mpc::types::Messages;
12use crate::mpc::types::RotationMessages;
13use crate::storage::PublicMessagesStore;
14
15pub struct EpochPublicMessagesStore {
16    db: Arc<Database>,
17    epoch: u64,
18}
19
20impl EpochPublicMessagesStore {
21    pub fn new(db: Arc<Database>, epoch: u64) -> Self {
22        Self { db, epoch }
23    }
24}
25
26impl PublicMessagesStore for EpochPublicMessagesStore {
27    fn store_dealer_message(
28        &mut self,
29        epoch: u64,
30        dealer: &Address,
31        message: &avss::Message,
32    ) -> anyhow::Result<()> {
33        self.db
34            .store_dealer_message(epoch, dealer, message)
35            .map_err(|e| anyhow::anyhow!("failed to store dealer message: {e}"))
36    }
37
38    fn get_dealer_message(
39        &self,
40        epoch: u64,
41        dealer: &Address,
42    ) -> anyhow::Result<Option<avss::Message>> {
43        self.db
44            .get_dealer_message(epoch, dealer)
45            .map_err(|e| anyhow::anyhow!("failed to get dealer message: {e}"))
46    }
47
48    fn list_all_dealer_messages(&self) -> anyhow::Result<Vec<(Address, Messages)>> {
49        self.db
50            .list_all_dealer_messages(self.epoch)
51            .map(|msgs| {
52                msgs.into_iter()
53                    .map(|(addr, msg)| (addr, Messages::Dkg(msg)))
54                    .collect()
55            })
56            .map_err(|e| anyhow::anyhow!("failed to list dealer messages: {e}"))
57    }
58
59    fn store_rotation_messages(
60        &mut self,
61        epoch: u64,
62        dealer: &Address,
63        messages: &RotationMessages,
64    ) -> anyhow::Result<()> {
65        self.db
66            .store_rotation_messages(epoch, dealer, messages)
67            .map_err(|e| anyhow::anyhow!("failed to store rotation messages: {e}"))
68    }
69
70    fn get_rotation_messages(
71        &self,
72        epoch: u64,
73        dealer: &Address,
74    ) -> anyhow::Result<Option<RotationMessages>> {
75        self.db
76            .get_rotation_messages(epoch, dealer)
77            .map_err(|e| anyhow::anyhow!("failed to get rotation messages: {e}"))
78    }
79
80    fn list_all_rotation_messages(&self) -> anyhow::Result<Vec<(Address, Messages)>> {
81        self.db
82            .list_all_rotation_messages(self.epoch)
83            .map(|msgs| {
84                msgs.into_iter()
85                    .map(|(addr, msg)| (addr, Messages::Rotation(msg)))
86                    .collect()
87            })
88            .map_err(|e| anyhow::anyhow!("failed to list rotation messages: {e}"))
89    }
90
91    fn store_nonce_message(
92        &mut self,
93        epoch: u64,
94        batch_index: u32,
95        dealer: &Address,
96        message: &batch_avss::Message,
97    ) -> anyhow::Result<()> {
98        self.db
99            .store_nonce_message(epoch, batch_index, dealer, message)
100            .map_err(|e| anyhow::anyhow!("failed to store nonce message: {e}"))
101    }
102
103    fn get_nonce_message(
104        &self,
105        epoch: u64,
106        batch_index: u32,
107        dealer: &Address,
108    ) -> anyhow::Result<Option<batch_avss::Message>> {
109        self.db
110            .get_nonce_message(epoch, batch_index, dealer)
111            .map_err(|e| anyhow::anyhow!("failed to get nonce message: {e}"))
112    }
113
114    fn list_nonce_messages(
115        &self,
116        batch_index: u32,
117    ) -> anyhow::Result<Vec<(Address, batch_avss::Message)>> {
118        self.db
119            .list_nonce_messages(self.epoch, batch_index)
120            .map_err(|e| anyhow::anyhow!("failed to list nonce messages: {e}"))
121    }
122}
