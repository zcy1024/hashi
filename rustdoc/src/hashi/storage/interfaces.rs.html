4use anyhow::Result;
5use fastcrypto_tbls::threshold_schnorr::avss;
6use fastcrypto_tbls::threshold_schnorr::batch_avss;
7use sui_sdk_types::Address;
8
9pub use crate::mpc::types::Messages;
10pub use crate::mpc::types::RotationMessages;
11
12pub trait PublicMessagesStore: Send + Sync {
13    /// Store a dealer's DKG message at the given epoch.
14    /// If a message already exists for this dealer, it will be overwritten.
15    fn store_dealer_message(
16        &mut self,
17        epoch: u64,
18        dealer: &Address,
19        message: &avss::Message,
20    ) -> Result<()>;
21
22    /// Retrieve a dealer's DKG message for the given epoch.
23    /// Returns None if no message exists for this dealer.
24    fn get_dealer_message(&self, epoch: u64, dealer: &Address) -> Result<Option<avss::Message>>;
25
26    /// List all stored dealer messages for the current epoch.
27    fn list_all_dealer_messages(&self) -> Result<Vec<(Address, Messages)>>;
28
29    /// Store a dealer's rotation messages at the given epoch.
30    /// If messages already exist for this dealer, they will be overwritten.
31    fn store_rotation_messages(
32        &mut self,
33        epoch: u64,
34        dealer: &Address,
35        messages: &RotationMessages,
36    ) -> Result<()>;
37
38    /// Retrieve a dealer's rotation messages for the given epoch.
39    /// Returns None if no messages exist for this dealer.
40    fn get_rotation_messages(
41        &self,
42        epoch: u64,
43        dealer: &Address,
44    ) -> Result<Option<RotationMessages>>;
45
46    /// List all stored rotation messages for the current epoch.
47    fn list_all_rotation_messages(&self) -> Result<Vec<(Address, Messages)>>;
48
49    /// Store a dealer's nonce message at the given epoch.
50    /// If a message already exists for this dealer and batch, it will be overwritten.
51    fn store_nonce_message(
52        &mut self,
53        epoch: u64,
54        batch_index: u32,
55        dealer: &Address,
56        message: &batch_avss::Message,
57    ) -> Result<()>;
58
59    /// Retrieve a dealer's nonce message for the given epoch and batch.
60    ///
61    ///  Returns None if no message exists for this dealer.
62    fn get_nonce_message(
63        &self,
64        epoch: u64,
65        batch_index: u32,
66        dealer: &Address,
67    ) -> Result<Option<batch_avss::Message>>;
68
69    /// List all nonce messages for the current epoch and given batch.
70    fn list_nonce_messages(&self, batch_index: u32) -> Result<Vec<(Address, batch_avss::Message)>>;
71}
