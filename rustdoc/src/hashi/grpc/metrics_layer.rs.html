4//! Tower callback layer plumbing for HTTP/gRPC metrics.
5//!
6//! This module owns every [`sui_http::middleware::callback`] handler the
7//! crate uses. A single [`RpcMetricsMakeCallbackHandler`] type drives
8//! both the server-side handler attached to the axum router in
9//! `crate::grpc::HttpService::start` and the client-side handler attached
10//! to outbound `tonic_rustls::Channel`s in `crate::grpc::client::Client`;
11//! the per-role behavior is selected via the [`Role`] passed at
12//! construction. The actual prometheus registration lives in
13//! [`crate::metrics::Metrics`]; this module is purely the wiring that
14//! observes traffic and writes into those metrics.
15

28/// Which RFC 9110 role this node is playing on the observed connection.
29///
30/// Selects the value written to the `role` label dimension and the
31/// direction each body travels relative to this node.
32#[derive(Clone, Copy, Debug)]
33pub enum Role {
34    Client,
35    Server,
36}
37
38impl Role {
39    /// Value for the `role` label dimension on every metric this module
40    /// writes.
41    fn label(self) -> &'static str {
42        match self {
43            Role::Client => "client",
44            Role::Server => "server",
45        }
46    }
47
48    /// Direction the request body travels from this node's perspective:
49    /// the client sends the request (outbound), the server receives it
50    /// (inbound).
51    fn request_direction(self) -> Direction {
52        match self {
53            Role::Client => Direction::Outbound,
54            Role::Server => Direction::Inbound,
55        }
56    }
57
58    /// Direction the response body travels from this node's perspective;
59    /// the inverse of [`Self::request_direction`].
60    fn response_direction(self) -> Direction {
61        match self {
62            Role::Client => Direction::Inbound,
63            Role::Server => Direction::Outbound,
64        }
65    }
66}
67
68/// Direction of HTTP body bytes from the perspective of this node.
69/// Selects which byte counter (`bytes_sent_total` vs
70/// `bytes_received_total`) the observed bytes are flushed to, and —
71/// together with [`Role`] — which body (request vs response) the tally
72/// was tracking.
73#[derive(Clone, Copy, Debug)]
74enum Direction {
75    Outbound,
76    Inbound,
77}
78
79/// Accumulates HTTP body bytes and, when the handler is dropped, flushes
80/// the total to both the per-path size histogram and the aggregate
81/// sent/received byte counter.
82///
83/// Flushing on `Drop` covers both the clean end-of-stream path and the
84/// partial-body path (cancellation, body error) in one place, so bytes
85/// that were actually transferred are always observed.
86struct SizeTally {
87    metrics: Arc<Metrics>,
88    path: Cow<'static, str>,
89    role: Role,
90    direction: Direction,
91    bytes: u64,
92}
93
94impl SizeTally {
95    fn new(
96        metrics: Arc<Metrics>,
97        path: Cow<'static, str>,
98        role: Role,
99        direction: Direction,
100    ) -> Self {
101        Self {
102            metrics,
103            path,
104            role,
105            direction,
106            bytes: 0,
107        }
108    }
109
110    fn add(&mut self, n: usize) {
111        self.bytes += n as u64;
112    }
113}
114
115impl Drop for SizeTally {
116    fn drop(&mut self) {
117        let labels = &[self.path.as_ref(), self.role.label()];
118
119        // (role, direction) uniquely determines whether this tally was
120        // tracking the request or the response body:
121        //   client + outbound = request sent; server + inbound = request received
122        //   client + inbound  = response received; server + outbound = response sent
123        let size_histogram = match (self.role, self.direction) {
124            (Role::Client, Direction::Outbound) | (Role::Server, Direction::Inbound) => {
125                &self.metrics.request_size_bytes
126            }
127            (Role::Client, Direction::Inbound) | (Role::Server, Direction::Outbound) => {
128                &self.metrics.response_size_bytes
129            }
130        };
131        size_histogram
132            .with_label_values(labels)
133            .observe(self.bytes as f64);
134
135        let bytes_counter = match self.direction {
136            Direction::Outbound => &self.metrics.bytes_sent_total,
137            Direction::Inbound => &self.metrics.bytes_received_total,
138        };
139        bytes_counter.with_label_values(labels).inc_by(self.bytes);
140    }
141}

143/// Single factory for the per-request handler pair used on both ends of
144/// the wire. Construct with [`Self::server`] for the axum-side
145/// CallbackLayer or [`Self::client`] for the tonic-channel-side
146/// CallbackLayer.
147#[derive(Clone)]
148pub struct RpcMetricsMakeCallbackHandler {
149    metrics: Arc<Metrics>,
150    role: Role,
151}
152
153impl RpcMetricsMakeCallbackHandler {
154    pub fn new(metrics: Arc<Metrics>, role: Role) -> Self {
155        Self { metrics, role }
156    }
157
158    pub fn server(metrics: Arc<Metrics>) -> Self {
159        Self::new(metrics, Role::Server)
160    }
161
162    pub fn client(metrics: Arc<Metrics>) -> Self {
163        Self::new(metrics, Role::Client)
164    }
165}
166
167impl callback::MakeCallbackHandler for RpcMetricsMakeCallbackHandler {
168    type RequestHandler = RequestHandler;
169    type ResponseHandler = ResponseHandler;
170
171    fn make_handler(
172        &self,
173        request: &request::Parts,
174    ) -> (Self::RequestHandler, Self::ResponseHandler) {
175        let metrics = self.metrics.clone();
176        let label = self.role.label();
177        let path = extract_path(request, self.role);
178
179        metrics
180            .inflight_requests
181            .with_label_values(&[path.as_ref(), label])
182            .inc();
183
184        let request_handler = RequestHandler {
185            tally: SizeTally::new(
186                metrics.clone(),
187                path.clone(),
188                self.role,
189                self.role.request_direction(),
190            ),
191        };
192        let response_handler = ResponseHandler {
193            tally: SizeTally::new(
194                metrics.clone(),
195                path.clone(),
196                self.role,
197                self.role.response_direction(),
198            ),
199            metrics,
200            role: self.role,
201            path,
202            start: Instant::now(),
203            counted_response: false,
204        };
205        (request_handler, response_handler)
206    }
207}

209/// Request-side handler. Tallies request body bytes; path-level metrics
210/// live on [`ResponseHandler`].
211pub struct RequestHandler {
212    tally: SizeTally,
213}
214
215impl callback::RequestHandler for RequestHandler {
216    fn on_body_chunk<B: Buf>(&mut self, chunk: &B) {
217        self.tally.add(chunk.remaining());
218    }
219}

221/// Response-side handler. Records path-level RPC metrics
222/// (`inflight_requests`, `request_latency`, the per-status `requests`
223/// counter) labeled by `role`, and tallies response body bytes.
224pub struct ResponseHandler {
225    metrics: Arc<Metrics>,
226    role: Role,
227    path: Cow<'static, str>,
228    start: Instant,
229    // Indicates if we successfully counted the response. In some cases when a request is
230    // prematurely canceled this will remain false.
231    counted_response: bool,
232    tally: SizeTally,
233}
234
235impl callback::ResponseHandler for ResponseHandler {
236    fn on_response(&mut self, response: &response::Parts) {
237        const GRPC_STATUS: http::HeaderName = http::HeaderName::from_static("grpc-status");
238
239        let status = if response
240            .headers
241            .get(&http::header::CONTENT_TYPE)
242            .is_some_and(|content_type| {
243                content_type
244                    .as_bytes()
245                    // check if the content-type starts_with 'application/grpc' in order to
246                    // consider this as a gRPC request. A prefix comparison is done instead of a
247                    // full equality check in order to account for the various types of
248                    // content-types that are considered as gRPC traffic.
249                    .starts_with(tonic::metadata::GRPC_CONTENT_TYPE.as_bytes())
250            }) {
251            let code = response
252                .headers
253                .get(&GRPC_STATUS)
254                .map(http::HeaderValue::as_bytes)
255                .map(tonic::Code::from_bytes)
256                .unwrap_or(tonic::Code::Ok);
257
258            code_as_str(code)
259        } else {
260            response.status.as_str()
261        };
262
263        self.metrics
264            .requests
265            .with_label_values(&[self.path.as_ref(), status, self.role.label()])
266            .inc();
267
268        self.counted_response = true;
269    }
270
271    fn on_service_error<E>(&mut self, _error: &E)
272    where
273        E: std::fmt::Display + 'static,
274    {
275        // Only reachable on the client role: axum requires server
276        // services to be infallible (Error = Infallible), so no service
277        // error can propagate to the server handler. Record under a
278        // distinct status so service-level failures are visible and
279        // aren't lumped in with canceled requests by the `Drop` fallback.
280        self.metrics
281            .requests
282            .with_label_values(&[self.path.as_ref(), "service-error", self.role.label()])
283            .inc();
284        self.counted_response = true;
285    }
286
287    fn on_body_chunk<B: Buf>(&mut self, chunk: &B) {
288        self.tally.add(chunk.remaining());
289    }
290}
291
292impl Drop for ResponseHandler {
293    fn drop(&mut self) {
294        let label = self.role.label();
295
296        self.metrics
297            .inflight_requests
298            .with_label_values(&[self.path.as_ref(), label])
299            .dec();
300
301        let latency = self.start.elapsed().as_secs_f64();
302        self.metrics
303            .request_latency
304            .with_label_values(&[self.path.as_ref(), label])
305            .observe(latency);
306
307        if !self.counted_response {
308            self.metrics
309                .requests
310                .with_label_values(&[self.path.as_ref(), "canceled", label])
311                .inc();
312        }
313    }
314}

316fn extract_path(request: &request::Parts, role: Role) -> Cow<'static, str> {
317    match role {
318        Role::Server => extract_server_path(request),
319        Role::Client => extract_client_path(request),
320    }
321}
322
323fn extract_server_path(request: &request::Parts) -> Cow<'static, str> {
324    if let Some(matched_path) = request.extensions.get::<axum::extract::MatchedPath>() {
325        if request
326            .headers
327            .get(&http::header::CONTENT_TYPE)
328            .is_some_and(|header| {
329                header
330                    .as_bytes()
331                    // check if the content-type starts_with 'application/grpc' in order to
332                    // consider this as a gRPC request. A prefix comparison is done instead of a
333                    // full equality check in order to account for the various types of
334                    // content-types that are considered as gRPC traffic.
335                    .starts_with(tonic::metadata::GRPC_CONTENT_TYPE.as_bytes())
336            })
337        {
338            Cow::Owned(request.uri.path().to_owned())
339        } else {
340            Cow::Owned(matched_path.as_str().to_owned())
341        }
342    } else {
343        Cow::Borrowed("unknown")
344    }
345}

347/// Build the client-side path label from tonic's [`GrpcMethod`]
348/// extension. Tonic-generated clients populate it with the gRPC service
349/// and method name before the request reaches our tower stack, giving us
350/// a stable label that matches the URI path the server sees. Falls back
351/// to the raw URI path for non-tonic callers.
352///
353/// [`GrpcMethod`]: tonic::GrpcMethod
354fn extract_client_path(request: &request::Parts) -> Cow<'static, str> {
355    if let Some(method) = request.extensions.get::<tonic::GrpcMethod<'static>>() {
356        Cow::Owned(format!("/{}/{}", method.service(), method.method()))
357    } else {
358        Cow::Owned(request.uri.path().to_owned())
359    }
360}

362fn code_as_str(code: tonic::Code) -> &'static str {
363    match code {
364        tonic::Code::Ok => "ok",
365        tonic::Code::Cancelled => "canceled",
366        tonic::Code::Unknown => "unknown",
367        tonic::Code::InvalidArgument => "invalid-argument",
368        tonic::Code::DeadlineExceeded => "deadline-exceeded",
369        tonic::Code::NotFound => "not-found",
370        tonic::Code::AlreadyExists => "already-exists",
371        tonic::Code::PermissionDenied => "permission-denied",
372        tonic::Code::ResourceExhausted => "resource-exhausted",
373        tonic::Code::FailedPrecondition => "failed-precondition",
374        tonic::Code::Aborted => "aborted",
375        tonic::Code::OutOfRange => "out-of-range",
376        tonic::Code::Unimplemented => "unimplemented",
377        tonic::Code::Internal => "internal",
378        tonic::Code::Unavailable => "unavailable",
379        tonic::Code::DataLoss => "data-loss",
380        tonic::Code::Unauthenticated => "unauthenticated",
381    }
