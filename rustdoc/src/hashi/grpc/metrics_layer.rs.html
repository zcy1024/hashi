4//! Tower callback layer plumbing for HTTP/gRPC metrics.
5//!
6//! This module owns every [`sui_http::middleware::callback`] handler the
7//! crate uses. A single [`RpcMetricsMakeCallbackHandler`] type drives
8//! both the server-side handler attached to the axum router in
9//! `crate::grpc::HttpService::start` and the client-side handler attached
10//! to outbound `tonic_rustls::Channel`s in `crate::grpc::client::Client`;
11//! the per-role behavior is selected via the [`Role`] passed at
12//! construction. The actual prometheus registration lives in
13//! [`crate::metrics::Metrics`]; this module is purely the wiring that
14//! observes traffic and writes into those metrics.
15

28/// Which RFC 9110 role this node is playing on the observed connection.
29///
30/// Selects the value written to the `role` label dimension and the
31/// direction each body travels relative to this node.
32#[derive(Clone, Copy, Debug)]
33pub enum Role {
34    Client,
35    Server,
36}
37
38impl Role {
39    /// Value for the `role` label dimension on every metric this module
40    /// writes.
41    fn label(self) -> &'static str {
42        match self {
43            Role::Client => "client",
44            Role::Server => "server",
45        }
46    }
47
48    /// Direction the request body travels from this node's perspective:
49    /// the client sends the request (outbound), the server receives it
50    /// (inbound).
51    fn request_direction(self) -> Direction {
52        match self {
53            Role::Client => Direction::Outbound,
54            Role::Server => Direction::Inbound,
55        }
56    }
57
58    /// Direction the response body travels from this node's perspective;
59    /// the inverse of [`Self::request_direction`].
60    fn response_direction(self) -> Direction {
61        match self {
62            Role::Client => Direction::Inbound,
63            Role::Server => Direction::Outbound,
64        }
65    }
66}
67
68/// Direction of HTTP body bytes from the perspective of this node.
69/// Selects which byte counter (`bytes_sent_total` vs
70/// `bytes_received_total`) the observed bytes are flushed to, and —
71/// together with [`Role`] — which body (request vs response) the tally
72/// was tracking.
73#[derive(Clone, Copy, Debug)]
74enum Direction {
75    Outbound,
76    Inbound,
77}
78
79/// Accumulates HTTP body bytes and, when the handler is dropped, flushes
80/// the total to both the per-path size histogram and the aggregate
81/// sent/received byte counter.
82///
83/// Flushing on `Drop` covers both the clean end-of-stream path and the
84/// partial-body path (cancellation, body error) in one place, so bytes
85/// that were actually transferred are always observed.
86struct SizeTally {
87    metrics: Arc<Metrics>,
88    path: Cow<'static, str>,
89    role: Role,
90    direction: Direction,
91    /// MPC protocol label
92    mpc_protocol: Option<&'static str>,
93    bytes: u64,
94}
95
96impl SizeTally {
97    fn new(
98        metrics: Arc<Metrics>,
99        path: Cow<'static, str>,
100        role: Role,
101        direction: Direction,
102        mpc_protocol: Option<&'static str>,
103    ) -> Self {
104        Self {
105            metrics,
106            path,
107            role,
108            direction,
109            mpc_protocol,
110            bytes: 0,
111        }
112    }
113
114    fn add(&mut self, n: usize) {
115        self.bytes += n as u64;
116    }
117}
118
119impl Drop for SizeTally {
120    fn drop(&mut self) {
121        let labels = &[self.path.as_ref(), self.role.label()];
122
123        // (role, direction) uniquely determines whether this tally was
124        // tracking the request or the response body:
125        //   client + outbound = request sent; server + inbound = request received
126        //   client + inbound  = response received; server + outbound = response sent
127        let is_request = matches!(
128            (self.role, self.direction),
129            (Role::Client, Direction::Outbound) | (Role::Server, Direction::Inbound)
130        );
131        let size_histogram = if is_request {
132            &self.metrics.request_size_bytes
133        } else {
134            &self.metrics.response_size_bytes
135        };
136        size_histogram
137            .with_label_values(labels)
138            .observe(self.bytes as f64);
139
140        let bytes_counter = match self.direction {
141            Direction::Outbound => &self.metrics.bytes_sent_total,
142            Direction::Inbound => &self.metrics.bytes_received_total,
143        };
144        bytes_counter.with_label_values(labels).inc_by(self.bytes);
145
146        if let Some(protocol) = self.mpc_protocol {
147            let mpc_labels = &[protocol];
148            let mpc_size_histogram = if is_request {
149                &self.metrics.mpc_request_size_bytes
150            } else {
151                &self.metrics.mpc_response_size_bytes
152            };
153            mpc_size_histogram
154                .with_label_values(mpc_labels)
155                .observe(self.bytes as f64);
156            let mpc_counter = match self.direction {
157                Direction::Outbound => &self.metrics.mpc_bytes_sent_total,
158                Direction::Inbound => &self.metrics.mpc_bytes_received_total,
159            };
160            mpc_counter.with_label_values(mpc_labels).inc_by(self.bytes);
161        }
162    }
163}

165/// Single factory for the per-request handler pair used on both ends of
166/// the wire. Construct with [`Self::server`] for the axum-side
167/// CallbackLayer or [`Self::client`] for the tonic-channel-side
168/// CallbackLayer.
169#[derive(Clone)]
170pub struct RpcMetricsMakeCallbackHandler {
171    metrics: Arc<Metrics>,
172    role: Role,
173}
174
175impl RpcMetricsMakeCallbackHandler {
176    pub fn new(metrics: Arc<Metrics>, role: Role) -> Self {
177        Self { metrics, role }
178    }
179
180    pub fn server(metrics: Arc<Metrics>) -> Self {
181        Self::new(metrics, Role::Server)
182    }
183
184    pub fn client(metrics: Arc<Metrics>) -> Self {
185        Self::new(metrics, Role::Client)
186    }
187}
188
189impl callback::MakeCallbackHandler for RpcMetricsMakeCallbackHandler {
190    type RequestHandler = RequestHandler;
191    type ResponseHandler = ResponseHandler;
192
193    fn make_handler(
194        &self,
195        request: &request::Parts,
196    ) -> (Self::RequestHandler, Self::ResponseHandler) {
197        let metrics = self.metrics.clone();
198        let label = self.role.label();
199        let path = extract_path(request, self.role);
200        let mpc_protocol = extract_mpc_protocol(request);
201
202        metrics
203            .inflight_requests
204            .with_label_values(&[path.as_ref(), label])
205            .inc();
206
207        let request_handler = RequestHandler {
208            tally: SizeTally::new(
209                metrics.clone(),
210                path.clone(),
211                self.role,
212                self.role.request_direction(),
213                mpc_protocol,
214            ),
215        };
216        let response_handler = ResponseHandler {
217            tally: SizeTally::new(
218                metrics.clone(),
219                path.clone(),
220                self.role,
221                self.role.response_direction(),
222                mpc_protocol,
223            ),
224            metrics,
225            role: self.role,
226            path,
227            start: Instant::now(),
228            counted_response: false,
229        };
230        (request_handler, response_handler)
231    }
232}
233
234fn extract_mpc_protocol(request: &request::Parts) -> Option<&'static str> {
235    let value = request
236        .headers
237        .get(crate::grpc::MPC_PROTOCOL_METADATA_KEY)?
238        .to_str()
239        .ok()?;
240    match value {
241        crate::metrics::MPC_LABEL_DKG => Some(crate::metrics::MPC_LABEL_DKG),
242        crate::metrics::MPC_LABEL_KEY_ROTATION => Some(crate::metrics::MPC_LABEL_KEY_ROTATION),
243        crate::metrics::MPC_LABEL_NONCE_GENERATION => {
244            Some(crate::metrics::MPC_LABEL_NONCE_GENERATION)
245        }
246        crate::metrics::MPC_LABEL_SIGNING => Some(crate::metrics::MPC_LABEL_SIGNING),
247        _ => None,
248    }
249}

251/// Request-side handler. Tallies request body bytes; path-level metrics
252/// live on [`ResponseHandler`].
253pub struct RequestHandler {
254    tally: SizeTally,
255}
256
257impl callback::RequestHandler for RequestHandler {
258    fn on_body_chunk<B: Buf>(&mut self, chunk: &B) {
259        self.tally.add(chunk.remaining());
260    }
261}

263/// Response-side handler. Records path-level RPC metrics
264/// (`inflight_requests`, `request_latency`, the per-status `requests`
265/// counter) labeled by `role`, and tallies response body bytes.
266pub struct ResponseHandler {
267    metrics: Arc<Metrics>,
268    role: Role,
269    path: Cow<'static, str>,
270    start: Instant,
271    // Indicates if we successfully counted the response. In some cases when a request is
272    // prematurely canceled this will remain false.
273    counted_response: bool,
274    tally: SizeTally,
275}
276
277impl callback::ResponseHandler for ResponseHandler {
278    fn on_response(&mut self, response: &response::Parts) {
279        const GRPC_STATUS: http::HeaderName = http::HeaderName::from_static("grpc-status");
280
281        let status = if response
282            .headers
283            .get(&http::header::CONTENT_TYPE)
284            .is_some_and(|content_type| {
285                content_type
286                    .as_bytes()
287                    // check if the content-type starts_with 'application/grpc' in order to
288                    // consider this as a gRPC request. A prefix comparison is done instead of a
289                    // full equality check in order to account for the various types of
290                    // content-types that are considered as gRPC traffic.
291                    .starts_with(tonic::metadata::GRPC_CONTENT_TYPE.as_bytes())
292            }) {
293            let code = response
294                .headers
295                .get(&GRPC_STATUS)
296                .map(http::HeaderValue::as_bytes)
297                .map(tonic::Code::from_bytes)
298                .unwrap_or(tonic::Code::Ok);
299
300            code_as_str(code)
301        } else {
302            response.status.as_str()
303        };
304
305        self.metrics
306            .requests
307            .with_label_values(&[self.path.as_ref(), status, self.role.label()])
308            .inc();
309
310        self.counted_response = true;
311    }
312
313    fn on_service_error<E>(&mut self, _error: &E)
314    where
315        E: std::fmt::Display + 'static,
316    {
317        // Only reachable on the client role: axum requires server
318        // services to be infallible (Error = Infallible), so no service
319        // error can propagate to the server handler. Record under a
320        // distinct status so service-level failures are visible and
321        // aren't lumped in with canceled requests by the `Drop` fallback.
322        self.metrics
323            .requests
324            .with_label_values(&[self.path.as_ref(), "service-error", self.role.label()])
325            .inc();
326        self.counted_response = true;
327    }
328
329    fn on_body_chunk<B: Buf>(&mut self, chunk: &B) {
330        self.tally.add(chunk.remaining());
331    }
332}
333
334impl Drop for ResponseHandler {
335    fn drop(&mut self) {
336        let label = self.role.label();
337
338        self.metrics
339            .inflight_requests
340            .with_label_values(&[self.path.as_ref(), label])
341            .dec();
342
343        let latency = self.start.elapsed().as_secs_f64();
344        self.metrics
345            .request_latency
346            .with_label_values(&[self.path.as_ref(), label])
347            .observe(latency);
348
349        if !self.counted_response {
350            self.metrics
351                .requests
352                .with_label_values(&[self.path.as_ref(), "canceled", label])
353                .inc();
354        }
355    }
356}

358fn extract_path(request: &request::Parts, role: Role) -> Cow<'static, str> {
359    match role {
360        Role::Server => extract_server_path(request),
361        Role::Client => extract_client_path(request),
362    }
363}
364
365fn extract_server_path(request: &request::Parts) -> Cow<'static, str> {
366    if let Some(matched_path) = request.extensions.get::<axum::extract::MatchedPath>() {
367        if request
368            .headers
369            .get(&http::header::CONTENT_TYPE)
370            .is_some_and(|header| {
371                header
372                    .as_bytes()
373                    // check if the content-type starts_with 'application/grpc' in order to
374                    // consider this as a gRPC request. A prefix comparison is done instead of a
375                    // full equality check in order to account for the various types of
376                    // content-types that are considered as gRPC traffic.
377                    .starts_with(tonic::metadata::GRPC_CONTENT_TYPE.as_bytes())
378            })
379        {
380            Cow::Owned(request.uri.path().to_owned())
381        } else {
382            Cow::Owned(matched_path.as_str().to_owned())
383        }
384    } else {
385        Cow::Borrowed("unknown")
386    }
387}

389/// Build the client-side path label from tonic's [`GrpcMethod`]
390/// extension. Tonic-generated clients populate it with the gRPC service
391/// and method name before the request reaches our tower stack, giving us
392/// a stable label that matches the URI path the server sees. Falls back
393/// to the raw URI path for non-tonic callers.
394///
395/// [`GrpcMethod`]: tonic::GrpcMethod
396fn extract_client_path(request: &request::Parts) -> Cow<'static, str> {
397    if let Some(method) = request.extensions.get::<tonic::GrpcMethod<'static>>() {
398        Cow::Owned(format!("/{}/{}", method.service(), method.method()))
399    } else {
400        Cow::Owned(request.uri.path().to_owned())
401    }
402}

404fn code_as_str(code: tonic::Code) -> &'static str {
405    match code {
406        tonic::Code::Ok => "ok",
407        tonic::Code::Cancelled => "canceled",
408        tonic::Code::Unknown => "unknown",
409        tonic::Code::InvalidArgument => "invalid-argument",
410        tonic::Code::DeadlineExceeded => "deadline-exceeded",
411        tonic::Code::NotFound => "not-found",
412        tonic::Code::AlreadyExists => "already-exists",
413        tonic::Code::PermissionDenied => "permission-denied",
414        tonic::Code::ResourceExhausted => "resource-exhausted",
415        tonic::Code::FailedPrecondition => "failed-precondition",
416        tonic::Code::Aborted => "aborted",
417        tonic::Code::OutOfRange => "out-of-range",
418        tonic::Code::Unimplemented => "unimplemented",
419        tonic::Code::Internal => "internal",
420        tonic::Code::Unavailable => "unavailable",
421        tonic::Code::DataLoss => "data-loss",
422        tonic::Code::Unauthenticated => "unauthenticated",
423    }
