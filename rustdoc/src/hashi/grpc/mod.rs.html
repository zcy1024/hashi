18/// Wrapper that triggers graceful HTTP server shutdown on drop.
19///
20/// The HTTP server is spawned on a detached tokio task by `sui_http::Builder::serve`.
21/// This guard ensures that `trigger_shutdown` is called when the owning `Service` is
22/// dropped (whether via explicit `Service::shutdown()` or implicit drop), so the server's
23/// accept loop breaks and in-flight connections are drained.
24struct ServerHandleGuard(Arc<ServerHandle>);
25
26impl Drop for ServerHandleGuard {
27    fn drop(&mut self) {
28        self.0.trigger_shutdown();
29    }
30}

42    pub async fn start(self) -> (std::net::SocketAddr, Service) {
43        let router = {
44            let max_decoding_message_size = self.inner.config.grpc_max_decoding_message_size();
45            let bridge_service =
46                hashi_types::proto::bridge_service_server::BridgeServiceServer::new(self.clone())
47                    .max_decoding_message_size(max_decoding_message_size);
48            let mpc_service =
49                hashi_types::proto::mpc_service_server::MpcServiceServer::new(self.clone())
50                    .max_decoding_message_size(max_decoding_message_size);
51
52            let (health_reporter, health_service) = tonic_health::server::health_reporter();
53
54            let mut reflection_v1 = tonic_reflection::server::Builder::configure();
55            let mut reflection_v1alpha = tonic_reflection::server::Builder::configure();
56
57            for file_descriptor_set in [
58                sui_rpc::proto::google::protobuf::FILE_DESCRIPTOR_SET,
59                sui_rpc::proto::google::rpc::FILE_DESCRIPTOR_SET,
60                tonic_health::pb::FILE_DESCRIPTOR_SET,
61                hashi_types::proto::FILE_DESCRIPTOR_SET,
62            ] {
63                reflection_v1 =
64                    reflection_v1.register_encoded_file_descriptor_set(file_descriptor_set);
65                reflection_v1alpha =
66                    reflection_v1alpha.register_encoded_file_descriptor_set(file_descriptor_set);
67            }
68
69            let reflection_v1 = reflection_v1.build_v1().unwrap();
70            let reflection_v1alpha = reflection_v1alpha.build_v1alpha().unwrap();
71
72            fn service_name<S: tonic::server::NamedService>(_service: &S) -> &'static str {
73                S::NAME
74            }
75
76            for service_name in [
77                service_name(&bridge_service),
78                service_name(&mpc_service),
79                service_name(&reflection_v1),
80                service_name(&reflection_v1alpha),
81            ] {
82                health_reporter
83                    .set_service_status(service_name, tonic_health::ServingStatus::Serving)
84                    .await;
85            }
86
87            axum::Router::new()
88                .add_grpc_service(bridge_service)
89                .add_grpc_service(mpc_service)
90                .add_grpc_service(reflection_v1)
91                .add_grpc_service(reflection_v1alpha)
92                .add_grpc_service(health_service)
93        };
94
95        let health_endpoint = axum::Router::new().route("/health", axum::routing::get(health));
96
97        let layers = ServiceBuilder::new()
98            // Add middleware for mapping a request to a known validator
99            .map_request(lookup_validator_middleware(self.inner.clone()))
100            .layer(sui_http::middleware::callback::CallbackLayer::new(
101                crate::metrics::RpcMetricsMakeCallbackHandler::new(self.inner.metrics.clone()),
102            ));
103
104        let router = router.merge(health_endpoint).layer(layers);
105
106        let tls_config =
107            crate::tls::make_server_config(self.inner.config.tls_private_key().unwrap());
108        // let tls_config =
109        //     crate::tls_rpk::make_server_config(self.inner.config.tls_private_key().unwrap());
110
111        let server_handle = Arc::new(
112            sui_http::Builder::new()
113                .tls_config(tls_config)
114                .serve(self.inner.config.listen_address(), router)
115                .unwrap(),
116        );
117        let local_addr = *server_handle.local_addr();
118
119        let guard = ServerHandleGuard(server_handle.clone());
120        let service = Service::new()
121            .spawn_aborting(async move {
122                guard.0.wait_for_shutdown().await;
123                Ok(())
124            })
125            .with_shutdown_signal(async move {
126                server_handle.trigger_shutdown();
127            });
128
129        (local_addr, service)
130    }

161trait RouterExt {
162    /// Add a new grpc service.
163    fn add_grpc_service<S>(self, svc: S) -> Self
164    where
165        S: tower::Service<
166                axum::extract::Request,
167                Response: axum::response::IntoResponse,
168                Error = std::convert::Infallible,
169            > + tonic::server::NamedService
170            + Clone
171            + Send
172            + Sync
173            + 'static,
174        S::Future: Send + 'static;
175}
176
177impl RouterExt for axum::Router {
178    /// Add a new grpc service.
179    fn add_grpc_service<S>(self, svc: S) -> Self
180    where
181        S: tower::Service<
182                axum::extract::Request,
183                Response: axum::response::IntoResponse,
184                Error = std::convert::Infallible,
185            > + tonic::server::NamedService
186            + Clone
187            + Send
188            + Sync
189            + 'static,
190        S::Future: Send + 'static,
191    {
192        self.route_service(&format!("/{}/{{*rest}}", S::NAME), svc)
193    }
194}

196// Given a TLS client cert, pull out the ed25519 public key and map it to a validator
197fn lookup_validator_middleware<B>(
198    hashi: Arc<Hashi>,
199) -> impl Fn(axum::http::Request<B>) -> axum::http::Request<B> + Clone {
200    move |mut request| {
201        if let Some(validator_address) = lookup_validator_address(&hashi, &request) {
202            request.extensions_mut().insert(validator_address);
203        }
204        request
205    }
206}
207
208fn lookup_validator_address<B>(
209    hashi: &Hashi,
210    request: &axum::http::Request<B>,
211) -> Option<sui_sdk_types::Address> {
212    let peer_certs = request.extensions().get::<sui_http::PeerCertificates>()?;
213    let cert = peer_certs.peer_certs().first()?;
214    let tls_public_key = crate::tls::public_key_from_certificate(cert).ok()?;
215    hashi
216        .onchain_state_opt()?
217        .state()
218        .hashi()
219        .committees
220        .lookup_address_by_tls_public_key(&tls_public_key)
221}
