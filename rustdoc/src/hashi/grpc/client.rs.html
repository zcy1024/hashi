28type Result<T, E = tonic::Status> = std::result::Result<T, E>;
29type BoxError = Box<dyn std::error::Error + Send + Sync + 'static>;
30
31const DEFAULT_MAX_DECODING_MESSAGE_SIZE: usize = 4 * 1024 * 1024;
32
33#[derive(Clone, Debug)]
34pub struct Client {
35    uri: http::Uri,
36    channel: Channel,
37    max_decoding_message_size: usize,
38}
39
40impl Client {
41    pub fn new<T>(uri: T, tls_config: rustls::ClientConfig) -> Result<Self>
42    where
43        T: TryInto<http::Uri>,
44        T::Error: Into<BoxError>,
45    {
46        let uri = uri
47            .try_into()
48            .map_err(Into::into)
49            .map_err(tonic::Status::from_error)?;
50        if uri.scheme() != Some(&http::uri::Scheme::HTTPS) {
51            return Err(tonic::Status::from_error(
52                "only https endpoints are supported".into(),
53            ));
54        }
55        let channel = Endpoint::from(uri.clone())
56            .tls_config(tls_config)
57            .map_err(Into::into)
58            .map_err(tonic::Status::from_error)?
59            .connect_timeout(Duration::from_secs(5))
60            .timeout(Duration::from_secs(40))
61            .http2_keep_alive_interval(Duration::from_secs(5))
62            .connect_lazy();
63
64        Ok(Self {
65            uri,
66            channel,
67            max_decoding_message_size: DEFAULT_MAX_DECODING_MESSAGE_SIZE,
68        })
69    }
70
71    pub fn max_decoding_message_size(mut self, limit: usize) -> Self {
72        self.max_decoding_message_size = limit;
73        self
74    }
75
76    pub fn new_no_auth<T>(uri: T) -> Result<Self>
77    where
78        T: TryInto<http::Uri>,
79        T::Error: Into<BoxError>,
80    {
81        Self::new(uri, make_client_config_no_verification())
82    }
83
84    pub fn uri(&self) -> &http::Uri {
85        &self.uri
86    }
87
88    pub fn bridge_service_client(&self) -> BridgeServiceClient<Channel> {
89        BridgeServiceClient::new(self.channel.clone())
90            .max_decoding_message_size(self.max_decoding_message_size)
91    }
92
93    pub fn mpc_service_client(&self) -> MpcServiceClient<Channel> {
94        MpcServiceClient::new(self.channel.clone())
95            .max_decoding_message_size(self.max_decoding_message_size)
96    }
97
98    pub async fn get_service_info(&self) -> Result<Response<GetServiceInfoResponse>> {
99        self.bridge_service_client()
100            .get_service_info(GetServiceInfoRequest::default())
101            .await
102    }
103
104    pub async fn send_messages(
105        &self,
106        epoch: u64,
107        request: &SendMessagesRequest,
108    ) -> Result<SendMessagesResponse> {
109        let proto_request = request.to_proto(epoch);
110        let response = self
111            .mpc_service_client()
112            .send_messages(proto_request)
113            .await?;
114        SendMessagesResponse::try_from(response.get_ref())
115            .map_err(|e| tonic::Status::internal(e.to_string()))
116    }

165    pub async fn get_reconfig_completion_signature(&self, epoch: u64) -> Result<Option<Vec<u8>>> {
166        let request = GetReconfigCompletionSignatureRequest { epoch: Some(epoch) };
167        let response = self
168            .mpc_service_client()
169            .get_reconfig_completion_signature(request)
170            .await?;
171        Ok(response.into_inner().signature.map(|b| b.to_vec()))
172    }
