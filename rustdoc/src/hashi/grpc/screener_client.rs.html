4use std::str::FromStr;
5use std::time::Duration;
6
7use hashi_types::proto::screener::ApproveRequest;
8use hashi_types::proto::screener::TransactionType;
9use hashi_types::proto::screener::screener_service_client::ScreenerServiceClient;
10use tonic::transport::Channel;
11use tonic::transport::Endpoint;
12
13type BoxError = Box<dyn std::error::Error + Send + Sync + 'static>;
14
15#[derive(Clone, Debug)]
16pub struct ScreenerClient {
17    endpoint: String,
18    channel: Channel,
19}
20
21impl ScreenerClient {
22    pub fn new(endpoint: &str) -> Result<Self, tonic::Status> {
23        let channel = Endpoint::from_shared(endpoint.to_string())
24            .map_err(Into::<BoxError>::into)
25            .map_err(tonic::Status::from_error)?
26            .connect_timeout(Duration::from_secs(5))
27            .http2_keep_alive_interval(Duration::from_secs(5))
28            .connect_lazy();
29        Ok(Self {
30            endpoint: endpoint.to_string(),
31            channel,
32        })
33    }
34
35    pub fn endpoint(&self) -> &str {
36        &self.endpoint
37    }
38
39    pub fn screener_service_client(&self) -> ScreenerServiceClient<Channel> {
40        ScreenerServiceClient::new(self.channel.clone())
41    }
42
43    pub async fn approve_deposit(
44        &self,
45        source_tx_hash: &str,
46        destination_address: &str,
47        source_chain_id: &str,
48        destination_chain_id: &str,
49    ) -> Result<bool, tonic::Status> {
50        let request = ApproveRequest {
51            source_transaction_hash: source_tx_hash.to_string(),
52            destination_address: destination_address.to_string(),
53            source_chain_id: btc_chain_id_to_caip2(source_chain_id)?,
54            destination_chain_id: sui_chain_id_to_caip2(destination_chain_id)?,
55            transaction_type: TransactionType::Deposit.into(),
56        };
57        let response = self.screener_service_client().approve(request).await?;
58        Ok(response.into_inner().approved)
59    }
60
61    pub async fn approve_withdrawal(
62        &self,
63        source_tx_hash: &str,
64        destination_address: &str,
65        source_chain_id: &str,
66        destination_chain_id: &str,
67    ) -> Result<bool, tonic::Status> {
68        let request = ApproveRequest {
69            source_transaction_hash: source_tx_hash.to_string(),
70            destination_address: destination_address.to_string(),
71            source_chain_id: sui_chain_id_to_caip2(source_chain_id)?,
72            destination_chain_id: btc_chain_id_to_caip2(destination_chain_id)?,
73            transaction_type: TransactionType::Withdrawal.into(),
74        };
75        let response = self.screener_service_client().approve(request).await?;
76        Ok(response.into_inner().approved)
77    }
78}

80/// Convert a full Bitcoin genesis block hash to CAIP-2 (BIP-122) chain ID.
81/// Format: "bip122:" + first 16 bytes of the hex encoded genesis block hash.
82fn btc_chain_id_to_caip2(genesis_block_hash: &str) -> Result<String, tonic::Status> {
83    if genesis_block_hash.len() < 32 {
84        return Err(tonic::Status::internal(format!(
85            "invalid bitcoin chain id: expected at least 32 hex characters, got {}",
86            genesis_block_hash.len()
87        )));
88    }
89    Ok(format!("bip122:{}", &genesis_block_hash[..32]))
90}
91
92/// Convert a full Sui genesis checkpoint digest (base58) to CAIP-2 chain ID.
93/// Format: "sui:" + first 4 bytes of decoded digest as hex.
94fn sui_chain_id_to_caip2(genesis_checkpoint_digest: &str) -> Result<String, tonic::Status> {
95    let digest = sui_sdk_types::Digest::from_str(genesis_checkpoint_digest)
96        .map_err(|e| tonic::Status::internal(format!("invalid sui chain id: {e}")))?;
97    Ok(format!("sui:{}", hex::encode(&digest.inner()[..4])))
98}
