14#[derive(Debug, Clone)]
15pub struct MonitorConfig {
16    /// Bitcoin network to connect to
17    pub network: Network,
18
19    /// Number of confirmations required for a transaction to be considered canonical
20    pub confirmation_threshold: u32,
21
22    /// Peers for P2P connections, identified by hostname (or IP) and port.
23    /// Re-resolved via DNS on each connection attempt, so IP changes
24    /// (e.g., Kubernetes pod rotation) are followed automatically.
25    pub dns_peers: Vec<kyoto::DnsPeer>,
26
27    /// Starting block height for synchronization
28    pub start_height: u32,
29
30    /// bitcoind JSON-RPC server URL to connect to
31    pub bitcoind_rpc_url: String,
32
33    /// bitcoind JSON-RPC server auth config
34    pub bitcoind_rpc_auth: corepc_client::client_sync::Auth,
35
36    /// Directory for storing BTC light client data
37    pub data_dir: Option<PathBuf>,
38}
39
40impl Default for MonitorConfig {
41    fn default() -> Self {
42        Self {
43            network: Network::Bitcoin,
44            confirmation_threshold: 6,
45            dns_peers: Vec::new(),
46            start_height: 800_000,
47            bitcoind_rpc_url: "http://localhost:8332".to_string(),
48            bitcoind_rpc_auth: corepc_client::client_sync::Auth::None,
49            data_dir: None,
50        }
51    }
52}

54impl MonitorConfig {

61/// Builder for constructing monitor configuration.
62#[derive(Debug, Default)]
63pub struct MonitorConfigBuilder {
64    network: Option<Network>,
65    confirmation_threshold: Option<u32>,
66    dns_peers: Vec<kyoto::DnsPeer>,
67    start_height: u32,
68    bitcoind_rpc_url: Option<String>,
69    bitcoind_rpc_auth: Option<corepc_client::client_sync::Auth>,
70    data_dir: Option<PathBuf>,
71}
72
73impl MonitorConfigBuilder {
74    /// Set the Bitcoin network.
75    pub fn network(mut self, network: Network) -> Self {
76        self.network = Some(network);
77        self
78    }
79
80    /// Set the confirmation threshold for deposits.
81    pub fn confirmation_threshold(mut self, confirmations: u32) -> Self {
82        self.confirmation_threshold = Some(confirmations);
83        self
84    }
85
86    /// Set peers for P2P connections. Accepts hostnames or IPs with port.
87    /// Hostnames are re-resolved via DNS on each connection attempt.
88    pub fn dns_peers(mut self, peers: Vec<kyoto::DnsPeer>) -> Self {
89        self.dns_peers = peers;
90        self
91    }
92
93    /// Set the starting block height for synchronization.
94    pub fn start_height(mut self, height: u32) -> Self {
95        self.start_height = height;
96        self
97    }
98
99    /// Set the bitcoind JSON-RPC server config.
100    pub fn bitcoind_rpc_config(
101        mut self,
102        url: String,
103        auth: corepc_client::client_sync::Auth,
104    ) -> Self {
105        self.bitcoind_rpc_url = Some(url);
106        self.bitcoind_rpc_auth = Some(auth);
107        self
108    }
109
110    /// Set the directory for storing BTC light client data.
111    pub fn data_dir(mut self, path: impl Into<PathBuf>) -> Self {
112        self.data_dir = Some(path.into());
113        self
114    }
115
116    pub fn build(self) -> MonitorConfig {
117        let default = MonitorConfig::default();
118
119        MonitorConfig {
120            network: self.network.unwrap_or(default.network),
121            confirmation_threshold: self
122                .confirmation_threshold
123                .unwrap_or(default.confirmation_threshold),
124            dns_peers: self.dns_peers,
125            start_height: self.start_height,
126            bitcoind_rpc_url: self.bitcoind_rpc_url.unwrap_or(default.bitcoind_rpc_url),
127            bitcoind_rpc_auth: self.bitcoind_rpc_auth.unwrap_or(default.bitcoind_rpc_auth),
128            data_dir: self.data_dir,
129        }
130    }
131}

133/// Wrapper around corepc_client::client_sync::Auth that we can serialize/deserialize from configs
134#[derive(Clone, Debug, serde_derive::Deserialize, serde_derive::Serialize)]
135pub enum BtcRpcAuth {
136    None,
137    UserPass(String, String),
138    CookieFile(PathBuf),
139}
140
141impl BtcRpcAuth {
142    pub fn to_corepc_auth(&self) -> corepc_client::client_sync::Auth {
143        match self {
144            BtcRpcAuth::None => corepc_client::client_sync::Auth::None,
145            BtcRpcAuth::UserPass(user, pass) => {
146                corepc_client::client_sync::Auth::UserPass(user.clone(), pass.clone())
147            }
148            BtcRpcAuth::CookieFile(path) => {
149                corepc_client::client_sync::Auth::CookieFile(path.clone())
150            }
151        }
152    }
153}

155/// Create a `corepc_client` RPC client, handling the `Auth::None` case
156/// (where `Client::new_with_auth` would error with `MissingUserPassword`).
157pub fn new_rpc_client(
158    url: &str,
159    auth: corepc_client::client_sync::Auth,
160) -> corepc_client::client_sync::Result<corepc_client::client_sync::v29::Client> {
161    match auth {
162        corepc_client::client_sync::Auth::None => {
163            Ok(corepc_client::client_sync::v29::Client::new(url))
164        }
165        auth => corepc_client::client_sync::v29::Client::new_with_auth(url, auth),
166    }
167}

169/// Parse a human-readable network name into a [`Network`].
170///
171/// Recognised values: `"mainnet"`, `"testnet4"`, `"regtest"`.
172/// Returns [`Network::Regtest`] when `name` is `None`.
173/// Returns an error for unrecognised network names.
174pub fn parse_btc_network(name: Option<&str>) -> anyhow::Result<Network> {
175    match name {
176        Some("mainnet") => Ok(Network::Bitcoin),
177        Some("testnet4") => Ok(Network::Testnet4),
178        Some("signet") => Ok(Network::Signet),
179        Some("regtest") | None => Ok(Network::Regtest),
180        Some(other) => anyhow::bail!(
181            "Unknown BTC network '{}'. Use mainnet, testnet4, signet, or regtest",
182            other
183        ),
184    }
185}
186
187pub fn network_from_chain_id(chain_id: &str) -> Option<Network> {
188    let hash = BlockHash::from_str(chain_id).ok()?;
189
190    [
191        Network::Bitcoin,
192        Network::Testnet4,
193        Network::Signet,
194        Network::Regtest,
195    ]
196    .into_iter()
197    .find(|&net| genesis_block(net).block_hash() == hash)
198}
