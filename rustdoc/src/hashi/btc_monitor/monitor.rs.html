53/// Monitor loop that tracks the state of the Bitcoin chain.
54///
55/// Client provides functions for querying for specific transactions,
56/// fee information, and transaction submission.
57pub struct Monitor {
58    config: MonitorConfig,
59    metrics: Arc<Metrics>,
60    bitcoind_rpc: Arc<bitcoincore_rpc::Client>,
61    client_tx: tokio::sync::mpsc::Sender<MonitorMessage>,
62    requester: kyoto::Requester,
63    tip: Option<HeaderCheckpoint>,
64    pending_deposits: Vec<PendingDeposit>,
65    pending_deposit_workers: JoinSet<()>,
66}

68impl Monitor {
69    fn build_kyoto_node(config: &MonitorConfig) -> Result<(NodeDefault, kyoto::Client)> {
70        let mut node_builder = kyoto::NodeBuilder::new(config.network)
71            .add_peers(config.trusted_peers.iter().cloned())
72            // Prevent Kyoto from storing additional peers via GetAddr.
73            .peer_db_size(kyoto::PeerStoreSizeConfig::Limit(0))
74            // TODO: should we set this higher than default?
75            // .required_peers(num_peers)
76            // Need a dummy script to prevent default match on every single block.
77            // TODO: Remove once commit
78            // https://github.com/rust-bitcoin/rust-bitcoin/commit/e7d992a5ff75807ec454655d112a671294a101dd
79            // is available in a released version of the bitcoin crate.
80            .add_scripts(vec![bitcoin::ScriptBuf::new()])
81            .after_checkpoint(kyoto::HeaderCheckpoint::closest_checkpoint_below_height(
82                config.start_height,
83                config.network,
84            ));
85        if let Some(data_dir) = &config.data_dir {
86            node_builder = node_builder.data_dir(data_dir.clone());
87        }
88        Ok(node_builder.build()?)
89    }

91    /// Run a BTC monitor with the given configuration.
92    /// Returns the client for interacting with the monitor and a Service for lifecycle management.
93    pub fn run(config: MonitorConfig, metrics: Arc<Metrics>) -> Result<(MonitorClient, Service)> {
94        let bitcoind_rpc = bitcoincore_rpc::Client::new(
95            config.bitcoind_rpc_url.as_str(),
96            config.bitcoind_rpc_auth.clone(),
97        )?;
98
99        let (client_tx, mut client_rx) = tokio::sync::mpsc::channel(100);
100
101        let service = Service::new().spawn_aborting({
102            let client_tx = client_tx.clone();
103            async move {
104                let bitcoind_rpc = Arc::new(bitcoind_rpc);
105
106                // Build initial Kyoto node.
107                let (kyoto_node, kyoto_client) = Self::build_kyoto_node(&config)?;
108
109                let mut monitor = Monitor {
110                    config,
111                    metrics,
112                    bitcoind_rpc,
113                    requester: kyoto_client.requester.clone(),
114                    client_tx,
115                    tip: None,
116                    pending_deposits: vec![],
117                    pending_deposit_workers: JoinSet::new(),
118                };
119
120                monitor
121                    .run_with_supervision(kyoto_node, kyoto_client, &mut client_rx)
122                    .await
123            }
124        });
125
126        Ok((MonitorClient { tx: client_tx }, service))
127    }

129    /// Run the monitor with automatic Kyoto restart on connectivity loss.
130    async fn run_with_supervision(
131        &mut self,
132        kyoto_node: NodeDefault,
133        kyoto_client: kyoto::Client,
134        client_rx: &mut tokio::sync::mpsc::Receiver<MonitorMessage>,
135    ) -> Result<()> {
136        let mut current_node = kyoto_node;
137        let mut current_client = kyoto_client;
138
139        loop {
140            info!(
141                "Starting Bitcoin monitor for network: {:?}",
142                self.config.network
143            );
144
145            // Spawn the Kyoto node as a background task. Node::run() takes
146            // ownership, so we move it in and get a JoinHandle back.
147            let kyoto_handle = tokio::spawn(async move { current_node.run().await });
148
149            let result = self.run_event_loop(&mut current_client, client_rx).await;
150
151            // Abort the Kyoto node task regardless of exit reason.
152            kyoto_handle.abort();
153
154            match result {
155                KyotoEventLoopExit::ConnectivityLost => {
156                    warn!(
157                        "Lost connectivity to Bitcoin peers after {KYOTO_MAX_CONSECUTIVE_FAILURES} \
158                         consecutive failures. Restarting Kyoto node..."
159                    );
160
161                    self.metrics.kyoto_restarts.inc();
162                    self.metrics.kyoto_connected_peers.set(0);
163                    self.metrics.kyoto_synced.set(0);
164                    self.metrics.kyoto_consecutive_failures.set(0);
165
166                    // Wait before restarting to avoid tight restart loops.
167                    tokio::time::sleep(KYOTO_RESTART_DELAY).await;
168
169                    // Build a fresh Kyoto node with the trusted peers re-added
170                    // to the whitelist.
171                    let (new_node, new_client) = Self::build_kyoto_node(&self.config)?;
172                    current_node = new_node;
173                    current_client = new_client;
174                    self.requester = current_client.requester.clone();
175
176                    info!("Kyoto node rebuilt, resuming monitor");
177                }
178                KyotoEventLoopExit::Shutdown => {
179                    info!("Bitcoin monitor stopped");
180                    return Ok(());
181                }
182            }
183        }
184    }

186    /// Map a Kyoto `Warning` variant to a short label for metrics.
187    fn warning_label(warning: &Warning) -> &'static str {
188        match warning {
189            Warning::NeedConnections { .. } => "need_connections",
190            Warning::PeerTimedOut => "peer_timed_out",
191            Warning::CouldNotConnect => "could_not_connect",
192            Warning::NoCompactFilters => "no_compact_filters",
193            Warning::PotentialStaleTip => "potential_stale_tip",
194            Warning::UnsolicitedMessage => "unsolicited_message",
195            Warning::InvalidStartHeight => "invalid_start_height",
196            Warning::CorruptedHeaders => "corrupted_headers",
197            Warning::TransactionRejected { .. } => "transaction_rejected",
198            Warning::FailedPersistence { .. } => "failed_persistence",
199            Warning::EvaluatingFork => "evaluating_fork",
200            Warning::EmptyPeerDatabase => "empty_peer_database",
201            Warning::UnexpectedSyncError { .. } => "unexpected_sync_error",
202            Warning::ChannelDropped => "channel_dropped",
203        }
204    }

206    /// Run the main event loop, returning the reason it exited.
207    async fn run_event_loop(
208        &mut self,
209        kyoto_client: &mut kyoto::Client,
210        client_rx: &mut tokio::sync::mpsc::Receiver<MonitorMessage>,
211    ) -> KyotoEventLoopExit {
212        let mut consecutive_failures: u32 = 0;
213        let mut required_peers: usize = 0;
214
215        loop {
216            tokio::select! {
217                Some(event) = kyoto_client.event_rx.recv() => {
218                    self.process_kyoto_event(event);
219                }
220                Some(msg) = client_rx.recv() => {
221                    self.process_client_message(msg);
222                }
223                Some(msg) = kyoto_client.log_rx.recv() => {
224                    debug!("Kyoto log: {msg}");
225                }
226                Some(msg) = kyoto_client.info_rx.recv() => {
227                    info!("Kyoto: {msg}");
228                    // Reset failure counter on any info message (successful
229                    // activity like syncing, handshakes, etc).
230                    consecutive_failures = 0;
231                    self.metrics.kyoto_consecutive_failures.set(0);
232
233                    // Parse info messages for metrics where possible.
234                    Self::update_info_metrics(&self.metrics, &msg, required_peers);
235                }
236                Some(warning) = kyoto_client.warn_rx.recv() => {
237                    warn!("Kyoto: {warning}");
238                    self.metrics.kyoto_warnings.with_label_values(&[Self::warning_label(&warning)]).inc();
239
240                    // Track connected peer count from NeedConnections
241                    if let Warning::NeedConnections { connected, required } = &warning {
242                        self.metrics.kyoto_connected_peers.set(*connected as i64);
243                        required_peers = *required;
244                    }
245
246                    let is_connectivity_failure = matches!(
247                        warning,
248                        Warning::CouldNotConnect
249                        | Warning::PeerTimedOut
250                        | Warning::NeedConnections { connected: 0, .. }
251                    );
252                    if is_connectivity_failure {
253                        consecutive_failures += 1;
254                        self.metrics.kyoto_consecutive_failures.set(consecutive_failures as i64);
255                        if consecutive_failures >= KYOTO_MAX_CONSECUTIVE_FAILURES {
256                            return KyotoEventLoopExit::ConnectivityLost;
257                        }
258                    }
259                }
260                Some(join_result) = self.pending_deposit_workers.join_next(), if !self.pending_deposit_workers.is_empty() => {
261                    if let Err(e) = join_result {
262                        error!("Pending deposit worker task failed: {e}");
263                    }
264                }
265                else => {
266                    return KyotoEventLoopExit::Shutdown;
267                }
268            }
269        }
270    }

272    /// Extract metrics from Kyoto info messages.
273    fn update_info_metrics(metrics: &Metrics, msg: &kyoto::Info, required_peers: usize) {
274        match msg {
275            kyoto::Info::ConnectionsMet => {
276                metrics.kyoto_connected_peers.set(required_peers as i64);
277            }
278            kyoto::Info::Progress(progress) => {
279                metrics
280                    .kyoto_sync_percent
281                    .set(progress.percentage_complete() as i64);
282            }
283            kyoto::Info::NewChainHeight(height) => {
284                metrics.kyoto_best_height.set(*height as i64);
285            }
286            _ => {}
287        }
288    }

290    fn process_kyoto_event(&mut self, event: kyoto::Event) {
291        match event {
292            kyoto::Event::Block(block) => self.process_block(block),
293            kyoto::Event::Synced(sync_update) => self.process_synced(sync_update),
294            kyoto::Event::BlocksDisconnected {
295                accepted,
296                disconnected,
297            } => self.process_blocks_disconnected(accepted, disconnected),
298        }
299    }
300
301    fn process_block(&mut self, block: kyoto::IndexedBlock) {
302        info!(
303            "Got block {} at height {} with {} transactions",
304            block.block.block_hash(),
305            block.height,
306            block.block.txdata.len()
307        );
308        self.metrics.kyoto_blocks_received.inc();
309        self.metrics.kyoto_best_height.set(block.height as i64);
310    }
311
312    fn process_synced(&mut self, sync_update: kyoto::messages::SyncUpdate) {
313        let tip = sync_update.tip;
314        info!(
315            "Synchronized to height {} ({}) with {} recent headers",
316            tip.height,
317            tip.hash,
318            sync_update.recent_history.len()
319        );
320        self.metrics.kyoto_synced.set(1);
321        self.metrics.kyoto_best_height.set(tip.height as i64);
322        self.metrics.kyoto_sync_percent.set(100);
323        self.tip = Some(tip);
324        self.process_pending_deposits();
325    }

327    fn process_blocks_disconnected(
328        &mut self,

340    fn process_client_message(&mut self, msg: MonitorMessage) {
341        match msg {
342            MonitorMessage::ConfirmDeposit(pending_deposit) => {
343                self.confirm_deposit(pending_deposit);
344            }
345            MonitorMessage::GetRecentFeeRate(conf_target, result_tx) => {
346                self.get_recent_fee_rate(conf_target, result_tx);
347            }
348            MonitorMessage::BroadcastTransaction(tx, result_tx) => {
349                self.broadcast_transaction(tx, result_tx);
350            }
351            MonitorMessage::GetTransactionStatus(txid, result_tx) => {
352                self.get_transaction_status(txid, result_tx);
353            }
354        }
355    }

357    fn confirm_deposit(&mut self, pending_deposit: PendingDeposit) {
358        debug!(
359            "Processing deposit confirmation for {}",
360            pending_deposit.outpoint.txid
361        );
362
363        let Some(tip) = &self.tip else {
364            debug!(
365                "Transaction {} confirmation on hold, pending chain tip update.",
366                pending_deposit.outpoint.txid
367            );
368            self.pending_deposits.push(pending_deposit);
369            return;
370        };
371
372        if pending_deposit.checked_at_height >= tip.height {
373            debug!(
374                "Transaction {} already checked at height {}. Waiting for more blocks.",
375                pending_deposit.outpoint.txid, pending_deposit.checked_at_height
376            );
377            self.pending_deposits.push(pending_deposit);
378            return;
379        }
380
381        self.pending_deposit_workers
382            .spawn(Monitor::process_pending_deposit(
383                tip.to_owned(),
384                self.config.confirmation_threshold,
385                self.bitcoind_rpc.clone(),
386                self.requester.clone(),
387                self.client_tx.clone(),
388                pending_deposit,
389            ));
390    }

392    fn get_recent_fee_rate(
393        &mut self,
394        conf_target: u16,
395        result_tx: oneshot::Sender<Result<FeeRate>>,
396    ) {
397        let result = self
398            .bitcoind_rpc
399            .estimate_smart_fee(conf_target, None)
400            .map_err(anyhow::Error::from)
401            .map(|res| {
402                let sat_per_kwu = match res.fee_rate {
403                    Some(amount) => {
404                        // Convert from BTC/kvB to sat/kwu (1 kvB = 4 kwu).
405                        amount.to_sat() / 4
406                    }
407                    None => {
408                        warn!(
409                            conf_target,
410                            fallback_sat_per_kwu = FALLBACK_FEE_RATE_SAT_PER_KWU,
411                            "Node could not estimate fee rate; falling back to minimum relay fee"
412                        );
413                        FALLBACK_FEE_RATE_SAT_PER_KWU
414                    }
415                };
416                self.metrics
417                    .btc_fee_rate_sat_per_kvb
418                    .set((sat_per_kwu * 4) as i64);
419                FeeRate::from_sat_per_kwu(sat_per_kwu)
420            });
421        let _ = result_tx.send(result);
422    }

424    fn broadcast_transaction(
425        &mut self,
426        tx: bitcoin::Transaction,
427        result_tx: oneshot::Sender<Result<()>>,
428    ) {
429        // Temp hack to get warning messages when a transaction would be rejected
430        // TODO: https://linear.app/mysten-labs/issue/IOP-216/better-error-reporting-for-failed-btc-broadcasts
431        let txid = tx.compute_txid();
432        match self.bitcoind_rpc.test_mempool_accept(&[&tx]) {
433            Ok(results) => match results.first() {
434                Some(result) if !result.allowed => {
435                    error!(
436                        "Bitcoin Core mempool will reject tx {txid}: {}",
437                        result.reject_reason.as_deref().unwrap_or("unknown reason")
438                    );
439                }
440                Some(_) => {
441                    debug!("Bitcoin Core mempool would accept tx {txid}");
442                }
443                None => {
444                    warn!("Bitcoin Core testmempoolaccept returned no result for tx {txid}");
445                }
446            },
447            Err(e) => {
448                warn!("Failed to run testmempoolaccept for tx {txid}: {e}");
449            }
450        }
451
452        let result = self.requester.broadcast_tx(kyoto::TxBroadcast {
453            tx,
454            broadcast_policy: kyoto::TxBroadcastPolicy::AllPeers,
455        });
456        let _ = result_tx.send(result.map_err(|e| anyhow::anyhow!(e)));
457    }

459    fn get_transaction_status(
460        &self,
461        txid: bitcoin::Txid,
462        result_tx: oneshot::Sender<Result<TxStatus>>,
463    ) {
464        let result = match getrawtransaction_brief(&self.bitcoind_rpc, &txid) {
465            Ok(tx_brief) => {
466                if tx_brief.blockhash.is_some() {
467                    let confirmations = tx_brief.confirmations.unwrap_or(0);
468                    Ok(TxStatus::Confirmed { confirmations })
469                } else {
470                    Ok(TxStatus::InMempool)
471                }
472            }
473            Err(bitcoincore_rpc::Error::JsonRpc(bitcoincore_rpc::jsonrpc::error::Error::Rpc(
474                ref e,
475            ))) if e.code == -5 => {
476                // RPC error -5: "No such mempool or blockchain transaction"
477                Ok(TxStatus::NotFound)
478            }
479            Err(e) => Err(anyhow::anyhow!("Failed to query transaction status: {e}")),
480        };
481        let _ = result_tx.send(result);
482    }

484    fn process_pending_deposits(&mut self) {
485        let Some(tip) = &self.tip else {
486            // Can't confirm deposits if we don't yet know the tip of the chain.
487            return;
488        };
489        if self.pending_deposits.is_empty() {
490            return;
491        }
492
493        info!(
494            "Processing {} pending deposits",
495            self.pending_deposits.len()
496        );
497        for pending_deposit in std::mem::take(&mut self.pending_deposits) {
498            self.pending_deposit_workers
499                .spawn(Monitor::process_pending_deposit(
500                    tip.to_owned(),
501                    self.config.confirmation_threshold,
502                    self.bitcoind_rpc.clone(),
503                    self.requester.clone(),
504                    self.client_tx.clone(),
505                    pending_deposit,
506                ));
507        }
508    }

510    async fn process_pending_deposit(
511        tip: HeaderCheckpoint,
512        confirmation_threshold: u32,
513        bitcoind_rpc: Arc<bitcoincore_rpc::Client>,
514        requester: kyoto::Requester,
515        client_tx: tokio::sync::mpsc::Sender<MonitorMessage>,
516        mut pending_deposit: PendingDeposit,
517    ) {
518        if pending_deposit.result_tx.is_closed() {
519            // Cancel if the receiver is no longer listening.
520            return;
521        }
522
523        pending_deposit.checked_at_height = tip.height;
524        let mut pending_deposit = PendingDepositGuard::new(pending_deposit, client_tx);
525
526        // Look up block from the txid.
527        let block_info = match pending_deposit.block_info {
528            Some(block_info) => block_info,
529            None => {
530                debug!(
531                    "Looking up block for transaction {}",
532                    pending_deposit.outpoint.txid
533                );
534                let tx_brief =
535                    match getrawtransaction_brief(&bitcoind_rpc, &pending_deposit.outpoint.txid) {
536                        Ok(tx_brief) => tx_brief,
537                        Err(e) => {
538                            error!(
539                                "Failed to look up txid {}: {e}",
540                                pending_deposit.outpoint.txid
541                            );
542                            return;
543                        }
544                    };
545                let Some(block_hash) = tx_brief.blockhash else {
546                    debug!(
547                        "Transaction {} is not yet included in a block",
548                        pending_deposit.outpoint.txid
549                    );
550                    return;
551                };
552                let block_header = match bitcoind_rpc.get_block_header_info(&block_hash) {
553                    Ok(block_header) => block_header,
554                    Err(e) => {
555                        error!("Failed to look up block header {}: {e}", block_hash);
556                        return;
557                    }
558                };
559                // Double check header with Kyoto to verify the height reported by bitcoind.
560                let kyoto_header = match requester.get_header(block_header.height as u32).await {
561                    Ok(kyoto_header) => kyoto_header,
562                    Err(e) => {
563                        error!(
564                            "Failed to look up header at height {}: {e}",
565                            block_header.height
566                        );
567                        return;
568                    }
569                };
570                if kyoto_header.block_hash() != block_hash {
571                    warn!(
572                        "Block hash mismatch at height {}! Possibly malicious behavior by the Bitcoin Core node. {} != {}",
573                        block_header.height,
574                        kyoto_header.block_hash(),
575                        block_hash
576                    );
577                    return;
578                }
579                let block_info = kyoto::HeaderCheckpoint {
580                    height: block_header.height as u32,
581                    hash: block_header.hash,
582                };
583                pending_deposit.block_info = Some(block_info);
584                block_info
585            }
586        };
587
588        // Check if the deposit has enough confirmations yet.
589        let confirmations = (tip.height + 1).saturating_sub(block_info.height);
590        if confirmations < confirmation_threshold {
591            debug!(
592                "Transaction {} has {confirmations}/{confirmation_threshold} confirmations. Waiting for more blocks.",
593                pending_deposit.outpoint.txid,
594            );
595            return;
596        }
597
598        // If deposit is confirmed, look up the TxOut info.
599        let block = match requester.get_block(block_info.hash).await {
600            Ok(block) => block,
601            Err(kyoto::error::FetchBlockError::UnknownHash) => {
602                // Error: The block is no longer in the current chain.
603                // TODO: Verify kyoto won't return blocks outside the chain of most work
604                warn!(
605                    "Pending deposit {:?}: Block is no longer in the current chain",
606                    pending_deposit.as_ref(),
607                );
608                pending_deposit.block_info = None;
609                return;
610            }
611            Err(e) => {
612                error!("Failed to look up block {}: {}", block_info.hash, e);
613                return;
614            }
615        };
616        let Some(transaction) = block
617            .block
618            .txdata
619            .iter()
620            .find(|tx| tx.compute_txid() == pending_deposit.outpoint.txid)
621        else {
622            // Error: The transaction is not actually in the block.
623            warn!(
624                "Pending deposit {:?}: Transaction not present in the block reported by the Bitcoin Core node! Possibly malicious behavior by the Bitcoin Core node.",
625                pending_deposit.as_ref(),
626            );
627            pending_deposit.block_info = None;
628            return;
629        };
630
631        let txout = match transaction.tx_out(pending_deposit.outpoint.vout.try_into().unwrap()) {
632            Ok(txout) => txout.clone(),
633            Err(e) => {
634                let pending_deposit = pending_deposit.take();
635                let _ = pending_deposit
636                    .result_tx
637                    .send(Err(anyhow::anyhow!(e).into()));
638                return;
639            }
640        };
641
642        // Verify the UTXO is still unspent in Bitcoin's UTXO set (including mempool).
643        match bitcoind_rpc.get_tx_out(
644            &pending_deposit.outpoint.txid,
645            pending_deposit.outpoint.vout,
646            Some(true),
647        ) {
648            Ok(Some(_)) => {
649                let pending_deposit = pending_deposit.take();
650                let _ = pending_deposit.result_tx.send(Ok(txout));
651            }
652            Ok(None) => {
653                warn!(
654                    "Deposit UTXO {}:{} has already been spent on Bitcoin. Rejecting deposit.",
655                    pending_deposit.outpoint.txid, pending_deposit.outpoint.vout,
656                );
657                let txid = pending_deposit.outpoint.txid;
658                let vout = pending_deposit.outpoint.vout;
659                let pending_deposit = pending_deposit.take();
660                let _ = pending_deposit
661                    .result_tx
662                    .send(Err(DepositConfirmError::UtxoSpent { txid, vout }));
663            }
664            Err(e) => {
665                error!("Failed to check UTXO spent status via gettxout: {e}");
666            }
667        }
668    }
669}

671#[derive(Debug)]
672struct PendingDeposit {
673    outpoint: bitcoin::OutPoint,
674    block_info: Option<kyoto::HeaderCheckpoint>,
675    result_tx: oneshot::Sender<Result<bitcoin::TxOut, DepositConfirmError>>,
676    checked_at_height: u32,
677}
678
679/// RAII guard that re-enqueues a pending deposit on drop unless marked as complete.
680struct PendingDepositGuard {
681    deposit: Option<PendingDeposit>,
682    client_tx: tokio::sync::mpsc::Sender<MonitorMessage>,
683}
684
685impl PendingDepositGuard {
686    fn new(deposit: PendingDeposit, client_tx: tokio::sync::mpsc::Sender<MonitorMessage>) -> Self {
687        Self {
688            deposit: Some(deposit),
689            client_tx,
690        }
691    }
692
693    fn take(mut self) -> PendingDeposit {
694        self.deposit.take().unwrap()
695    }
696}
697
698impl std::ops::Deref for PendingDepositGuard {
699    type Target = PendingDeposit;
700
701    fn deref(&self) -> &Self::Target {
702        self.deposit.as_ref().unwrap()
703    }
704}
705
706impl std::ops::DerefMut for PendingDepositGuard {
707    fn deref_mut(&mut self) -> &mut Self::Target {
708        self.deposit.as_mut().unwrap()
709    }
710}
711
712impl AsRef<PendingDeposit> for PendingDepositGuard {
713    fn as_ref(&self) -> &PendingDeposit {
714        self.deposit.as_ref().unwrap()
715    }
716}
717
718impl Drop for PendingDepositGuard {
719    fn drop(&mut self) {
720        if let Some(deposit) = self.deposit.take()
721            && let Err(e) = self
722                .client_tx
723                .try_send(MonitorMessage::ConfirmDeposit(deposit))
724        {
725            warn!("Failed to re-enqueue PendingDeposit on drop: {e}");
726        }
727    }
728}

746#[derive(Clone)]
747pub struct MonitorClient {
748    tx: tokio::sync::mpsc::Sender<MonitorMessage>,
749}
750
751impl MonitorClient {
752    pub async fn confirm_deposit(
753        &self,
754        outpoint: bitcoin::OutPoint,
755    ) -> Result<bitcoin::TxOut, DepositConfirmError> {
756        let (tx, rx) = oneshot::channel();
757        let pending_deposit = PendingDeposit {
758            outpoint,
759            block_info: None,
760            result_tx: tx,
761            checked_at_height: 0,
762        };
763        self.tx
764            .send(MonitorMessage::ConfirmDeposit(pending_deposit))
765            .await
766            .map_err(|e| anyhow::anyhow!(e))?;
767        tokio::time::timeout(Duration::from_secs(5), rx)
768            .await
769            .map_err(|_| {
770                anyhow::anyhow!("confirm_deposit timed out waiting for Bitcoin confirmation")
771            })?
772            .map_err(|e| anyhow::anyhow!(e))?
773    }
774
775    pub async fn get_recent_fee_rate(&self, conf_target: u16) -> Result<FeeRate> {
776        let (tx, rx) = oneshot::channel();
777        self.tx
778            .send(MonitorMessage::GetRecentFeeRate(conf_target, tx))
779            .await
780            .map_err(|e| anyhow::anyhow!(e))?;
781        rx.await.map_err(|e| anyhow::anyhow!(e))?
782    }
783
784    pub async fn broadcast_transaction(&self, transaction: bitcoin::Transaction) -> Result<()> {
785        let (tx, rx) = oneshot::channel();
786        self.tx
787            .send(MonitorMessage::BroadcastTransaction(transaction, tx))
788            .await
789            .map_err(|e| anyhow::anyhow!(e))?;
790        rx.await.map_err(|e| anyhow::anyhow!(e))?
791    }
792
793    pub async fn get_transaction_status(&self, txid: bitcoin::Txid) -> Result<TxStatus> {
794        let (tx, rx) = oneshot::channel();
795        self.tx
796            .send(MonitorMessage::GetTransactionStatus(txid, tx))
797            .await
798            .map_err(|e| anyhow::anyhow!(e))?;
799        rx.await.map_err(|e| anyhow::anyhow!(e))?
800    }
801}
802
803enum MonitorMessage {
804    // Locates the given OutPoint in a block, waits for it to have enough
805    // confirmations, and returns output information. Will keep trying to
806    // confirm indefinitely, unless the proivded channel is closed.
807    ConfirmDeposit(PendingDeposit),
808
809    // Returns an estimated fee rate targeting confirmation within `conf_target` blocks.
810    GetRecentFeeRate(u16, oneshot::Sender<Result<FeeRate>>),
811
812    // Broadcast a transaction to the network.
813    BroadcastTransaction(bitcoin::Transaction, oneshot::Sender<Result<()>>),
814
815    // Query the status of a transaction (confirmed, in mempool, or not found).
816    GetTransactionStatus(bitcoin::Txid, oneshot::Sender<Result<TxStatus>>),
817}
