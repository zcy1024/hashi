26pub async fn run(cfg: ProvisionerConfig) -> anyhow::Result<()> {
27    let s3_client = S3Logger::new_checked(&cfg.s3)
28        .await
29        .map_err(|e| anyhow::anyhow!(e))?;
30
31    // 1. Check no past enclave's heartbeats remain & gather the latest enclave's session id.
32    let session_id = heartbeat_checks::kp_heartbeat_audit(&s3_client).await?;
33    info!(session_id, "heartbeat checks passed for selected session");
34
35    // 2. Check that enclave's config is as expected (valid attestation, expected s3 bucket & share commitments)
36    let guardian_info = get_guardian_info_from_s3(&s3_client, &session_id).await?;
37    let expected_guardian_config = cfg.expected_guardian_config()?;
38    expected_guardian_config.ensure_matches_info(&guardian_info)?;
39    info!(session_id, "init checks passed for selected session");
40
41    // TODO: replace mock limiter state with actual state from S3 logs.
42    let committee = cfg.hashi_committee.try_into()?;
43    let mock_limiter_state = LimiterState {
44        num_tokens_available: cfg.withdrawal_config.max_bucket_capacity_sats,
45        last_updated_at: 0,
46        next_seq: 0,
47    };
48    let state = ProvisionerInitState::new(
49        committee,
50        cfg.withdrawal_config,
51        mock_limiter_state,
52        cfg.hashi_btc_master_pubkey,
53    )
54    .map_err(|e| anyhow::anyhow!(e))?;
55
56    let guardian_pub_key =
57        EncPubKey::from_bytes(&guardian_info.encryption_pubkey).map_err(|e| anyhow::anyhow!(e))?;
58    let request = ProvisionerInitRequest::build_from_share_and_state(
59        &cfg.share.to_domain()?,
60        &guardian_pub_key,
61        state,
62        &mut thread_rng(),
63    );
64    let share_id = request.encrypted_share().id.get();
65    let state_digest_hex = hex::encode(request.state().digest());
66    info!(
67        share_id,
68        state_digest = state_digest_hex,
69        "built provisioner-init request"
70    );
71
72    if let Some(endpoint) = cfg.guardian_endpoint {
73        submit_provisioner_init_to_guardian(
74            &endpoint,
75            &session_id,
76            &expected_guardian_config,
77            request,
78        )
79        .await?;
80    }
81    Ok(())
82}

84/// Implements check B of IOP-225.
85pub async fn get_guardian_info_from_s3(
86    s3_client: &S3Logger,
87    session_id: &str,
88) -> anyhow::Result<GuardianInfo> {
89    let (attestation, signing_pubkey) = s3_client
90        .get_attestation(session_id)
91        .await
92        .map_err(|e| anyhow::anyhow!(e))?;
93    verify_enclave_attestation(attestation).map_err(|e| anyhow::anyhow!(e))?;
94
95    s3_client
96        .get_guardian_info(session_id, &signing_pubkey)
97        .await
98        .map_err(|e| anyhow::anyhow!(e))
99}

101async fn submit_provisioner_init_to_guardian(
102    endpoint: &str,
103    expected_session_id: &str,
104    expected_guardian_config: &GuardianConfig,
105    request: ProvisionerInitRequest,
106) -> anyhow::Result<()> {
107    let mut client =
108        pb::guardian_service_client::GuardianServiceClient::connect(endpoint.to_string())
109            .await
110            .with_context(|| format!("failed to connect to guardian endpoint {endpoint}"))?;
111
112    prechecks(&mut client, expected_session_id, expected_guardian_config)
113        .await
114        .with_context(|| "guardian endpoint pre-check failed")?;
115
116    info!("prechecks passed, submitting ProvisionerInit");
117    let pb_request = provisioner_init_request_to_pb(request)?;
118    client
119        .provisioner_init(pb_request)
120        .await
121        .with_context(|| format!("guardian ProvisionerInit RPC failed at {endpoint}"))?;
122
123    info!("successfully submitted ProvisionerInit request");
124    Ok(())
125}

127async fn prechecks(
128    client: &mut pb::guardian_service_client::GuardianServiceClient<tonic::transport::Channel>,
129    expected_session_id: &str,
130    expected_guardian_config: &GuardianConfig,
131) -> anyhow::Result<()> {
132    let resp_pb = client
133        .get_guardian_info(pb::GetGuardianInfoRequest {})
134        .await
135        .with_context(|| "GetGuardianInfo RPC failed")?
136        .into_inner();
137
138    let resp = <GetGuardianInfoResponse as TryFrom<pb::GetGuardianInfoResponse>>::try_from(resp_pb)
139        .map_err(|e| anyhow::anyhow!(e.to_string()))?;
140
141    let signing_pub_key = resp.signing_pub_key;
142    let actual_session_id = session_id_from_signing_pubkey(&signing_pub_key);
143    anyhow::ensure!(
144        actual_session_id == expected_session_id,
145        "guardian endpoint session mismatch: expected {}, got {}",
146        expected_session_id,
147        actual_session_id
148    );
149
150    let info = resp
151        .signed_info
152        .verify(&signing_pub_key)
153        .map_err(|e| anyhow::anyhow!(e.to_string()))?;
154
155    expected_guardian_config.ensure_matches_info(&info)?;
156
157    Ok(())
158}
