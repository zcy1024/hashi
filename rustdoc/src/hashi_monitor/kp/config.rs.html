23#[derive(Debug, Clone, PartialEq)]
24pub struct GuardianConfig {
25    pub bucket_info: S3BucketInfo,
26    pub share_commitments: ShareCommitments,
27}
28
29impl GuardianConfig {
30    fn from_guardian_info(info: &GuardianInfo) -> anyhow::Result<Self> {
31        let bucket_info = info.bucket_info.clone().ok_or_else(|| {
32            anyhow::anyhow!("guardian info missing bucket_info; operator_init may be incomplete")
33        })?;
34        let share_commitments = info.share_commitments.clone().ok_or_else(|| {
35            anyhow::anyhow!(
36                "guardian info missing share_commitments; operator_init may be incomplete"
37            )
38        })?;
39        Ok(Self {
40            bucket_info,
41            share_commitments,
42        })
43    }
44
45    pub fn ensure_matches_info(&self, info: &GuardianInfo) -> anyhow::Result<()> {
46        let actual = Self::from_guardian_info(info)?;
47        anyhow::ensure!(
48            actual == *self,
49            "guardian config mismatch: expected {:?}, got {:?}",
50            self,
51            actual
52        );
53        Ok(())
54    }
55}

89impl ProvisionerConfig {
90    pub fn load_yaml(path: &Path) -> anyhow::Result<Self> {
91        let bytes = std::fs::read(path)
92            .with_context(|| format!("failed to read kp-init config at {}", path.display()))?;
93        serde_yaml::from_slice(&bytes)
94            .with_context(|| format!("failed to parse kp-init yaml at {}", path.display()))
95    }
96
97    pub fn expected_guardian_config(&self) -> anyhow::Result<GuardianConfig> {
98        let pb_commitments = self
99            .share_commitments
100            .iter()
101            .map(ShareCommitmentInput::to_pb)
102            .collect::<Vec<_>>();
103
104        // Domain validation checks
105        let share_commitments =
106            pb_share_commitments_to_domain(&pb_commitments).map_err(|e| anyhow::anyhow!(e))?;
107
108        Ok(GuardianConfig {
109            bucket_info: self.s3.bucket_info.clone(),
110            share_commitments,
111        })
112    }
113}

115impl ShareCommitmentInput {
116    fn to_pb(&self) -> pb::GuardianShareCommitment {
117        pb::GuardianShareCommitment {
118            id: Some(pb::GuardianShareId {
119                id: Some(self.id.into()),
120            }),
121            digest_hex: Some(self.digest_hex.clone()),
122        }
123    }
124}
125
126impl ShareInput {
127    pub fn to_domain(&self) -> anyhow::Result<Share> {
128        let id =
129            NonZeroU16::new(self.id).ok_or_else(|| anyhow::anyhow!("share id must be non-zero"))?;
130        let bytes = hex::decode(&self.value_hex)
131            .with_context(|| format!("invalid share value hex for id={}", self.id))?;
132        let scalar_bytes: [u8; 32] = bytes
133            .as_slice()
134            .try_into()
135            .map_err(|_| anyhow::anyhow!("share value must be 32 bytes"))?;
136        let scalar = Option::<Scalar>::from(Scalar::from_repr(FieldBytes::from(scalar_bytes)))
137            .ok_or_else(|| anyhow::anyhow!("invalid scalar in share value"))?;
138        Ok(Share {
139            id: ShareID::from(id),
140            value: scalar,
141        })
142    }
