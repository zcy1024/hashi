16/// Heartbeat-audit logic used by the KP flow.
17///
18/// The audit enforces a single-live-session invariant before a KP submits `ProvisionerInit`:
19/// - exactly one session is considered live (latest heartbeat is recent), and
20/// - all other sessions have been quiet long enough.
21///
22/// Maximum acceptable heartbeat age for the selected "live" session.
23/// Set to HEARTBEAT_INTERVAL + grace period to account for clock skew and retries.
24const LIVE_SESSION_MAX_AGE: Duration = Duration::from_mins(3);
25
26/// If another session has a heartbeat in the last X secs (X is given below), it is considered still active.
27/// MAX_HEARTBEAT_FAILURES * HEARTBEAT_INTERVAL + grace period to account for retries and clock skew.
28const OTHER_SESSION_QUIET_PERIOD: Duration = Duration::from_mins(10);
29
30#[derive(Debug, Clone)]
31pub struct GuardianSessionInfo {
32    /// Session identifier derived from the guardian signing key.
33    pub session_id: String,
34    /// Earliest heartbeat timestamp observed for this session.
35    pub first_heartbeat: UnixSeconds,
36    /// Latest heartbeat timestamp observed for this session.
37    pub last_heartbeat: UnixSeconds,
38}
39
40/// Implements check A of IOP-225.
41///
42/// Returns the selected live session id if all invariants pass.
43pub async fn kp_heartbeat_audit(s3_client: &S3Logger) -> anyhow::Result<String> {
44    let recent_heartbeats = read_recent_heartbeats(s3_client).await?;
45    let summary = summarize_heartbeats_by_session(recent_heartbeats)?;
46    let now = now_unix_seconds();
47    select_live_session(
48        &summary,
49        now,
50        LIVE_SESSION_MAX_AGE.as_secs(),
51        OTHER_SESSION_QUIET_PERIOD.as_secs(),
52    )
53}
54
55async fn read_recent_heartbeats(s3_client: &S3Logger) -> anyhow::Result<Vec<VerifiedLogRecord>> {
56    // Read from the current and next hour-scoped prefixes to cover clock-boundary cases.
57    let one_hour_ago = now_unix_seconds().saturating_sub(60 * 60);
58    let mut poller = GuardianPollerCore::from_s3_client(
59        s3_client.clone(),
60        one_hour_ago,
61        GuardianLogDir::Heartbeat,
62    );
63    let mut logs = Vec::new();
64    logs.extend(poller.read_cur_dir().await?);
65    poller.advance_cursor();
66    logs.extend(poller.read_cur_dir().await?);
67    Ok(logs)
68}

70/// Aggregates verified heartbeat logs into [first, last] bounds per session.
71fn summarize_heartbeats_by_session(
72    logs: Vec<VerifiedLogRecord>,
73) -> anyhow::Result<Vec<GuardianSessionInfo>> {
74    let mut map: BTreeMap<String, (UnixSeconds, UnixSeconds)> = BTreeMap::new();
75
76    for log in logs {
77        if !matches!(log.message, LogMessage::Heartbeat { .. }) {
78            anyhow::bail!("non-heartbeat logs found");
79        }
80
81        let ts = unix_millis_to_seconds(log.timestamp_ms);
82        map.entry(log.session_id)
83            .and_modify(|(first, last)| {
84                *first = (*first).min(ts);
85                *last = (*last).max(ts);
86            })
87            .or_insert((ts, ts));
88    }
89
90    Ok(map
91        .into_iter()
92        .map(
93            |(session_id, (first_heartbeat, last_heartbeat))| GuardianSessionInfo {
94                session_id,
95                first_heartbeat,
96                last_heartbeat,
97            },
98        )
99        .collect())
100}

102/// Checks that exactly one session is live and returns its ID.
103///
104/// A session is considered live when its latest heartbeat is not older than
105/// `live_session_max_age_secs`. Any other session with a heartbeat newer than
106/// `other_session_quiet_secs` is treated as still active and causes a failure.
107fn select_live_session(
108    summary: &[GuardianSessionInfo],
109    now: UnixSeconds,
110    live_session_max_age_secs: UnixSeconds,
111    other_session_quiet_secs: UnixSeconds,
112) -> anyhow::Result<String> {
113    let mut summary = summary.to_vec();
114
115    summary.sort_by_key(|s| s.last_heartbeat);
116    let live_session_info = summary
117        .last()
118        .ok_or_else(|| anyhow::anyhow!("no heartbeat logs found in the most recent 2 hours"))?;
119    let live_session_id = &live_session_info.session_id;
120
121    let live_session_age_secs = now.saturating_sub(live_session_info.last_heartbeat);
122    if live_session_age_secs > live_session_max_age_secs {
123        anyhow::bail!(
124            "latest session {} is stale: last heartbeat {}s ago (expected <= {}s)",
125            live_session_id,
126            live_session_age_secs,
127            live_session_max_age_secs
128        );
129    }
130
131    let active_other_sessions = summary
132        .iter()
133        .filter(|s| s.session_id != *live_session_id)
134        .filter_map(|s| {
135            let age_secs = now.saturating_sub(s.last_heartbeat);
136            (age_secs < other_session_quiet_secs)
137                .then(|| format!("{} ({}s ago)", s.session_id, age_secs))
138        })
139        .collect::<Vec<_>>();
140    if !active_other_sessions.is_empty() {
141        anyhow::bail!(
142            "other sessions are still active within {}s: {}",
143            other_session_quiet_secs,
144            active_other_sessions.join(", ")
145        );
146    }
147
148    info!(
149        "Selected session {} (first={}s, last={}s, age={}s)",
150        live_session_id,
151        live_session_info.first_heartbeat,
152        live_session_info.last_heartbeat,
153        now.saturating_sub(live_session_info.last_heartbeat)
154    );
155
156    Ok(live_session_id.clone())
157}

159#[cfg(test)]
160mod tests {
161    use super::*;
162
163    #[test]
164    fn select_live_session_picks_latest_session() {
165        let now = 1_000;
166        let summary = vec![
167            GuardianSessionInfo {
168                session_id: "old".to_string(),
169                first_heartbeat: 100,
170                last_heartbeat: 300,
171            },
172            GuardianSessionInfo {
173                session_id: "new".to_string(),
174                first_heartbeat: 600,
175                last_heartbeat: 990,
176            },
177        ];
178
179        let target =
180            select_live_session(&summary, now, 100, 600).expect("must select newest session");
181        assert_eq!(target, "new");
182    }
183
184    #[test]
185    fn select_live_session_fails_when_empty() {
186        let err = select_live_session(&[], 1_000, 100, 600).expect_err("must fail");
187        assert!(err.to_string().contains("no heartbeat logs found"));
188    }
189
190    #[test]
191    fn select_live_session_fails_when_latest_stale() {
192        let summary = vec![GuardianSessionInfo {
193            session_id: "only".to_string(),
194            first_heartbeat: 100,
195            last_heartbeat: 200,
196        }];
197
198        let err = select_live_session(&summary, 1_000, 100, 600).expect_err("must fail");
199        assert!(err.to_string().contains("stale"));
200    }
201
202    #[test]
203    fn select_live_session_fails_when_other_session_is_active() {
204        let now = 1_000;
205        let summary = vec![
206            GuardianSessionInfo {
207                session_id: "old".to_string(),
208                first_heartbeat: 100,
209                last_heartbeat: 950,
210            },
211            GuardianSessionInfo {
212                session_id: "new".to_string(),
213                first_heartbeat: 900,
214                last_heartbeat: 990,
215            },
216        ];
217
218        let err = select_live_session(&summary, now, 100, 100).expect_err("must fail");
219        assert!(err.to_string().contains("other sessions are still active"));
220    }
221}
