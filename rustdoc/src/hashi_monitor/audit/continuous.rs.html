17// TODO: Consider switching to a streaming API
18/// The frequency at which we poll sui, guardian and btc RPC
19const POLL_INTERVAL: Duration = Duration::from_secs(10 * 60);
20
21/// The frequency at which we do validation checks.
22const STATE_TICK_INTERVAL: Duration = Duration::from_secs(5 * 60);
23
24/// A continuous audit only requires a start time
25pub struct ContinuousAuditWindow {
26    pub user_start: UnixSeconds,
27    pub sui_start: UnixSeconds,
28    pub guardian_start: UnixSeconds,
29}
30
31/// A continuous auditor that runs indefinitely processing events as they arrive.
32/// Constructors accept a start time as input that acts as a starting point for the auditor.
33pub struct ContinuousAuditor {
34    pub inner: AuditorCore,
35    pub window: ContinuousAuditWindow,
36}
37
38impl AuditWindow for ContinuousAuditWindow {
39    fn in_window(&self, timestamp_secs: UnixSeconds) -> bool {
40        timestamp_secs >= self.user_start
41    }
42}
43
44impl ContinuousAuditWindow {
45    pub fn new(cfg: &Config, start: UnixSeconds) -> Self {
46        let e1_e2_delay_secs = cfg
47            .next_event_delay(WithdrawalEventType::E1HashiApproved)
48            .expect("should be Some");
49        let sui_start = start.saturating_sub(e1_e2_delay_secs); // guardian_e2@{start} might match sui_e1@{start-e1_e2_delay_secs}
50        let guardian_start = start;
51
52        Self {
53            user_start: start,
54            sui_start,
55            guardian_start,
56        }
57    }
58}
59
60impl ContinuousAuditor {
61    pub async fn new(cfg: &Config, start: UnixSeconds) -> anyhow::Result<Self> {
62        let cur_time = now_unix_seconds();
63        anyhow::ensure!(
64            start <= cur_time,
65            "start is in the future: start={start} > cur_time={cur_time}"
66        );
67        let audit_window = ContinuousAuditWindow::new(cfg, start);
68        let cursors = Cursors {
69            sui: audit_window.sui_start,
70            guardian: audit_window.guardian_start,
71        };
72
73        Ok(Self {
74            inner: AuditorCore::new(cfg, cursors).await?,
75            window: audit_window,
76        })
77    }
78
79    pub fn ingest_batch(&mut self, events: Vec<MonitorEvent>) {
80        let errors = self.inner.ingest_batch(events);
81        log_findings("continuous", "ingest", &errors);
82    }
83
84    async fn tick_sui(&mut self) -> anyhow::Result<()> {
85        if let PollOutcome::CursorAdvanced(events) = self.inner.poll_sui().await? {
86            self.ingest_batch(events);
87        }
88        Ok(())
89    }
90
91    async fn tick_guardian(&mut self) -> anyhow::Result<()> {
92        if let PollOutcome::CursorAdvanced(events) = self.inner.poll_guardian().await? {
93            self.ingest_batch(events);
94        }
95        Ok(())
96    }
97
98    /// Throws an error if BTC RPC infra fails.
99    fn tick_btc(&mut self) -> anyhow::Result<()> {
100        let errors = self.inner.fetch_btc_info(&self.window)?;
101        log_findings("continuous", "btc", &errors);
102        Ok(())
103    }
104
105    fn tick_state_checks_and_gc(&mut self) {
106        let violations = self.inner.detect_violations(&self.window);
107        // TODO: If a violation is detected, we keep logging it on every call to this. Decide if that's the behavior we want.
108        log_findings("continuous", "violations", &violations);
109
110        // Garbage collect
111        self.inner.garbage_collect(&self.window);
112
113        let progress = self.inner.progress_watermarks(&self.window);
114        tracing::info!(
115            guardian_cursor = self.inner.get_guardian_cursor(),
116            sui_cursor = self.inner.get_sui_cursor(),
117            verified_up_to_withdrawals = progress.verified_up_to_withdrawals,
118            verified_up_to_deposits = progress.verified_up_to_deposits,
119            restart_start = progress.restart_start,
120            "continuous progress checkpoint"
121        );
122    }
123
124    pub async fn run(&mut self) -> anyhow::Result<()> {
125        let mut sui_ticker = tokio::time::interval(POLL_INTERVAL);
126        let mut guardian_ticker = tokio::time::interval(POLL_INTERVAL);
127        let mut btc_ticker = tokio::time::interval(POLL_INTERVAL);
128        let mut state_checks_ticker = tokio::time::interval(STATE_TICK_INTERVAL);
129
130        sui_ticker.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
131        guardian_ticker.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
132        btc_ticker.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
133        state_checks_ticker.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
134
135        // Consume the immediate first tick and then run on a stable cadence.
136        sui_ticker.tick().await;
137        guardian_ticker.tick().await;
138        btc_ticker.tick().await;
139        state_checks_ticker.tick().await;
140
141        loop {
142            // TODO: Make it multi-threaded?
143            tokio::select! {
144                _ = sui_ticker.tick() => {
145                    if let Err(error) = self.tick_sui().await {
146                        tracing::warn!(source = "sui", ?error, "poll failed; continuing");
147                    }
148                }
149                _ = guardian_ticker.tick() => {
150                    if let Err(error) = self.tick_guardian().await {
151                        tracing::warn!(source = "guardian", ?error, "poll failed; continuing");
152                    }
153                }
154                _ = btc_ticker.tick() => {
155                    if let Err(error) = self.tick_btc() {
156                        tracing::warn!(source = "btc", ?error, "btc tick failed; continuing");
157                    }
158                }
159                _ = state_checks_ticker.tick() => {
160                    self.tick_state_checks_and_gc();
161                }
162            }
163        }
164    }
165}
