4use crate::domain::MonitorEvent;
5use crate::domain::MonitorEventType;
6use hashi_types::guardian::time_utils::UnixSeconds;
7use std::fmt;
8
9/// Findings emitted by the monitor.
10#[derive(Clone, Debug, Eq, PartialEq)]
11pub enum MonitorError {
12    InvalidEventAdded(String),
13    EventOccurredAfterDeadline {
14        event: MonitorEvent,
15        deadline: UnixSeconds,
16        occurred_at: UnixSeconds, // same as event.timestamp
17    },
18    ExpectedEventMissing {
19        event_type: MonitorEventType,
20        deadline: UnixSeconds,
21        cursor: UnixSeconds,
22    },
23}
24
25impl fmt::Display for MonitorError {
26    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
27        write!(f, "{self:?}")
28    }
29}
