13pub struct BtcRpcClient {
14    client: bitcoincore_rpc::Client,
15}

17impl BtcRpcClient {
18    pub fn new(cfg: &Config) -> anyhow::Result<Self> {
19        let client = bitcoincore_rpc::Client::new(
20            &cfg.btc.rpc_url,
21            cfg.btc.rpc_auth.to_bitcoincore_rpc_auth(),
22        )
23        .with_context(|| format!("failed to connect to bitcoin rpc at {}", cfg.btc.rpc_url))?;
24        Ok(Self { client })
25    }
26
27    /// Query BTC RPC to check if a transaction is confirmed.
28    /// Returns
29    /// - `Ok(Some(block_time))` if txid is confirmed,
30    /// - `Ok(None)` if txid is not seen or txid is seen but not confirmed,
31    /// - `Err(...)` for all other errors
32    pub fn lookup_confirmation(&self, txid: Txid) -> anyhow::Result<Option<UnixSeconds>> {
33        // Note: rpc returns Ok(...) even for unconfirmed txid in the mempool.
34        let tx_info = match self.client.get_raw_transaction_info(&txid, None) {
35            Ok(tx_info) => tx_info,
36            Err(e) if txid_not_found(&e) => {
37                debug!(%txid, "bitcoin tx not found in mempool or chain yet");
38                return Ok(None);
39            }
40            Err(e) => {
41                warn!(%txid, error = %e, "failed to fetch bitcoin tx from rpc");
42                anyhow::bail!("failed to fetch bitcoin tx {} from rpc: {}", txid, e)
43            }
44        };
45
46        let Some(block_hash) = tx_info.blockhash else {
47            debug!(%txid, "bitcoin tx found but not mined yet");
48            return Ok(None);
49        };
50
51        let block_header = self
52            .client
53            .get_block_header_info(&block_hash)
54            .with_context(|| format!("failed to fetch block header for {}", block_hash))?;
55
56        let block_time = u64::try_from(block_header.time)
57            .with_context(|| format!("invalid block timestamp for {}", block_hash))?;
58
59        Ok(Some(block_time))
60    }
61}
62
63// https://github.com/bitcoin/bitcoin/blob/v25.0/src/rpc/protocol.h#L41
64fn txid_not_found(error: &bitcoincore_rpc::Error) -> bool {
65    matches!(
66        error,
67        bitcoincore_rpc::Error::JsonRpc(bitcoincore_rpc::jsonrpc::Error::Rpc(rpc_error))
68            if rpc_error.code == -5
69    )
70}

95    fn init_test_tracing() {
96        TRACING_INIT.call_once(|| {
97            let _ = tracing_subscriber::fmt()
98                .with_test_writer()
99                .with_env_filter(tracing_subscriber::EnvFilter::from_default_env())
100                .try_init();
101        });
102    }

104    fn test_config(rpc_url: String) -> Config {
105        Config {
106            next_event_delays: NextEventDelays::new(vec![
107                (WithdrawalEventType::E1HashiApproved, 100),
108                (WithdrawalEventType::E2GuardianApproved, 200),
109            ])
110            .expect("valid next event delays"),
111            clock_skew: 10,
112            guardian: hashi_types::guardian::S3Config::mock_for_testing(),
113            sui: SuiConfig {
114                rpc_url: "http://sui".to_string(),
115            },
116            btc: BtcConfig {
117                rpc_url,
118                rpc_auth: BtcRpcAuth::UserPass {
119                    username: RPC_USER.to_string(),
120                    password: RPC_PASSWORD.to_string(),
121                },
122            },
123        }
124    }
125
126    // Note that this test requires local bitcoind running.
127    #[tokio::test]
128    async fn lookup_btc_confirmation_with_local_regtest() -> Result<()> {
129        init_test_tracing();
130
131        let temp_dir = TempDir::new()?;
132        let node = BitcoinNodeBuilder::new()
133            .dir(temp_dir.path())
134            .build()
135            .await?;
136        let cfg = test_config(node.rpc_url().to_string());
137        let btc_rpc_client = BtcRpcClient::new(&cfg)?;
138
139        let unknown_txid = Txid::from_slice(&[7u8; 32])?;
140        let unknown = btc_rpc_client.lookup_confirmation(unknown_txid)?;
141        assert!(
142            unknown.is_none(),
143            "expected unknown tx lookup to return none"
144        );
145
146        let destination = node.get_new_address()?;
147        let txid = node.send_to_address(&destination, Amount::from_sat(50_000))?;
148
149        let unconfirmed = btc_rpc_client.lookup_confirmation(txid)?;
150        assert!(unconfirmed.is_none(), "expected unconfirmed transaction");
151
152        node.generate_blocks(1)?;
153
154        let confirmed = btc_rpc_client.lookup_confirmation(txid)?;
155        assert!(confirmed.is_some(), "expected confirmed transaction");
156
157        Ok(())
158    }
