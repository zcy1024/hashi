30    Withdraw,
31    Heartbeat,

34enum VerifiedWithdrawal {
35    Success(MonitorWithdrawalEvent),
36    Failure,

39/// Reusable S3 poller core with attestation and signature checks. Meant to be used for either withdrawal or heartbeat logs.
40/// Idea: Since guardian can write out of order and S3 ListObjectVersions only supports lexicographic cursors, we
41///       read from an S3 directory only after we are certain that all writes to it finish.
42/// E.g., 12-1 PM bucket is read at 1 PM + DIR_WRITES_COMPLETION_DELAY, e.g., 1:10 PM.
43pub struct GuardianPollerCore {
44    s3_client: S3Logger,
45    cursor: S3HourScopedDirectory,
46    enclave_pub_keys: HashMap<String, GuardianPubKey>,
47}
48
49impl TryFrom<VerifiedLogRecord> for VerifiedWithdrawal {
50    type Error = anyhow::Error;
51
52    fn try_from(log: VerifiedLogRecord) -> Result<Self, Self::Error> {
53        let LogMessage::Withdrawal(withdrawal_message) = log.message else {
54            anyhow::bail!("non-withdrawal logs found");
55        };
56
57        match *withdrawal_message {
58            WithdrawalLogMessage::Success {
59                txid, request_data, ..
60            } => {
61                debug!(
62                    wid = request_data.wid,
63                    txid = %txid,
64                    "successful guardian withdrawal log"
65                );
66                Ok(VerifiedWithdrawal::Success(MonitorWithdrawalEvent {
67                    event_type: WithdrawalEventType::E2GuardianApproved,
68                    wid: request_data.wid,
69                    timestamp_secs: unix_millis_to_seconds(log.timestamp_ms),
70                    btc_txid: txid,
71                }))
72            }
73            failure @ WithdrawalLogMessage::Failure { .. } => {
74                info!(?failure, "failed guardian withdrawal log");
75                Ok(VerifiedWithdrawal::Failure)
76            }
77        }
78    }
79}

81impl GuardianLogDir {
82    fn as_prefix(self) -> &'static str {
83        match self {
84            GuardianLogDir::Withdraw => S3_DIR_WITHDRAW,
85            GuardianLogDir::Heartbeat => S3_DIR_HEARTBEAT,
86        }
87    }
88}

90impl GuardianPollerCore {
91    pub async fn new(
92        config: &S3Config,
93        start: UnixSeconds,
94        log_dir: GuardianLogDir,
95    ) -> anyhow::Result<Self> {
96        let s3_client = S3Logger::new_checked(config)
97            .await
98            .map_err(|e| anyhow::anyhow!(e))
99            .context("failed to verify guardian S3 connectivity")?;
100        Ok(Self::from_s3_client(s3_client, start, log_dir))
101    }
102
103    pub fn from_s3_client(
104        s3_client: S3Logger,
105        start: UnixSeconds,
106        log_dir: GuardianLogDir,
107    ) -> Self {
108        Self {
109            s3_client,
110            cursor: S3HourScopedDirectory::new(log_dir.as_prefix(), start),
111            enclave_pub_keys: HashMap::new(),
112        }
113    }
114
115    pub fn is_readable(&self) -> bool {
116        now_unix_seconds() >= self.cursor.write_completion_time()
117    }
118
119    pub fn cursor_seconds(&self) -> UnixSeconds {
120        self.cursor.to_unix_seconds()
121    }
122
123    pub fn advance_cursor(&mut self) {
124        self.cursor = self.cursor.next_dir();
125    }
126
127    /// Read and verify the signatures on all the records in the current directory.
128    pub async fn read_cur_dir(&mut self) -> anyhow::Result<Vec<VerifiedLogRecord>> {
129        let all_logs = self
130            .s3_client
131            .list_all_objects_in_dir::<LogRecord>(&self.cursor)
132            .await
133            .with_context(|| format!("failed to list guardian logs in {}", self.cursor))?;
134
135        let mut out = Vec::with_capacity(all_logs.len());
136        for log in all_logs {
137            self.ensure_session_loaded(&log.session_id).await?;
138
139            let signing_pubkey = self
140                .enclave_pub_keys
141                .get(&log.session_id)
142                .ok_or_else(|| anyhow::anyhow!("missing session signing pubkey"))?;
143
144            let verified = log
145                .verify(signing_pubkey)
146                .with_context(|| "failed to verify guardian enclave signature")?;
147
148            out.push(verified);
149        }
150
151        Ok(out)
152    }
153
154    async fn ensure_session_loaded(&mut self, session_id: &str) -> anyhow::Result<()> {
155        if self.enclave_pub_keys.contains_key(session_id) {
156            return Ok(());
157        }
158
159        let (attestation, signing_pubkey) = self.s3_client.get_attestation(session_id).await?;
160        verify_enclave_attestation(attestation)?;
161
162        self.enclave_pub_keys
163            .insert(session_id.to_string(), signing_pubkey);
164        Ok(())
165    }
166}

168// Note: current design does not check if multiple concurrent sessions are running.
169//       one way to impl this: store the first & last observed session timestamp & ensure no overlap between time ranges.
170pub struct GuardianWithdrawalsPoller(GuardianPollerCore);
171
172impl GuardianWithdrawalsPoller {
173    // Note: Throws an error if there is a S3 connectivity issue
174    pub async fn new(config: &Config, start: UnixSeconds) -> anyhow::Result<Self> {
175        Ok(Self(
176            GuardianPollerCore::new(&config.guardian, start, GuardianLogDir::Withdraw).await?,
177        ))
178    }
179
180    pub fn cursor_seconds(&self) -> UnixSeconds {
181        self.0.cursor_seconds()
182    }
183
184    /// Polls the Guardian S3 bucket for one hour worth of events.
185    /// A more aggressive fetch, e.g., one day at a time, can also be done if needed.
186    pub async fn poll_one_hour(&mut self) -> anyhow::Result<PollOutcome> {
187        if !self.0.is_readable() {
188            return Ok(PollOutcome::CursorUnmoved);
189        }
190
191        let verified_logs = self.0.read_cur_dir().await?;
192        let withdrawal_events = verified_logs
193            .into_iter()
194            .map(VerifiedWithdrawal::try_from)
195            .collect::<anyhow::Result<Vec<_>>>()?
196            .into_iter()
197            .filter_map(|e| match e {
198                VerifiedWithdrawal::Success(event) => Some(MonitorEvent::Withdrawal(event)),
199                VerifiedWithdrawal::Failure => None,
200            })
201            .collect::<Vec<MonitorEvent>>();
202
203        self.0.advance_cursor();
204        Ok(PollOutcome::CursorAdvanced(withdrawal_events))
205    }
