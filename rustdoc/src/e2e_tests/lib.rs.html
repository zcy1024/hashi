5//!
6//! The general bootstrapping process is as follows:
7//! 1. Stand up a Bitcoin regtest

201    pub async fn build(self) -> Result<TestNetworks> {
202        let dir = tempfile::Builder::new()
203            .prefix("hashi-test-env-")
204            .tempdir()?;
205
206        tracing::info!("test env: {}", dir.path().display());
207
208        let bitcoin_node = self.bitcoin_builder.dir(dir.as_ref()).build().await?;
209
210        let mut sui_network = self
211            .sui_builder
212            .dir(&dir.path().join("sui"))
213            .build()
214            .await?;
215        Self::cp_packages(dir.as_ref())?;
216
217        let hashi_ids = publish(
218            dir.as_ref(),
219            &mut sui_network.client,
220            sui_network.user_keys.first().unwrap(),
221        )
222        .await?;
223
224        let hashi_network = self
225            .hashi_builder
226            .build(
227                &dir.path().join("hashi"),
228                &sui_network,
229                &bitcoin_node,
230                hashi_ids,
231            )
232            .await?;
233
234        let mut test_networks = TestNetworks {
235            dir,
236            sui_network,
237            hashi_network,
238            bitcoin_node,
239        };
240
241        tracing::info!("rpc url: {}", test_networks.sui_network().rpc_url);
242
243        if !self.onchain_config_overrides.is_empty() {
244            apply_onchain_config_overrides(&mut test_networks, &self.onchain_config_overrides)
245                .await?;
246        }
247
248        Ok(test_networks)
249    }

268/// Apply on-chain config overrides by running the full propose/vote/execute
269/// cycle for each `(key, value)` pair. Called from `TestNetworksBuilder::build`
270/// when overrides are present.
271///
272/// Waits for DKG to complete first so the committee is ready to vote.
273/// All nodes vote on every proposal, ensuring quorum is always reached
274/// regardless of the number of nodes or their weight distribution.
275pub(crate) async fn apply_onchain_config_overrides(
276    networks: &mut TestNetworks,
277    overrides: &[(String, hashi_types::move_types::ConfigValue)],
278) -> Result<()> {
279    use hashi::cli::client::CreateProposalParams;
280    use hashi::cli::client::build_create_proposal_transaction;
281    use hashi::cli::client::build_vote_transaction;
282    use hashi::cli::upgrade::build_execute_proposal_transaction;
283    use hashi::cli::upgrade::extract_proposal_id_from_response;
284    use hashi::sui_tx_executor::SuiTxExecutor;
285    use sui_sdk_types::Identifier;
286    use sui_sdk_types::StructTag;
287    use sui_sdk_types::TypeTag;
288
289    let nodes = networks.hashi_network.nodes();
290
291    // The committee is only available after DKG. Wait on the first node; the
292    // others are guaranteed to be ready too once DKG completes.
293    nodes[0]
294        .wait_for_mpc_key(std::time::Duration::from_secs(120))
295        .await?;
296
297    let hashi_ids = networks.hashi_network.ids();
298
299    // Build one executor per node, reused across all overrides.
300    let mut executors: Vec<SuiTxExecutor> = nodes
301        .iter()
302        .map(|node| {
303            let hashi = node.hashi();
304            SuiTxExecutor::from_config(&hashi.config, hashi.onchain_state())
305        })
306        .collect::<anyhow::Result<_>>()?;
307
308    // Updated to the checkpoint of each execute response; used after the loop
309    // to wait for all nodes to catch up to the last applied override.
310    let mut exec_checkpoint: u64 = 0;
311
312    let update_config_type_tag = TypeTag::Struct(Box::new(StructTag::new(
313        hashi_ids.package_id,
314        Identifier::from_static("update_config"),
315        Identifier::from_static("UpdateConfig"),
316        vec![],
317    )));
318
319    //TODO could we build the proposals and vote/execute on them all at the same time vs doing them
320    //one at a time?
321    for (key, value) in overrides {
322        tracing::info!("applying on-chain config override: {key} = {value:?}");
323
324        // 1. Node 0 creates the proposal (and automatically casts its own vote).
325        let create_tx = build_create_proposal_transaction(
326            hashi_ids,
327            CreateProposalParams::UpdateConfig {
328                key: key.clone(),
329                value: value.clone(),
330                metadata: vec![],
331            },
332        );
333        let response = executors[0].execute(create_tx).await?;
334        anyhow::ensure!(
335            response.transaction().effects().status().success(),
336            "create UpdateConfig proposal for '{key}' failed"
337        );
338
339        let proposal_id = extract_proposal_id_from_response(&response)?;
340        tracing::info!("proposal {proposal_id} created for '{key}'; collecting votes");
341
342        // 2. All remaining nodes vote. This gives 100% of total weight,
343        //    guaranteeing the 66.67% quorum threshold is met.
344        for executor in &mut executors[1..] {
345            let vote_tx =
346                build_vote_transaction(hashi_ids, proposal_id, update_config_type_tag.clone());
347            let vote_resp = executor.execute(vote_tx).await?;
348            anyhow::ensure!(
349                vote_resp.transaction().effects().status().success(),
350                "vote on UpdateConfig proposal {proposal_id} for '{key}' failed"
351            );
352        }
353
354        // 3. Node 0 executes the proposal now that quorum is reached.
355        let execute_tx = build_execute_proposal_transaction(
356            hashi_ids,
357            proposal_id,
358            hashi_ids.package_id,
359            "update_config",
360        )?;
361        let exec_resp = executors[0].execute(execute_tx).await?;
362        anyhow::ensure!(
363            exec_resp.transaction().effects().status().success(),
364            "execute UpdateConfig proposal {proposal_id} for '{key}' failed"
365        );
366
367        exec_checkpoint = exec_resp
368            .transaction()
369            .checkpoint_opt()
370            .ok_or_else(|| anyhow::anyhow!("execute transaction response missing checkpoint"))?;
371
372        tracing::info!("on-chain config override applied: {key} (checkpoint {exec_checkpoint})");
373    }
374
375    // Wait for all nodes' watchers to process the checkpoint that contains the
376    // last execute transaction. The watcher re-fetches config on each
377    // ProposalExecutedEvent<UpdateConfig>, so once a node reaches this
378    // checkpoint its in-memory config will reflect the override.
379    let futs = networks.hashi_network().nodes().iter().map(|node| {
380        let mut subscription = node.hashi().onchain_state().subscribe_checkpoint();
381        async move {
382            while subscription.borrow().height < exec_checkpoint {
383                subscription.changed().await.unwrap();
384            }
385        }
386    });
387    tokio::time::timeout(
388        std::time::Duration::from_secs(30),
389        futures::future::join_all(futs),
390    )
391    .await?;
392
393    Ok(())
394}

432    /// Wait for all nodes to reach at least `target_epoch`.
433    /// Returns the actual epoch of `nodes[0]` after the wait (may exceed `target_epoch`).
434    async fn wait_for_rotation(nodes: &[HashiNodeHandle], target_epoch: u64) -> u64 {
435        let futures: Vec<_> = nodes
436            .iter()
437            .map(|node| node.wait_for_epoch(target_epoch, ROTATION_TIMEOUT))
438            .collect();
439        let results: Vec<Result<()>> = futures::future::join_all(futures).await;
440        for (i, result) in results.into_iter().enumerate() {
441            result.unwrap_or_else(|e| panic!("Node {i} failed to reach epoch {target_epoch}: {e}"));
442        }
443        nodes[0].current_epoch().unwrap()
444    }

446    async fn wait_for_signing_manager(
447        nodes: &[HashiNodeHandle],
448        epoch: u64,
449        timeout: std::time::Duration,
450    ) -> Result<()> {
451        let deadline = tokio::time::Instant::now() + timeout;
452        loop {
453            let all_ready = nodes
454                .iter()
455                .all(|node| node.hashi().signing_manager_for(epoch).is_some());
456            if all_ready {
457                return Ok(());
458            }
459            if tokio::time::Instant::now() >= deadline {
460                let statuses: Vec<_> = nodes
461                    .iter()
462                    .enumerate()
463                    .map(|(i, node)| (i, node.hashi().signing_manager_for(epoch).is_some()))
464                    .collect();
465                return Err(anyhow::anyhow!(
466                    "Timed out waiting for SigningManager for epoch {epoch} on all nodes: {statuses:?}"
467                ));
468            }
469            tokio::time::sleep(std::time::Duration::from_millis(200)).await;
470        }
471    }

675    /// Have all nodes call sign() concurrently, returning per-node results.
676    async fn sign_on_all_nodes(
677        nodes: &[HashiNodeHandle],
678        message: &[u8],
679        epoch: u64,
680        sui_request_id: sui_sdk_types::Address,
681        global_presig_index: u64,
682    ) -> Vec<
683        hashi::mpc::types::SigningResult<fastcrypto::groups::secp256k1::schnorr::SchnorrSignature>,
684    > {
685        let beacon_value = S::rand(&mut rand::thread_rng());
686        let sign_futures: Vec<_> = nodes
687            .iter()
688            .map(|node| {
689                let signing_manager = node
690                    .hashi()
691                    .signing_manager_for(epoch)
692                    .unwrap_or_else(|| panic!("SigningManager not initialized for epoch {epoch}"));
693                let onchain_state = node.hashi().onchain_state().clone();
694                let p2p_channel = hashi::mpc::rpc::RpcP2PChannel::new(
695                    onchain_state,
696                    epoch,
697                    hashi::metrics::MPC_LABEL_SIGNING,
698                );
699                let beacon = beacon_value;
700                let message = message.to_vec();
701                let metrics = node.hashi().metrics.clone();
702                async move {
703                    signing_manager
704                        .sign(
705                            &p2p_channel,
706                            sui_request_id,
707                            &message,
708                            global_presig_index,
709                            &beacon,
710                            None,
711                            SIGNING_TIMEOUT,
712                            &metrics,
713                        )
714                        .await
715                }
716            })
717            .collect();
718        futures::future::join_all(sign_futures).await
719    }
