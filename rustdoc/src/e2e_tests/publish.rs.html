5use hashi::config::HashiIds;

7use sui_crypto::ed25519::Ed25519PrivateKey;
8use sui_rpc::Client;
9

12pub async fn publish(
13    dir: &Path,
14    client: &mut Client,
15    private_key: &Ed25519PrivateKey,
16) -> Result<HashiIds> {
17    let params = hashi::publish::BuildParams {
18        sui_binary: sui_binary(),
19        package_path: &dir.join("packages/hashi"),
20        client_config: Some(&dir.join("sui/client.yaml")),
21        environment: Some("testnet"),
22    };
23    let compiled = hashi::publish::build_package(&params)?;
24    let bitcoin_chain_id = hashi::constants::BITCOIN_REGTEST_CHAIN_ID;
25    hashi::publish::publish_and_init(client, private_key, compiled, bitcoin_chain_id).await
26}
