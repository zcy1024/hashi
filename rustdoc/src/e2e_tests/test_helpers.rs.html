51pub async fn get_hbtc_balance(
52    sui_client: &mut sui_rpc::Client,
53    package_id: Address,
54    owner: Address,
55) -> Result<u64> {
56    let btc_type = format!("{package_id}::btc::BTC");
57    let btc_struct_tag: StructTag = btc_type.parse()?;
58    let request = GetBalanceRequest::default()
59        .with_owner(owner.to_string())
60        .with_coin_type(btc_struct_tag.to_string());
61
62    let response = sui_client
63        .state_client()
64        .get_balance(request)
65        .await?
66        .into_inner();
67
68    let balance = response.balance().balance_opt().unwrap_or(0);
69    debug!("hBTC balance for {owner}: {balance} sats");
70    Ok(balance)
71}

73pub async fn wait_for_deposit_confirmation(
74    sui_client: &mut sui_rpc::Client,
75    request_id: Address,
76    timeout: Duration,
77) -> Result<()> {
78    info!("Waiting for deposit confirmation for request_id: {request_id}");
79
80    let start = std::time::Instant::now();
81    let read_mask = FieldMask::from_paths([Checkpoint::path_builder()
82        .transactions()
83        .events()
84        .events()
85        .contents()
86        .finish()]);
87    let mut subscription = sui_client
88        .subscription_client()
89        .subscribe_checkpoints(SubscribeCheckpointsRequest::default().with_read_mask(read_mask))
90        .await?
91        .into_inner();
92
93    while let Some(item) = subscription.next().await {
94        if start.elapsed() > timeout {
95            return Err(anyhow!(
96                "Timeout waiting for deposit confirmation after {timeout:?}"
97            ));
98        }
99
100        let checkpoint = match item {
101            Ok(cp) => cp,
102            Err(e) => {
103                debug!("Error in checkpoint stream: {e}");
104                continue;
105            }
106        };
107
108        for txn in checkpoint.checkpoint().transactions() {
109            for event in txn.events().events() {
110                if event.contents().name().contains("DepositConfirmedEvent")
111                    && let Ok(evt) = DepositConfirmedEvent::from_bcs(event.contents().value())
112                    && evt.request_id == request_id
113                {
114                    info!("Deposit confirmed for request_id: {request_id}");
115                    return Ok(());
116                }
117            }
118        }
119    }
120
121    Err(anyhow!("Checkpoint subscription ended unexpectedly"))
122}

124pub fn lookup_vout(
125    networks: &TestNetworks,
126    txid: Txid,
127    address: bitcoin::Address,
128    amount: u64,
129) -> Result<usize> {
130    let tx = networks
131        .bitcoin_node
132        .rpc_client()
133        .get_raw_transaction(txid)
134        .and_then(|r| r.transaction().map_err(Into::into))?;
135    let vout = tx
136        .output
137        .iter()
138        .position(|output| {
139            output.value == Amount::from_sat(amount)
140                && output.script_pubkey == address.script_pubkey()
141        })
142        .ok_or_else(|| anyhow!("Could not find output with amount {amount} and deposit address"))?;
143    debug!("Found deposit in tx output {vout}");
144    Ok(vout)
145}

147/// Deposit BTC and wait for the validators to auto-confirm it via the full
148/// observe → sign → confirm path. Returns the hBTC recipient address.
149pub async fn create_deposit_and_wait(
150    networks: &mut TestNetworks,
151    amount_sats: u64,
152) -> Result<Address> {
153    let user_key = networks.sui_network.user_keys.first().unwrap();
154    let hbtc_recipient = user_key.public_key().derive_address();
155    let hashi = networks.hashi_network.nodes()[0].hashi().clone();
156    // Use the on-chain MPC key rather than the local key-ready channel.
157    // The on-chain key is set during end_reconfig and is guaranteed
158    // available once HashiNetworkBuilder::build() returns.
159    let deposit_address =
160        hashi.get_deposit_address(&hashi.get_onchain_mpc_pubkey()?, Some(&hbtc_recipient))?;
161
162    info!("Sending Bitcoin to deposit address...");
163    let txid = networks
164        .bitcoin_node
165        .send_to_address(&deposit_address, Amount::from_sat(amount_sats))?;
166    info!("Transaction sent: {txid}");
167
168    info!("Mining blocks for confirmation...");
169    let blocks_to_mine = 10;
170    networks.bitcoin_node.generate_blocks(blocks_to_mine)?;
171    info!("{blocks_to_mine} blocks mined");
172
173    info!("Creating deposit request on Sui...");
174    let vout = lookup_vout(networks, txid, deposit_address, amount_sats)?;
175    let mut executor = SuiTxExecutor::from_config(&hashi.config, hashi.onchain_state())?
176        .with_signer(user_key.clone());
177    let request_id = executor
178        .execute_create_deposit_request(
179            txid_to_address(&txid),
180            vout as u32,
181            amount_sats,
182            Some(hbtc_recipient),
183        )
184        .await?;
185    info!("Deposit request created: {request_id}");
186
187    // Mine blocks in the background so the leader's BTC-block-driven
188    // deposit processing loop fires.
189    let _miner = BackgroundMiner::start(&networks.bitcoin_node);
190    wait_for_deposit_confirmation(
191        &mut networks.sui_network.client,
192        request_id,
193        Duration::from_secs(300),
194    )
195    .await?;
196    info!("Deposit confirmed on Sui");
197
198    Ok(hbtc_recipient)
199}

201/// Mines one block per second on Bitcoin regtest until stopped.
202/// Stops automatically when dropped.
203pub struct BackgroundMiner {
204    stop_flag: Arc<AtomicBool>,
205    handle: Option<std::thread::JoinHandle<()>>,
206}
207
208impl BackgroundMiner {
209    pub fn start(bitcoin_node: &BitcoinNodeHandle) -> Self {
210        let stop_flag = Arc::new(AtomicBool::new(false));
211        let stop_clone = stop_flag.clone();
212        let rpc_url = bitcoin_node.rpc_url().to_string();
213        let handle = std::thread::spawn(move || {
214            let rpc = corepc_client::client_sync::v29::Client::new_with_auth(
215                &rpc_url,
216                corepc_client::client_sync::Auth::UserPass(
217                    crate::bitcoin_node::RPC_USER.to_string(),
218                    crate::bitcoin_node::RPC_PASSWORD.to_string(),
219                ),
220            )
221            .expect("failed to create mining RPC client");
222            let addr = rpc.new_address().expect("failed to get mining address");
223            while !stop_clone.load(Ordering::Relaxed) {
224                let _ = rpc.generate_to_address(1, &addr);
225                std::thread::sleep(Duration::from_secs(1));
226            }
227        });
228        Self {
229            stop_flag,
230            handle: Some(handle),
231        }
232    }
233}
234
235impl Drop for BackgroundMiner {
236    fn drop(&mut self) {
237        self.stop_flag.store(true, Ordering::Relaxed);
238        if let Some(handle) = self.handle.take() {
239            let _ = handle.join();
240        }
241    }
