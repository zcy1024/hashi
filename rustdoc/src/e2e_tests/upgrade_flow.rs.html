31/// Prepare an upgrade package by copying the deployed source and patching it.
32///
33/// 1. Copies `<test_dir>/packages/hashi` to `<test_dir>/packages/hashi-upgrade`
34/// 2. Bumps `PACKAGE_VERSION` from 1 to 2 in `config.move`
35/// 3. Sets `published-at` in `Move.toml` to the original package ID
36///
37/// Returns the path to the patched package directory.
38pub fn prepare_upgrade_package(test_dir: &Path, original_package_id: Address) -> Result<PathBuf> {
39    let src = test_dir.join("packages/hashi");
40    let dst = test_dir.join("packages/hashi-upgrade");
41
42    anyhow::ensure!(
43        src.exists(),
44        "source package not found at {}",
45        src.display()
46    );
47
48    // Copy the package
49    let output = std::process::Command::new("cp")
50        .args(["-r", &src.to_string_lossy(), &dst.to_string_lossy()])
51        .output()?;
52    anyhow::ensure!(
53        output.status.success(),
54        "failed to copy package: {}",
55        String::from_utf8_lossy(&output.stderr)
56    );
57
58    // Patch config.move: bump PACKAGE_VERSION from 1 to 2
59    let config_path = dst.join("sources/core/config/config.move");
60    let config_src = std::fs::read_to_string(&config_path)?;
61    let patched = config_src.replace(
62        "const PACKAGE_VERSION: u64 = 1;",
63        "const PACKAGE_VERSION: u64 = 2;",
64    );
65    anyhow::ensure!(
66        patched != config_src,
67        "PACKAGE_VERSION replacement failed — pattern not found in config.move"
68    );
69    std::fs::write(&config_path, patched)?;
70
71    // Patch Move.toml: add published-at
72    let move_toml_path = dst.join("Move.toml");
73    let move_toml = std::fs::read_to_string(&move_toml_path)?;
74    let patched_toml = move_toml.replace(
75        "[package]",
76        &format!("[package]\npublished-at = \"{}\"", original_package_id),
77    );
78    std::fs::write(&move_toml_path, patched_toml)?;
79
80    // Add a trivial v2-only module to prove new code is callable post-upgrade
81    let test_module_path = dst.join("sources/upgrade_canary.move");
82    std::fs::write(
83        &test_module_path,
84        "module hashi::upgrade_canary;\n\npublic fun version(): u64 { 2 }\n",
85    )?;
86
87    // Clean build artifacts from the copy
88    let _ = std::fs::remove_dir_all(dst.join("build"));
89
90    tracing::info!(
91        "upgrade package prepared at {} (published-at = {})",
92        dst.display(),
93        original_package_id
94    );
95
96    Ok(dst)
97}

99/// Run the full upgrade lifecycle: prepare → build → propose → vote → execute+publish+finalize.
100///
101/// Returns the new package ID on success.
102pub async fn execute_full_upgrade(networks: &mut TestNetworks) -> Result<Address> {
103    let nodes = networks.hashi_network.nodes();
104    let hashi_ids = networks.hashi_network.ids();
105
106    let mut executors: Vec<SuiTxExecutor> = nodes
107        .iter()
108        .map(|node| {
109            let hashi = node.hashi();
110            SuiTxExecutor::from_config(&hashi.config, hashi.onchain_state())
111        })
112        .collect::<Result<_>>()?;
113
114    // 1. Prepare the upgrade package (copy + patch)
115    let test_dir = networks.dir();
116    let upgrade_path = prepare_upgrade_package(test_dir, hashi_ids.package_id)?;
117
118    let client_config_path = test_dir.join("sui/client.yaml");
119    let client_config = client_config_path
120        .exists()
121        .then_some(client_config_path.as_path());
122
123    // 2. Build the upgrade
124    tracing::info!("building upgrade package from {}", upgrade_path.display());
125    let current_version = nodes[0]
126        .hashi()
127        .onchain_state()
128        .state()
129        .package_versions()
130        .keys()
131        .copied()
132        .max()
133        .ok_or_else(|| anyhow::anyhow!("onchain state has no package versions yet"))?;
134    let (compiled, digest) = build_upgrade_package(
135        sui_binary(),
136        &upgrade_path,
137        client_config,
138        current_version + 1,
139    )?;
140    tracing::info!("upgrade package built, digest: {digest:?}");
141
142    // 3. Propose the upgrade
143    tracing::info!("proposing upgrade...");
144    let create_tx = build_create_proposal_transaction(
145        hashi_ids,
146        CreateProposalParams::Upgrade {
147            digest: digest.clone(),
148            metadata: vec![("reason".to_string(), "upgrade test".to_string())],
149        },
150    );
151    let response = executors[0].execute(create_tx).await?;
152    anyhow::ensure!(
153        response.transaction().effects().status().success(),
154        "create Upgrade proposal failed"
155    );
156
157    let proposal_id = extract_proposal_id_from_response(&response)?;
158    tracing::info!("upgrade proposal {proposal_id} created");
159
160    // 4. All other nodes vote (upgrade requires 100% quorum)
161    let upgrade_type_tag = TypeTag::Struct(Box::new(StructTag::new(
162        hashi_ids.package_id,
163        Identifier::from_static("upgrade"),
164        Identifier::from_static("Upgrade"),
165        vec![],
166    )));
167
168    for executor in &mut executors[1..] {
169        let vote_tx = build_vote_transaction(hashi_ids, proposal_id, upgrade_type_tag.clone());
170        let vote_resp = executor.execute(vote_tx).await?;
171        anyhow::ensure!(
172            vote_resp.transaction().effects().status().success(),
173            "vote on Upgrade proposal failed"
174        );
175    }
176    tracing::info!("all nodes voted on upgrade proposal");
177
178    // 5. Execute upgrade + publish + finalize in one PTB
179    tracing::info!("executing upgrade (execute + publish + finalize in one PTB)...");
180    let upgrade_tx = build_upgrade_execution_transaction(hashi_ids, proposal_id, compiled);
181    let upgrade_resp = executors[0].execute(upgrade_tx).await?;
182    anyhow::ensure!(
183        upgrade_resp.transaction().effects().status().success(),
184        "upgrade execute+publish+finalize failed: {:?}",
185        upgrade_resp.transaction().effects().status()
186    );
187
188    let new_package_id = extract_new_package_id_from_response(&upgrade_resp)?;
189    tracing::info!("upgrade complete! new package: {new_package_id}");
190    Ok(new_package_id)
191}

193/// Propose + vote + execute a DisableVersion governance action.
194///
195/// `execute_package_id` is the package whose `disable_version::execute` is called.
196/// When disabling an old version after upgrade, this must be the NEW package ID
197/// (whose `PACKAGE_VERSION` differs from the version being disabled).
198pub async fn disable_version(
199    executors: &mut [SuiTxExecutor],
200    hashi_ids: HashiIds,
201    version: u64,
202    execute_package_id: Address,
203) -> Result<()> {
204    let create_tx = build_create_proposal_transaction(
205        hashi_ids,
206        CreateProposalParams::DisableVersion {
207            version,
208            metadata: vec![],
209        },
210    );
211    let response = executors[0].execute(create_tx).await?;
212    anyhow::ensure!(
213        response.transaction().effects().status().success(),
214        "create DisableVersion proposal failed"
215    );
216
217    let proposal_id = extract_proposal_id_from_response(&response)?;
218
219    let disable_version_type = TypeTag::Struct(Box::new(StructTag::new(
220        hashi_ids.package_id,
221        Identifier::from_static("disable_version"),
222        Identifier::from_static("DisableVersion"),
223        vec![],
224    )));
225
226    for executor in &mut executors[1..] {
227        let vote_tx = build_vote_transaction(hashi_ids, proposal_id, disable_version_type.clone());
228        let vote_resp = executor.execute(vote_tx).await?;
229        anyhow::ensure!(
230            vote_resp.transaction().effects().status().success(),
231            "vote on DisableVersion proposal failed"
232        );
233    }
234
235    let execute_tx = build_execute_proposal_transaction(
236        hashi_ids,
237        proposal_id,
238        execute_package_id,
239        "disable_version",
240    )?;
241    let exec_resp = executors[0].execute(execute_tx).await?;
242    anyhow::ensure!(
243        exec_resp.transaction().effects().status().success(),
244        "execute DisableVersion proposal failed"
245    );
246
247    tracing::info!("version {version} disabled");
248    Ok(())
249}
