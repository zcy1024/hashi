47    pub async fn new(config: &S3Config) -> Self {
48        info!("S3 Configuration:");
49        info!("   Bucket: {}", config.bucket_name());
50        info!("   Region: {}", config.region());
51
52        let creds = CredentialsBuilder::default()
53            .access_key_id(config.access_key.clone())
54            .secret_access_key(config.secret_key.clone())
55            .provider_name("hashi-guardian")
56            .build();
57
58        let retry_config = RetryConfig::standard().with_max_attempts(MAX_RETRY_ATTEMPTS); // default is 3
59
60        let aws_config = aws_config::defaults(aws_config::BehaviorVersion::latest())
61            .region(aws_config::Region::new(config.region().to_string()))
62            .credentials_provider(SharedCredentialsProvider::new(creds))
63            .retry_config(retry_config)
64            .load()
65            .await;
66        let client = S3Client::new(&aws_config);
67
68        Self {
69            client,
70            config: config.clone(),
71        }
72    }
73
74    pub async fn new_checked(config: &S3Config) -> GuardianResult<Self> {
75        let logger = Self::new(config).await;
76        logger.test_s3_connectivity().await?;
77        Ok(logger)

99    pub async fn write_log_record(&self, log: LogRecord) -> GuardianResult<()> {
100        let object_lock_duration = log.object_lock_duration();
101        let key = log.object_key();
102        self.write_at_key(&key, &log, object_lock_duration).await
103    }

105    /// Write a value to S3 at an explicit key.
106    ///
107    /// This is intended for ordered log streams where the caller determines the key.
108    async fn write_at_key<T: Serialize>(
109        &self,
110        key: &str,
111        value: &T,
112        object_lock_duration: Duration,
113    ) -> GuardianResult<()> {
114        let s3_client = &self.client;
115        let s3_config = &self.config;
116
117        info!("Logging to {}", key);
118
119        let expiry_time = SystemTime::now()
120            .checked_add(object_lock_duration)
121            .expect("Cant overflow");
122
123        let body = serde_json::to_string(value).expect("Cant serialize to JSON");
124        info!("Log message: {}", body);
125
126        // TODO(integration-test): Verify on a real S3 bucket with Object Lock enabled that an object written with `Compliance` + `retain_until` cannot be deleted/overwritten before expiry.
127        let _result = s3_client
128            .put_object()
129            .bucket(s3_config.bucket_name())
130            .key(key)
131            .content_type("application/json")
132            .object_lock_mode(ObjectLockMode::Compliance)
133            .object_lock_retain_until_date(DateTime::from(expiry_time))
134            .body(ByteStream::from(body.into_bytes()))
135            .send()
136            .await
137            // DisplayErrorContext displays the full error returned by the SDK
138            .map_err(|e| {
139                S3Error(format!(
140                    "Failed to write to s3: {}",
141                    DisplayErrorContext(&e)
142                ))
143            })?;
144
145        info!("Logged entry to immutable storage");
146        info!("Object locked until: {:?}", expiry_time);
147        info!(
148            "Public URL: https://{}.s3.amazonaws.com/{}",
149            s3_config.bucket_name(),
150            key
151        );
152
153        Ok(())
154    }

164    /// Verify that the S3 bucket has object lock enabled and returns an Err if not.
165    /// Can be used as a test for S3 connectivity.
166    pub async fn assert_object_lock_enabled(&self) -> GuardianResult<()> {
167        let s3_client = &self.client;
168        let s3_config = &self.config;
169
170        // Verify bucket exists and has Object Lock enabled
171        let bucket_config = s3_client
172            .get_object_lock_configuration()
173            .bucket(s3_config.bucket_name())
174            .send()
175            .await;
176
177        match bucket_config {
178            Ok(config) => {
179                let object_lock_config = config.object_lock_configuration().ok_or_else(|| {
180                    S3Error("Object lock configuration missing in S3 response".into())
181                })?;
182
183                let object_lock_enabled_config =
184                    object_lock_config.object_lock_enabled().ok_or_else(|| {
185                        S3Error("Object lock enabled field missing in S3 response".into())
186                    })?;
187
188                match object_lock_enabled_config {
189                    ObjectLockEnabled::Enabled => {
190                        info!("Bucket {} has Object Lock enabled", s3_config.bucket_name());
191                    }
192                    other => {
193                        return Err(S3Error(format!(
194                            "Unexpected object lock enabled config: {:?}",
195                            other
196                        )))
197                    }
198                }
199            }
200            Err(e) => {
201                return Err(S3Error(format!(
202                    "Failed to verify Object Lock configuration: {}",
203                    DisplayErrorContext(&e)
204                )));
205            }
206        }
207
208        Ok(())
209    }

211    /// List up to 10 objects in the bucket.
212    /// This is intended as a lightweight connectivity/debug helper (primarily for testing).
213    pub async fn list_objects_sample(&self) -> GuardianResult<()> {
214        let s3_client = &self.client;
215        let s3_config = &self.config;
216
217        let bucket_objects = s3_client
218            .list_objects_v2()
219            .bucket(s3_config.bucket_name())
220            .max_keys(10)
221            .send()
222            .await
223            .map_err(|e| {
224                S3Error(format!(
225                    "Failed to list objects: {}",
226                    DisplayErrorContext(&e)
227                ))
228            })?;
229
230        let objects = bucket_objects.contents();
231
232        if objects.is_empty() {
233            info!(
234                "Bucket {} has no objects (or no access to list)",
235                s3_config.bucket_name()
236            );
237            return Ok(());
238        }
239
240        info!(
241            "Bucket {}: listing {} object(s) (max 10)",
242            s3_config.bucket_name(),
243            objects.len()
244        );
245
246        for (i, obj) in objects.iter().enumerate() {
247            let key = obj.key().unwrap_or("<missing key>");
248            info!(
249                "  {}. key={} size={:?} last_modified={:?} etag={:?}",
250                i + 1,
251                key,
252                obj.size(),
253                obj.last_modified(),
254                obj.e_tag()
255            );
256        }
257
258        Ok(())
259    }

265    /// Checks that all matching object keys do not have either deletions or overwrites.
266    /// The prefix can either correspond to a directory or a complete object key.
267    ///
268    /// Returns: list of keys.
269    async fn ensure_no_duplicates_or_deletions(&self, prefix: &str) -> GuardianResult<Vec<String>> {
270        let s3_client = &self.client;
271        let s3_config = &self.config;
272
273        let mut key_marker: Option<String> = None;
274        let mut version_id_marker: Option<String> = None;
275        let mut seen_keys: BTreeSet<String> = BTreeSet::new();
276
277        loop {
278            let mut req = s3_client
279                .list_object_versions()
280                .bucket(s3_config.bucket_name())
281                .prefix(prefix);
282            if let Some(ref marker) = key_marker {
283                req = req.key_marker(marker);
284            }
285            if let Some(ref marker) = version_id_marker {
286                req = req.version_id_marker(marker);
287            }
288
289            let response = req.send().await.map_err(|e| {
290                S3Error(format!(
291                    "Failed to list object versions for prefix {}: {}",
292                    prefix,
293                    DisplayErrorContext(&e)
294                ))
295            })?;
296
297            if !response.delete_markers().is_empty() {
298                return Err(S3Error(format!(
299                    "Delete marker found under prefix {}",
300                    prefix
301                )));
302            }
303
304            // https://docs.aws.amazon.com/AmazonS3/latest/API/API_ObjectVersion.html
305            for version in response.versions() {
306                let key = version.key().ok_or_else(|| {
307                    S3Error("Missing key in list_object_versions response".into())
308                })?;
309
310                // NOTE: If an object's lock expires, then all bets are off.
311                // For example, is_latest could be true even though an older version of it was deleted (post lock expiry).
312                if version.is_latest() != Some(true) {
313                    return Err(S3Error(format!(
314                        "Non-latest version found for key {} under prefix {}",
315                        key, prefix
316                    )));
317                }
318
319                if !seen_keys.insert(key.to_string()) {
320                    // this check is redundant as we ensure is_latest = true above
321                    return Err(S3Error(format!(
322                        "Duplicate version found for key {} under prefix {}",
323                        key, prefix
324                    )));
325                }
326            }
327
328            if response.is_truncated() != Some(true) {
329                break;
330            }
331
332            key_marker = response.next_key_marker().map(ToString::to_string);
333            version_id_marker = response.next_version_id_marker().map(ToString::to_string);
334
335            if key_marker.is_none() {
336                return Err(S3Error(format!(
337                    "Truncated response but no next_key_marker for prefix {}",
338                    prefix
339                )));
340            }
341        }
342
343        Ok(seen_keys.into_iter().collect())
344    }

346    /// Batch read. Callers must ensure that all objects with prefix `dir.to_string()` have
347    /// unexpired compliance-mode object locks.
348    ///
349    /// Returns: List of objects.
350    pub async fn list_all_objects_in_dir<T: DeserializeOwned>(
351        &self,
352        dir: &S3HourScopedDirectory,
353    ) -> GuardianResult<Vec<T>> {
354        let prefix = dir.to_string();
355        let keys = self.ensure_no_duplicates_or_deletions(&prefix).await?;
356        let mut out = Vec::with_capacity(keys.len());
357        for key in &keys {
358            let obj: T = self.get_object_unsafe(key).await?;
359            out.push(obj);
360        }
361        Ok(out)
362    }

364    /// Point read. This method is unsafe to use since the bucket operator might've overwritten objects.
365    async fn get_object_unsafe<T: DeserializeOwned>(&self, key: &str) -> GuardianResult<T> {
366        let s3_client = &self.client;
367        let s3_config = &self.config;
368
369        let response = s3_client
370            .get_object()
371            .bucket(s3_config.bucket_name())
372            .key(key)
373            .send()
374            .await
375            .map_err(|e| {
376                S3Error(format!(
377                    "Failed to get object {}: {}",
378                    key,
379                    DisplayErrorContext(&e)
380                ))
381            })?;
382
383        // NOTE: Here we are explicitly assuming locks are unexpired.
384        if response.object_lock_mode() != Some(&ObjectLockMode::Compliance)
385            || response.object_lock_retain_until_date().is_none()
386        {
387            return Err(S3Error(format!(
388                "Missing or invalid object lock metadata for key {}",
389                key
390            )));
391        }
392
393        let bytes = response.body.collect().await.map_err(|e| {
394            S3Error(format!(
395                "Failed to read object body for key {}: {}",
396                key,
397                DisplayErrorContext(&e)
398            ))
399        })?;
400
401        serde_json::from_slice::<T>(&bytes.into_bytes()).map_err(|e| {
402            S3Error(format!(
403                "Failed to deserialize object {} into target type: {}",
404                key, e
405            ))
406        })
407    }

409    /// Caller must ensure that the object has unexpired compliance-mode object lock.
410    pub async fn get_object<T: DeserializeOwned>(&self, key: &str) -> GuardianResult<T> {
411        let keys = self.ensure_no_duplicates_or_deletions(key).await?;
412        if keys.len() != 1 || keys[0] != key {
413            return Err(S3Error(format!(
414                "expected exactly one object for key {}, found {:?}",
415                key, keys
416            )));
417        }
418
419        self.get_object_unsafe::<T>(key).await
420    }

422    /// Note: Callers must set signing_pubkey to None only for unsigned messages.
423    pub async fn get_verified_log_record(
424        &self,
425        key: &str,
426        expected_session_id: &str,
427        signing_pubkey: Option<&GuardianPubKey>,
428    ) -> GuardianResult<VerifiedLogRecord> {
429        let log = self.get_object::<LogRecord>(key).await?;
430        if log.session_id != expected_session_id {
431            return Err(S3Error(format!("log session_id mismatch for key {}", key)));
432        }
433        match signing_pubkey {
434            Some(pk) => log.verify(pk),
435            None => log.verify_unsigned(),
436        }
437    }

439    pub async fn get_attestation(
440        &self,
441        session_id: &str,
442    ) -> GuardianResult<(Attestation, GuardianPubKey)> {
443        let key = InitLogMessage::attestation_object_key(session_id);
444        self.get_verified_log_record(&key, session_id, None)
445            .await?
446            .message
447            .into_init_log()
448            .and_then(|x| match x {
449                InitLogMessage::OIAttestationUnsigned {
450                    attestation,
451                    signing_public_key,
452                } => Some((attestation, signing_public_key)),
453                _ => None,
454            })
455            .ok_or_else(|| S3Error(format!("expected OIAttestationUnsigned at key {}", key)))
456    }

458    pub async fn get_guardian_info(
459        &self,
460        session_id: &str,
461        signing_pubkey: &GuardianPubKey,
462    ) -> GuardianResult<GuardianInfo> {
463        let key = InitLogMessage::guardian_info_object_key(session_id);
464        self.get_verified_log_record(&key, session_id, Some(signing_pubkey))
465            .await?
466            .message
467            .into_init_log()
468            .and_then(|x| match x {
469                InitLogMessage::OIGuardianInfo(info) => Some(info),
470                _ => None,
471            })
472            .ok_or_else(|| S3Error(format!("expected OIGuardianInfo at key {}", key)))
473    }

502    #[tokio::test]

504        let put_ok = mock!(Client::put_object)
505            .match_requests(|req| {
506                req.bucket() == Some("bucket")
507                    && req.key() == Some("init/session-oi-attestation-unsigned.json")
508                    && req.content_type() == Some("application/json")
509                    && req.object_lock_mode() == Some(&ObjectLockMode::Compliance)
510                    && req.object_lock_retain_until_date().is_some()
511            })
512            .then_output(|| PutObjectOutput::builder().build());
513
514        let client = mock_client!(aws_sdk_s3, RuleMode::MatchAny, &[&put_ok]);
515        let logger = mk_logger_with_client(client);
516        let object_lock_duration = Duration::from_mins(5);
