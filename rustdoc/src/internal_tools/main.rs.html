4//! Internal operator tools for the Hashi bridge.
5
6use std::path::PathBuf;
7
8use anyhow::Context;
9use anyhow::anyhow;
10use clap::Parser;
11use clap::Subcommand;
12use hashi::config::Config;
13use hashi::onchain::OnchainState;
14
15mod dump_utxos;
16mod key_recovery;
17mod sweep_utxos;
18mod utxo_csv;
19
20#[derive(Parser)]
21#[command(name = "internal-tools", about = "Internal operator tools for Hashi")]
22struct Cli {
23    #[command(subcommand)]
24    command: Commands,
25}
26
27/// Shared arguments for subcommands that connect to a Sui node.
28#[derive(Parser)]
29struct ConfigArgs {
30    /// Path to a node config TOML file (provides sui-rpc, chain-id, hashi-ids).
31    #[arg(long)]
32    config: PathBuf,
33}
34
35impl ConfigArgs {
36    fn load(&self) -> anyhow::Result<Config> {
37        let s = std::fs::read_to_string(&self.config)
38            .with_context(|| format!("failed to read config: {}", self.config.display()))?;
39        toml::from_str(&s).context("failed to parse config TOML")
40    }
41}
42
43#[derive(Subcommand)]
44enum Commands {
45    KeyRecovery {
46        #[command(flatten)]
47        config: ConfigArgs,
48        #[command(flatten)]
49        args: key_recovery::Args,
50    },
51    DumpUtxos {
52        #[command(flatten)]
53        config: ConfigArgs,
54    },
55    SweepUtxos(sweep_utxos::Args),
56}
57
58#[tokio::main]
59async fn main() -> anyhow::Result<()> {
60    tracing_subscriber::fmt::init();
61    let cli = Cli::parse();
62
63    match cli.command {
64        Commands::SweepUtxos(args) => sweep_utxos::run(args).await,
65        Commands::KeyRecovery { config, args } => {
66            let cfg = config.load()?;
67            let sui_rpc = cfg
68                .sui_rpc
69                .as_deref()
70                .ok_or_else(|| anyhow!("config missing sui-rpc"))?;
71            let chain_id = cfg
72                .sui_chain_id
73                .as_deref()
74                .ok_or_else(|| anyhow!("config missing sui-chain-id"))?;
75            println!("Connecting to Sui RPC: {sui_rpc}");
76            println!("Chain ID: {chain_id}");
77            let (onchain_state, _watcher) =
78                OnchainState::new(sui_rpc, cfg.hashi_ids(), None, None, None)
79                    .await
80                    .context("failed to connect to Sui RPC")?;
81            key_recovery::run(args, &onchain_state, chain_id).await
82        }
83        Commands::DumpUtxos { config } => {
84            let cfg = config.load()?;
85            let sui_rpc = cfg
86                .sui_rpc
87                .as_deref()
88                .ok_or_else(|| anyhow!("config missing sui-rpc"))?;
89            println!("Connecting to Sui RPC: {sui_rpc}");
90            let (onchain_state, _watcher) =
91                OnchainState::new(sui_rpc, cfg.hashi_ids(), None, None, None)
92                    .await
93                    .context("failed to connect to Sui RPC")?;
94            dump_utxos::run(&onchain_state)
95        }
96    }
97}
