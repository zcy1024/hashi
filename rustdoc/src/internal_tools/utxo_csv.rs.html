14pub fn write_csv(utxos: &[(UtxoId, Utxo)], mut w: impl Write) -> anyhow::Result<()> {
15    writeln!(w, "txid,vout,derivation_path,amount")?;
16    for (id, utxo) in utxos {
17        let deriv = utxo
18            .derivation_path
19            .map_or(String::new(), |p| p.to_string());
20        writeln!(w, "{},{},{},{}", id.txid, id.vout, deriv, utxo.amount)?;
21    }
22    Ok(())
23}

25pub fn parse_csv(path: &Path) -> anyhow::Result<Vec<Utxo>> {
26    let mut reader = csv::Reader::from_path(path).context("failed to open CSV")?;
27    let mut utxos = Vec::new();
28
29    for (i, result) in reader.records().enumerate() {
30        let row = i + 1;
31        let record = result.with_context(|| format!("failed to read CSV row {row}"))?;
32
33        let txid_str = record
34            .get(0)
35            .ok_or_else(|| anyhow!("row {row}: missing txid"))?;
36        let vout_str = record
37            .get(1)
38            .ok_or_else(|| anyhow!("row {row}: missing vout"))?;
39        let deriv_str = record
40            .get(2)
41            .ok_or_else(|| anyhow!("row {row}: missing derivation_path"))?;
42        let amount_str = record
43            .get(3)
44            .ok_or_else(|| anyhow!("row {row}: missing amount"))?;
45
46        let txid: BitcoinTxid = txid_str
47            .parse()
48            .with_context(|| format!("row {row}: bad txid"))?;
49        let vout: u32 = vout_str
50            .parse()
51            .with_context(|| format!("row {row}: bad vout"))?;
52        let derivation_path: Option<Address> = if deriv_str.is_empty() {
53            None
54        } else {
55            Some(
56                deriv_str
57                    .parse()
58                    .with_context(|| format!("row {row}: bad derivation_path"))?,
59            )
60        };
61        let amount: u64 = amount_str
62            .parse()
63            .with_context(|| format!("row {row}: bad amount"))?;
64
65        utxos.push(Utxo {
66            id: UtxoId { txid, vout },
67            amount,
68            derivation_path,
69        });
70    }
71
72    Ok(utxos)
73}
