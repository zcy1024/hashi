4use hashi::onchain::OnchainState;
5
6pub fn run(onchain_state: &OnchainState) -> anyhow::Result<()> {
7    let state = onchain_state.state();
8    let utxo_pool = &state.hashi().utxo_pool;
9    let utxos: Vec<_> = utxo_pool
10        .active_utxos()
11        .map(|(id, utxo)| (*id, utxo.clone()))
12        .collect();
13
14    eprintln!("{} active UTXOs", utxos.len());
15
16    let total_amount: u64 = utxos.iter().map(|(_, u)| u.amount).sum();
17    eprintln!(
18        "Total: {} sat ({:.8} BTC)",
19        total_amount,
20        total_amount as f64 / 1e8
21    );
22
23    crate::utxo_csv::write_csv(&utxos, std::io::stdout().lock())
24}
