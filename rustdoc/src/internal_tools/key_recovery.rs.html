31#[derive(Parser)]
32pub struct Args {
33    /// Comma-separated paths to validator DB backups.
34    #[arg(long, value_delimiter = ',')]
35    db_paths: Vec<PathBuf>,
36
37    /// Expected public key in hex (for verification).
38    #[arg(long)]
39    expected_pubkey: Option<String>,
40}
41
42pub async fn run(args: Args, onchain_state: &OnchainState, chain_id: &str) -> anyhow::Result<()> {
43    if args.db_paths.is_empty() {
44        bail!("--db-paths is required");
45    }
46
47    let source_epoch = detect_epoch(&args.db_paths[0])?;
48    let reconstruction_epoch = source_epoch + 1;
49
50    let committee = {
51        let state = onchain_state.state();
52        let committees = state.hashi().committees.committees();
53        committees
54            .get(&reconstruction_epoch)
55            .or_else(|| committees.get(&source_epoch))
56            .ok_or_else(|| {
57                anyhow!("no committee found for epoch {reconstruction_epoch} or {source_epoch}")
58            })?
59            .clone()
60    };
61
62    println!(
63        "Source epoch {source_epoch}: {} validators, fetching certificates...",
64        committee.members().len()
65    );
66
67    // Fetch certificates for the source epoch
68    let raw_certs = fetch_certificates(onchain_state, source_epoch, None)
69        .await
70        .map_err(|e| anyhow!("failed to fetch certificates: {e}"))?;
71    let certificates: Vec<CertificateV1> = raw_certs.into_iter().map(|(_, cert)| cert).collect();
72    println!(
73        "Fetched {} certificates for epoch {source_epoch}",
74        certificates.len()
75    );
76
77    if certificates.is_empty() {
78        bail!("No certificates found for epoch {source_epoch}. Try a different epoch.");
79    }
80
81    // For each validator DB, reconstruct their key shares
82    let dummy_signing_key = Bls12381PrivateKey::generate(&mut rand::thread_rng());
83    let mut all_shares: Vec<Eval<S>> = Vec::new();
84    let mut recovered_pubkey: Option<G> = None;
85
86    for (i, db_path) in args.db_paths.iter().enumerate() {
87        println!("\n=== Validator {i}: {} ===", db_path.display());
88
89        let db = Arc::new(
90            Database::open(db_path)
91                .with_context(|| format!("failed to open DB: {}", db_path.display()))?,
92        );
93        let encryption_key = db
94            .get_encryption_key(source_epoch)
95            .map_err(|e| anyhow!("failed to read encryption key: {e}"))?
96            .ok_or_else(|| {
97                anyhow!(
98                    "no encryption key found for epoch {source_epoch} in {}",
99                    db_path.display()
100                )
101            })?;
102
103        let my_enc_pk = EncryptionPublicKey::from_private_key(&encryption_key);
104        let validator_address = find_validator_by_encryption_key(&committee, &my_enc_pk)
105            .ok_or_else(|| {
106                anyhow!(
107                    "DB encryption key doesn't match any committee member (epoch {source_epoch})"
108                )
109            })?;
110        println!("  Validator address: {validator_address}");
111
112        let store = EpochPublicMessagesStore::new(db.clone(), source_epoch);
113        let session_id = SessionId::new(chain_id, reconstruction_epoch, &ProtocolType::KeyRotation);
114        let mut manager = {
115            let state = onchain_state.state();
116            let hashi = state.hashi();
117            MpcManager::new(
118                validator_address,
119                &hashi.committees,
120                session_id,
121                encryption_key,
122                dummy_signing_key.clone(),
123                Box::new(store),
124                chain_id,
125                None, // weight_divisor
126                0,    // batch_size_per_weight (unused for reconstruction)
127                None, // test_corrupt_shares_for
128            )
129            .map_err(|e| anyhow!("failed to create MpcManager: {e}"))?
130        };
131
132        // Override source_epoch to match the backed-up DB's epoch, since
133        // the on-chain epoch may have advanced past the backup.
134        manager.set_source_epoch(source_epoch);
135
136        // Pass an empty complaint cache: the recovery tool runs reconstruction
137        // once without retrying through complaint recovery, so there are no
138        // recovered outputs to reuse across attempts.
139        let outcome = manager
140            .reconstruct_previous_output(&certificates, &std::collections::HashMap::new())
141            .map_err(|e| anyhow!("reconstruction failed: {e}"))?;
142
143        match outcome {
144            ReconstructionOutcome::Success(output) => {
145                println!(
146                    "  Public key: {}",
147                    hex::encode(output.public_key.to_byte_array())
148                );
149                println!("  Shares: {}", output.key_shares.shares.len());
150                println!("  Threshold: {}", output.threshold);
151
152                if let Some(ref pk) = recovered_pubkey
153                    && *pk != output.public_key
154                {
155                    bail!("Public key mismatch between validators!");
156                }
157                recovered_pubkey = Some(output.public_key);
158                all_shares.extend(output.key_shares.shares.iter().cloned());
159            }
160            ReconstructionOutcome::NeedsDkgComplaintRecovery { dealer_address, .. } => {
161                println!(
162                    "  Warning: needs DKG complaint recovery for dealer {dealer_address}, skipping"
163                );
164            }
165            ReconstructionOutcome::NeedsRotationComplaintRecovery {
166                dealer_address,
167                share_index,
168                ..
169            } => {
170                println!(
171                    "  Warning: needs rotation complaint recovery for dealer {dealer_address} share {share_index}, skipping"
172                );
173            }
174        }
175    }
176
177    // Lagrange interpolation to recover full private key
178    let pubkey =
179        recovered_pubkey.ok_or_else(|| anyhow!("no shares recovered from any validator"))?;
180    println!("\n=== Lagrange Interpolation ===");
181    println!("Total shares collected: {}", all_shares.len());
182
183    let secret_key = lagrange_interpolate_at_zero(&all_shares)?;
184
185    // Verify
186    let recovered_pk = G::generator() * secret_key;
187    println!(
188        "Recovered public key:  {}",
189        hex::encode(recovered_pk.to_byte_array())
190    );
191    println!(
192        "Expected public key:   {}",
193        hex::encode(pubkey.to_byte_array())
194    );
195
196    if recovered_pk != pubkey {
197        bail!("VERIFICATION FAILED: recovered key does not match expected public key");
198    }
199    println!("\nVerification PASSED!");
200
201    if let Some(ref expected) = args.expected_pubkey {
202        let expected_bytes = hex::decode(expected)?;
203        let expected_pk = G::from_byte_array(
204            &expected_bytes
205                .try_into()
206                .map_err(|_| anyhow!("invalid pubkey length"))?,
207        )
208        .map_err(|e| anyhow!("invalid expected pubkey: {e}"))?;
209        if recovered_pk != expected_pk {
210            bail!("MISMATCH with --expected-pubkey!");
211        }
212        println!("Matches --expected-pubkey!");
213    }
214
215    println!(
216        "\nRecovered private key (hex): {}",
217        hex::encode(secret_key.to_byte_array())
218    );
219
220    Ok(())
221}

223fn detect_epoch(db_path: &std::path::Path) -> anyhow::Result<u64> {
224    let db = Database::open(db_path).with_context(|| {
225        format!(
226            "failed to open DB for epoch detection: {}",
227            db_path.display()
228        )
229    })?;
230    let epoch = db
231        .latest_encryption_key_epoch()
232        .map_err(|e| anyhow!("failed to scan encryption keys: {e}"))?
233        .ok_or_else(|| anyhow!("no encryption keys found in DB: {}", db_path.display()))?;
234    println!("Auto-detected source epoch: {epoch}");
235    Ok(epoch)
236}

238fn find_validator_by_encryption_key(
239    committee: &Committee,
240    enc_pk: &EncryptionPublicKey,
241) -> Option<Address> {
242    committee.members().iter().find_map(|m| {
243        if m.encryption_public_key().as_element().to_byte_array()
244            == enc_pk.as_element().to_byte_array()
245        {
246            Some(m.validator_address())
247        } else {
248            None
249        }
250    })
251}

253fn lagrange_interpolate_at_zero(shares: &[Eval<S>]) -> anyhow::Result<S> {
254    if shares.is_empty() {
255        bail!("no shares to interpolate");
256    }
257    let indices: Vec<S> = shares
258        .iter()
259        .map(|s| S::from(s.index.get() as u128))
260        .collect();
261    let mut result = S::zero();
262    for (i, share) in shares.iter().enumerate() {
263        let xi = indices[i];
264        let one = S::generator();
265        let mut numerator = one;
266        let mut denominator = one;
267        for (j, xj) in indices.iter().enumerate() {
268            if i == j {
269                continue;
270            }
271            numerator *= -*xj;
272            denominator *= xi - *xj;
273        }
274        let inv = denominator
275            .inverse()
276            .map_err(|e| anyhow!("Lagrange denominator inversion failed: {e}"))?;
277        let lagrange_coeff = numerator * inv;
278        result += share.value * lagrange_coeff;
279    }
280    Ok(result)
281}
