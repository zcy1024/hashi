73pub type DigestBytes = Vec<u8>;

75impl ShareCommitments {
76    pub fn new(commitments: Vec<ShareCommitment>) -> GuardianResult<Self> {
77        if commitments.len() != NUM_OF_SHARES {
78            return Err(InvalidInputs(format!(
79                "expected {} share commitments, got {}",
80                NUM_OF_SHARES,
81                commitments.len()
82            )));
83        }
84        let mut map = BTreeMap::new();
85        for commitment in commitments {
86            if map.insert(commitment.id, commitment.digest).is_some() {
87                return Err(InvalidInputs("duplicate share id".into()));
88            }
89        }
90        Ok(Self(map))
91    }
92
93    pub fn from_shares(shares: &[Share]) -> GuardianResult<Self> {
94        Self::new(shares.iter().map(commit_share).collect())
95    }
96
97    pub fn len(&self) -> usize {
98        self.0.len()
99    }
100
101    pub fn is_empty(&self) -> bool {
102        self.0.is_empty()
103    }
104
105    pub fn contains(&self, commitment: &ShareCommitment) -> bool {
106        self.0
107            .get(&commitment.id)
108            .is_some_and(|digest| digest == &commitment.digest)
109    }
110
111    pub fn iter(&self) -> impl Iterator<Item = ShareCommitment> + '_ {
112        self.0.iter().map(|(id, digest)| ShareCommitment {
113            id: *id,
114            digest: digest.clone(),
115        })
116    }
117}
118
119impl IntoIterator for ShareCommitments {
120    type Item = ShareCommitment;
121    type IntoIter = std::vec::IntoIter<ShareCommitment>;
122
123    fn into_iter(self) -> Self::IntoIter {
124        self.0
125            .into_iter()
126            .map(|(id, digest)| ShareCommitment { id, digest })
127            .collect::<Vec<_>>()
128            .into_iter()
129    }
130}

165/// Encrypts plaintext. Returns InvalidInputs if plaintext / aad is extraordinarily long (~2^36).
166pub fn encrypt<R: CryptoRng + RngCore>(
167    bytes: &[u8],
168    pk: &EncPubKey,
169    aad: Option<&[u8; 32]>,
170    rng: &mut R,
171) -> GuardianResult<Ciphertext> {
172    let (encapsulated_key, aes_ciphertext) =
173        hpke::single_shot_seal::<AesGcm256, HkdfSha384, X25519HkdfSha256, _>(
174            &hpke::OpModeS::Base,
175            pk,
176            &[],
177            bytes,
178            aad.unwrap_or(&[0; 32]),
179            rng,
180        )
181        .map_err(|e| InvalidInputs(format!("Encryption failed: {}", e)))?;
182    Ok(Ciphertext {
183        encapsulated_key: encapsulated_key.to_bytes().to_vec(),
184        aes_ciphertext,
185    })
186}
187
188/// Decrypts ciphertext. Returns InvalidInputs if aad is invalid.
189pub fn decrypt(
190    ciphertext: &Ciphertext,
191    sk: &EncSecKey,
192    aad: Option<&[u8; 32]>,
193) -> GuardianResult<Vec<u8>> {
194    let encapsulated_key = EncapsulatedKey::from_bytes(&ciphertext.encapsulated_key)
195        .map_err(|e| InvalidInputs(format!("Failed to deserialize encapsulated key: {}", e)))?;
196    hpke::single_shot_open::<AesGcm256, HkdfSha384, X25519HkdfSha256>(
197        &hpke::OpModeR::Base,
198        sk,
199        &encapsulated_key,
200        &[],
201        &ciphertext.aes_ciphertext,
202        aad.unwrap_or(&[0; 32]),
203    )
204    .map_err(|e| InvalidInputs(format!("Decryption failed: {}", e)))
205}

211/// Split a k256 SecretKey into shares using Shamir's secret sharing
212pub fn split_secret<R: CryptoRng + RngCore>(sk: &k256::SecretKey, rng: &mut R) -> Vec<Share> {
213    let secret = *sk.to_nonzero_scalar().as_ref();
214    let mut coefficients = vec![secret];
215    for _ in 0..(THRESHOLD - 1) {
216        coefficients.push(Scalar::random(&mut *rng))
217    }
218
219    // Evaluate
220    (1..=NUM_OF_SHARES)
221        .map(|i| NonZeroU16::new(i as u16).expect("Not zeroes!"))
222        .map(|i| Share {
223            id: i,
224            value: eval_poly(i, &coefficients),
225        })
226        .collect()
227}
228
229// Coefficients: [c0, c1, c2, c3]
230// Returns: c0 + c1 * x + c2 * x^2 + c3 * x^3
231pub fn eval_poly(pos: ShareID, coefficients: &[Scalar]) -> Scalar {
232    let x = to_scalar(pos);
233    let mut out = Scalar::ZERO;
234    let mut xpow = Scalar::ONE;
235    for c in coefficients {
236        out = out.add(&c.mul(&xpow));
237        xpow = xpow.mul(&x);
238    }
239    out
240}

242/// Combine secret shares to a secp256k1 secret key
243/// Throws an error if duplicate share IDs exist or <t shares are input
244pub fn combine_shares(shares: &[Share]) -> GuardianResult<bitcoin::secp256k1::Keypair> {
245    // Validation: ensure no duplicates
246    let mut seen_ids = std::collections::HashSet::new();
247    for share in shares {
248        if !seen_ids.insert(share.id) {
249            return Err(InvalidInputs("Duplicate share ID".into()));
250        }
251    }
252    if seen_ids.len() < THRESHOLD {
253        return Err(InvalidInputs(format!(
254            "Received only {} out of {} shares",
255            seen_ids.len(),
256            THRESHOLD
257        )));
258    }
259
260    let ids = shares.iter().map(|s| to_scalar(s.id)).collect::<Vec<_>>();
261    let mut result = Scalar::ZERO;
262    for share in shares {
263        let cur_share_id = to_scalar(share.id);
264        let numerator: Scalar = ids
265            .iter()
266            .filter(|&id| cur_share_id != *id)
267            .map(|id| id.negate())
268            .product();
269        let denominator: Scalar = ids
270            .iter()
271            .filter(|&id| cur_share_id != *id)
272            .map(|id| cur_share_id.sub(id))
273            .product();
274
275        // Lagrange basis polynomial evaluated at x=0
276        // L_i(0) = product_{j != i} (-x_j) / (x_i - x_j)
277        let lagrange_basis = numerator.mul(
278            &denominator
279                .invert()
280                .expect("Denominator is never zero because share IDs are unique"),
281        );
282        result = result.add(&share.value.mul(&lagrange_basis));
283    }
284
285    info!("Bitcoin key created with fingerprint {:x}", exp_g(&result));
286
287    // Note: Library switching works because k256's to_bytes and secp256k1's from_slice both
288    //       use big-endian representation. We are juggling between two libraries because secp256k1
289    //       does not expose the arithmetic tools needed to implement secret-sharing.
290    let sk = bitcoin::secp256k1::SecretKey::from_slice(&result.to_bytes())
291        .expect("casting secret key into secp256k1 failed");
292    Ok(bitcoin::secp256k1::Keypair::from_secret_key(&BTC_LIB, &sk))
293}

295/// Create a commitment (hash) for a share
296pub fn commit_share(share: &Share) -> ShareCommitment {
297    let commitment = ProjectivePoint::GENERATOR * share.value;
298    ShareCommitment {
299        id: share.id,
300        digest: commitment.to_bytes().to_vec(),
301    }
302}

304/// Encrypt a share with optional AAD
305pub fn encrypt_share<R: CryptoRng + RngCore>(
306    share: &Share,
307    pk: &EncPubKey,
308    aad: Option<&[u8; 32]>,
309    rng: &mut R,
310) -> EncryptedShare {
311    EncryptedShare {
312        id: share.id,
313        ciphertext: encrypt(&share.value.to_bytes(), pk, aad, rng)
314            .expect("neither plaintext nor aad are long"),
315    }
316}
317
318/// Decrypt an encrypted share with optional AAD
319pub fn decrypt_share(
320    encrypted_share: &EncryptedShare,
321    sk: &EncSecKey,
322    aad: Option<&[u8; 32]>,
323) -> GuardianResult<Share> {
324    let serialized_share = decrypt(&encrypted_share.ciphertext, sk, aad)?;
325    let result: Option<Scalar> =
326        Scalar::from_repr(*FieldBytes::from_slice(&serialized_share)).into();
327    match result {
328        Some(x) => Ok(Share {
329            id: encrypted_share.id,
330            value: x,
331        }),
332        None => Err(InvalidInputs("Failed to deserialize share".into())),
333    }
334}

340/// Methods for `Signed<T>` wrapper - signing and verification
341impl<T: Serialize + SigningIntent> GuardianSigned<T> {
342    /// Create a new signed payload (used by enclave)
343    /// Includes intent byte for domain separation to prevent cross-type signature attacks
344    pub fn new(data: T, signing_key: &SigningKey, timestamp_ms: UnixMillis) -> Self {
345        let tuple = (T::INTENT, &data, timestamp_ms);
346        let signing_payload = bcs::to_bytes(&tuple).expect("serialization should not fail");
347        let signature = signing_key.sign(&signing_payload);
348        Self {
349            data,
350            timestamp_ms,
351            signature,
352        }
353    }
354
355    /// Verify signature and extract payload
356    /// Checks intent byte to ensure signature is for the correct type
357    pub fn verify(self, pub_key: &VerificationKey) -> GuardianResult<T> {
358        let tuple = (T::INTENT, &self.data, self.timestamp_ms);
359        let msg_bytes = bcs::to_bytes(&tuple).expect("serialization should not fail");
360        pub_key
361            .verify(&self.signature, &msg_bytes)
362            .map_err(|_| InvalidInputs("signature invalid".into()))?;
363        Ok(self.data)
364    }
365}

367pub fn fingerprint(sk: &k256::SecretKey) -> CompressedPoint {
368    exp_g(&Scalar::from(sk.as_scalar_primitive()))
369}
370
371pub fn exp_g(scalar: &Scalar) -> CompressedPoint {
372    (ProjectivePoint::GENERATOR * scalar).to_bytes()
373}

381    fn test_encrypt_and_decrypt() {
382        let bytes = b"Let's encrypt some stuff!";
383        let keypair = GuardianEncKeyPair::random(&mut rand::thread_rng());
384        let aad = Some(&[0; 32]);
385        let ciphertext =
386            encrypt(bytes, keypair.public_key(), aad, &mut rand::thread_rng()).unwrap();
387        assert!(decrypt(&ciphertext, keypair.secret_key(), aad).is_ok_and(|x| x == bytes));

396    // Verify secret reconstruction with varying number of shares (0 to limit)
397    // Tests that:
398    // - With insufficient shares (< threshold): either error or wrong reconstruction
399    // - Threshold shares can reconstruct the original secret
400    // - Correct conversion to bitcoin::secp256k1::SecretKey
401    // - Full round-trip produces equivalent keys
402    #[test]
403    fn test_varying_share_count() {
404        // Start with a k256::SecretKey
405        let original_k256_sk = SecretKey::random(&mut rand::thread_rng());
406        let original_bytes = original_k256_sk.to_bytes();
407
408        // Split the secret into shares
409        let shares = split_secret(&original_k256_sk, &mut rand::thread_rng());
410
411        // Test reconstruction with varying numbers of shares from 0 to LIMIT
412        for num_shares in 0..=NUM_OF_SHARES {
413            let shares_subset = &shares[0..num_shares];
414            let result = combine_shares(shares_subset);
415
416            if num_shares < THRESHOLD {
417                // With insufficient shares, either:
418                // 1. The combine operation fails (returns error), OR
419                // 2. The combine operation succeeds but produces wrong secret
420                match result {
421                    Err(_) => {
422                        // Good: operation failed as expected
423                    }
424                    Ok(reconstructed) => {
425                        // Operation succeeded but should produce wrong secret
426                        let reconstructed_bytes = reconstructed.secret_bytes();
427                        assert_ne!(
428                            original_bytes.as_slice(),
429                            &reconstructed_bytes,
430                            "With {} shares (less than threshold {}), should not reconstruct correct secret",
431                            num_shares,
432                            THRESHOLD
433                        );
434                    }
435                }
436            } else {
437                // With threshold or more shares, reconstruction should succeed and match original
438                let reconstructed_secp_sk = result.unwrap();
439                let reconstructed_bytes = reconstructed_secp_sk.secret_bytes();
440
441                // Verify the reconstructed secret matches the original
442                assert_eq!(
443                    original_bytes.as_slice(),
444                    &reconstructed_bytes,
445                    "Reconstructed secret should match original (using {} shares)",
446                    num_shares
447                );
448            }
449        }
450    }
