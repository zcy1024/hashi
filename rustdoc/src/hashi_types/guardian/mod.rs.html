76/// All possible signing intent types.
77/// Using an enum ensures no two types can accidentally share the same intent value.
78#[repr(u8)]
79#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
80pub enum IntentType {
81    /// Intent for all LogMessage's
82    LogMessage = 0,
83    /// Intent for SetupNewKeyResponse
84    SetupNewKeyResponse = 1,
85    /// Intent for StandardWithdrawalResponse
86    StandardWithdrawalResponse = 2,
87    /// Intent for GuardianInfo
88    GuardianInfo = 3,
89}
90
91/// Trait for types that can be signed, providing domain separation via an intent.
92pub trait SigningIntent {
93    const INTENT: IntentType;
94}

222/// All log messages emitted by the guardian enclave.
223/// Uses enum discriminator for automatic domain separation between variants.
224#[derive(Debug, Serialize, Deserialize)]
225pub enum LogMessage {
226    Heartbeat { seq: u64 },
227    Init(Box<InitLogMessage>),
228    Withdrawal(Box<WithdrawalLogMessage>),
229}
230
231/// OI: operator_init
232/// PI: provisioner_init
233/// Init messages are expected to be logged in the following order:
234/// OIAttestationUnsigned -> OIGuardianInfo -> PISuccess (T times) -> PIEnclaveFullyInitialized.
235#[derive(Debug, Serialize, Deserialize)]
236pub enum InitLogMessage {
237    /// Attestation and signing public key posted in /operator_init
238    OIAttestationUnsigned {
239        attestation: Attestation,
240        signing_public_key: GuardianPubKey,
241    },
242    /// Share commitments given in /operator_init
243    OIGuardianInfo(GuardianInfo),
244    /// A successful /setup_new_key call
245    SetupNewKeySuccess {
246        encrypted_shares: Vec<EncryptedShare>,
247        share_commitments: Vec<ShareCommitment>,
248    },
249    /// A single successful /provisioner_init call (happens N times)
250    PISuccess {
251        share_id: ShareID,
252        state_hash: [u8; 32],
253    },
254    /// Threshold reached - enclave fully initialized (happens once)
255    PIEnclaveFullyInitialized,
256}

258#[derive(Debug, Serialize, Deserialize)]
259pub enum WithdrawalLogMessage {
260    /// Immediate withdraw success
261    Success {
262        txid: Txid,
263        request_data: StandardWithdrawalRequestWire,
264        request_sign: CommitteeSignature,
265        response: StandardWithdrawalResponse,
266    },
267    /// Immediate withdraw failure
268    Failure {
269        request_data: StandardWithdrawalRequestWire,
270        request_sign: CommitteeSignature,
271        error: GuardianError,
272    },
273}

455    /// Create a new ProvisionerInitRequest by encrypting the share to the enclave's public key.
456    /// In addition, it sets the state hash as AAD for the encryption effectively
457    /// allowing the enclave to trust that state is indeed coming from the KP.
458    pub fn build_from_share_and_state<R: CryptoRng + RngCore>(
459        share: &Share,
460        enclave_pub_key: &EncPubKey,
461        state: ProvisionerInitState,
462        rng: &mut R,
463    ) -> Self {
464        let state_hash = state.digest();
465        let encrypted_share = encrypt_share(share, enclave_pub_key, Some(&state_hash), rng);
466        ProvisionerInitRequest::new(encrypted_share, state)
467    }

470        &self.encrypted_share
471    }

473    pub fn state(&self) -> &ProvisionerInitState {
474        &self.state
475    }

477    pub fn into_state(self) -> ProvisionerInitState {
478        self.state
479    }

523            }
524            InitLogMessage::PIEnclaveFullyInitialized => Self::PI_FULLY_INITIALIZED.to_string(),
525        };

566impl LogMessage {
567    pub fn is_allowed_unsigned(&self) -> bool {
568        if let LogMessage::Init(init_message) = self {
569            matches!(**init_message, InitLogMessage::OIAttestationUnsigned { .. })
570        } else {
571            false
572        }
573    }
574
575    pub fn must_be_signed(&self) -> bool {
576        !self.is_allowed_unsigned()
577    }
578
579    /// The directory under which logs are written. Ends with a slash.
580    pub fn log_dir(&self, timestamp_ms: UnixMillis) -> String {
581        match self {
582            LogMessage::Init(_) => format!("{}/", S3_DIR_INIT),
583            LogMessage::Heartbeat { .. } => {
584                S3HourScopedDirectory::new(S3_DIR_HEARTBEAT, unix_millis_to_seconds(timestamp_ms))
585                    .to_string()
586            }
587            LogMessage::Withdrawal(..) => {
588                S3HourScopedDirectory::new(S3_DIR_WITHDRAW, unix_millis_to_seconds(timestamp_ms))
589                    .to_string()
590            }
591        }
592    }
593
594    /// The name of the log.
595    pub fn log_name(&self, prefix: &str) -> String {
596        match self {
597            LogMessage::Init(init_message) => init_message.log_name(prefix),
598            LogMessage::Heartbeat { seq } => format!("{}-{:020}.json", prefix, seq),
599            LogMessage::Withdrawal(withdrawal_message) => withdrawal_message.log_name(prefix),
600        }
601    }
602
603    pub fn into_init_log(self) -> Option<InitLogMessage> {
604        match self {
605            LogMessage::Init(init_message) => Some(*init_message),
606            _ => None,
607        }
608    }
609}

611impl LogRecord {
612    pub fn new(session_id: String, message: LogMessage, signing_key: &GuardianSignKeyPair) -> Self {
613        let timestamp_ms = now_timestamp_ms();
614        if message.is_allowed_unsigned() {
615            Self::unsigned(session_id, message, timestamp_ms)
616        } else {
617            Self::signed(session_id, message, signing_key, timestamp_ms)
618        }
619    }
620
621    /// Object key format:
622    /// - Init: `init/{session_id}-{suffix}.json`
623    /// - Heartbeats & Withdrawals: `{prefix}/{yyyy}/{mm}/{dd}/{hh}/{session_id}-{suffix}.json`.
624    pub fn object_key(&self) -> String {
625        let dir = self.message.log_dir(self.timestamp_ms);
626        let log_name = self.message.log_name(&self.session_id);
627        format!("{}{}", dir, log_name)
628    }
629
630    pub fn object_lock_duration(&self) -> Duration {
631        match &self.message {
632            LogMessage::Init(..) => S3_OBJECT_LOCK_DURATION_INIT,
633            LogMessage::Heartbeat { .. } => S3_OBJECT_LOCK_DURATION_HEARTBEAT,
634            LogMessage::Withdrawal(..) => S3_OBJECT_LOCK_DURATION_WITHDRAW,
635        }
636    }
637
638    fn signed(
639        session_id: String,
640        message: LogMessage,
641        signing_key: &GuardianSignKeyPair,
642        timestamp_ms: UnixMillis,
643    ) -> Self {
644        let signed = GuardianSigned::new(message, signing_key, timestamp_ms);
645        Self {
646            session_id,
647            timestamp_ms: signed.timestamp_ms,
648            message: signed.data,
649            signature: Some(signed.signature),
650        }
651    }
652
653    fn unsigned(session_id: String, message: LogMessage, timestamp_ms: UnixMillis) -> Self {
654        assert!(
655            message.is_allowed_unsigned(),
656            "message must be Init(OIAttestationUnsigned)"
657        );
658        Self {
659            session_id,
660            timestamp_ms,
661            message,
662            signature: None,
663        }
664    }
665
666    pub fn verify(self, pub_key: &GuardianPubKey) -> GuardianResult<VerifiedLogRecord> {
667        let session_id = self.session_id;
668        let timestamp_ms = self.timestamp_ms;
669        let message = self.message;
670
671        let message = if message.is_allowed_unsigned() {
672            message
673        } else {
674            let signature = self
675                .signature
676                .ok_or_else(|| InvalidInputs("missing log signature".into()))?;
677            GuardianSigned {
678                data: message,
679                timestamp_ms,
680                signature,
681            }
682            .verify(pub_key)?
683        };
684
685        Ok(VerifiedLogRecord {
686            session_id,
687            timestamp_ms,
688            message,
689        })
690    }
691
692    pub fn verify_unsigned(self) -> GuardianResult<VerifiedLogRecord> {
693        if !self.message.is_allowed_unsigned() {
694            return Err(InvalidInputs(
695                "expected unsigned log record but message requires a signature".into(),
696            ));
697        }
698        Ok(VerifiedLogRecord {
699            session_id: self.session_id,
700            timestamp_ms: self.timestamp_ms,
701            message: self.message,
702        })
703    }
704}

746/// Serializable representation of ProvisionerInitState. Used for computing its digest.
747#[derive(Serialize)]
748struct ProvisionerInitStateRepr {
749    pub committee: crate::move_types::Committee,
750    pub withdrawal_config: WithdrawalConfig,
751    pub limiter_state: LimiterState,
752    pub hashi_btc_master_pubkey: BitcoinPubkey,
753}
754
755/// Converter from T -> Self that internally validates addresses
756pub trait AddressValidation<T>: Sized {
757    fn validate_addr(value: T, network: Network) -> GuardianResult<Self>;
758}
759
760impl AddressValidation<SignedStandardWithdrawalRequestWire>
761    for HashiSigned<StandardWithdrawalRequest>
762{
763    fn validate_addr(
764        wire_value: SignedStandardWithdrawalRequestWire,
765        network: Network,
766    ) -> GuardianResult<Self> {
767        HashiSigned::<StandardWithdrawalRequest>::new(
768            wire_value.signature.epoch,
769            StandardWithdrawalRequest::validate_addr(wire_value.data, network)?,
770            &wire_value.signature.signature,
771            &wire_value.signature.bitmap,
772        )
773        .map_err(|e| InvalidInputs(format!("{:?}", e)))
774    }
775}
776
777impl AddressValidation<StandardWithdrawalRequestWire> for StandardWithdrawalRequest {
778    fn validate_addr(
779        value: StandardWithdrawalRequestWire,
780        network: Network,
781    ) -> GuardianResult<Self> {
782        let utxos = value.utxos;
783        let inputs = utxos
784            .inputs
785            .into_iter()
786            .map(|utxo| InputUTXO::from_wire(utxo, network))
787            .collect::<Result<Vec<_>, _>>()?;
788
789        let outputs = utxos
790            .outputs
791            .into_iter()
792            .map(|utxo| OutputUTXO::from_wire(utxo, network))
793            .collect::<Result<Vec<_>, _>>()?;
794
795        Ok(Self {
796            wid: value.wid,
797            utxos: TxUTXOs::new(inputs, outputs)?,
798            timestamp_secs: value.timestamp_secs,
799            seq: value.seq,
800        })
801    }
802}
803
804impl From<StandardWithdrawalRequest> for StandardWithdrawalRequestWire {
805    fn from(m: StandardWithdrawalRequest) -> Self {
806        Self {
807            wid: m.wid,
808            utxos: m.utxos.into(),
809            timestamp_secs: m.timestamp_secs,
810            seq: m.seq,
811        }
812    }
813}

815impl From<&ProvisionerInitState> for ProvisionerInitStateRepr {

821            limiter_state,
