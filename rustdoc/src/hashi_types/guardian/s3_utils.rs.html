4use crate::guardian::time_utils::UnixSeconds;
5use std::convert::TryFrom;
6use std::fmt;
7use std::time::Duration;
8use time::Date;
9use time::OffsetDateTime;
10use time::PrimitiveDateTime;
11use time::Time;
12
13pub const SECONDS_PER_HOUR: UnixSeconds = 60 * 60;
14const DIR_WRITES_COMPLETION_DELAY: Duration = Duration::from_mins(10);
15
16type Year = i32;
17type Month = u8;
18type Day = u8;
19type Hour = u8;
20
21/// An S3 directory: prefix/YYYY/MM/DD/HH.
22/// All logs emitted within an hour are stored in the same directory, e.g., logs emitted between 12-1 PM are in `<prefix>`/12 directory.
23#[derive(Clone, Debug, Eq, PartialEq)]
24pub struct S3HourScopedDirectory {
25    prefix: String,
26    year: Year,
27    month: Month,
28    day: Day,
29    hour: Hour,
30}
31
32impl S3HourScopedDirectory {
33    pub fn new(prefix: &str, t: UnixSeconds) -> Self {
34        let unix_seconds = i64::try_from(t).expect("timestamp should fit i64");
35        let datetime =
36            OffsetDateTime::from_unix_timestamp(unix_seconds).expect("timestamp should be valid");
37        Self {
38            prefix: prefix.to_string(),
39            year: datetime.year(),
40            month: u8::from(datetime.month()),
41            day: datetime.day(),
42            hour: datetime.hour(),
43        }
44    }
45
46    pub fn next_dir(&self) -> Self {
47        Self::new(
48            &self.prefix,
49            self.to_unix_seconds().saturating_add(SECONDS_PER_HOUR),
50        )
51    }
52
53    /// The time at which writes to current S3 directory finish.
54    /// DIR_WRITES_COMPLETION_DELAY accounts for any in-flight retries and clock skew.
55    pub fn write_completion_time(&self) -> UnixSeconds {
56        self.next_dir()
57            .to_unix_seconds()
58            .saturating_add(DIR_WRITES_COMPLETION_DELAY.as_secs())
59    }
60
61    pub fn to_unix_seconds(&self) -> UnixSeconds {
62        let month = time::Month::try_from(self.month).expect("month should be valid");
63        let date =
64            Date::from_calendar_date(self.year, month, self.day).expect("date should be valid");
65        let time = Time::from_hms(self.hour, 0, 0).expect("hour should be valid");
66        let ts = PrimitiveDateTime::new(date, time)
67            .assume_utc()
68            .unix_timestamp();
69        UnixSeconds::try_from(ts).expect("timestamp should be non-negative")
70    }
71}
72
73impl fmt::Display for S3HourScopedDirectory {
74    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
75        write!(
76            f,
77            "{}/{:04}/{:02}/{:02}/{:02}/",
78            self.prefix, self.year, self.month, self.day, self.hour
79        )
80    }
81}

83#[cfg(test)]
84mod tests {
85    use super::DIR_WRITES_COMPLETION_DELAY;
86    use super::S3HourScopedDirectory;
87
88    #[test]
89    fn test_epoch_directory_format() {
90        let dir = S3HourScopedDirectory::new("heartbeat", 0);
91        assert_eq!(dir.to_string(), "heartbeat/1970/01/01/00/");
92    }
93
94    #[test]
95    fn test_hour_and_day_rollover_format() {
96        let before_hour_boundary = S3HourScopedDirectory::new("withdraw", 3_599);
97        assert_eq!(before_hour_boundary.to_string(), "withdraw/1970/01/01/00/");
98
99        let next_hour = S3HourScopedDirectory::new("withdraw", 3_600);
100        assert_eq!(next_hour.to_string(), "withdraw/1970/01/01/01/");
101
102        let next_day = S3HourScopedDirectory::new("withdraw", 86_400);
103        assert_eq!(next_day.to_string(), "withdraw/1970/01/02/00/");
104    }
105
106    #[test]
107    fn test_next_dir_and_completion_time() {
108        let mut dir = S3HourScopedDirectory::new("withdraw", 3_599);
109        assert_eq!(dir.to_string(), "withdraw/1970/01/01/00/");
110        assert_eq!(dir.to_unix_seconds(), 0);
111        assert_eq!(
112            dir.write_completion_time(),
113            3_600 + DIR_WRITES_COMPLETION_DELAY.as_secs()
114        );
115
116        for i in 0..24 {
117            assert_eq!(dir.to_string(), format!("withdraw/1970/01/01/{:02}/", i));
118            dir = dir.next_dir();
119        }
120        assert_eq!(dir.to_string(), "withdraw/1970/01/02/00/");
121    }
122}
