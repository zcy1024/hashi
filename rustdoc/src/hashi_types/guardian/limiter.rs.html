4use super::GuardianError::InvalidInputs;
5use super::GuardianError::RateLimitExceeded;
6use super::GuardianResult;
7use serde::Serialize;
8
9/// Immutable configuration for the token bucket rate limiter.
10#[derive(Debug, Copy, Clone, PartialEq, Serialize)]
11pub struct LimiterConfig {
12    /// Refill rate in sats per second.
13    pub refill_rate: u64,
14    /// Maximum bucket capacity in sats.
15    pub max_bucket_capacity: u64,
16}
17
18/// Serializable state for the token bucket rate limiter.
19/// Provisioners provide this when initializing the enclave.
20#[derive(Debug, Copy, Clone, PartialEq, Serialize)]
21pub struct LimiterState {
22    /// Available tokens in sats.
23    pub num_tokens_available: u64,
24    /// Last refill timestamp in unix seconds.
25    pub last_updated_at: u64,
26    /// Next expected withdrawal sequence number.
27    pub next_seq: u64,
28}
29
30/// Token bucket rate limiter. Tokens refill linearly over time.
31///
32/// Pure data structure — concurrency is handled by the caller via a Mutex.
33pub struct RateLimiter {
34    config: LimiterConfig,
35    state: LimiterState,
36    /// Snapshot of state before the most recent `consume`, used for revert.
37    prev_state: LimiterState,
38}
39
40impl RateLimiter {
41    pub fn new(config: LimiterConfig, state: LimiterState) -> GuardianResult<Self> {
42        if state.num_tokens_available > config.max_bucket_capacity {
43            return Err(InvalidInputs(
44                "num_tokens_available exceeds max_bucket_capacity".into(),
45            ));
46        }
47        let prev_state = state;
48        Ok(Self {
49            config,
50            state,
51            prev_state,
52        })
53    }
54
55    pub fn config(&self) -> &LimiterConfig {
56        &self.config
57    }
58
59    pub fn state(&self) -> &LimiterState {
60        &self.state
61    }
62
63    pub fn next_seq(&self) -> u64 {
64        self.state.next_seq
65    }
66
67    /// Consume tokens from the bucket. Validates seq and timestamp ordering,
68    /// refills based on elapsed time, then debits the requested amount.
69    pub fn consume(&mut self, seq: u64, timestamp: u64, amount_sats: u64) -> GuardianResult<()> {
70        if seq != self.state.next_seq {
71            return Err(InvalidInputs(format!(
72                "seq mismatch: expected {}, got {}",
73                self.state.next_seq, seq
74            )));
75        }
76        if timestamp < self.state.last_updated_at {
77            return Err(InvalidInputs(format!(
78                "timestamp {} < last_updated_at {}",
79                timestamp, self.state.last_updated_at
80            )));
81        }
82
83        // Refill tokens based on elapsed time.
84        let elapsed = timestamp
85            .checked_sub(self.state.last_updated_at)
86            .expect("timestamp checked above");
87        let refilled = elapsed.saturating_mul(self.config.refill_rate);
88        let capacity = self
89            .state
90            .num_tokens_available
91            .saturating_add(refilled)
92            .min(self.config.max_bucket_capacity);
93
94        if capacity < amount_sats {
95            return Err(RateLimitExceeded);
96        }
97
98        // Snapshot for revert, then mutate.
99        self.prev_state = self.state;
100        self.state.last_updated_at = timestamp;
101        self.state.num_tokens_available = capacity - amount_sats;
102        self.state.next_seq += 1;
103        Ok(())
104    }
105
106    /// Revert a previous `consume`. Restores state to pre-consume snapshot.
107    pub fn revert(&mut self) {
108        self.state = self.prev_state;
109    }
110}

112#[cfg(test)]
113mod test {
114    use super::*;
115
116    fn make_limiter() -> (LimiterConfig, LimiterState) {
117        let config = LimiterConfig {
118            refill_rate: 1_000,
119            max_bucket_capacity: 2_000_000,
120        };
121        let state = LimiterState {
122            num_tokens_available: 0,
123            last_updated_at: 0,
124            next_seq: 0,
125        };
126
127        (config, state)
128    }
129
130    #[test]
131    fn test_basic() {
132        let (config, state) = make_limiter();
133        let mut limiter = RateLimiter::new(config, state).unwrap();
134        assert!(limiter.consume(0, 1, config.refill_rate).is_ok());
135
136        let target_amount = 1_000_000u64;
137        let num_secs_required = target_amount.div_ceil(config.refill_rate);
138        assert!(
139            limiter
140                .consume(1, num_secs_required, target_amount)
141                .is_err()
142        );
143        assert!(
144            limiter
145                .consume(1, 1 + num_secs_required, target_amount)
146                .is_ok()
147        );
148    }
149
150    #[test]
151    fn test_limits() {
152        let (config, state) = make_limiter();
153        let mut limiter = RateLimiter::new(config, state).unwrap();
154        assert!(
155            limiter
156                .consume(0, u64::MAX, config.max_bucket_capacity + 1)
157                .is_err()
158        );
159        assert!(
160            limiter
161                .consume(0, u64::MAX, config.max_bucket_capacity)
162                .is_ok()
163        );
164    }
165
166    #[test]
167    fn test_revert_restores_pre_refill_state() {
168        let (config, state) = make_limiter();
169        let mut limiter = RateLimiter::new(config, state).unwrap();
170        // Consume after refill, then revert — should restore original state.
171        limiter.consume(0, 100, 50_000).unwrap();
172        assert_eq!(limiter.state().num_tokens_available, 50_000); // 100*1000 - 50_000
173        limiter.revert();
174        assert_eq!(limiter.state().num_tokens_available, 0);
175        assert_eq!(limiter.state().last_updated_at, 0);
176        assert_eq!(limiter.state().next_seq, 0);
177    }
178
179    #[test]
180    fn test_rejects_wrong_seq_and_old_timestamp() {
181        let (config, state) = make_limiter();
182        let mut limiter = RateLimiter::new(config, state).unwrap();
183        // Wrong seq.
184        assert!(limiter.consume(1, 0, 0).is_err());
185        // Advance state.
186        limiter.consume(0, 100, 1_000).unwrap();
187        // Old timestamp.
188        assert!(limiter.consume(1, 50, 1_000).is_err());
189    }
190}
