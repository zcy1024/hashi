4use std::fmt;
5
6use base64ct::Encoding;
7
8/// Debug formatting helper that renders bytes as `TypeName("base64...")`.
9pub struct Base64<'a>(pub &'static str, pub &'a [u8]);
10
11impl fmt::Debug for Base64<'_> {
12    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
13        f.debug_tuple(self.0)
14            .field(&base64ct::Base64::encode_string(self.1))
15            .finish()
16    }
17}
